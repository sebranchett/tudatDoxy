20 Eigen::Vector3d computePartialOfUnitVectorWrtLinkEndTime(
21  const Eigen::Vector3d& vectorToReceiver,
22  const Eigen::Vector3d& unitVectorToReceiver,
23  const double linkEndDistance,
24  const Eigen::Vector3d linkEndVelocity )

31 double computePartialOfProjectedLinkEndVelocityWrtAssociatedTime(
32  const Eigen::Vector3d& vectorToReceiver,
33  const Eigen::Vector3d& projectedLinkEndVelocity,
34  const Eigen::Vector3d& variableLinkEndVelocity,
35  const Eigen::Vector3d& projectedLinkEndAcceleration,
36  const bool linkEndIsReceiver,
37  const bool projectedLinkEndIsVariableLinkEnd )
38 {
39  Eigen::Vector3d normalizedVector = vectorToReceiver.normalized( );
40  double distance = vectorToReceiver.norm( );
41 
42  return static_cast< double >( linkEndIsReceiver ? 1.0 : -1.0 ) * computePartialOfUnitVectorWrtLinkEndTime(
43  vectorToReceiver, normalizedVector, distance, variableLinkEndVelocity ).dot( projectedLinkEndVelocity ) +
44  static_cast< double >( projectedLinkEndIsVariableLinkEnd ) * normalizedVector.dot( projectedLinkEndAcceleration );
45 }

48 void OneWayDopplerDirectFirstOrderProperTimeComponentScaling::update( const std::vector< Eigen::Vector6d >& linkEndStates,
49  const std::vector< double >& times,
50  const observation_models::LinkEndType fixedLinkEnd,
51  const Eigen::VectorXd currentObservation )
52 {
53  // Get relative state
54  Eigen::Vector6d relativeState = properTimeRateModel_->getComputationPointRelativeState(
55  times, linkEndStates );
56  currentDistance_ = relativeState.segment( 0, 3 ).norm( );
57  currentGravitationalParameter_ = properTimeRateModel_->getGravitationalParameter( );
58 
59  currentLinkEndTime_ == ( linkEndWithPartial_ == observation_models::transmitter ) ? ( times.at( 0 ) ) : ( times.at( 1 ) );
60 
61  // Compute partials w.r.t. position and velocity
62  if( computeStatePartials_ )
63  {
64  partialWrPosition_ = -physical_constants::INVERSE_SQUARE_SPEED_OF_LIGHT *
65  ( 1.0 + relativity::equivalencePrincipleLpiViolationParameter ) *
66  currentGravitationalParameter_ / ( currentDistance_ * currentDistance_ ) *
67  ( relativeState.segment( 0, 3 ).normalized( ) ).transpose( );
68  partialWrtVelocity_ = physical_constants::INVERSE_SQUARE_SPEED_OF_LIGHT *
69  ( relativeState.segment( 3, 3 ) ).transpose( );
70  }
71 }

136 void OneWayDopplerScaling::update( const std::vector< Eigen::Vector6d >& linkEndStates,
137  const std::vector< double >& times,
138  const observation_models::LinkEndType fixedLinkEnd,
139  const Eigen::VectorXd currentObservation )
140 {
141  // Compute geometry.
142  double distance = ( linkEndStates.at( 1 ) - linkEndStates.at( 0 ) ).segment( 0, 3 ).norm( );
143  Eigen::Vector3d lineOfSightVector = ( linkEndStates.at( 1 ) - linkEndStates.at( 0 ) ).segment( 0, 3 ).normalized( );
144  Eigen::Vector3d receiverVelocity = linkEndStates.at( 1 ).segment( 3, 3 );
145  Eigen::Vector3d transmitterVelocity = linkEndStates.at( 0 ).segment( 3, 3 );
146 
147  double lineOfSightVelocityReceiver = observation_models::calculateLineOfSightVelocityAsCFraction< double >(
148  lineOfSightVector, receiverVelocity );
149  double lineOfSightVelocityTransmitter = observation_models::calculateLineOfSightVelocityAsCFraction< double >(
150  lineOfSightVector, transmitterVelocity );
151 
152  // Compute scaling terms of transmitter/receiver
153  double transmitterPartialScalingTerm = 0.0;
154  double receiverPartialScalingTerm = -1.0;
155 
156  double currentTaylorSeriesTerm = 1.0;
157 
158  for( int i = 1; i <= 3; i++ )
159  {
160  transmitterPartialScalingTerm += static_cast< double >( i ) * currentTaylorSeriesTerm;
161  currentTaylorSeriesTerm *= lineOfSightVelocityTransmitter;
162  receiverPartialScalingTerm -= currentTaylorSeriesTerm;
163  }
164 
165  transmitterPartialScalingTerm *= ( 1.0 - lineOfSightVelocityReceiver );
166 
167  // Compute position partial scaling term,
168  positionScalingFactor_ =
169  ( receiverVelocity.transpose( ) * receiverPartialScalingTerm +
170  transmitterVelocity.transpose( ) * transmitterPartialScalingTerm ) /
171  physical_constants::SPEED_OF_LIGHT *
172  ( Eigen::Matrix3d::Identity( ) - lineOfSightVector * lineOfSightVector.transpose( ) ) / distance;
173 
174  if( fixedLinkEnd == observation_models::receiver )
175  {
176  lightTimeEffectPositionScalingFactor_ =
177  -1.0 / ( ( physical_constants::SPEED_OF_LIGHT - lineOfSightVector.dot( transmitterVelocity ) ) *
178  physical_constants::SPEED_OF_LIGHT ) *
179  ( transmitterPartialScalingTerm *
180  computePartialOfProjectedLinkEndVelocityWrtAssociatedTime(
181  ( linkEndStates.at( 1 ) - linkEndStates.at( 0 ) ).segment( 0, 3 ),
182  transmitterVelocity, transmitterVelocity, transmitterAccelerationFunction_( times.at( 0 ) ), false, true ) +
183  receiverPartialScalingTerm *
184  computePartialOfProjectedLinkEndVelocityWrtAssociatedTime(
185  ( linkEndStates.at( 1 ) - linkEndStates.at( 0 ) ).segment( 0, 3 ),
186  receiverVelocity, transmitterVelocity, Eigen::Vector3d::Zero( ), false, false ) );
187  }
188  else if( fixedLinkEnd == observation_models::transmitter )
189  {
190  lightTimeEffectPositionScalingFactor_ =
191  1.0 / ( ( physical_constants::SPEED_OF_LIGHT - lineOfSightVector.dot( receiverVelocity ) ) *
192  physical_constants::SPEED_OF_LIGHT ) *
193  ( receiverPartialScalingTerm *
194  computePartialOfProjectedLinkEndVelocityWrtAssociatedTime(
195  ( linkEndStates.at( 1 ) - linkEndStates.at( 0 ) ).segment( 0, 3 ),
196  receiverVelocity, receiverVelocity, receiverAccelerationFunction_( times.at( 1 ) ), true, true ) +
197  transmitterPartialScalingTerm *
198  computePartialOfProjectedLinkEndVelocityWrtAssociatedTime(
199  ( linkEndStates.at( 1 ) - linkEndStates.at( 0 ) ).segment( 0, 3 ),
200  transmitterVelocity, receiverVelocity, Eigen::Vector3d::Zero( ), true, true ) );
201  }
202 
203 
204  positionScalingFactor_ += lineOfSightVector.transpose( ) * lightTimeEffectPositionScalingFactor_;
205 
206  // Compute velocity scaling terms
207  receiverVelocityScalingFactor_ = -lineOfSightVector.transpose( ) * receiverPartialScalingTerm /
208  physical_constants::SPEED_OF_LIGHT;
209  transmitterVelocityScalingFactor_ = -lineOfSightVector.transpose( ) * transmitterPartialScalingTerm /
210  physical_constants::SPEED_OF_LIGHT;
211 
212  // Update proper time scaling objects.
213  currentLinkEndType_ = fixedLinkEnd;
214 
215  if( transmitterProperTimePartials_ != nullptr )
216  {
217  transmitterProperTimePartials_->update(
218  linkEndStates, times, fixedLinkEnd, currentObservation );
219  }
220 
221  if( receiverProperTimePartials_ != nullptr )
222  {
223  receiverProperTimePartials_->update(
224  linkEndStates, times, fixedLinkEnd, currentObservation );
225  }
226

230 Eigen::Matrix< double, 1, 3 > OneWayDopplerScaling::getPositionScalingFactor( const observation_models::LinkEndType linkEndType )
231 {
232  Eigen::Matrix< double, 1, 3 > scalingFactor =
233  positionScalingFactor_ * ( ( linkEndType == observation_models::receiver ) ? ( 1.0 ) : ( -1.0 ) );
234 
235  if( transmitterProperTimePartials_ != nullptr )
236  {
237  scalingFactor += transmitterProperTimePartials_->getPositionScalingFactor( linkEndType );

239 
240  if( receiverProperTimePartials_ != nullptr )
241  {
242  scalingFactor -= receiverProperTimePartials_->getPositionScalingFactor( linkEndType );
243  }
244 
245  return scalingFactor;

269 int OneWayDopplerScaling::getProperTimeParameterDependencySize(
270  const estimatable_parameters::EstimatebleParameterIdentifier parameterType )
271 {
272  int transmitterDependencySize = 0;
273  if( transmitterProperTimePartials_ != nullptr )
274  {
275  transmitterDependencySize = transmitterProperTimePartials_->getParameterDependencySize( parameterType );
276  }
277 
278  int receiverDependencySize = 0;
279  if( receiverProperTimePartials_ != nullptr )
280  {
281  receiverDependencySize = receiverProperTimePartials_->getParameterDependencySize( parameterType );
282  }
283 
284 
285  int totalDependencySize = 0;
286  if( transmitterDependencySize == receiverDependencySize )
287  {
288  totalDependencySize = transmitterDependencySize;
289  }
290  else if( transmitterDependencySize == 0 )
291  {
292  totalDependencySize = receiverDependencySize;
293  }
294  else if( receiverDependencySize == 0 )
295  {
296  totalDependencySize = transmitterDependencySize;
297  }
298  else
299  {
300  throw std::runtime_error( "Error, proper time parameter dependency has inconsistent size" );
301  }
302  return totalDependencySize;

306 std::vector< std::pair< Eigen::Matrix< double, 1, Eigen::Dynamic >, double > > OneWayDopplerScaling::getProperTimeParameterPartial(
307  const estimatable_parameters::EstimatebleParameterIdentifier parameterType )
308 {
309  std::pair< Eigen::Matrix< double, 1, Eigen::Dynamic >, double > transmitterPartial =
310  transmitterProperTimePartials_->getProperTimeParameterPartial( parameterType );
311  std::pair< Eigen::Matrix< double, 1, Eigen::Dynamic >, double > receiverPartial =
312  receiverProperTimePartials_->getProperTimeParameterPartial( parameterType );
313 
314  std::vector< std::pair< Eigen::Matrix< double, 1, Eigen::Dynamic >, double > > totalPartial;
315  if( transmitterPartial.first.rows( ) == receiverPartial.first.rows( ) )
316  {
317  totalPartial.push_back( transmitterPartial );
318  totalPartial.push_back( receiverPartial );
319 
320  totalPartial[ totalPartial.size( ) - 1 ].first *= -1.0;
321  }
322  else if( transmitterPartial.first.rows( ) == 0 )
323  {
324  totalPartial.push_back( receiverPartial );
325  totalPartial[ totalPartial.size( ) - 1 ].first *= -1.0;
326  }
327  else if( receiverPartial.first.rows( ) == 0 )
328  {
329  totalPartial.push_back( transmitterPartial );
330  }
331  else
332  {
333  throw std::runtime_error( "Error, proper time parameter partials have inconsistent size" );
334  }
335  return totalPartial;
336 }

373 OneWayDopplerPartial::OneWayDopplerPartialReturnType OneWayDopplerPartial::calculatePartial(
374  const std::vector< Eigen::Vector6d >& states,
375  const std::vector< double >& times,
376  const observation_models::LinkEndType linkEndOfFixedTime,
377  const Eigen::Vector1d& currentObservation )
378 {
379  if( linkEndOfFixedTime != oneWayDopplerScaler_->getCurrentLinkEndType( ) )
380  {
381  throw std::runtime_error( "Error one-way doppler partial and scaling are inconsistent" );
382  }
383 
384  OneWayDopplerPartialReturnType returnPartial;
385 
386  // Iterate over all link ends
387  for( positionPartialIterator_ = positionPartialList_.begin( ); positionPartialIterator_ != positionPartialList_.end( );
388  positionPartialIterator_++ )
389  {
390  if( positionPartialIterator_->first == observation_models::transmitter )
391  {
392  currentState_ = states[ 0 ];
393  currentTime_ = times[ 0 ];
394  }
395  else if( positionPartialIterator_->first == observation_models::receiver )
396  {
397  currentState_ = states[ 1 ];
398  currentTime_ = times[ 1 ];
399  }
400 
401  // Scale position partials
402  returnPartial.push_back(
403  std::make_pair(
404  oneWayDopplerScaler_->getPositionScalingFactor( positionPartialIterator_->first ) *
405  ( positionPartialIterator_->second->calculatePartialOfPosition(
406  currentState_ , currentTime_ ) ) +
407  oneWayDopplerScaler_->getVelocityScalingFactor( positionPartialIterator_->first ) *
408  ( positionPartialIterator_->second->calculatePartialOfVelocity(
409  currentState_ , currentTime_ ) ), currentTime_ ) );
410  }
411 
412  // Add scaled light-time correcion partials.
413  for( unsigned int i = 0; i < lighTimeCorrectionPartialsFunctions_.size( ); i++ )
414  {
415 
416  returnPartial.push_back( lighTimeCorrectionPartialsFunctions_.at( i )( states, times ) );
417  returnPartial[ returnPartial.size( ) - 1 ].first *=
418  physical_constants::SPEED_OF_LIGHT * oneWayDopplerScaler_->getLightTimeCorrectionPartialScaling( );
419  }
420 
421  if( addProperTimeParameterPartials_ )
422  {
423  OneWayDopplerPartialReturnType properTimeReturnPartials = oneWayDopplerScaler_->getProperTimeParameterPartial(
424  parameterIdentifier_ );
425  for( unsigned int i = 0; i < properTimeReturnPartials.size( ); i++ )
426  {
427  returnPartial.push_back( properTimeReturnPartials.at( i ) );
428  }
429  }
430 
431  return returnPartial;
432 }
433 
434 }
