40 void LibrationPoint::computeLocationOfLibrationPoint(
41  LagrangeLibrationPoints lagrangeLibrationPoint )
42 {
43  using std::pow;
44  using std::sqrt;
45 
46  // Set functions for Newton-Raphson based on collinear libration point passed as input
47  // parameter, or computed locations directly of equilateral libration points.
48  switch( lagrangeLibrationPoint )
49  {
50  case l1:
51  {
52  // Create an object containing the function of which we whish to obtain the root from.
53  UnivariateProxyPointer rootFunction = std::make_shared< UnivariateProxy >(
54  std::bind( &LibrationPoint::computeL1LocationFunction, this, std::placeholders::_1 ) );
55 
56  // Add the first derivative of the root function.
57  rootFunction->addBinding( -1, std::bind( &LibrationPoint::
58  computeL1FirstDerivativeLocationFunction, this, std::placeholders::_1 ) );
59 
60  // Set position vector of L1 in Cartesian elements based on result of Newton-Raphson
61  // root-finding algorithm.
62  positionOfLibrationPoint_ << rootFinder->execute( rootFunction, 1.0 ), 0.0, 0.0;
63  }
64  break;
65 
66  case l2:
67  {
68  // Create an object containing the function of which we whish to obtain the root from.
69  UnivariateProxyPointer rootFunction = std::make_shared< UnivariateProxy >(
70  std::bind( &LibrationPoint::computeL2LocationFunction, this, std::placeholders::_1 ) );
71 
72  // Add the first derivative of the root function.
73  rootFunction->addBinding( -1, std::bind( &LibrationPoint::
74  computeL2FirstDerivativeLocationFunction, this, std::placeholders::_1 ) );
75 
76  // Set position vector of L1 in Cartesian elements based on result of Newton-Raphson
77  // root-finding algorithm.
78  positionOfLibrationPoint_ << rootFinder->execute( rootFunction, 1.0 ), 0.0, 0.0;
79  }
80  break;
81 
82  case l3:
83  {
84  // Create an object containing the function of which we whish to obtain the root from.
85  UnivariateProxyPointer rootFunction = std::make_shared< UnivariateProxy >(
86  std::bind( &LibrationPoint::computeL3LocationFunction, this, std::placeholders::_1 ) );
87 
88  // Add the first derivative of the root function.
89  rootFunction->addBinding( -1, std::bind( &LibrationPoint::
90  computeL3FirstDerivativeLocationFunction, this, std::placeholders::_1 ) );
91 
92  // Set position vector of L1 in Cartesian elements based on result of Newton-Raphson
93  // root-finding algorithm.
94  positionOfLibrationPoint_ << rootFinder->execute( rootFunction, -1.0 ), 0.0, 0.0;
95  }
96  break;
97 
98  case l4:
99 
100  // Set position vector of L4 in Cartesian elements.
101  positionOfLibrationPoint_.x( ) = 0.5 - massParameter;
102  positionOfLibrationPoint_.y( ) = 0.5 * sqrt( 3.0 );
103  positionOfLibrationPoint_.z( ) = 0.0;
104 
105  break;
106 
107  case l5:
108 
109  // Set position vector of L5 in Cartesian elements.
110  positionOfLibrationPoint_.x( ) = 0.5 - massParameter;
111  positionOfLibrationPoint_.y( ) = -0.5 * sqrt( 3.0 );
112  positionOfLibrationPoint_.z( ) = 0.0;
113 
114  break;
115 
116  default:
117 
118  throw std::runtime_error(
119  "The Lagrange libration point requested does not exist." );
120  };
121 }
