25 SeparatedParser::SeparatedParser( std::string separator, int numberOfFields, ... )
26  : TextParser ( false ), doTrim( true )

45 void SeparatedParser::parseLine( std::string& line )
46 {
47  // Some short hand notations.
48  using namespace parsed_data_vector_utilities;
49  typedef std::pair< FieldType, FieldValuePtr > FieldDataPair;
50 
51  // Define a new type: vector of strings.
52  typedef std::vector< std::string > vectorOfIndividualStrings;
53 
54  // Create a vector of individual strings.
55  vectorOfIndividualStrings vectorOfIndividualStrings_;
56 
57  // Split string into multiple strings based on provided separator and place in vector.
58  typedef boost::algorithm::split_iterator< std::string::iterator > string_split_iterator;
59  for( string_split_iterator stringIterator =
60  boost::algorithm::make_split_iterator( line, boost::algorithm::first_finder(
61  separator_, boost::algorithm::is_iequal( ) ) );
62  stringIterator!=string_split_iterator( );
63  ++stringIterator )
64  {
65  // Prevent empty rows due to a double separator.
66  if ( !boost::copy_range< std::string >( *stringIterator ).empty( ) )
67  {
68  vectorOfIndividualStrings_.push_back( boost::copy_range< std::string >( *stringIterator ) );
69  }
70  }
71 
72  // Verify that number of individual vectors corresponds to the specified number of fields.
73  if ( vectorOfIndividualStrings_.size( ) != numberOfFields_ )
74  {
75  std::cerr << "Number of elements in the line (" << vectorOfIndividualStrings_.size( )
76  << ") does not match the specified number of fields (" << numberOfFields_ << ")"
77  << std::endl;
78  }
79 
80  // Create a new data line
81  ParsedDataLineMapPtr currentLineData = std::make_shared< ParsedDataLineMap >(
82  std::map< FieldType, FieldValuePtr >( ) );
83 
84  // Register the data line with the global current parsed data vector
85  parsedData->push_back( currentLineData );
86 
87  // Loop over all fields, until the end position pointer is located at the end of the string
88  for ( int unsigned currentFieldNumber = 0; currentFieldNumber < numberOfFields_;
89  currentFieldNumber++ )
90  {
91  // If we need to trim whitespace, do so.
92  if ( doTrim )
93  {
94  boost::trim( vectorOfIndividualStrings_.at( currentFieldNumber ) );
95  }
96 
97  // Get the corresponding field type.
98  FieldType type( typeList.at( currentFieldNumber ) );
99 
100  // Define unit transformer.
101  std::shared_ptr< FieldTransform > transformer;
102 
103  // If type corresponds to one of the entries of the unit transformation map.
104  if ( unitTransformationMap_.find( type ) != unitTransformationMap_.end( ) )
105  {
106  // Set corresponding transformer.
107  transformer = unitTransformationMap_.find( type )->second;
108  }
109  // Else, do nothing.
110  else
111  {
112  transformer = std::shared_ptr< FieldTransform >( );
113  }
114 
115  // Store the resulting string.
116  FieldValuePtr value(
117  new FieldValue(type, vectorOfIndividualStrings_.at( currentFieldNumber ),
118  transformer ) );
119 
120  // Store the type and value in the current line data
121  currentLineData->insert( FieldDataPair( type, value ) );
122  }
123 }
