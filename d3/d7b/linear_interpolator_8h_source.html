84  if ( dataMap.size( ) == 0 )

89  // Resize data vectors of independent/dependent values.
90  independentValues_.resize( dataMap.size( ) );
91  dependentValues_.resize( dataMap.size( ) );
92 
93  // Fill data vectors with data from map.
94  int counter = 0;
95  for ( typename std::map< IndependentVariableType, DependentVariableType >::const_iterator
96  mapIterator = dataMap.begin( ); mapIterator != dataMap.end( ); mapIterator++ )
97  {
98  independentValues_[ counter ] = std::move( mapIterator->first );
99  dependentValues_[ counter ] = std::move( mapIterator->second );
100  counter++;
101  }

218  DependentVariableType interpolate( const IndependentVariableType independentVariableValue )
219  {
220  // Check whether boundary handling needs to be applied, if independent variable is beyond its defined range.
221  DependentVariableType interpolatedValue;
222  bool useValue = false;
223  this->checkBoundaryCase( interpolatedValue, useValue, independentVariableValue );
224  if( useValue )
225  {
226  return interpolatedValue;
227  }
228 
229  // Lookup nearest lower index.
230  int newNearestLowerIndex = lookUpScheme_->findNearestLowerNeighbour(
231  independentVariableValue );
232 
233  // Perform linear interpolation.
234  interpolatedValue = dependentValues_[ newNearestLowerIndex ] +
235  ( independentVariableValue - independentValues_[ newNearestLowerIndex ] ) /
236  ( independentValues_[ newNearestLowerIndex + 1 ] -
237  independentValues_[ newNearestLowerIndex ] ) *
238  ( dependentValues_[ newNearestLowerIndex + 1 ] -
239  dependentValues_[ newNearestLowerIndex ] );
240 
241  return interpolatedValue;
242  }
