37 Eigen::Vector6d convertCartesianToSphericalOrbitalState(
38  const Eigen::Vector6d& bodyFixedCartesianState )
39 {
40  Eigen::Vector6d sphericalOrbitalState;
41 
42  // Compute and set spherical position
43  Eigen::Vector3d sphericalPosition = coordinate_conversions::convertCartesianToSpherical< double >(
44  bodyFixedCartesianState.segment( 0, 3 ) );
45  sphericalOrbitalState( radiusIndex ) = sphericalPosition( 0 );
46  sphericalOrbitalState( latitudeIndex ) = mathematical_constants::PI / 2.0 - sphericalPosition( 1 );
47  sphericalOrbitalState( longitudeIndex ) = sphericalPosition( 2 );
48 
49  // Convert velocity to vertical frame.
50  Eigen::Vector3d verticalFrameVelocity =
51  reference_frames::getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
52  sphericalOrbitalState( longitudeIndex ), sphericalOrbitalState( latitudeIndex ) ) *
53  bodyFixedCartesianState.segment( 3, 3 );
54 
55  // Set velocity in spherical orbital state.
56  sphericalOrbitalState( speedIndex ) = verticalFrameVelocity.norm( );
57  sphericalOrbitalState( flightPathIndex ) = calculateFlightPathAngle( verticalFrameVelocity );
58  sphericalOrbitalState( headingAngleIndex ) = calculateHeadingAngle( verticalFrameVelocity );
59 
60  return sphericalOrbitalState;
