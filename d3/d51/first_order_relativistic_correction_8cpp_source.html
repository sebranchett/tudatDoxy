27  const double transmissionTime,
28  const double receptionTime )
29 {
30  // Retrieve ppn parameter gamma.
31  double ppnParameterGamma = ppnParameterGammaFunction_( );
32 
33  // Initialize correction to zero.
34  currentTotalLightTimeCorrection_ = 0.0;
35 
36  double evaluationTime = TUDAT_NAN;
37  // Iterate over all gravitating bodies.
38  for( unsigned int i = 0; i < perturbingBodyStateFunctions_.size( ); i++ )
39  {
40  evaluationTime = transmissionTime + lightTimeEvaluationContribution_.at( i ) * ( receptionTime - transmissionTime );
41  // Calculate correction due to current body and add to total.
42  currentLighTimeCorrectionComponents_[ i ] = relativity::calculateFirstOrderLightTimeCorrectionFromCentralBody(
43  perturbingBodyGravitationalParameterFunctions_[ i ]( ),
44  transmitterState.segment( 0, 3 ), receiverState.segment( 0, 3 ),
45  perturbingBodyStateFunctions_[ i ]( evaluationTime ).segment( 0, 3 ),
46  ppnParameterGamma );
47  currentTotalLightTimeCorrection_ += currentLighTimeCorrectionComponents_[ i ];
48  }
49

51 }
