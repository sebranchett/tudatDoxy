95 SolarActivityDataMap readSolarActivityData( std::string filePath )
96 {
97  // Data Vector container
98  tudat::input_output::parsed_data_vector_utilities::ParsedDataVectorPtr parsedDataVector;
99 
100  // datafile Parser and Extractor
101  tudat::input_output::solar_activity::ParseSolarActivityData solarActivityParser;
102  tudat::input_output::solar_activity::ExtractSolarActivityData solarActivityExtractor;
103 
104 
105  // Open dataFile and Parse
106  std::ifstream dataFile;
107  dataFile.open( filePath.c_str( ), std::ifstream::in );
108  parsedDataVector = solarActivityParser.parse( dataFile );
109  dataFile.close( );
110 
111  int numberOfLines = parsedDataVector->size( );
112  SolarActivityDataMap dataMap;
113  double julianDate = TUDAT_NAN;
114 
115  // Save each line to datamap
116  for(int i = 0 ; i < numberOfLines ; i++ ){
117  julianDate = tudat::basic_astrodynamics::convertCalendarDateToJulianDay(
118  solarActivityExtractor.extract( parsedDataVector->at( i ) )->year,
119  solarActivityExtractor.extract( parsedDataVector->at( i ) )->month,
120  solarActivityExtractor.extract( parsedDataVector->at( i ) )->day,
121  0, 0, 0.0 ) ;
122  dataMap[ julianDate ] = solarActivityExtractor.extract( parsedDataVector->at( i ) ) ;
123  }
124 
125  return dataMap;
126 
127 }
