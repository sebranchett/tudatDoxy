729 class AccelerationLimitedThrottleGuidance: public ThrustInputParameterGuidance
730 {
731 public:
732 
734 
745  AccelerationLimitedThrottleGuidance(
746  const NamedBodyMap& bodyMap,
747  const std::string nameOfBodyWithGuidance,
748  const std::string nameOfCentralBody,
749  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
750  const std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator,
751  const double maximumAcceleration ): ThrustInputParameterGuidance( 1, 0, true, 0 ),
752  bodyWithGuidance_( bodyMap.at( nameOfBodyWithGuidance ) ), thrustInterpolator_( thrustInterpolator ),
753  maximumAcceleration_( maximumAcceleration )
754  {
755  // Split independent variables into environmental/guidance.
756  int numberOfThrottles = 0;
757  std::vector< propulsion::ThrustIndependentVariables > guidanceFreeIndependentVariables;
758  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
759  {
760  if( independentVariables.at( i ) == propulsion::throttle_dependent_thrust )
761  {
762  numberOfThrottles ++;
763  }
764  else if( independentVariables.at( i ) == propulsion::guidance_input_dependent_thrust )
765  {
766  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, guidance input not supported" );
767  }
768  else
769  {
770  guidanceFreeIndependentVariables.push_back( independentVariables.at( i ) );
771  }
772  }
773 
774  // Check input consistency.
775  if( numberOfThrottles > 1 )
776  {
777  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, multiple throttles detected" );
778  }
779 
780  // Create thrust input.
781  if( bodyWithGuidance_->getFlightConditions( ) == nullptr && nameOfCentralBody != "" )
782  {
783  bodyWithGuidance_->setFlightConditions(
784  createAtmosphericFlightConditions( bodyWithGuidance_,
785  bodyMap.at( nameOfCentralBody ),
786  nameOfBodyWithGuidance,
787  nameOfCentralBody ) );
788  }
789  thrustInputFunctions_ = getPropulsionInputVariables(
790  bodyWithGuidance_, guidanceFreeIndependentVariables );
791 
792  currentThrustInput_.resize( thrustInputFunctions_.size( ) );
793  }
794 
796 
800  void updateGuidanceParameters( )
801  {
802  // Compute environmental input variables for thrust.
803  for( unsigned int i = 0; i < thrustInputFunctions_.size( ); i++ )
804  {
805  currentThrustInput_[ i ] = thrustInputFunctions_.at( i )( );
806  }
807 
808  // Compute maximum thrust
809  double currentThrust = thrustInterpolator_->interpolate( currentThrustInput_ );
810  double currentMass = bodyWithGuidance_->getBodyMass( );
811 
812  // Set throttle (to < 1 if necessary)
813  if( currentThrust / currentMass < maximumAcceleration_ )
814  {
815  currentThrustGuidanceParameters_[ 0 ] = 1.0;
816  }
817  else
818  {
819  currentThrustGuidanceParameters_[ 0 ] = maximumAcceleration_ * currentMass / currentThrust;
820  }
821  }
822 
823 private:
824 
826  std::shared_ptr< simulation_setup::Body > bodyWithGuidance_;
827 
829  std::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator_;
830 
832  double maximumAcceleration_;
833 
835  std::vector< std::function< double( ) > > thrustInputFunctions_;
836 
838  std::vector< double > currentThrustInput_;
839 
840 };
