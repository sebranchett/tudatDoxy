38 template< int ObservationSize = 1 >
39 void performObservationParameterEstimationClosureForSingleModelSet(
40  const std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameter,
41  const std::shared_ptr< ObservationBias< ObservationSize > > observationBias,
42  const LinkEnds linkEnds,
43  const ObservableType observableType )
44 {
45  ObservationBiasTypes biasType = getObservationBiasType( observationBias );
46 
47  // Check if bias type is multi-type
48  if( biasType == multiple_observation_biases )
49  {
50  // Test input consistency
51  std::shared_ptr< MultiTypeObservationBias< ObservationSize > > multiTypeBias =
52  std::dynamic_pointer_cast< MultiTypeObservationBias< ObservationSize > >( observationBias );
53  if( multiTypeBias == nullptr )
54  {
55  throw std::runtime_error( "Error, cannot perform bias closure for multi-type bias, inconsistent bias types" );
56  }
57 
58  // Perform closure for each constituent bias object
59  for( unsigned int i = 0; i < multiTypeBias->getBiasList( ).size( ); i++ )
60  {
61  performObservationParameterEstimationClosureForSingleModelSet(
62  parameter, multiTypeBias->getBiasList( ).at( i ), linkEnds, observableType );
63  }
64  }
65  else
66  {
67  // Check bias type
68  switch( parameter->getParameterName( ).first )
69  {
70  case estimatable_parameters::constant_additive_observation_bias:
71  {
72  // Test input consistency
73  std::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > biasParameter =
74  std::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
75  parameter );
76  if( biasParameter == nullptr )
77  {
78  throw std::runtime_error( "Error, cannot perform bias closure for additive bias, inconsistent bias types" );
79  }
80 
81  // Check if bias object is of same type as estimated parameter
82  std::shared_ptr< ConstantObservationBias< ObservationSize > > constantBiasObject =
83  std::dynamic_pointer_cast< ConstantObservationBias< ObservationSize > >( observationBias );
84  if( constantBiasObject != nullptr )
85  {
86  // Check if bias and parameter link properties are equal
87  if( linkEnds == biasParameter->getLinkEnds( ) &&
88  observableType == biasParameter->getObservableType( ) )
89  {
90  biasParameter->setObservationBiasFunctions(
91  std::bind( &ConstantObservationBias< ObservationSize >::getTemplateFreeConstantObservationBias,
92  constantBiasObject ),
93  std::bind( &ConstantObservationBias< ObservationSize >::resetConstantObservationBiasTemplateFree,
94  constantBiasObject, std::placeholders::_1 ) );
95  }
96  }
97  break;
98  }
99  case estimatable_parameters::arcwise_constant_additive_observation_bias:
100  {
101 
102  // Test input consistency
103  std::shared_ptr< estimatable_parameters::ArcWiseObservationBiasParameter > biasParameter =
104  std::dynamic_pointer_cast< estimatable_parameters::ArcWiseObservationBiasParameter >(
105  parameter );
106  if( biasParameter == nullptr )
107  {
108  throw std::runtime_error( "Error, cannot perform bias closure for arc-wise additive bias, inconsistent bias types" );
109  }
110 
111  // Check if bias object is of same type as estimated parameter
112  std::shared_ptr< ConstantArcWiseObservationBias< ObservationSize > > constantBiasObject =
113  std::dynamic_pointer_cast< ConstantArcWiseObservationBias< ObservationSize > >( observationBias );
114  if( constantBiasObject != nullptr )
115  {
116  // Check if bias and parameter link properties are equal
117  if( ( linkEnds == biasParameter->getLinkEnds( ) ) &&
118  ( observableType == biasParameter->getObservableType( ) ) &&
119  ( constantBiasObject->getLinkEndIndexForTime( ) == biasParameter->getLinkEndIndex( ) ) &&
120  ( constantBiasObject->getArcStartTimes( ).size( ) == biasParameter->getArcStartTimes( ).size( ) ) )
121  {
122  bool doTimesMatch = true;
123  for( unsigned int i = 0; i < constantBiasObject->getArcStartTimes( ).size( ); i++ )
124  {
125  if( std::fabs( constantBiasObject->getArcStartTimes( ).at( i ) -
126  biasParameter->getArcStartTimes( ).at( i ) ) > std::max(
127  1.0E-15 * std::fabs( constantBiasObject->getArcStartTimes( ).at( i ) ),
128  1.0E-15 * std::fabs( biasParameter->getArcStartTimes( ).at( i ) ) ) )
129  {
130  doTimesMatch = false;
131  }
132  }
133 
134  if( doTimesMatch == true )
135  {
136  biasParameter->setObservationBiasFunctions(
137  std::bind( &ConstantArcWiseObservationBias< ObservationSize >::getTemplateFreeConstantObservationBias,
138  constantBiasObject ),
139  std::bind( &ConstantArcWiseObservationBias< ObservationSize >::resetConstantObservationBiasTemplateFree,
140  constantBiasObject, std::placeholders::_1 ) );
141  biasParameter->setLookupScheme( constantBiasObject->getLookupScheme( ) );
142  }
143  }
144  }
145  break;
146  }
147  case estimatable_parameters::constant_relative_observation_bias:
148  {
149  // Test input consistency
150  std::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > biasParameter =
151  std::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
152  parameter );
153  if( biasParameter == nullptr )
154  {
155  throw std::runtime_error( "Error, cannot perform bias closure for additive bias, inconsistent bias types" );
156  }
157 
158  // Check if bias object is of same type as estimated parameter
159  std::shared_ptr< ConstantRelativeObservationBias< ObservationSize > > constantBiasObject =
160  std::dynamic_pointer_cast< ConstantRelativeObservationBias< ObservationSize > >( observationBias );
161  if( constantBiasObject != nullptr )
162  {
163  // Check if bias and parameter link properties are equal
164  if( linkEnds == biasParameter->getLinkEnds( ) &&
165  observableType == biasParameter->getObservableType( ) )
166  {
167  biasParameter->setObservationBiasFunctions(
168  std::bind( &ConstantRelativeObservationBias< ObservationSize >::getTemplateFreeConstantObservationBias,
169  constantBiasObject ),
170  std::bind( &ConstantRelativeObservationBias< ObservationSize >::resetConstantObservationBiasTemplateFree,
171  constantBiasObject, std::placeholders::_1 ) );
172  }
173  }
174  break;
175  }
176  case estimatable_parameters::arcwise_constant_relative_observation_bias:
177  {
178 
179  // Test input consistency
180  std::shared_ptr< estimatable_parameters::ArcWiseObservationBiasParameter > biasParameter =
181  std::dynamic_pointer_cast< estimatable_parameters::ArcWiseObservationBiasParameter >(
182  parameter );
183  if( biasParameter == nullptr )
184  {
185  throw std::runtime_error( "Error, cannot perform bias closure for arc-wise relative bias, inconsistent bias types" );
186  }
187 
188  // Check if bias object is of same type as estimated parameter
189  std::shared_ptr< ConstantRelativeArcWiseObservationBias< ObservationSize > > constantBiasObject =
190  std::dynamic_pointer_cast< ConstantRelativeArcWiseObservationBias< ObservationSize > >( observationBias );
191  if( constantBiasObject != nullptr )
192  {
193  // Check if bias and parameter link properties are equal
194  if( ( linkEnds == biasParameter->getLinkEnds( ) ) &&
195  ( observableType == biasParameter->getObservableType( ) ) &&
196  ( constantBiasObject->getLinkEndIndexForTime( ) == biasParameter->getLinkEndIndex( ) ) &&
197  ( constantBiasObject->getArcStartTimes( ).size( ) == biasParameter->getArcStartTimes( ).size( ) ) )
198  {
199  bool doTimesMatch = true;
200  for( unsigned int i = 0; i < constantBiasObject->getArcStartTimes( ).size( ); i++ )
201  {
202  if( std::fabs( constantBiasObject->getArcStartTimes( ).at( i ) -
203  biasParameter->getArcStartTimes( ).at( i ) ) > std::max(
204  1.0E-15 * std::fabs( constantBiasObject->getArcStartTimes( ).at( i ) ),
205  1.0E-15 * std::fabs( biasParameter->getArcStartTimes( ).at( i ) ) ) )
206  {
207  doTimesMatch = false;
208  }
209  }
210 
211  if( doTimesMatch == true )
212  {
213  biasParameter->setObservationBiasFunctions(
214  std::bind( &ConstantRelativeArcWiseObservationBias< ObservationSize >::getTemplateFreeConstantObservationBias,
215  constantBiasObject ),
216  std::bind( &ConstantRelativeArcWiseObservationBias< ObservationSize >::resetConstantObservationBiasTemplateFree,
217  constantBiasObject, std::placeholders::_1 ) );
218  biasParameter->setLookupScheme( constantBiasObject->getLookupScheme( ) );
219  }
220  }
221  }
222  break;
223  }
224  default:
225  std::string errorMessage = "Error when closing observation bias/estimation loop, did not recognize bias type " +
226  std::to_string( parameter->getParameterName( ).first );
227  throw std::runtime_error( errorMessage );
228 
229  }
230  }

239 template< int ObservationSize = 1, typename ObservationScalarType = double, typename TimeType = double >
240 void performObservationParameterEstimationClosure(
241  std::shared_ptr< ObservationSimulator< ObservationSize, ObservationScalarType, TimeType > > observationSimulator ,
242  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > >
243  parametersToEstimate )
244 {
245  // Retrieve observation models and parameter
246  std::map< LinkEnds, std::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > >
247  observationModels = observationSimulator->getObservationModels( );
248  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > > vectorParameters =
249  parametersToEstimate->getEstimatedVectorParameters( );
250 
251  // Retrieve estimated bias parameters.
252  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > > vectorBiasParameters;
253  for( unsigned int i = 0; i < vectorParameters.size( ); i++ )
254  {
255  if( estimatable_parameters::isParameterObservationLinkProperty( vectorParameters.at( i )->getParameterName( ).first ) )
256  {
257  vectorBiasParameters.push_back( vectorParameters.at( i ) );
258  }
259  }
260 
261  if( vectorBiasParameters.size( ) > 0 )
262  {
263  // Retrieve bias objects
264  std::map< LinkEnds, std::shared_ptr< ObservationBias< ObservationSize > > > observationBiases =
265  extractObservationBiasList( observationModels );
266 
267  // Iterate over all combinations of parameters and biases and perform closure for each (if needed)
268  for( unsigned int i = 0; i < vectorBiasParameters.size( ); i++ )
269  {
270  for( typename std::map< LinkEnds, std::shared_ptr< ObservationBias< ObservationSize > > >::const_iterator
271  biasIterator = observationBiases.begin( ); biasIterator != observationBiases.end( ); biasIterator++ )
272  {
273  performObservationParameterEstimationClosureForSingleModelSet(
274  vectorBiasParameters.at( i ), biasIterator->second, biasIterator->first,
275  observationSimulator->getObservableType( ) );
276  }
277  }
278  }
279 }

292 template< int ObservationSize = 1, typename ObservationScalarType, typename TimeType >
293 std::shared_ptr< ObservationManagerBase< ObservationScalarType, TimeType > > createObservationManager(
294  const ObservableType observableType,
295  const std::map< LinkEnds, std::shared_ptr< ObservationSettings > > settingsPerLinkEnds,
296  const simulation_setup::NamedBodyMap &bodyMap,
297  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > >
298  parametersToEstimate,
299  const std::shared_ptr< propagators::CombinedStateTransitionAndSensitivityMatrixInterface >
300  stateTransitionMatrixInterface )
301 {
302  using namespace observation_models;
303  using namespace observation_partials;
304 
305  // Create observation simulator
306  std::shared_ptr< ObservationSimulator< ObservationSize, ObservationScalarType, TimeType > > observationSimulator =
307  createObservationSimulator< ObservationSize, ObservationScalarType, TimeType >(
308  observableType, settingsPerLinkEnds, bodyMap );
309 
310  performObservationParameterEstimationClosure(
311  observationSimulator, parametersToEstimate );
312 
313  // Create observation partials for all link ends/parameters
314  std::shared_ptr< ObservationPartialCreator< ObservationSize, ObservationScalarType, TimeType > > observationPartialCreator =
315  std::make_shared< ObservationPartialCreator< ObservationSize, ObservationScalarType, TimeType > >( );
316  std::map< LinkEnds, std::pair< std::map< std::pair< int, int >,
317  std::shared_ptr< ObservationPartial< ObservationSize > > >,
318  std::shared_ptr< PositionPartialScaling > > > observationPartialsAndScaler;
319  if( parametersToEstimate != nullptr )
320  {
321  observationPartialsAndScaler =
322  observationPartialCreator->createObservationPartials(
323  observableType, observationSimulator->getObservationModels( ), bodyMap, parametersToEstimate );
324  }
325 
326  // Split position partial scaling and observation partial objects.
327  std::map< LinkEnds, std::map< std::pair< int, int >,
328  std::shared_ptr< observation_partials::ObservationPartial< ObservationSize > > > > observationPartials;
329  std::map< LinkEnds, std::shared_ptr< observation_partials::PositionPartialScaling > > observationPartialScalers;
330  splitObservationPartialsAndScalers( observationPartialsAndScaler, observationPartials, observationPartialScalers );
331 
332 
333  return std::make_shared< ObservationManager< ObservationSize, ObservationScalarType, TimeType > >(
334  observableType, observationSimulator, observationPartials,
335  observationPartialScalers, stateTransitionMatrixInterface );
336 }

350 template< typename ObservationScalarType, typename TimeType >
351 std::shared_ptr< ObservationManagerBase< ObservationScalarType, TimeType > > createObservationManagerBase(
352  const ObservableType observableType,
353  const std::map< LinkEnds, std::shared_ptr< ObservationSettings > > settingsPerLinkEnds,
354  const simulation_setup::NamedBodyMap &bodyMap,
355  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > > parametersToEstimate,
356  const std::shared_ptr< propagators::CombinedStateTransitionAndSensitivityMatrixInterface > stateTransitionMatrixInterface )
357 {
358  std::shared_ptr< ObservationManagerBase< ObservationScalarType, TimeType > > observationManager;
359  switch( observableType )
360  {
361  case one_way_range:
362  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
363  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
364  stateTransitionMatrixInterface );
365  break;
366  case n_way_range:
367  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
368  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
369  stateTransitionMatrixInterface );
370  break;
371  case one_way_doppler:
372  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
373  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
374  stateTransitionMatrixInterface );
375  break;
376  case two_way_doppler:
377  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
378  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
379  stateTransitionMatrixInterface );
380  break;
381  case one_way_differenced_range:
382  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
383  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
384  stateTransitionMatrixInterface );
385  break;
386  case angular_position:
387  observationManager = createObservationManager< 2, ObservationScalarType, TimeType >(
388  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
389  stateTransitionMatrixInterface );
390  break;
391  case position_observable:
392  observationManager = createObservationManager< 3, ObservationScalarType, TimeType >(
393  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
394  stateTransitionMatrixInterface );
395  break;
396  case euler_angle_313_observable:
397  observationManager = createObservationManager< 3, ObservationScalarType, TimeType >(
398  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
399  stateTransitionMatrixInterface );
400  break;
401  case velocity_observable:
402  observationManager = createObservationManager< 3, ObservationScalarType, TimeType >(
403  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
404  stateTransitionMatrixInterface );
405  break;
406  default:
407  throw std::runtime_error(
408  "Error when making observation manager, could not identify observable type " +
409  std::to_string( observableType ) );
410  }
411  return observationManager;
412 }
