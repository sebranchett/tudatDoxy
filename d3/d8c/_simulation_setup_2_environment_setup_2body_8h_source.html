302  template< typename StateScalarType = double, typename TimeType = double >
303  void setStateFromEphemeris( const TimeType& time )
304  {
305  if( !( static_cast< Time >( time ) == timeOfCurrentState_ ) )
306  {
307  // If body is not global frame origin, set state.
308  if( bodyIsGlobalFrameOrigin_ == 0 )
309  {
310  if( sizeof( StateScalarType ) == 8 )
311  {
312  currentState_ =
313  ( bodyEphemeris_->getTemplatedStateFromEphemeris< StateScalarType, TimeType >( time ) +
314  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ) ).
315  template cast< double >( );
316  currentLongState_ = currentState_.template cast< long double >( );
317  }
318  else
319  {
320  currentLongState_ =
321  ( bodyEphemeris_->getTemplatedStateFromEphemeris< StateScalarType, TimeType >( time ) +
322  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ) ).
323  template cast< long double >( );
324  currentState_ = currentLongState_.template cast< double >( );
325  }
326  }
327  // If body is global frame origin, set state to zeroes, and barycentric state value.
328  else if( bodyIsGlobalFrameOrigin_ == 1 )
329  {
330  currentState_.setZero( );
331  currentLongState_.setZero( );
332 
333  if( sizeof( StateScalarType ) == 8 )
334  {
335  currentBarycentricState_ =
336  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ).
337  template cast< double >( );
338  currentBarycentricLongState_ = currentBarycentricState_.template cast< long double >( );
339  }
340  else
341  {
342  currentBarycentricLongState_ =
343  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ).
344  template cast< long double >( );
345  currentBarycentricState_ = currentBarycentricLongState_.template cast< double >( );
346  }
347  }
348  else
349  {
350  throw std::runtime_error( "Error when setting body state, global origin not yet defined." );
351  }
352 
353  timeOfCurrentState_ = static_cast< TimeType >( time );
354  }
355  }
