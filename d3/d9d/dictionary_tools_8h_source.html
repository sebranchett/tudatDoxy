115 template< typename DataType >
116 DataType extractParameterValue( const DataLineIterator& firstDataLine,
117  const DataLineIterator& lastDataLine,
118  const DictionaryIterator& dictionaryEntry,
119  const DataType& defaultValue = DataType( ),
120  const std::function< DataType( DataType ) >& convert
121  = &convertDummy< DataType > )
122 {
123  // Attempt to match dictionary entry with any data lines.
124  DataLineIterator parsedDataVectorIterator
125  = std::find_if( firstDataLine, lastDataLine, DictionaryComparer( *dictionaryEntry ) );
126 
127  // If the data line found is the last data line (typically iterator to the end of the
128  // container), then check if the dictionary entry is required.
129  if ( parsedDataVectorIterator == lastDataLine )
130  {
131  // If the entry is required, throw an error indicating that the parameter cannot be found
132  // in the input stream.
133  if ( ( *dictionaryEntry )->isRequired )
134  {
135  throw std::runtime_error(
136  "Required parameter " + ( *dictionaryEntry )->parameterName
137  + " not found in input stream! " );
138  }
139 
140  // Else, return the default value specified.
141  else
142  {
143  return defaultValue;
144  }
145  }
146 
147  // Else, set the flag in the dictionary entry indicating that it is extracted to true and
148  // return the extract value using the getField() function.
149  else
150  {
151  ( *dictionaryEntry )->isExtracted = true;
152 
153  return convert( parsed_data_vector_utilities::getField< DataType >(
154  *parsedDataVectorIterator, field_types::general::parameterValue ) );
155  }
156 
157  // Dummy return to ensure no warnings are output.
158  return DataType( );
159 }
