34 double getConditionNumberOfDecomposedMatrix( const Eigen::JacobiSVD< Eigen::MatrixXd >& singularValueDecomposition )
35 {
36  Eigen::VectorXd singularValues = singularValueDecomposition.singularValues( );
37  return singularValues( 0 ) / singularValues( singularValues.rows( ) - 1 );

41 Eigen::VectorXd solveSystemOfEquationsWithSvd( const Eigen::MatrixXd matrixToInvert,
42  const Eigen::VectorXd rightHandSideVector,
43  const bool checkConditionNumber,
44  const double maximumAllowedConditionNumber )
45 {
46  Eigen::JacobiSVD< Eigen::MatrixXd > svdDecomposition = matrixToInvert.jacobiSvd(
47  Eigen::ComputeThinU | Eigen::ComputeThinV );
48  if( checkConditionNumber )
49  {
50  double conditionNumber = getConditionNumberOfDecomposedMatrix( svdDecomposition );
51 
52  if( conditionNumber > maximumAllowedConditionNumber )
53  {
54  std::cerr << "Warning when performing least squares, condition number is " << conditionNumber << std::endl;
55  }
56  }
57  return svdDecomposition.solve( rightHandSideVector );
58 }

61 Eigen::MatrixXd multiplyInformationMatrixByDiagonalWeightMatrix(
62  const Eigen::MatrixXd& informationMatrix,
63  const Eigen::VectorXd& diagonalOfWeightMatrix )
64 {
65  Eigen::MatrixXd weightedInformationMatrix = Eigen::MatrixXd::Zero( informationMatrix.rows( ), informationMatrix.cols( ) );
66 
67  for( int i = 0; i < informationMatrix.cols( ); i++ )
68  {
69  weightedInformationMatrix.block( 0, i, informationMatrix.rows( ), 1 ) =
70  informationMatrix.block( 0, i, informationMatrix.rows( ), 1 ).cwiseProduct( diagonalOfWeightMatrix );
71  }
72 
73  return weightedInformationMatrix;
74 }

98 std::pair< Eigen::VectorXd, Eigen::MatrixXd > performLeastSquaresAdjustmentFromInformationMatrix(
99  const Eigen::MatrixXd& informationMatrix,
100  const Eigen::VectorXd& observationResiduals,
101  const Eigen::VectorXd& diagonalOfWeightMatrix,
102  const Eigen::MatrixXd& inverseOfAPrioriCovarianceMatrix,
103  const bool checkConditionNumber,
104  const double maximumAllowedConditionNumber,
105  const Eigen::MatrixXd& constraintMultiplier,
106  const Eigen::VectorXd& constraintRightHandside )
107 {
108 // std::cout<<"Residuals "<<observationResiduals.transpose( )<<std::endl;
109 // std::cout<<"Weight diag. "<<diagonalOfWeightMatrix.transpose( )<<std::endl;
110 // std::cout<<"Partials "<<informationMatrix.transpose( )<<std::endl;
111 
112  Eigen::VectorXd rightHandSide = informationMatrix.transpose( ) *
113  ( diagonalOfWeightMatrix.cwiseProduct( observationResiduals ) );
114  Eigen::MatrixXd inverseOfCovarianceMatrix = calculateInverseOfUpdatedCovarianceMatrix(
115  informationMatrix, diagonalOfWeightMatrix, inverseOfAPrioriCovarianceMatrix );
116 
117  // Add constraints to inverse covariance matrix if required
118  if( constraintMultiplier.rows( ) != 0 )
119  {
120  if( constraintMultiplier.rows( ) != constraintRightHandside.rows( ) )
121  {
122  throw std::runtime_error( "Error when performing constrained least-squares, constraints are incompatible" );
123  }
124 
125  if( constraintMultiplier.cols( ) != informationMatrix.cols( ) )
126  {
127  throw std::runtime_error( "Error when performing constrained least-squares, constraints are incompatible with partials" );
128  }
129 
130  int numberOfConstraints = constraintMultiplier.rows( );
131  int numberOfParameters = constraintMultiplier.cols( );
132 
133  inverseOfCovarianceMatrix.conservativeResize(
134  numberOfParameters + numberOfConstraints, numberOfParameters + numberOfConstraints );
135  inverseOfCovarianceMatrix.block( numberOfParameters, 0, numberOfConstraints, numberOfParameters ) =
136  constraintMultiplier;
137  inverseOfCovarianceMatrix.block( 0, numberOfParameters, numberOfParameters, numberOfConstraints ) =
138  constraintMultiplier.transpose( );
139  inverseOfCovarianceMatrix.block(
140  numberOfParameters, numberOfParameters, numberOfConstraints, numberOfConstraints ).setZero( );
141 
142  rightHandSide.conservativeResize( numberOfParameters + numberOfConstraints );
143  rightHandSide.segment( numberOfParameters, numberOfConstraints ) = constraintRightHandside;
144  }
145 
146 // std::cout<<"RHS "<<rightHandSide.transpose( )<<std::endl;
147 // std::cout<<"Inv cov "<<inverseOfCovarianceMatrix<<std::endl;
148 
149  return std::make_pair( solveSystemOfEquationsWithSvd(
150  inverseOfCovarianceMatrix, rightHandSide, checkConditionNumber, maximumAllowedConditionNumber ),
151  inverseOfCovarianceMatrix );
152 
153 }

182 Eigen::VectorXd getLeastSquaresPolynomialFit(
183  const Eigen::VectorXd& independentValues,
184  const Eigen::VectorXd& dependentValues,
185  const std::vector< double >& polynomialPowers )
186 {
187  if( independentValues.rows( ) != dependentValues.rows( ) )
188  {
189  throw std::runtime_error( "Error when doing least squares polynomial fit, size of dependent and independent "
190  "variable vectors is not equal." );
191  }
192 
193  Eigen::MatrixXd informationMatrix = Eigen::MatrixXd::Zero( dependentValues.rows( ), polynomialPowers.size( ) );
194 
195  // Compute information matrix
196  for( int i = 0; i < independentValues.rows( ); i++ )
197  {
198  for( unsigned int j = 0; j < polynomialPowers.size( ); j++ )
199  {
200  informationMatrix( i, j ) = std::pow( independentValues( i ), polynomialPowers.at( j ) );
201  }
202  }
203 
204  return performLeastSquaresAdjustmentFromInformationMatrix( informationMatrix, dependentValues ).first;
205 }

222 Eigen::VectorXd nonLinearLeastSquaresFit(
223  const std::function< std::pair< Eigen::VectorXd, Eigen::MatrixXd >( const Eigen::VectorXd& ) >& observationAndJacobianFunctions,
224  const Eigen::VectorXd& initialEstimate, const Eigen::VectorXd& actualObservations, const double initialScaling,
225  const double convergenceTolerance, const unsigned int maximumNumberOfIterations )
226 {
227  // Set current estimate to initial value
228  Eigen::VectorXd currentEstimate = initialEstimate;
229 
230  // Initialize variables
231  std::pair< Eigen::VectorXd, Eigen::MatrixXd > pairOfEstimatedObservationsAndDesignMatrix;
232  Eigen::MatrixXd designMatrix;
233  Eigen::VectorXd offsetInObservations;
234  Eigen::VectorXd updateInEstimate;
235 
236  // Initial parameters for Levenberg–Marquardt method
237  double levenbergMarquardtDampingParameter = 0.0;
238  double scalingParameterUpdate = 2.0;
239  double levenbergMarquardtGainRatio = 0.0;
240 
241  // Start iterative loop
242  unsigned int iteration = 0;
243  do
244  {
245  // Compute current system and jacobian functions
246  pairOfEstimatedObservationsAndDesignMatrix = observationAndJacobianFunctions( currentEstimate );
247  designMatrix = pairOfEstimatedObservationsAndDesignMatrix.second;
248 
249  // Offset in observation
250  offsetInObservations = actualObservations - pairOfEstimatedObservationsAndDesignMatrix.first;
251 
252  // Compute damping parameter for first iteration
253  if ( iteration == 0 )
254  {
255  levenbergMarquardtDampingParameter = initialScaling * ( designMatrix.transpose( ) * designMatrix ).diagonal( ).maxCoeff( );
256  }
257 
258  // Compute update in estimate
259  Eigen::VectorXd diagonalOfWeightMatrix = Eigen::VectorXd::Ones( offsetInObservations.rows( ) );
260  Eigen::MatrixXd inverseOfAPrioriCovarianceMatrix = levenbergMarquardtDampingParameter *
261 // Eigen::MatrixXd( ( designMatrix.transpose( ) * designMatrix ).diagonal( ).asDiagonal( ) ); // Marquardt’s update
262  Eigen::MatrixXd::Identity( currentEstimate.rows( ), currentEstimate.rows( ) );
263  updateInEstimate = linear_algebra::performLeastSquaresAdjustmentFromInformationMatrix(
264  designMatrix, offsetInObservations, diagonalOfWeightMatrix, inverseOfAPrioriCovarianceMatrix, false ).first;
265 
266  // Check that update is real
267  if ( ( !updateInEstimate.allFinite( ) ) || ( updateInEstimate.hasNaN( ) ) )
268  {
269  throw std::runtime_error( "Error in non-linear least squares estimation. Iterative process diverges." );
270  }
271 
272  // Compute gain ratio
273  levenbergMarquardtGainRatio =
274  ( offsetInObservations.squaredNorm( ) -
275  ( actualObservations - observationAndJacobianFunctions( currentEstimate + updateInEstimate ).first ).squaredNorm( ) ) /
276  ( updateInEstimate.transpose( ) * ( levenbergMarquardtDampingParameter * updateInEstimate +
277  designMatrix.transpose( ) * offsetInObservations ) );
278 
279  // Update damping parameter
280  if ( levenbergMarquardtGainRatio > 0 )
281  {
282  // Reduce damping parameter, since good approximation
283  levenbergMarquardtDampingParameter *= std::max( 1.0 / 3.0, 1.0 - std::pow( 2.0 * levenbergMarquardtGainRatio - 1.0, 3 ) );
284  scalingParameterUpdate = 2; // reset
285 
286  // Correct estimate
287  currentEstimate += updateInEstimate;
288  }
289  else
290  {
291  // Increase damping parameter, since bad approximation and reject step
292  levenbergMarquardtDampingParameter *= scalingParameterUpdate;
293  scalingParameterUpdate *= 2.0;
294  }
295 
296  // Increase iteration counter
297  iteration++;
298  }
299  while ( ( updateInEstimate.norm( ) > convergenceTolerance ) && ( iteration <= maximumNumberOfIterations ) );
300 
301  // Warn user of exceeded maximum number of iterations
302  if ( iteration > maximumNumberOfIterations )
303  {
304  std::cerr << "Warning in non-linear least squares estimation. Maximum number of iterations exceeded." << std::endl;
305  }
306 
307  // Give out new estimate in parameters
308  return currentEstimate;
309 }
