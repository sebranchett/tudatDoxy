116 double ZeroRevolutionLambertTargeterIzzo::getSemiMajorAxis( )
117 // Based on lambertTargeterIzzo class.
118 {
119  // If execute has not been called yet, execute.
120  if ( !solved ) execute( );
121 
122  // Compute specific orbital energy: eps = v^2/ - mu/r.
123  const double specificOrbitalEnergy = cartesianVelocityAtDeparture.squaredNorm( ) / 2.0
124  - gravitationalParameter / cartesianPositionAtDeparture.norm( );
125 
126  // Compute semi-major axis: a = -mu / 2*eps.
127  return -gravitationalParameter / ( 2.0 * specificOrbitalEnergy );
128 }

181 void ZeroRevolutionLambertTargeterIzzo::transformDimensions( )
182 // Created using theory from PyKEP toolbox and LambertTargeterIzzo class
183 {
184  // Compute normalizing values.
185  const double distanceNormalizingValue = cartesianPositionAtDeparture.norm( );
186  velocityNormalizingValue = std::sqrt( gravitationalParameter /
187  distanceNormalizingValue );
188  const double timeNormalizingValue = distanceNormalizingValue / velocityNormalizingValue;
189 
190  // Compute transfer geometry parameters in adimensional units.
191  // Time of Flight.
192  normalizedTimeOfFlight = timeOfFlight / timeNormalizingValue;
193 
194  // Cosine of transfer angle.
195  const double cosineOfTransferAngle =
196  cartesianPositionAtDeparture.dot( cartesianPositionAtArrival )
197  / (distanceNormalizingValue * cartesianPositionAtArrival.norm( ) );
198 
199  // Normalized Cartesian position at arrival.
200  normalizedRadiusAtArrival = cartesianPositionAtArrival.norm( ) / distanceNormalizingValue;
201 
202  // Chord.
203  normalizedChord = std::sqrt( 1.0 + normalizedRadiusAtArrival
204  * ( normalizedRadiusAtArrival
205  - 2.0 * cosineOfTransferAngle ) );
206 
207  // Semi-perimeter.
208  normalizedSemiPerimeter = ( 1.0 + normalizedRadiusAtArrival + normalizedChord ) / 2.0;
209 
210  // Assuming a prograde motion, determine whether the transfer corresponds to the long- or the
211  // short-way solution: longway if x1*y2 - x2*y1 < 0.
212  isLongway = false;
213  if ( cartesianPositionAtDeparture.x( ) * cartesianPositionAtArrival.y( )
214  - cartesianPositionAtDeparture.y( ) * cartesianPositionAtArrival.x( ) < 0.0 )
215  {
216  isLongway = true;
217  }
218 
219  // If retrograde is true, switch longway flag.
220  if ( isRetrograde )
221  {
222  isLongway = !isLongway;
223  }
224 
225  // Semi-major axis of the minimum energy ellipse.
226  normalizedMinimumEnergySemiMajorAxis = normalizedSemiPerimeter / 2.0;
227 
228  // Transfer angle.
229  transferAngle = std::acos( cosineOfTransferAngle );
230  if ( isLongway )
231  {
232  transferAngle = 2.0 * mathematical_constants::PI - transferAngle;
233  }
234 
235  // Lambda parameter.
236  lambdaParameter = std::sqrt( normalizedRadiusAtArrival )
237  * std::cos( transferAngle / 2.0 ) / normalizedSemiPerimeter;
238 
239  // Set transformed flag to true, as the dimension transformation has been performed
240  transformed = true;
241 }

244 double ZeroRevolutionLambertTargeterIzzo::computeTimeOfFlight( const double xParameter )
245 // Created using theory from PyKEP toolbox and LambertTargeterIzzo class
246 {
247  // Determine semi-major axis.
248  const double semiMajorAxis = normalizedMinimumEnergySemiMajorAxis
249  / ( 1.0 - xParameter * xParameter );
250 
251  // If x < 1, the solution is an ellipse.
252  if ( xParameter < 1.0 )
253  {
254  // Alpha parameter in Lagrange's equation (no explanation available).
255  const double alphaParameter = 2.0 * std::acos( xParameter );
256 
257  // Beta parameter in Lagrange's equation (no explanation available).
258  double betaParameter;
259  // If long transfer arc
260  if ( isLongway )
261  {
262  betaParameter = -2.0 * std::asin(
263  std::sqrt( ( normalizedSemiPerimeter - normalizedChord )
264  / ( 2.0 * semiMajorAxis ) ) );
265  }
266  // Otherwise short transfer arc
267  else
268  {
269  betaParameter = 2.0 * std::asin(
270  std::sqrt( ( normalizedSemiPerimeter - normalizedChord )
271  / ( 2.0 * semiMajorAxis ) ) );
272  }
273 
274  // Time-of-flight according to Lagrange including multiple revolutions.
275  const double timeOfFlight = semiMajorAxis * std::sqrt( semiMajorAxis ) *
276  ( ( alphaParameter - std::sin( alphaParameter ) )
277  - ( betaParameter - std::sin( betaParameter ) ) );
278 
279  return timeOfFlight;
280  }
281  // Otherwise it is a hyperbola.
282  else
283  {
284  // Alpha parameter in Lagrange's equation (no explanation available).
285  const double alphaParameter = 2.0 * boost::math::acosh( xParameter );
286 
287  // Beta parameter in Lagrange's equation (no explanation available).
288  double betaParameter;
289  // If long transfer arc
290  if ( isLongway )
291  {
292  betaParameter = -2.0 * boost::math::asinh ( std::sqrt( ( normalizedSemiPerimeter
293  - normalizedChord )
294  / ( -2.0 * semiMajorAxis ) ) );
295  }
296  // Otherwise short transfer arc
297  else
298  {
299  betaParameter = 2.0 * boost::math::asinh ( std::sqrt( ( normalizedSemiPerimeter
300  - normalizedChord )
301  / ( -2.0 * semiMajorAxis ) ) );
302  }
303 
304  // Time-of-flight according to Lagrange.
305  const double timeOfFlightLagrange = -semiMajorAxis * std::sqrt( -semiMajorAxis ) *
306  ( ( std::sinh( alphaParameter ) - alphaParameter )
307  - ( std::sinh( betaParameter ) - betaParameter ) );
308 
309  return timeOfFlightLagrange;
310  }

314 double ZeroRevolutionLambertTargeterIzzo::computeRootTimeOfFlight( )
315 // Created using theory from PyKEP toolbox and LambertTargeterIzzo class
316 {
317  // Find root (secant method, currently hard coded)
318  // Optimize log(t_spec).
319  const double logarithmOfTheSpecifiedTimeOfFlight = std::log( normalizedTimeOfFlight );
320 
321  // Define initial guesses for abcissae (x) and ordinates (y).
322  double x1 = std::log( 0.5 ), x2 = std::log( 1.5 );
323 
324  double y1 = std::log( computeTimeOfFlight( -0.5 ) ) - logarithmOfTheSpecifiedTimeOfFlight;
325 
326  double y2 = std::log( computeTimeOfFlight( 0.5 ) ) - logarithmOfTheSpecifiedTimeOfFlight;
327 
328  // Declare and initialize root-finding parameters.
329  double rootFindingError = 1.0, xNew = 0.0, yNew = 0.0;
330  int iterator = 0;
331 
332  // Root-finding loop.
333  while ( ( rootFindingError > convergenceTolerance ) && (y1 != y2)
334  && ( iterator < maximumNumberOfIterations ) )
335  {
336  // Update iterator.
337  iterator++;
338 
339  // Compute new x-value.
340  xNew = ( x1 * y2 - y1 * x2 ) / ( y2 - y1 );
341 
342  // Compute corresponding y-value.
343  yNew = std::log( computeTimeOfFlight( std::exp( xNew ) - 1.0 ) )
344  - logarithmOfTheSpecifiedTimeOfFlight;
345 
346  // Update abcissae and ordinates.
347  x1 = x2;
348  y1 = y2;
349  x2 = xNew;
350  y2 = yNew;
351 
352  // Compute root-finding error.
353  rootFindingError = std::fabs( x1 - xNew );
354  }
355 
356  // Verify that root-finder has converged.
357  if ( iterator == maximumNumberOfIterations )
358  {
359  throw std::runtime_error(
360  "Multi-Revolution Lambert targeter failed to converge to a solution. Reached the maximum number of iterations: " +
361  std::to_string( maximumNumberOfIterations ) );
362  }
363 
364  // Recovering x parameter and returning it.
365  double xParameter = std::exp( xNew ) - 1.0;
366  return xParameter;
367 }

370 void ZeroRevolutionLambertTargeterIzzo::computeVelocities( const double xParameter )
371 // Created using theory from PyKEP toolbox and LambertTargeterIzzo class
372 {
373  // Then it is possible to retrieve a sensible value from the x-parameter computed)
374  // Determine semi-major axis of the conic.
375  const double semiMajorAxis = normalizedMinimumEnergySemiMajorAxis
376  / ( 1.0 - xParameter * xParameter );
377 
378  // Declare variables.
379  double etaParameter, etaParameterSquared, psiParameter;
380 
381  // If x < 1, the solution is an ellipse.
382  if ( xParameter < 1.0 )
383  {
384  // Alpha parameter in Lagrange's equation (no explanation available).
385  const double alphaParameter = 2.0 * std::acos( xParameter );
386 
387  // Beta parameter in Lagrange's equation (no explanation available).
388  double betaParameter = 2.0 * std::asin( std::sqrt( ( normalizedSemiPerimeter
389  - normalizedChord )
390  / ( 2.0 * semiMajorAxis ) ) );
391 
392  if ( isLongway )
393  {
394  betaParameter = -betaParameter;
395  }
396 
397  // Psi parameter in Izzo's approach (no explanation available).
398  psiParameter = ( alphaParameter - betaParameter ) / 2.0;
399 
400  // Eta parameter in Izzo's approach (no explanation available).
401  etaParameterSquared = 2.0 * semiMajorAxis * std::sin( psiParameter )
402  * std::sin( psiParameter ) / normalizedSemiPerimeter;
403  etaParameter = std::sqrt( etaParameterSquared );
404  }
405 
406  // Otherwise it is a hyperbola.
407  else
408  {
409  // Alpha parameter in Lagrange's equation (no explanation available).
410  const double alphaParameter = 2.0 * boost::math::acosh( xParameter );
411 
412  // Beta parameter in Lagrange's equation (no explanation available).
413  double betaParameter = 2.0 * boost::math::asinh (
414  std::sqrt( ( normalizedSemiPerimeter - normalizedChord )
415  / ( -2.0 * semiMajorAxis ) ) );
416 
417  if ( isLongway )
418  {
419  betaParameter = -betaParameter;
420  }
421 
422  // Psi parameter in Izzo's approach (no explanation available).
423  psiParameter = (alphaParameter - betaParameter ) / 2.0;
424 
425  // Eta parameter in Izzo's approach (no explanation available).
426  etaParameterSquared = -2.0 * semiMajorAxis * std::sinh( psiParameter )
427  * std::sinh( psiParameter ) / normalizedSemiPerimeter;
428  etaParameter = std::sqrt( etaParameterSquared );
429  }
430 
431  // Determine semi-latus rectum, p.
432  const double semiLatusRectum = ( normalizedRadiusAtArrival
433  / ( normalizedMinimumEnergySemiMajorAxis
434  * etaParameterSquared ) )
435  * std::sin( transferAngle / 2.0 )
436  * std::sin( transferAngle / 2.0 );
437 
438  // Velocity components at departure.
439  const double radialVelocityAtDeparture =
440  ( 1.0 / ( etaParameter * std::sqrt( normalizedMinimumEnergySemiMajorAxis ) ) )
441  * ( 2.0 * lambdaParameter * normalizedMinimumEnergySemiMajorAxis
442  - ( lambdaParameter + xParameter * etaParameter ) );
443 
444  const double transverseVelocityAtDeparture = std::sqrt( semiLatusRectum );
445 
446  // Velocity components at arrival.
447  const double transverseVelocityAtArrival = transverseVelocityAtDeparture
448  / normalizedRadiusAtArrival;
449  const double radialVelocityAtArrival = ( transverseVelocityAtDeparture
450  - transverseVelocityAtArrival )
451  / std::tan( transferAngle / 2.0 )
452  - radialVelocityAtDeparture;
453 
454  // Determining inertial vectors
455  // Determine radial unit vectors.
456  const Eigen::Vector3d radialUnitVectorAtDeparture = cartesianPositionAtDeparture.normalized( );
457  const Eigen::Vector3d radialUnitVectorAtArrival = cartesianPositionAtArrival.normalized( );
458 
459  // Determine plane of motion.
460  Eigen::Vector3d angularMomentumVector;
461 
462  if ( isLongway )
463  {
464  angularMomentumVector = radialUnitVectorAtArrival.cross( radialUnitVectorAtDeparture );
465  }
466  else
467  {
468  angularMomentumVector = radialUnitVectorAtDeparture.cross( radialUnitVectorAtArrival );
469  }
470 
471  // Compute normalized angular momentum vector.
472  const Eigen::Vector3d angularMomentumUnitVector = angularMomentumVector.normalized( );
473 
474  // Compute transverse unit vectors.
475  const Eigen::Vector3d transverseUnitVectorAtDeparture =
476  radialUnitVectorAtDeparture.cross( angularMomentumUnitVector );
477  const Eigen::Vector3d transverseUnitVectorAtArrival =
478  radialUnitVectorAtArrival.cross( angularMomentumUnitVector );
479 
480  // Reconstruct non-dimensional velocity vectors.
481  cartesianVelocityAtDeparture << radialVelocityAtDeparture * radialUnitVectorAtDeparture
482  - transverseVelocityAtDeparture * transverseUnitVectorAtDeparture;
483 
484  cartesianVelocityAtArrival << radialVelocityAtArrival * radialUnitVectorAtArrival
485  - transverseVelocityAtArrival * transverseUnitVectorAtArrival;
486 
487  // Return to dimensions of initial problem definition.
488  cartesianVelocityAtDeparture *= velocityNormalizingValue;
489  cartesianVelocityAtArrival *= velocityNormalizingValue;
490 }
