20 Eigen::Vector6d computeGaussPlanetaryEquationsForKeplerElements(
21  const Eigen::Vector6d& currentOsculatingKeplerElements,
22  const Eigen::Vector3d& accelerationsInRswFrame,
23  const double semiLatusRectum,
24  const double distance,
25  const double meanMotion,
26  const double orbitalAngularMomentum )
27 {
28  // Retrieve Kepler elements and precompute reused terms
29  double semiMajorAxis = currentOsculatingKeplerElements( 0 );
30  double eccentricity = currentOsculatingKeplerElements( 1 );
31  double inclination = currentOsculatingKeplerElements( 2 );
32 
33  double meanMotionTimesASquared = meanMotion * semiMajorAxis * semiMajorAxis;
34  double eccentricityTerm = std::sqrt( 1.0 - eccentricity * eccentricity );
35  double sineTrueAnomaly = std::sin( currentOsculatingKeplerElements( 5 ) );
36  double cosineTrueAnomaly = std::cos( currentOsculatingKeplerElements( 5 ) );
37  double argumentOfLatitude = currentOsculatingKeplerElements( 5 ) + currentOsculatingKeplerElements( 3 );
38 
39  Eigen::Vector6d stateDerivative;
40 
41  // Evaluate Gauss equations.
42  stateDerivative( 0 ) = 2.0 / ( meanMotion * eccentricityTerm ) * (
43  eccentricity * sineTrueAnomaly * accelerationsInRswFrame( 0 ) +
44  semiLatusRectum / distance * accelerationsInRswFrame( 1 ) );
45 
46  stateDerivative( 1 ) = eccentricityTerm / ( meanMotion * semiMajorAxis ) * (
47  sineTrueAnomaly * accelerationsInRswFrame( 0 ) +
48  ( cosineTrueAnomaly + ( eccentricity + cosineTrueAnomaly ) /
49  ( 1.0 + eccentricity * cosineTrueAnomaly ) ) * accelerationsInRswFrame( 1 ) );
50 
51  stateDerivative( 2 ) = distance * std::cos( argumentOfLatitude ) /
52  ( meanMotionTimesASquared * eccentricityTerm ) *
53  accelerationsInRswFrame( 2 );
54 
55  stateDerivative( 3 ) = eccentricityTerm /
56  ( meanMotion * semiMajorAxis * eccentricity ) * (
57  -cosineTrueAnomaly * accelerationsInRswFrame( 0 ) +
58  sineTrueAnomaly * ( 1.0 + distance / semiLatusRectum ) * accelerationsInRswFrame( 1 ) ) -
59  distance * std::sin( argumentOfLatitude ) /
60  ( orbitalAngularMomentum * std::tan( inclination ) ) *
61  accelerationsInRswFrame( 2 );
62 
63  stateDerivative( 4 ) = distance * std::sin( argumentOfLatitude ) /
64  ( meanMotionTimesASquared * eccentricityTerm * std::sin( inclination ) ) *
65  accelerationsInRswFrame( 2 );
66 
67  stateDerivative( 5 ) = meanMotion + 1.0 / ( meanMotionTimesASquared * eccentricity ) * (
68  ( semiLatusRectum * cosineTrueAnomaly - 2.0 * eccentricity * distance ) * accelerationsInRswFrame( 0 ) -
69  ( semiLatusRectum + distance ) * sineTrueAnomaly * accelerationsInRswFrame( 1 ) );
70 
71  return stateDerivative;
72 
73 }

76 Eigen::Vector6d computeGaussPlanetaryEquationsForKeplerElements(
77  const Eigen::Vector6d& currentOsculatingKeplerElements,
78  const Eigen::Vector3d& accelerationsInRswFrame,
79  const double centralBodyGravitationalParameter )
80 {
81  // Compute orbit properties
82  double semiLatusRectum = orbital_element_conversions::computeSemiLatusRectum(
83  currentOsculatingKeplerElements( 1 ),
84  currentOsculatingKeplerElements( 0 ),
85  std::numeric_limits< double >::epsilon( ) );
86  double orbitalAngularMomentum = orbital_element_conversions::computeOrbitalAngularMomentumPerUnitMass(
87  semiLatusRectum, centralBodyGravitationalParameter );
88  double meanMotion = basic_astrodynamics::computeKeplerMeanMotion(
89  currentOsculatingKeplerElements( 0 ), centralBodyGravitationalParameter );
90  double distance = semiLatusRectum /
91  ( 1.0 + currentOsculatingKeplerElements( 1 ) * std::cos( currentOsculatingKeplerElements( 5 ) ) );
92 
93  // Evaluate Gauss equations
94  return computeGaussPlanetaryEquationsForKeplerElements(
95  currentOsculatingKeplerElements, accelerationsInRswFrame, semiLatusRectum, distance, meanMotion,
96  orbitalAngularMomentum );
97 }
