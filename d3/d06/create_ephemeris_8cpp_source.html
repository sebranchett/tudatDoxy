35 std::shared_ptr< ephemerides::Ephemeris > createBodyEphemeris(
36  const std::shared_ptr< EphemerisSettings > ephemerisSettings,
37  const std::string& bodyName )
38 {
39  // Declare return object.
40  std::shared_ptr< ephemerides::Ephemeris > ephemeris;
41 
42  if( ephemerisSettings->getMakeMultiArcEphemeris( ) )
43  {
44  std::map< double, std::shared_ptr< Ephemeris > > singleArcEphemerides;
45  ephemerisSettings->resetMakeMultiArcEphemeris( false );
46 
47  singleArcEphemerides[ -std::numeric_limits< double >::lowest( ) ] = createBodyEphemeris(
48  ephemerisSettings, bodyName );
49 
50  ephemeris = std::make_shared< MultiArcEphemeris >(
51  singleArcEphemerides, ephemerisSettings->getFrameOrigin( ), ephemerisSettings->getFrameOrientation( ) );
52  }
53  else
54  {
55 
56  // Check which type of ephemeris model is to be created.
57  switch( ephemerisSettings->getEphemerisType( ) )
58  {
59 #if USE_CSPICE
60  case direct_spice_ephemeris:
61  {
62  // Check consistency of type and class.
63  std::shared_ptr< DirectSpiceEphemerisSettings > directEphemerisSettings =
64  std::dynamic_pointer_cast< DirectSpiceEphemerisSettings >( ephemerisSettings );
65  if( directEphemerisSettings == nullptr )
66  {
67  throw std::runtime_error(
68  "Error, expected direct spice ephemeris settings for body " + bodyName );
69  }
70  else
71  {
72  std::string inputName = bodyName;
73 
74  // Create corresponding ephemeris object.
75  ephemeris = std::make_shared< SpiceEphemeris >(
76  bodyName,
77  directEphemerisSettings->getFrameOrigin( ),
78  directEphemerisSettings->getCorrectForStellarAberration( ),
79  directEphemerisSettings->getCorrectForLightTimeAberration( ),
80  directEphemerisSettings->getConvergeLighTimeAberration( ),
81  directEphemerisSettings->getFrameOrientation( ) );
82  }
83  break;
84  }
85  case interpolated_spice:
86  {
87  // Check consistency of type and class.
88  std::shared_ptr< InterpolatedSpiceEphemerisSettings > interpolatedEphemerisSettings =
89  std::dynamic_pointer_cast< InterpolatedSpiceEphemerisSettings >(
90  ephemerisSettings );
91  if( interpolatedEphemerisSettings == nullptr )
92  {
93  throw std::runtime_error(
94  "Error, expected tabulated spice ephemeris settings for body " + bodyName );
95  }
96  else
97  {
98  // Since only the barycenters of planetary systems are included in the standard DE
99  // ephemerides, append 'Barycenter' to body name.
100  std::string inputName = bodyName;
101 
102  // Create corresponding ephemeris object.
103  if( !interpolatedEphemerisSettings->getUseLongDoubleStates( ) )
104  {
105  ephemeris = createTabulatedEphemerisFromSpice< double, double >(
106  inputName,
107  interpolatedEphemerisSettings->getInitialTime( ),
108  interpolatedEphemerisSettings->getFinalTime( ),
109  interpolatedEphemerisSettings->getTimeStep( ),
110  interpolatedEphemerisSettings->getFrameOrigin( ),
111  interpolatedEphemerisSettings->getFrameOrientation( ),
112  interpolatedEphemerisSettings->getInterpolatorSettings( ) );
113  }
114  else
115  {
116 #if( BUILD_WITH_EXTENDED_PRECISION_PROPAGATION_TOOLS )
117 
118  ephemeris = createTabulatedEphemerisFromSpice< long double, double >(
119  inputName,
120  static_cast< long double >( interpolatedEphemerisSettings->getInitialTime( ) ),
121  static_cast< long double >( interpolatedEphemerisSettings->getFinalTime( ) ),
122  static_cast< long double >( interpolatedEphemerisSettings->getTimeStep( ) ),
123  interpolatedEphemerisSettings->getFrameOrigin( ),
124  interpolatedEphemerisSettings->getFrameOrientation( ),
125  interpolatedEphemerisSettings->getInterpolatorSettings( ) );
126 #else
127  throw std::runtime_error( "Error, long double compilation is turned off; requested long doubel tabulated ephemeris" );
128 #endif
129  }
130  }
131  break;
132  }
133 #endif
134  case tabulated_ephemeris:
135  {
136  // Check consistency of type and class.
137  std::shared_ptr< TabulatedEphemerisSettings > tabulatedEphemerisSettings =
138  std::dynamic_pointer_cast< TabulatedEphemerisSettings >( ephemerisSettings );
139  if( tabulatedEphemerisSettings == nullptr )
140  {
141  throw std::runtime_error(
142  "Error, expected tabulated ephemeris settings for body " + bodyName );
143  }
144  else
145  {
146  // Create corresponding ephemeris object.
147  if( !tabulatedEphemerisSettings->getUseLongDoubleStates( ) )
148  {
149  if( tabulatedEphemerisSettings->getBodyStateHistory( ).size( ) != 0 )
150  {
151  ephemeris = std::make_shared< TabulatedCartesianEphemeris< > >(
152  std::make_shared<
153  interpolators::LagrangeInterpolator< double, Eigen::Vector6d > >
154  ( tabulatedEphemerisSettings->getBodyStateHistory( ), 6,
155  interpolators::huntingAlgorithm,
156  interpolators::lagrange_cubic_spline_boundary_interpolation ),
157  tabulatedEphemerisSettings->getFrameOrigin( ),
158  tabulatedEphemerisSettings->getFrameOrientation( ) );
159  }
160  else
161  {
162  ephemeris = std::make_shared< TabulatedCartesianEphemeris< > >(
163  std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::Vector6d > >( ),
164  tabulatedEphemerisSettings->getFrameOrigin( ),
165  tabulatedEphemerisSettings->getFrameOrientation( ) );
166  }
167  }
168  else
169  {
170 #if( BUILD_WITH_EXTENDED_PRECISION_PROPAGATION_TOOLS )
171 
172  // Cast input history to required type.
173  if( tabulatedEphemerisSettings->getBodyStateHistory( ).size( ) != 0 )
174  {
175  std::map< double, Eigen::Vector6d > originalStateHistory =
176  tabulatedEphemerisSettings->getBodyStateHistory( );
177  std::map< double, Eigen::Matrix< long double, 6, 1 > > longStateHistory;
178 
179  for( std::map< double, Eigen::Vector6d >::const_iterator stateIterator =
180  originalStateHistory.begin( ); stateIterator != originalStateHistory.end( ); stateIterator++ )
181  {
182  longStateHistory[ stateIterator->first ] = stateIterator->second.cast< long double >( );
183  ephemeris =
184  std::make_shared< TabulatedCartesianEphemeris< long double, double > >(
185  std::make_shared< interpolators::LagrangeInterpolator<
186  double, Eigen::Matrix< long double, 6, 1 > > >
187  ( longStateHistory, 6,
188  interpolators::huntingAlgorithm,
189  interpolators::lagrange_cubic_spline_boundary_interpolation ),
190  tabulatedEphemerisSettings->getFrameOrigin( ),
191  tabulatedEphemerisSettings->getFrameOrientation( ) );
192  }
193  }
194  else
195  {
196  ephemeris = std::make_shared< TabulatedCartesianEphemeris< long double, double > >(
197  std::shared_ptr< interpolators::OneDimensionalInterpolator<
198  double, Eigen::Matrix< long double, 6, 1 > > >( ),
199  tabulatedEphemerisSettings->getFrameOrigin( ),
200  tabulatedEphemerisSettings->getFrameOrientation( ) );
201  }
202 #else
203  throw std::runtime_error( "Error, long double compilation is turned off; requested long doubel tabulated ephemeris" );
204 #endif
205  }
206  }
207  break;
208  }
209  case constant_ephemeris:
210  {
211  // Check consistency of type and class.
212  std::shared_ptr< ConstantEphemerisSettings > constantEphemerisSettings =
213  std::dynamic_pointer_cast< ConstantEphemerisSettings >( ephemerisSettings );
214  if( constantEphemerisSettings == nullptr )
215  {
216  throw std::runtime_error( "Error, expected constant ephemeris settings for " + bodyName );
217  }
218  else
219  {
220  // Create ephemeris
221  ephemeris = std::make_shared< ConstantEphemeris >(
222  [ = ]( ){ return constantEphemerisSettings->getConstantState( ); },
223  constantEphemerisSettings->getFrameOrigin( ),
224  constantEphemerisSettings->getFrameOrientation( ) );
225  }
226  break;
227  }
228  case custom_ephemeris:
229  {
230  // Check consistency of type and class.
231  std::shared_ptr< CustomEphemerisSettings > customEphemerisSettings =
232  std::dynamic_pointer_cast< CustomEphemerisSettings >( ephemerisSettings );
233  if( customEphemerisSettings == nullptr )
234  {
235  throw std::runtime_error( "Error, expected constant ephemeris settings for " + bodyName );
236  }
237  else
238  {
239  // Create ephemeris
240  ephemeris = std::make_shared< CustomEphemeris >(
241  customEphemerisSettings->getCustomStateFunction( ),
242  customEphemerisSettings->getFrameOrigin( ),
243  customEphemerisSettings->getFrameOrientation( ) );
244  }
245  break;
246  }
247 
248  case kepler_ephemeris:
249  {
250  // Check consistency of type and class.
251  std::shared_ptr< KeplerEphemerisSettings > keplerEphemerisSettings =
252  std::dynamic_pointer_cast< KeplerEphemerisSettings >( ephemerisSettings );
253  if( keplerEphemerisSettings == nullptr )
254  {
255  throw std::runtime_error( "Error, expected Kepler ephemeris settings for " + bodyName );
256  }
257  else
258  {
259  // Create ephemeris
260  ephemeris = std::make_shared< KeplerEphemeris >(
261  keplerEphemerisSettings->getInitialStateInKeplerianElements( ),
262  keplerEphemerisSettings->getEpochOfInitialState( ),
263  keplerEphemerisSettings->getCentralBodyGravitationalParameter( ),
264  keplerEphemerisSettings->getFrameOrigin( ),
265  keplerEphemerisSettings->getFrameOrientation( ),
266  keplerEphemerisSettings->getRootFinderAbsoluteTolerance( ),
267  keplerEphemerisSettings->getRootFinderMaximumNumberOfIterations( ) );
268  }
269  break;
270  }
271  case approximate_planet_positions:
272  {
273  // Check consistency of type and class.
274  std::shared_ptr< ApproximatePlanetPositionSettings > approximateEphemerisSettings =
275  std::dynamic_pointer_cast< ApproximatePlanetPositionSettings >(
276  ephemerisSettings );
277  if( approximateEphemerisSettings == nullptr )
278  {
279  throw std::runtime_error(
280  "Error, expected approximate ephemeris settings for body " + bodyName );
281  }
282  else
283  {
284  // Create corresponding ephemeris object.
285  if( approximateEphemerisSettings->getUseCircularCoplanarApproximation( ) )
286  {
287  ephemeris = std::make_shared< ApproximatePlanetPositionsCircularCoplanar >(
288  approximateEphemerisSettings->getBodyIdentifier( ) );
289  }
290  else
291  {
292  ephemeris = std::make_shared< ApproximatePlanetPositions >(
293  approximateEphemerisSettings->getBodyIdentifier( ) );
294  }
295  }
296  break;
297  }
298  default:
299  {
300  throw std::runtime_error(
301  "Error, did not recognize ephemeris model settings type " +
302  std::to_string( ephemerisSettings->getEphemerisType( ) ) );
303  }
304  }
305  }
306  return ephemeris;

311 std::pair< double, double > getSafeInterpolationInterval( const std::shared_ptr< ephemerides::Ephemeris > ephemerisModel )
312 {
313  // Make default output pair
314  std::pair< double, double > safeInterval = std::make_pair(
315  std::numeric_limits< double >::lowest( ), std::numeric_limits< double >::max( ) );
316 
317  // Check if model is tabulated, and retrieve safe interval from model
318  if( isTabulatedEphemeris( ephemerisModel ) )
319  {
320  safeInterval = getTabulatedEphemerisSafeInterval( ephemerisModel );
321  }
322  // Check if model is multi-arc, and retrieve safe intervals from first and last arc.
323  else if( std::dynamic_pointer_cast< ephemerides::MultiArcEphemeris >( ephemerisModel ) != nullptr )
324  {
325  std::shared_ptr< ephemerides::MultiArcEphemeris > multiArcEphemerisModel =
326  std::dynamic_pointer_cast< ephemerides::MultiArcEphemeris >( ephemerisModel );
327  safeInterval.first = getSafeInterpolationInterval( multiArcEphemerisModel->getSingleArcEphemerides( ).at( 0 ) ).first;
328  safeInterval.second = getSafeInterpolationInterval(
329  multiArcEphemerisModel->getSingleArcEphemerides( ).at(
330  multiArcEphemerisModel->getSingleArcEphemerides( ).size( ) - 1 ) ).second;
331  }
332  return safeInterval;
333 }
