38 template< typename StateScalarType, typename TimeType >
39 std::map< propagators::IntegratedStateType, orbit_determination::StateDerivativePartialsMap > createStateDerivativePartials(
40  const std::unordered_map< propagators::IntegratedStateType,
41  std::vector< std::shared_ptr< propagators::SingleStateTypeDerivative< StateScalarType, TimeType > > > >
42  stateDerivativeModels,
43  const simulation_setup::NamedBodyMap& bodyMap,
44  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< StateScalarType > >
45  parametersToEstimate )
46 {
47  std::map< propagators::IntegratedStateType, orbit_determination::StateDerivativePartialsMap > stateDerivativePartials;
48 
49  // Iterate over all state types
50  for( typename std::unordered_map< propagators::IntegratedStateType,
51  std::vector< std::shared_ptr< propagators::SingleStateTypeDerivative< StateScalarType, TimeType > > > >::
52  const_iterator stateDerivativeIterator = stateDerivativeModels.begin( );
53  stateDerivativeIterator != stateDerivativeModels.end( );
54  stateDerivativeIterator++ )
55  {
56  // Identify state types
57  switch( stateDerivativeIterator->first )
58  {
59  case propagators::translational_state:
60  {
61  if( stateDerivativeIterator->second.size( ) > 1 )
62  {
63  throw std::runtime_error(
64  "Error, cannot yet process multiple separate same type propagators when making partial "
65  "derivatives of translational state." );
66  }
67  else
68  {
69  // Retrieve acceleration models and create partials
70  basic_astrodynamics::AccelerationMap accelerationModelList =
71  std::dynamic_pointer_cast< propagators::NBodyStateDerivative< StateScalarType, TimeType > >(
72  stateDerivativeIterator->second.at( 0 ) )->getFullAccelerationsMap( );
73  stateDerivativePartials[ propagators::translational_state ] =
74  createAccelerationPartialsMap< StateScalarType >(
75  accelerationModelList, bodyMap, parametersToEstimate );
76  }
77  break;
78  }
79  case propagators::rotational_state:
80  {
81  if( stateDerivativeIterator->second.size( ) > 1 )
82  {
83  throw std::runtime_error(
84  "Error, cannot yet process multiple separate same type propagators when making partial derivatives of rotational state." );
85  }
86  else
87  {
88  // Retrieve acceleration models and create partials
89  basic_astrodynamics::TorqueModelMap torqueModelList =
90  std::dynamic_pointer_cast< propagators::RotationalMotionStateDerivative< StateScalarType, TimeType > >(
91  stateDerivativeIterator->second.at( 0 ) )->getTorquesMap( );
92  stateDerivativePartials[ propagators::rotational_state ] =
93  createTorquePartialsMap< StateScalarType >(
94  torqueModelList, bodyMap, parametersToEstimate );
95  }
96  break;
97  }
98  default:
99  std::string errorMessage = "Cannot yet create state derivative partial models for type " +
100  std::to_string( stateDerivativeIterator->first );
101  throw std::runtime_error( errorMessage );
102  }
103  }
104 
105  return stateDerivativePartials;
106 }
