22 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< AccelerationSettings >& accelerationSettings )
23 {
24  if ( ! accelerationSettings )
25  {
26  return;
27  }
28  using namespace basic_astrodynamics;
29  using namespace json_interface;
30  using K = Keys::Propagator::Acceleration;
31 
32  const AvailableAcceleration accelerationType = accelerationSettings->accelerationType_;
33  jsonObject[ K::type ] = accelerationType;
34 
35  switch ( accelerationType ) {
36  case undefined_acceleration:
37  case aerodynamic:
38  case cannon_ball_radiation_pressure:
39  {
40  return;
41  }
42  case point_mass_gravity:
43  case third_body_point_mass_gravity:
44  {
45  jsonObject[ K::type ] = point_mass_gravity;
46  return;
47  }
48  case spherical_harmonic_gravity:
49  case third_body_spherical_harmonic_gravity:
50  {
51  jsonObject[ K::type ] = spherical_harmonic_gravity;
52  std::shared_ptr< SphericalHarmonicAccelerationSettings > sphericalHarmonicAccelerationSettings
53  = std::dynamic_pointer_cast< SphericalHarmonicAccelerationSettings >( accelerationSettings );
54  assertNonnullptrPointer( sphericalHarmonicAccelerationSettings );
55  jsonObject[ K::maximumDegree ] = sphericalHarmonicAccelerationSettings->maximumDegree_;
56  jsonObject[ K::maximumOrder ] = sphericalHarmonicAccelerationSettings->maximumOrder_;
57  return;
58  }
59  case mutual_spherical_harmonic_gravity:
60  case third_body_mutual_spherical_harmonic_gravity:
61  {
62  jsonObject[ K::type ] = mutual_spherical_harmonic_gravity;
63  std::shared_ptr< MutualSphericalHarmonicAccelerationSettings > mutualSphericalHarmonicAccelerationSettings
64  = std::dynamic_pointer_cast< MutualSphericalHarmonicAccelerationSettings >(
65  accelerationSettings );
66  assertNonnullptrPointer( mutualSphericalHarmonicAccelerationSettings );
67  jsonObject[ K::maximumDegreeOfBodyExertingAcceleration ] =
68  mutualSphericalHarmonicAccelerationSettings->maximumDegreeOfBodyExertingAcceleration_;
69  jsonObject[ K::maximumOrderOfBodyExertingAcceleration ] =
70  mutualSphericalHarmonicAccelerationSettings->maximumOrderOfBodyExertingAcceleration_;
71  jsonObject[ K::maximumDegreeOfBodyUndergoingAcceleration ] =
72  mutualSphericalHarmonicAccelerationSettings->maximumDegreeOfBodyUndergoingAcceleration_;
73  jsonObject[ K::maximumOrderOfBodyUndergoingAcceleration ] =
74  mutualSphericalHarmonicAccelerationSettings->maximumOrderOfBodyUndergoingAcceleration_;
75  jsonObject[ K::maximumDegreeOfCentralBody ] =
76  mutualSphericalHarmonicAccelerationSettings->maximumDegreeOfCentralBody_;
77  jsonObject[ K::maximumOrderOfCentralBody ] =
78  mutualSphericalHarmonicAccelerationSettings->maximumOrderOfCentralBody_;
79  return;
80  }
81  case thrust_acceleration:
82  {
83  std::shared_ptr< ThrustAccelerationSettings > thrustAccelerationSettings
84  = std::dynamic_pointer_cast< ThrustAccelerationSettings >( accelerationSettings );
85  assertNonnullptrPointer( thrustAccelerationSettings );
86  jsonObject = thrustAccelerationSettings;
87  return;
88  }
89  case relativistic_correction_acceleration:
90  {
91  std::shared_ptr< RelativisticAccelerationCorrectionSettings > relativisticAccelerationCorrectionSettings
92  = std::dynamic_pointer_cast< RelativisticAccelerationCorrectionSettings >( accelerationSettings );
93  assertNonnullptrPointer( relativisticAccelerationCorrectionSettings );
94  jsonObject[ K::calculateSchwarzschildCorrection ] =
95  relativisticAccelerationCorrectionSettings->calculateSchwarzschildCorrection_;
96  jsonObject[ K::calculateLenseThirringCorrection ] =
97  relativisticAccelerationCorrectionSettings->calculateLenseThirringCorrection_;
98  jsonObject[ K::calculateDeSitterCorrection ] =
99  relativisticAccelerationCorrectionSettings->calculateDeSitterCorrection_;
100  jsonObject[ K::primaryBody ] =
101  relativisticAccelerationCorrectionSettings->primaryBody_;
102  jsonObject[ K::centralBodyAngularMomentum ] =
103  relativisticAccelerationCorrectionSettings->centralBodyAngularMomentum_;
104  return;
105  }
106  case empirical_acceleration:
107  {
108  std::shared_ptr< EmpiricalAccelerationSettings > empiricalAccelerationSettings
109  = std::dynamic_pointer_cast< EmpiricalAccelerationSettings >( accelerationSettings );
110  assertNonnullptrPointer( empiricalAccelerationSettings );
111  jsonObject[ K::constantAcceleration ] = empiricalAccelerationSettings->constantAcceleration_;
112  jsonObject[ K::sineAcceleration ] = empiricalAccelerationSettings->sineAcceleration_;
113  jsonObject[ K::cosineAcceleration ] = empiricalAccelerationSettings->cosineAcceleration_;
114  return;
115  }
116  default:
117  handleUnimplementedEnumValue( accelerationType, accelerationTypes, unsupportedAccelerationTypes );
118  }

122 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< AccelerationSettings >& accelerationSettings )
123 {
124  using namespace json_interface;
125  using namespace basic_astrodynamics;
126  using K = Keys::Propagator::Acceleration;
127 
128  // Get acceleration type
129  const AvailableAcceleration accelerationType = getValue< AvailableAcceleration >( jsonObject, K::type );
130 
131  switch ( accelerationType ) {
132  case undefined_acceleration:
133  case point_mass_gravity:
134  case aerodynamic:
135  case cannon_ball_radiation_pressure:
136  {
137  accelerationSettings = std::make_shared< AccelerationSettings >( accelerationType );
138  return;
139  }
140  case spherical_harmonic_gravity:
141  {
142  accelerationSettings = std::make_shared< SphericalHarmonicAccelerationSettings >(
143  getValue< int >( jsonObject, K::maximumDegree ),
144  getValue< int >( jsonObject, K::maximumOrder ) );
145  return;
146  }
147  case mutual_spherical_harmonic_gravity:
148  {
149  MutualSphericalHarmonicAccelerationSettings defaults( -1, -1, -1, -1 );
150  accelerationSettings = std::make_shared< MutualSphericalHarmonicAccelerationSettings >(
151  getValue< int >( jsonObject, K::maximumDegreeOfBodyExertingAcceleration ),
152  getValue< int >( jsonObject, K::maximumOrderOfBodyExertingAcceleration ),
153  getValue< int >( jsonObject, K::maximumDegreeOfBodyUndergoingAcceleration ),
154  getValue< int >( jsonObject, K::maximumOrderOfBodyUndergoingAcceleration ),
155  getValue( jsonObject, K::maximumDegreeOfCentralBody, defaults.maximumDegreeOfCentralBody_ ),
156  getValue( jsonObject, K::maximumOrderOfCentralBody, defaults.maximumOrderOfCentralBody_ ) );
157  return;
158  }
159  case thrust_acceleration:
160  {
161  accelerationSettings = getAs< std::shared_ptr< ThrustAccelerationSettings > >( jsonObject );
162  return;
163  }
164  case relativistic_correction_acceleration:
165  {
166  RelativisticAccelerationCorrectionSettings defaults;
167  accelerationSettings = std::make_shared< RelativisticAccelerationCorrectionSettings >(
168  getValue( jsonObject, K::calculateSchwarzschildCorrection,
169  defaults.calculateSchwarzschildCorrection_ ),
170  getValue( jsonObject, K::calculateLenseThirringCorrection,
171  defaults.calculateLenseThirringCorrection_ ),
172  getValue( jsonObject, K::calculateDeSitterCorrection, defaults.calculateDeSitterCorrection_ ),
173  getValue( jsonObject, K::primaryBody, defaults.primaryBody_ ),
174  getValue( jsonObject, K::centralBodyAngularMomentum, defaults.centralBodyAngularMomentum_ ) );
175  return;
176  }
177  case empirical_acceleration:
178  {
179  EmpiricalAccelerationSettings defaults;
180  accelerationSettings = std::make_shared< EmpiricalAccelerationSettings >(
181  getValue( jsonObject, K::constantAcceleration, defaults.constantAcceleration_ ),
182  getValue( jsonObject, K::sineAcceleration, defaults.sineAcceleration_ ),
183  getValue( jsonObject, K::cosineAcceleration, defaults.cosineAcceleration_ ) );
184  return;
185  }
186  default:
187  {
188  if ( accelerationType == third_body_point_mass_gravity ||
189  accelerationType == third_body_spherical_harmonic_gravity ||
190  accelerationType == third_body_mutual_spherical_harmonic_gravity )
191  {
192  std::cerr << "Whether a body will cause a third-body acceleration is determined internally "
193  << "by Tudat based on the propagation settings." << std::endl;
194  }
195  handleUnimplementedEnumValue( accelerationType, accelerationTypes, unsupportedAccelerationTypes );
196  }
197  }
198 }
