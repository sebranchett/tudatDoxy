30 Eigen::Vector6d propagateClohessyWiltshire(
31  const Eigen::Vector6d& initialState,
32  const double propagationDuration,
33  const double centralBodyGravitationalParameter,
34  const double referenceOrbitRadius )
35 {
36  // Initialize final state vector.
37  Eigen::Vector6d finalState;
38 
39  // Calculate mean angular motion of reference orbit.
40  const double meanAngularMotion =
41  sqrt( centralBodyGravitationalParameter
42  / ( referenceOrbitRadius * referenceOrbitRadius * referenceOrbitRadius ) );
43 
44  // Calculate integration constants of Clohessy-Wiltshire equations.
45  const double integrationConstant1 = 4.0 * initialState( 0 )
46  + 2.0 * initialState( 4 ) / meanAngularMotion;
47  const double integrationConstant2 = initialState( 1 )
48  - 2.0 * initialState( 3 ) / meanAngularMotion;
49  const double integrationConstant3 = 3.0 * initialState( 0 )
50  + 2.0 * initialState( 4 ) / meanAngularMotion;
51  const double integrationConstant4 = initialState( 3 ) / meanAngularMotion;
52  const double integrationConstant5 = initialState( 2 );
53  const double integrationConstant6 = initialState( 5 ) / meanAngularMotion;
54 
55  // Calculate dynamical terms of Clohessy-Wiltshire equations.
56  const double cosineTerm = cos( meanAngularMotion * propagationDuration );
57  const double sineTerm = sin( meanAngularMotion * propagationDuration );
58 
59  // Calculate Cartesian position components with Clohessy-Wilthsire equations as given by
60  // by Vallado [2001, p. 382].
61  finalState( 0 ) = integrationConstant1 - integrationConstant3 * cosineTerm
62  + integrationConstant4 * sineTerm;
63  finalState( 1 ) = integrationConstant2
64  - 1.5 * integrationConstant1 * meanAngularMotion * propagationDuration
65  + 2.0 * integrationConstant3 * sineTerm
66  + 2.0 * integrationConstant4 * cosineTerm;
67  finalState( 2 ) = integrationConstant5 * cosineTerm + integrationConstant6 * sineTerm;
68 
69  // Calculate Cartesian velocity components with Clohessy-Wilthsire equations as given by
70  // by Vallado [2001, p. 382].
71  finalState( 3 ) = meanAngularMotion * integrationConstant3 * sineTerm
72  + meanAngularMotion * integrationConstant4 * cosineTerm;
73  finalState( 4 ) = - 1.5 * meanAngularMotion * integrationConstant1
74  + 2.0 * meanAngularMotion * integrationConstant3 * cosineTerm
75  - 2.0 * meanAngularMotion * integrationConstant4 * sineTerm;
76  finalState( 5 ) = - meanAngularMotion * integrationConstant5 * sineTerm
77  + meanAngularMotion * integrationConstant6 * cosineTerm;
78 
79  // Return final state.
80  return finalState;
81 }
