111 template< typename StateScalarType = double, typename TimeType = double >
112 void setGlobalFrameBodyEphemerides( const NamedBodyMap& bodyMap,
113  const std::string& globalFrameOrigin,
114  const std::string& globalFrameOrientation )
115 {
116  using namespace tudat::simulation_setup;
117  std::string ephemerisFrameOrigin;
118  std::string ephemerisFrameOrientation;
119  std::string rotationModelFrame;
120 
121  std::vector< std::string > globalFrameOriginChain;
122 
123  // Get chain of ephemeris frame origins of global frame origin (if it is not SSB
124  if( globalFrameOrigin != "SSB" )
125  {
126  if( bodyMap.count( globalFrameOrigin ) == 0 )
127  {
128  throw std::runtime_error(
129  "Error, body non-barycentric global frame origin selected, but this body " + globalFrameOrigin +
130  " is not found." );
131  }
132  else
133  {
134  std::string currentOrigin = globalFrameOrigin;
135  while( currentOrigin != "SSB" )
136  {
137  std::shared_ptr< ephemerides::Ephemeris > currentEphemeris =
138  bodyMap.at( currentOrigin )->getEphemeris( );
139  if( currentEphemeris == nullptr )
140  {
141  throw std::runtime_error(
142  "Error, body non-barycentric global frame origin selected, but body " + currentOrigin +
143  " in chain has no ephemeris." );
144  }
145  else
146  {
147  currentOrigin = currentEphemeris->getReferenceFrameOrigin( );
148  ephemerisFrameOrientation = currentEphemeris->getReferenceFrameOrientation( );
149  if( ephemerisFrameOrientation != globalFrameOrientation )
150  {
151  throw std::runtime_error(
152  "Error, ephemeris orientation of body " + currentOrigin
153  + " is not the same as global orientation " + ephemerisFrameOrientation
154  + ", " + globalFrameOrientation );
155  }
156  }
157 
158  if( std::find( globalFrameOriginChain.begin( ), globalFrameOriginChain.end( ), currentOrigin ) !=
159  globalFrameOriginChain.end( ) )
160  {
161  throw std::runtime_error(
162  "Error, body non-barycentric global frame origin selected, but body " + currentOrigin +
163  " already found in origin chain." );
164  }
165  else
166  {
167  globalFrameOriginChain.push_back( currentOrigin );
168  }
169  }
170  }
171  }
172 
173  // Iterate over all bodies
174  for( NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
175  bodyIterator != bodyMap.end( ); bodyIterator++ )
176  {
177  // Check id body contains an ephemeris
178  if( bodyIterator->second->getEphemeris( ) != nullptr )
179  {
180  // Retrieve ephemeris origin
181  ephemerisFrameOrigin = bodyIterator->second->getEphemeris( )->getReferenceFrameOrigin( );
182 
183  // Check if ephemeris origin differs from global origin.
184  if( ephemerisFrameOrigin != globalFrameOrigin )
185  {
186  // Make correction to SSB if it is global frame origin
187  if( globalFrameOrigin == "SSB" )
188  {
189  // Check if correction can be made
190  if( bodyMap.count( ephemerisFrameOrigin ) == 0 )
191  {
192  throw std::runtime_error(
193  "Error, body " + bodyIterator->first + " has ephemeris in frame " +
194  ephemerisFrameOrigin + ", but no conversion to frame " + globalFrameOrigin +
195  " can be made" );
196  }
197  else
198  {
199  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > stateFunction =
200  std::bind( &Body::getStateInBaseFrameFromEphemeris< StateScalarType, TimeType >,
201  bodyMap.at( ephemerisFrameOrigin ), std::placeholders::_1 );
202  std::shared_ptr< BaseStateInterface > baseStateInterface =
203  std::make_shared< BaseStateInterfaceImplementation< TimeType, StateScalarType > >(
204  ephemerisFrameOrigin, stateFunction );
205  bodyIterator->second->setEphemerisFrameToBaseFrame( baseStateInterface );
206  }
207  }
208  // Make correction to global frame origin (if not SSB)
209  else
210  {
211  // Set barycentric state function of global frame origin
212  if( globalFrameOrigin == bodyIterator->first )
213  {
214  std::shared_ptr< ephemerides::ReferenceFrameManager > frameManager =
215  propagators::createFrameManager( bodyMap );
216  frameManager->getEphemeris( globalFrameOrigin, "SSB" );
217 
218  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > stateFunction =
219  std::bind( &ephemerides::Ephemeris::getTemplatedStateFromEphemeris< StateScalarType, TimeType >,
220  frameManager->getEphemeris( globalFrameOrigin, "SSB" ), std::placeholders::_1 );
221 
222  std::shared_ptr< BaseStateInterface > baseStateInterface =
223  std::make_shared< BaseStateInterfaceImplementation< TimeType, StateScalarType > >(
224  globalFrameOrigin, stateFunction, true );
225  bodyIterator->second->setEphemerisFrameToBaseFrame( baseStateInterface );
226 
227  }
228  // Set correction function if ephemeris origin is SSB
229  else if( ephemerisFrameOrigin == "SSB" )
230  {
231 
232  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > stateFunction =
233  std::bind( &Body::getGlobalFrameOriginBarycentricStateFromEphemeris< StateScalarType, TimeType >,
234  bodyMap.at( globalFrameOrigin ), std::placeholders::_1 );
235  std::shared_ptr< BaseStateInterface > baseStateInterface =
236  std::make_shared< BaseStateInterfaceImplementation< TimeType, StateScalarType > >(
237  globalFrameOrigin, stateFunction, true );
238  bodyIterator->second->setEphemerisFrameToBaseFrame( baseStateInterface );
239  }
240  else
241  {
242  // Check if correction can be made
243  if( bodyMap.count( ephemerisFrameOrigin ) == 0 )
244  {
245  throw std::runtime_error(
246  "Error, body " + bodyIterator->first + " has ephemeris in frame " +
247  ephemerisFrameOrigin + ", but no conversion to frame " + globalFrameOrigin +
248  " can be made" );
249  }
250  else
251  {
252  // Set correction function from ephemeris origin to global frame origin
253  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > stateFunction =
254  std::bind( &Body::getStateInBaseFrameFromEphemeris< StateScalarType, TimeType >,
255  bodyMap.at( ephemerisFrameOrigin ), std::placeholders::_1 );
256  std::shared_ptr< BaseStateInterface > baseStateInterface =
257  std::make_shared< BaseStateInterfaceImplementation< TimeType, StateScalarType > >(
258  ephemerisFrameOrigin, stateFunction, false );
259  bodyIterator->second->setEphemerisFrameToBaseFrame( baseStateInterface );
260  }
261  }
262  }
263  }
264 
265  // Retrieve ephemeris orientation
266  ephemerisFrameOrientation = bodyIterator->second->getEphemeris( )->getReferenceFrameOrientation( );
267  // If two are not equal, throw error.
268  if( ephemerisFrameOrientation != globalFrameOrientation )
269  {
270  throw std::runtime_error(
271  "Error, ephemeris orientation of body " + bodyIterator->first
272  + " is not the same as global orientation " + ephemerisFrameOrientation
273  + ", " + globalFrameOrientation );
274  }
275 
276 
277  }
278 
279  // Set global frame origin identifiers
280  if( globalFrameOrigin == bodyIterator->first )
281  {
282  bodyIterator->second->setIsBodyGlobalFrameOrigin( 1 );
283  }
284  else
285  {
286  bodyIterator->second->setIsBodyGlobalFrameOrigin( 0 );
287  }
288 
289  // Check if body has rotational ephemeris.
290  if( bodyIterator->second->getRotationalEphemeris( ) != nullptr )
291  {
292  // Check if rotational ephemeris base frame orienatation is equal to to global orientation.
293  rotationModelFrame = bodyIterator->second->getRotationalEphemeris( )->getBaseFrameOrientation( );
294 
295  // Throw error if two frames are not equal.
296  if( rotationModelFrame != globalFrameOrientation )
297  {
298  throw std::runtime_error(
299  "Error, rotation base orientation of body " + bodyIterator->first +
300  " is not the same as global orientation " + rotationModelFrame + ", " +
301  globalFrameOrientation );
302  }
303  }
304  }
305 
306  // Set body state-dependent environment variables
307  for( NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
308  bodyIterator != bodyMap.end( ); bodyIterator++ )
309  {
310  bodyIterator->second->updateConstantEphemerisDependentMemberQuantities( );
311  }
312 
313 }
