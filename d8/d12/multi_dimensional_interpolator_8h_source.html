54  MultiDimensionalInterpolator(
55  const std::vector< BoundaryInterpolationType >& boundaryHandling =
56  std::vector< BoundaryInterpolationType >( NumberOfDimensions, extrapolate_at_boundary ),
57  const std::vector< std::pair< DependentVariableType, DependentVariableType > >& defaultExtrapolationValue =
58  std::vector< std::pair< DependentVariableType, DependentVariableType > >(
59  NumberOfDimensions, std::make_pair( IdentityElement::getAdditionIdentity< DependentVariableType >( ),
60  IdentityElement::getAdditionIdentity< DependentVariableType >( ) ) ) ) :
61  boundaryHandling_( boundaryHandling ), defaultExtrapolationValue_( defaultExtrapolationValue )
62  {
63  // Check that the user-defined default value does not correspond to a boundary method that does not use the
64  // default value
65  std::pair< DependentVariableType, DependentVariableType > defaultValueForBoundaryHandling =
66  std::make_pair( IdentityElement::getAdditionIdentity< DependentVariableType >( ),
67  IdentityElement::getAdditionIdentity< DependentVariableType >( ) );
68  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
69  {
70  if ( defaultExtrapolationValue_.at( i ) != defaultValueForBoundaryHandling )
71  {
72  if ( !( ( boundaryHandling_.at( i ) == use_default_value ) ||
73  ( boundaryHandling_.at( i ) == use_default_value_with_warning ) ) )
74  {
75  std::cerr << "Warning in multi-dimensional interpolator. A default value has been set (for when the "
76  "independent variable is out-of-range) but the boundary handling method is not "
77  "use_default_value or use_default_value_with_warning. The method selected for dimension " <<
78  std::to_string( i ) << " is: " <<
79  std::to_string( boundaryHandling_.at( i ) ) << std::endl;
80  }
81  }

183  int checkInterpolationBoundary( const IndependentVariableType& currentIndependentVariable,
184  const unsigned int& currentDimension )

209  void checkBoundaryCase(
210  const unsigned int currentDimension,
211  bool& useValue,
212  IndependentVariableType& currentIndependentVariable,
213  DependentVariableType& dependentVariable )
214  {
215  // If extrapolation outside domain is not allowed
216  if ( boundaryHandling_.at( currentDimension ) != extrapolate_at_boundary )
217  {
218  // If independent variable is out of range
219  int isAtBoundary = checkInterpolationBoundary( currentIndependentVariable, currentDimension );
220  if ( isAtBoundary != 0 )
221  {
222  // Select course of action based on boundary handling method selected
223  switch ( boundaryHandling_.at( currentDimension ) )
224  {
225  case throw_exception_at_boundary:
226  {
227  // Throw exception
228  std::string errorMessage = "Error in interpolator, requesting data point outside of boundaries, requested data of dimension " +
229  std::to_string( currentDimension ) + " at " +
230  std::to_string( currentIndependentVariable ) + " but limit values are " +
231  std::to_string( independentValues_.at( currentDimension ).front( ) ) + " and " +
232  std::to_string( independentValues_.at( currentDimension ).back( ) );
233  throw std::runtime_error( errorMessage );
234  break;
235  }
236  case extrapolate_at_boundary_with_warning:
237  {
238  // Warn user
239  std::string errorMessage = "Warning in interpolator, requesting data point outside of boundaries, requested data of dimension " +
240  std::to_string( currentDimension ) + " at " +
241  std::to_string( currentIndependentVariable ) + " but limit values are " +
242  std::to_string( independentValues_.at( currentDimension ).front( ) ) + " and " +
243  std::to_string( independentValues_.at( currentDimension ).back( ) ) + ", applying extrapolation instead.";
244  std::cerr << errorMessage << std::endl;
245  break;
246  }
247  case use_boundary_value:
248  case use_boundary_value_with_warning:
249  {
250  // Warn user, if requested
251  if ( boundaryHandling_.at( currentDimension ) == use_boundary_value_with_warning )
252  {
253  std::string errorMessage = "Warning in interpolator, requesting data point outside of boundaries, requested data of dimension " +
254  std::to_string( currentDimension ) + " at " +
255  std::to_string( currentIndependentVariable ) + " but limit values are " +
256  std::to_string( independentValues_.at( currentDimension ).front( ) ) + " and " +
257  std::to_string( independentValues_.at( currentDimension ).back( ) ) + ", taking boundary value instead.";
258  std::cerr << errorMessage << std::endl;
259  }
260 
261  // Replace current independent variable with boundary value
262  if ( isAtBoundary == -1 )
263  {
264  currentIndependentVariable = independentValues_.at( currentDimension ).front( );
265  }
266  else if ( isAtBoundary == 1 )
267  {
268  currentIndependentVariable = independentValues_.at( currentDimension ).back( );
269  }
270  break;
271  }
272  case use_default_value:
273  case use_default_value_with_warning:
274  {
275  // Warn user, if requested
276  if ( boundaryHandling_.at( currentDimension ) == use_default_value_with_warning )
277  {
278  std::string errorMessage = "Warning in interpolator, requesting data point outside of boundaries, requested data of dimension " +
279  std::to_string( currentDimension ) + " at " +
280  std::to_string( currentIndependentVariable ) + " but limit values are " +
281  std::to_string( independentValues_.at( currentDimension ).front( ) ) + " and " +
282  std::to_string( independentValues_.at( currentDimension ).back( ) ) + ", taking default value instead.";
283  std::cerr << errorMessage << std::endl;
284  }
285 
286  // Get default value
287  useValue = true;
288  if ( isAtBoundary == -1 )
289  {
290  dependentVariable = defaultExtrapolationValue_.at( currentDimension ).first;
291  }
292  else if ( isAtBoundary == 1 )
293  {
294  dependentVariable = defaultExtrapolationValue_.at( currentDimension ).second;
295  }
296  else
297  {
298  throw std::runtime_error( "Error when checking interpolation boundary, inconsistent data encountered" );
299  }
300  break;
301  }
302  default:
303  throw std::runtime_error( "Error when checking interpolation boundary, boundary handling method not recognized." );
304  }
305  }
306  }

317  void makeLookupSchemes( const AvailableLookupScheme selectedScheme )
318  {
319  // Find which type of scheme is used.
320  lookUpSchemes_.resize( NumberOfDimensions );
321  switch ( selectedScheme )
322  {
323  case binarySearch:
324  {
325  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
326  {
327  // Create binary search look up scheme.
328  lookUpSchemes_[ i ] = std::shared_ptr< LookUpScheme< IndependentVariableType > >
329  ( new BinarySearchLookupScheme< IndependentVariableType >(
330  independentValues_[ i ] ) );
331  }
332  break;
333  }
334  case huntingAlgorithm:
335  {
336  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
337  {
338  // Create hunting scheme, which uses an intial guess from previous look-ups.
339  lookUpSchemes_[ i ] = std::shared_ptr< LookUpScheme< IndependentVariableType > >
340  ( new HuntingAlgorithmLookupScheme< IndependentVariableType >(
341  independentValues_[ i ] ) );
342  }
343  break;
344  }
345  default:
346  throw std::runtime_error( "Error: lookup scheme not found when making scheme for N-D interpolator." );
347  }
348  }
