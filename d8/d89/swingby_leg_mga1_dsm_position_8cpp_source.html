133 void SwingbyLegMga1DsmPosition::calculateDsmLocation( )
134 {
135  // Calculate the required unit vectors
136  const Eigen::Vector3d unitVector1 = departureBodyPosition_ / departureBodyPosition_.norm( ) ;
137  const Eigen::Vector3d unitVector3 = unitVector1.cross( departureBodyVelocity_ ) /
138  ( unitVector1.cross( departureBodyVelocity_ ) ).norm( );
139  const Eigen::Vector3d unitVector2 = unitVector3.cross( unitVector1 );
140 
141  // Calculate the absolute DSM radius.
142  const double absoluteRadiusDsm = dimensionlessRadiusDsm_ * departureBodyPosition_.norm( );
143 
144  // Calculate the radius in the central body reference frame.
145  dsmLocation_ = std::cos( inPlaneAngle_ ) * std::cos( outOfPlaneAngle_ ) * absoluteRadiusDsm *
146  unitVector1 +
147  std::sin( inPlaneAngle_ ) * std::cos( outOfPlaneAngle_ ) * absoluteRadiusDsm *
148  unitVector2 +
