74 template< typename IndependentVariableType >
75 int computeNearestLeftNeighborUsingBinarySearch(
76  const std::vector< IndependentVariableType > vectorOfSortedData,
77  const IndependentVariableType targetValueInVectorOfSortedData )
78 {
79  // Declare local variables.
80  // Declare bounds of vector of sorted data and current position.
81  int leftLimitOfVectorOfSortedData = 0;
82  int rightLimitOfVectorOfSortedData = vectorOfSortedData.size( ) - 1;
83  int currentPositionInVectorOfSortedData;
84 
85  // Check if data is sorted in ascending order.
86  // ( true if ascending, else false ).
87  bool isVectorOfSortedDataAscending
88  = ( vectorOfSortedData[ rightLimitOfVectorOfSortedData ]
89  >= vectorOfSortedData[ leftLimitOfVectorOfSortedData ] );
90 
91  // Loop through vector of sorted data until left and right limits
92  // are neighbours.
93  while ( rightLimitOfVectorOfSortedData
94  - leftLimitOfVectorOfSortedData > 1 )
95  {
96  // Compute midpoint ( bitshift is same as division by 2.0 ).
97  currentPositionInVectorOfSortedData
98  = ( rightLimitOfVectorOfSortedData
99  + leftLimitOfVectorOfSortedData ) >> 1;
100 
101  // Check which limit to replace ( if ascending and target datum
102  // is in right half, replace left limit ).
103  if ( targetValueInVectorOfSortedData
104  >= vectorOfSortedData[ currentPositionInVectorOfSortedData ]
105  && isVectorOfSortedDataAscending )
106  {
107  // Set left limit to current position in vector of sorted data.
108  leftLimitOfVectorOfSortedData = currentPositionInVectorOfSortedData;
109  }
110 
111  else
112  {
113  // Set right limit to current position in vector of sorted data.
114  rightLimitOfVectorOfSortedData = currentPositionInVectorOfSortedData;
115  }
116  }
117 
118  // Set current position to left limit.
119  currentPositionInVectorOfSortedData = leftLimitOfVectorOfSortedData;
120 
121  // Return current position in vector.
122  return currentPositionInVectorOfSortedData;
123 }

169 template< typename IndependentVariableType >
170 int findNearestLeftNeighbourUsingHuntingAlgorithm(
171  const IndependentVariableType independentVariableValue,
172  const int previousNearestLowerIndex_,
173  const std::vector< IndependentVariableType >& independentValues_ )
174 {
175  // Initialize return variable for new nearest left neighbor.
176  int newNearestLowerIndex = 0;
177 
178  // Initialize boolean denoting whether the new value has been found.
179  bool isFound = 0;
180 
181  int independentValueVectorSize = static_cast< int >( independentValues_.size( ) );
182 
183  if( !( independentVariableValue == independentVariableValue ) )
184  {
185  throw std::runtime_error( "Error in nearest left neighbour search, input is NaN" );
186  }
187 
188  if( independentValueVectorSize < 2 )
189  {
190  throw std::runtime_error( "Error in nearest neighbour search, size of input vector is " +
191  std::to_string( independentValueVectorSize ) );
192  }
193 
194  // Check whether initial estimate is possible.
195  if ( previousNearestLowerIndex_ < 0 ||
196  previousNearestLowerIndex_ > static_cast< int >( independentValues_.size( ) - 2 ) )
197  {
198  throw std::runtime_error( "Error, initial guess for nearest neighbour search not within allowable bounds." );
199  }
200 
201  // Check if independent variable value falls within region of values provided.
202  if ( independentVariableValue <= independentValues_[ 0 ] )
203  {
204  newNearestLowerIndex = 0;
205  isFound = 1;
206  }
207 
208  else if ( independentVariableValue >= independentValues_[ independentValueVectorSize - 1 ] )
209  {
210  newNearestLowerIndex = independentValueVectorSize - 2;
211  isFound = 1;
212  }
213 
214  // Perform hunting algorithm if requested value is not outside given range.
215  if ( !isFound )
216  {
217  // Create indices delimiting interval under consideration.
218  int upperIndex, lowerIndex;
219 
220  // Create size and direction of jumps.
221  int jumpSize = 1;
222  int jumpDirection;
223 
224  // Check direction of jumps that is needed.
225  if ( independentVariableValue >= independentValues_[ previousNearestLowerIndex_ ] )
226  {
227  jumpDirection = 1;
228  lowerIndex = previousNearestLowerIndex_;
229  upperIndex = previousNearestLowerIndex_ + 1;
230  }
231  else
232  {
233  jumpDirection = -1;
234  lowerIndex = previousNearestLowerIndex_ - 1;
235  upperIndex = previousNearestLowerIndex_;
236  }
237 
238  // Boolean to denote whether the value is in te interval under consideration.
239  bool isRegionReached = 0;
240 
241  // While nearest lower neighbour is not found, continue algorithm.
242  while ( !isFound )
243  {
244  // Check if value is in regio under consideration.
245  if( independentVariableValue >= independentValues_[ lowerIndex ] &&
246  independentVariableValue <= independentValues_[ upperIndex ] )
247  {
248  isRegionReached = 1;
249  }
250 
251  // If not in current region:
252  else
253  {
254  // Double jump size.
255  jumpSize *= 2;
256  if ( jumpDirection == 1 )
257  {
258  // Reset interval under consideration.
259  lowerIndex = upperIndex;
260  upperIndex += jumpSize;
261 
262  // Correct jump size if it would bring interval out of bounds.
263  if ( upperIndex > independentValueVectorSize - 1 )
264  {
265  upperIndex = independentValueVectorSize - 1;
266  jumpSize = upperIndex - lowerIndex;
267  }
268  }
269  else
270  {
271  // Reset interval under consideration.
272  upperIndex = lowerIndex;
273  lowerIndex -= jumpSize;
274 
275  // Correct jump size if it would bring interval out of bounds.
276  if ( lowerIndex < 0 )
277  {
278  lowerIndex = 0;
279  jumpSize = upperIndex;
280  }
281  }
282  }
283 
284  // Independent variable value has been identified to be between upperIndex and
285  // lowerIndex. Perform bisection of interval to decrease interval to size 1.
286  if ( isRegionReached )
287  {
288  while( !isFound )
289  {
290 
291  int middleIndex;
292 
293  if( !( upperIndex - lowerIndex > 0 ) )
294  {
295  throw std::runtime_error( "Error, upper and lower indices are inconsistent in nearest neighbour search" +
296  std::to_string( upperIndex ) + " " +
297  std::to_string( lowerIndex ) );
298  }
299 
300  // If the upper and lower indices have a difference of exactly one, the
301  // interval has been found.
302  if ( upperIndex - lowerIndex == 1 )
303  {
304  isFound = 1;
305  newNearestLowerIndex = lowerIndex;
306  }
307  else
308  {
309  // Half size of interval.
310  jumpSize /= 2;
311  middleIndex = lowerIndex + jumpSize;
312 
313  // Check which half of interval is to be new interval.
314  if ( independentVariableValue < independentValues_[ middleIndex ] )
315  {
316  upperIndex = middleIndex;
317  }
318  else
319  {
320  lowerIndex = middleIndex;
321  }
322 
323  jumpSize = upperIndex - lowerIndex;
324  }
325  }
326  }
327  }
328  }
329 
330  return newNearestLowerIndex;
331 }
