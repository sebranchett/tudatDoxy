60 ObservationViabilitySettingsList filterObservationViabilitySettings(
61  const ObservationViabilitySettingsList& observationViabilitySettings,
62  const LinkEnds& linkEnds )
63 {
64  ObservationViabilitySettingsList filteredViabilitySettings;
65 
66  // Iterate over all viability settings
67  for( unsigned int i = 0; i < observationViabilitySettings.size( ); i++ )
68  {
69  // Iterate over all link ends
70  for( LinkEnds::const_iterator linkEndIterator = linkEnds.begin( ); linkEndIterator != linkEnds.end( ); linkEndIterator++ )
71  {
72  // Check if present viabilitytt setting is relevant
73  if( linkEndIterator->second == observationViabilitySettings.at( i )->getAssociatedLinkEnd( ) ||
74  ( ( observationViabilitySettings.at( i )->getAssociatedLinkEnd( ).second == "" ) &&
75  ( observationViabilitySettings.at( i )->getAssociatedLinkEnd( ).first == linkEndIterator->second.first ) ) )
76  {
77  filteredViabilitySettings.push_back( observationViabilitySettings.at( i ) );
78  break;
79  }
80  }
81  }
82 
83  return filteredViabilitySettings;
84 }

87 std::vector< std::pair< int, int > > getLinkEndIndicesForObservationViability(
88  const LinkEnds& linkEnds, const ObservableType observableType, const LinkEndId linkEndToCheck )
89 {
90  std::vector< std::pair< int, int > > linkEndIndices;
91 
92  switch( observableType )
93  {
94  case one_way_range:
95  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) ||
96  ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) && ( linkEndToCheck.second == "" ) ) )
97  {
98  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
99  }
100  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
101  linkEndToCheck.second == "" ) )
102  {
103  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
104  }
105  else
106  {
107  throw std::runtime_error( "Error, parsed irrelevant 1-way range viability indices" );
108  }
109  break;
110  case one_way_doppler:
111  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
112  ( linkEndToCheck.second == "" ) ) )
113  {
114  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
115  }
116  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
117  linkEndToCheck.second == "" ) )
118  {
119  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
120  }
121  else
122  {
123  throw std::runtime_error( "Error, parsed irrelevant 1-way doppler viability indices" );
124  }
125  break;
126  case two_way_doppler:
127  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
128  ( linkEndToCheck.second == "" ) ) )
129  {
130  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
131  }
132 
133  if( linkEnds.at( reflector1 ) == linkEndToCheck || ( ( linkEnds.at( reflector1 ).first == linkEndToCheck.first ) &&
134  linkEndToCheck.second == "" ) )
135  {
136  linkEndIndices.push_back( std::make_pair( 2, 3 ) );
137  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
138  }
139 
140  if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
141  linkEndToCheck.second == "" ) )
142  {
143  linkEndIndices.push_back( std::make_pair( 3, 2 ) );
144  }
145 
146  if( linkEndIndices.size( ) == 0 )
147  {
148  throw std::runtime_error( "Error, parsed irrelevant 1-way doppler viability indices" );
149  }
150  break;
151  case one_way_differenced_range:
152  if( linkEnds.at( transmitter ) == linkEndToCheck || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
153  linkEndToCheck.second == "" ) )
154  {
155  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
156  linkEndIndices.push_back( std::make_pair( 2, 3 ) );
157  }
158  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
159  linkEndToCheck.second == "" ) )
160  {
161  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
162  linkEndIndices.push_back( std::make_pair( 3, 2 ) );
163  }
164  else
165  {
166  throw std::runtime_error( "Error, parsed irrelevant 1-way differenced range viability indices" );
167  }
168  break;
169  case n_way_range:
170  {
171  std::vector< int > matchingLinkEndIndices = getNWayLinkEndIndicesFromLinkEndId( linkEndToCheck, linkEnds );
172  if( matchingLinkEndIndices.size( ) > 0 )
173  {
174  for( unsigned int i = 0; i < matchingLinkEndIndices.size( ); i++ )
175  {
176  if( matchingLinkEndIndices.at( i ) == 0 )
177  {
178  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
179  }
180  else if( matchingLinkEndIndices.at( i ) == static_cast< int >( linkEnds.size( ) ) - 1 )
181  {
182  linkEndIndices.push_back( std::make_pair( 2 * ( linkEnds.size( ) - 1 ) - 1,
183  2 * ( linkEnds.size( ) - 1 ) - 2 ) );
184  }
185  else
186  {
187  linkEndIndices.push_back(
188  std::make_pair( 2 * matchingLinkEndIndices.at( i ),
189  2 * matchingLinkEndIndices.at( i ) + 1 ) );
190  linkEndIndices.push_back(
191  std::make_pair( 2 * matchingLinkEndIndices.at( i ) - 1,
192  2 * matchingLinkEndIndices.at( i ) - 2 ) );
193  }
194  }
195  }
196  else
197  {
198  throw std::runtime_error( "Error, parsed irrelevant n-way range viability indices" );
199  }
200  break;
201  }
202  case angular_position:
203  if( ( linkEnds.at( transmitter ) == linkEndToCheck ) || ( ( linkEnds.at( transmitter ).first == linkEndToCheck.first ) &&
204  ( linkEndToCheck.second == "" ) ) )
205  {
206  linkEndIndices.push_back( std::make_pair( 0, 1 ) );
207  }
208  else if( linkEnds.at( receiver ) == linkEndToCheck || ( ( linkEnds.at( receiver ).first == linkEndToCheck.first ) &&
209  linkEndToCheck.second == "" ) )
210  {
211  linkEndIndices.push_back( std::make_pair( 1, 0 ) );
212  }
213  else
214  {
215  throw std::runtime_error( "Error, parsed irrelevant angular position viability indices" );
216  }
217  break;
218  case position_observable:
219 
220  throw std::runtime_error( "Error, parsed irrelevant position observable viability indices" );
221  break;
222  case euler_angle_313_observable:
223 
224  throw std::runtime_error( "Error, parsed irrelevant euler angle observable viability indices" );
225  break;
226  case velocity_observable:
227 
228  throw std::runtime_error( "Error, parsed irrelevant position observable viability indices" );
229  break;
230  default:
231  throw std::runtime_error( "Error, observable type " + std::to_string(
232  observableType ) + " not recognized when making viability link ends" );
233 
234  }
235 
236  return linkEndIndices;
237 }

240 std::shared_ptr< MinimumElevationAngleCalculator > createMinimumElevationAngleCalculator(
241  const simulation_setup::NamedBodyMap& bodyMap,
242  const LinkEnds linkEnds,
243  const ObservableType observationType,
244  const std::shared_ptr< ObservationViabilitySettings > observationViabilitySettings,
245  const std::string& stationName )
246 {
247  if( observationViabilitySettings->observationViabilityType_ != minimum_elevation_angle )
248  {
249  throw std::runtime_error( "Error when making minimum elevation angle calculator, inconsistent input" );
250  }
251 
252  // If specific link end is specified
253  std::string groundStationNameToUse;
254  if( observationViabilitySettings->getAssociatedLinkEnd( ).second != "" )
255  {
256  if( groundStationNameToUse != stationName )
257  {
258  throw std::runtime_error( "Error when making minimum elevation angle calculator, inconsistent station input" );
259  }
260  groundStationNameToUse = observationViabilitySettings->getAssociatedLinkEnd( ).second;
261  }
262  else
263  {
264  groundStationNameToUse = stationName;
265  }
266 
267  if( bodyMap.count( observationViabilitySettings->getAssociatedLinkEnd( ).first ) == 0 )
268  {
269  throw std::runtime_error( "Error when making minimum elevation angle calculator, body " +
270  observationViabilitySettings->getAssociatedLinkEnd( ).first + " not found." );
271  }
272 
273  // Retrieve pointing angles calculator
274  std::shared_ptr< ground_stations::PointingAnglesCalculator > pointingAngleCalculator =
275  bodyMap.at( observationViabilitySettings->getAssociatedLinkEnd( ).first )->
276  getGroundStation( groundStationNameToUse )->getPointingAnglesCalculator( );
277 
278  // Create check object
279  double minimumElevationAngle = observationViabilitySettings->getDoubleParameter( );
280  return std::make_shared< MinimumElevationAngleCalculator >(
281  getLinkEndIndicesForObservationViability(
282  linkEnds,observationType, observationViabilitySettings->getAssociatedLinkEnd( ) ),
283  minimumElevationAngle, pointingAngleCalculator );

355 std::vector< std::shared_ptr< ObservationViabilityCalculator > > createObservationViabilityCalculators(
356  const simulation_setup::NamedBodyMap& bodyMap,
357  const LinkEnds linkEnds,
358  const ObservableType observationType,
359  const std::vector< std::shared_ptr< ObservationViabilitySettings > >& observationViabilitySettings )
360 {
361  std::vector< std::shared_ptr< ObservationViabilityCalculator > > linkViabilityCalculators;
362 
363  std::vector< std::shared_ptr< ObservationViabilitySettings > > relevantObservationViabilitySettings =
364  filterObservationViabilitySettings( observationViabilitySettings, linkEnds );
365 
366  for( unsigned int i = 0; i < relevantObservationViabilitySettings.size( ); i++ )
367  {
368  switch( relevantObservationViabilitySettings.at( i )->observationViabilityType_ )
369  {
370  case minimum_elevation_angle:
371  {
372  // Create list of ground stations for which elevation angle check is to be made.
373  std::vector< std::string > listOfGroundStations;
374  for( LinkEnds::const_iterator linkEndIterator = linkEnds.begin( );
375  linkEndIterator != linkEnds.end( ); linkEndIterator++ )
376  {
377  if( linkEndIterator->second.first == relevantObservationViabilitySettings.at( i )->getAssociatedLinkEnd( ).first )
378  {
379  if( std::find( listOfGroundStations.begin( ), listOfGroundStations.end( ), linkEndIterator->second.second ) ==
380  listOfGroundStations.end( ) )
381  {
382  listOfGroundStations.push_back( linkEndIterator->second.second );
383  }
384  }
385  }
386 
387  // Create elevation angle check separately for eah ground station: check requires different pointing angles calculator
388  for( unsigned int j = 0; j < listOfGroundStations.size( ); j++ )
389  {
390  linkViabilityCalculators.push_back(
391  createMinimumElevationAngleCalculator(
392  bodyMap, linkEnds, observationType, relevantObservationViabilitySettings.at( i ),
393  listOfGroundStations.at( j ) ) );
394  }
395  break;
396  }
397  case body_avoidance_angle:
398 
399  linkViabilityCalculators.push_back(
400  createBodyAvoidanceAngleCalculator(
401  bodyMap, linkEnds, observationType, relevantObservationViabilitySettings.at( i ) ) );
402  break;
403  case body_occultation:
404 
405  linkViabilityCalculators.push_back(
406  createOccultationCalculator(
407  bodyMap, linkEnds, observationType, relevantObservationViabilitySettings.at( i ) ) );
408  break;
409  default:
410  throw std::runtime_error(
411  "Error when making observation viability calculator, type not recognized " +
412  std::to_string(
413  relevantObservationViabilitySettings.at( i )->observationViabilityType_ ) );
414  }
415 
416  }
417 
418  return linkViabilityCalculators;
419 }
