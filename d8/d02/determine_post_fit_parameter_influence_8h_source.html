49 template< typename TimeType = double, typename StateScalarType = double >
50 std::pair< std::shared_ptr< PodOutput< StateScalarType > >, Eigen::VectorXd > determinePostfitParameterInfluence(
51  const NamedBodyMap& bodyMap,
52  const std::shared_ptr< numerical_integrators::IntegratorSettings< TimeType > > integratorSettings,
53  const std::shared_ptr< propagators::PropagatorSettings< StateScalarType > > propagatorSettings,
54  const std::shared_ptr< estimatable_parameters::EstimatableParameterSettings > perturbedParameterSettings,
55  const double simulatedObservationInterval,
56  const std::vector< double > parameterPerturbations,
57  const std::vector< int > parameterIndices,
58  const int numberOfIterations = 2 )
59 {
60  using namespace observation_models;
61  using namespace estimatable_parameters;

63  // Check input consistency
64  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< StateScalarType > > translationalPropagatorSettings =
65  std::dynamic_pointer_cast< propagators::TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings ) ;
66  if( translationalPropagatorSettings == nullptr )
67  {
68  throw std::runtime_error(
69  "Error in determinePostfitParameterInfluence, only single-arc translational dynamics currently supported" );
70  }
71 
72  // Getlist of bodies for which the dynamics is to be fit
73  std::vector< std::string > observedBodies = translationalPropagatorSettings->bodiesToIntegrate_;
74 
75  // Create list of ideal observation settings and initial states to estimate
76  std::vector< LinkEnds > linkEndsList;
77  ObservationSettingsMap observationSettingsMap;
78  std::vector< std::shared_ptr< EstimatableParameterSettings > > initialStateParameterNames;
79 
80  for( unsigned int i = 0; i < observedBodies.size( ); i++ )
81  {
82  // Add current body to list of observed bodies
83  LinkEnds observationLinkEnds;
84  observationLinkEnds[ observed_body ] = std::make_pair( observedBodies.at( i ), "" );
85  linkEndsList.push_back( observationLinkEnds );
86  observationSettingsMap.insert(
87  std::make_pair( observationLinkEnds, std::make_shared< ObservationSettings >(
88  position_observable ) ) );
89 
90  // Add current body to list of estimated bodies
91  initialStateParameterNames.push_back(
92  std::make_shared< InitialTranslationalStateEstimatableParameterSettings< StateScalarType > >(
93  observedBodies.at( i ), translationalPropagatorSettings->getInitialStates( ).segment( i * 6, 6 ),
94  translationalPropagatorSettings->centralBodies_.at( i ) ) );
95  }
96 
97  // Create initial state estimation objects
98  std::shared_ptr< EstimatableParameterSet< StateScalarType > > initialStateParametersToEstimate =
99  createParametersToEstimate< StateScalarType >( initialStateParameterNames, bodyMap );
100 
101  // Create orbit determination object.
102  OrbitDeterminationManager< StateScalarType, TimeType > orbitDeterminationManager =
103  OrbitDeterminationManager< StateScalarType, TimeType >(
104  bodyMap, initialStateParametersToEstimate, observationSettingsMap,
105  integratorSettings, propagatorSettings );
106 
107  // Retrieve nominal (e.g. pre-fit) body states
108  Eigen::VectorXd nominalBodyStates = initialStateParametersToEstimate->template getFullParameterValues< double >( );
109 
110  // Get range over which observations are to be simulated.
111  std::pair< double, double > dataTimeInterval =
112  getTabulatedEphemerisSafeInterval( bodyMap.at( observedBodies.at( 0 ) )->getEphemeris( ) );
113  double startTime = dataTimeInterval.first;
114  double endTime = dataTimeInterval.second;
115 
116  // Define list of times at which observations are to be simulated
117  std::vector< TimeType > baseTimeList;
118  double currentTime = startTime;
119  while( currentTime < endTime )
120  {
121  baseTimeList.push_back( currentTime );
122  currentTime += simulatedObservationInterval;
123  }
124  std::map< ObservableType, std::map< LinkEnds, std::pair< std::vector< TimeType >, LinkEndType > > > measurementSimulationInput;
125  for( unsigned int i = 0; i < linkEndsList.size( ); i++ )
126  {
127  measurementSimulationInput[ position_observable ][ linkEndsList.at( i ) ] =
128  std::make_pair( baseTimeList, observed_body );
129  }
130 
131  // Simulate ideal observations

133  typedef std::map< LinkEnds, std::pair< ObservationVectorType, std::pair< std::vector< TimeType >, LinkEndType > > > SingleObservablePodInputType;
134  typedef std::map< ObservableType, SingleObservablePodInputType > PodInputDataType;
135  PodInputDataType observationsAndTimes = simulateObservations< StateScalarType, TimeType >(
136  measurementSimulationInput, orbitDeterminationManager.getObservationSimulators( ) );

138 
139  // Define estimation input
140  std::shared_ptr< PodInput< StateScalarType, TimeType > > podInput =
141  std::make_shared< PodInput< StateScalarType, TimeType > >(
142  observationsAndTimes, initialStateParametersToEstimate->getParameterSetSize( ) );
143  podInput->defineEstimationSettings( true, true, false, true, true );
144 
145  // Create parameters that are to be perturbed
146  std::vector< std::shared_ptr< EstimatableParameterSettings > > perturbedParameterSettingsList;
147  perturbedParameterSettingsList.push_back( perturbedParameterSettings );
148  std::shared_ptr< EstimatableParameterSet< StateScalarType > > perturbedParameters =
149  createParametersToEstimate< StateScalarType >( perturbedParameterSettingsList, bodyMap );
150 
151  // Perturb parameters by required amount
152  Eigen::VectorXd parameterVectorToPerturb = perturbedParameters->template getFullParameterValues< double >( );
153  Eigen::VectorXd unperturbedParameterVector = parameterVectorToPerturb;
154  for( unsigned int i = 0; i < parameterPerturbations.size( ); i++ )
155  {
156  if( parameterIndices.at( i ) >= parameterVectorToPerturb.rows( ) )
157  {
158  throw std::runtime_error( "Error, inconsistent input to determinePostfitParameterInfluence function, perturbation index is too high" );
159  }
160  else
161  {
162  parameterVectorToPerturb( parameterIndices.at( i ) ) += parameterPerturbations.at( i );
163  }
164  }
165  perturbedParameters->resetParameterValues( parameterVectorToPerturb );
166 
167  // Fit nominal dynamics to pertrubed dynamical model
168  std::shared_ptr< PodOutput< StateScalarType > > podOutput = orbitDeterminationManager.estimateParameters(
169  podInput, std::make_shared< EstimationConvergenceChecker >( numberOfIterations ) );
170 
171  // Reset parameter to nominal value
172  perturbedParameters->resetParameterValues( unperturbedParameterVector );
173 
174  return std::make_pair(
175  podOutput, initialStateParametersToEstimate->template getFullParameterValues< double >( ) - nominalBodyStates );
