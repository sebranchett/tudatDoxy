63 ReplaceElements::ReplaceElements( std::string searchFilter, std::string replaceString,
64  bool isOmitIfEmpty )
65  // line_filter is set to true, to indicate to Boost not to append newline characters.
66  : boost::iostreams::line_filter( true ),
67  replaceString_( replaceString ), isOmitIfEmpty_( isOmitIfEmpty )
68 {
69  // The following are all characters with special meaning in regex, so escape them:
70  std::vector< std::string > replaceCharacters_ =
71  { "\\", ".", "[", "]", "{", "}", "(", ")", "*", "+", "?", "|", "^", "$" };
72 
73  // Iterate over each possible character.
74  for ( unsigned int i = 0; i < replaceCharacters_.size( ); i++ )
75  {
76  // For each of the special characters, replace it with the escaped '\' version of that
77  // character.

79  "\\" + replaceCharacters_[ i ] );
80  }
