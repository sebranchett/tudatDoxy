42 void MissileDatcomData::convertDatcomData( const std::vector< double >& datcomData )
43 {
44  std::vector< double > datcomDataToBeConverted = datcomData;
45 
46  // Add an extra item at the begin, such that the first real value is at entry 1.
47  // The same indices as described in the MissileDatcom user manual can now be used.
48  std::vector< double >::iterator it;
49  it = datcomDataToBeConverted.begin( );
50  datcomDataToBeConverted.insert( it,-0.0 );
51 
52  // The first "card" is the Flight condition data array, which is the same for every "case".
53  numberOfAnglesOfAttack_ = datcomDataToBeConverted[ 1 ];
54 
55  for ( int iteratorDataVector_= 2; iteratorDataVector_ < ( 2 + numberOfAnglesOfAttack_ );
56  iteratorDataVector_++ )
57  {
58  angleOfAttack_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
59  }
60 
61  sideslipAngle_ = datcomDataToBeConverted[ 22 ];
62 
63  rollAngle_ = datcomDataToBeConverted[ 23 ];
64 
65  numberOfMachNumbers_ = datcomDataToBeConverted[ 24 ];
66 
67  for ( int iteratorDataVector_= 25; iteratorDataVector_ < ( 25 + numberOfMachNumbers_ );
68  iteratorDataVector_++ )
69  {
70  machNumber_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
71  }
72 
73  for ( int iteratorDataVector_= 45; iteratorDataVector_ < ( 25 + numberOfMachNumbers_ );
74  iteratorDataVector_++ )
75  {
76  altitudes_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
77  }
78 
79  for ( int iteratorDataVector_= 66; iteratorDataVector_ < ( 66 + numberOfMachNumbers_ );
80  iteratorDataVector_++ )
81  {
82  reynoldsNumbers_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
83  }
84 
85  for ( int iteratorDataVector_= 86; iteratorDataVector_ < (86 +numberOfMachNumbers_);
86  iteratorDataVector_++ )
87  {
88  freeStreamVelocities_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
89  }
90 
91  for ( int iteratorDataVector_= 106; iteratorDataVector_ < (106 + numberOfMachNumbers_);
92  iteratorDataVector_++ )
93  {
94  freeStreamStaticTemperatures_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
95  }
96 
97  // NOTE apperently the last entry of flight card is not written to de for004.dat file.
98  // According to the user manual the last entry should be 145, this entry is not found in the
99  // for004.dat file.
100  for ( int iteratorDataVector_= 126; iteratorDataVector_ <= 144; iteratorDataVector_++ )
101  {
102  freeStreamStaticPressure_.push_back( datcomDataToBeConverted[ iteratorDataVector_ ] );
103  }
104 
105  // Loop over Mach, angle of attack and coefficient index to fill the static coefficient array.
106  for (int machIndex = 0; machIndex < numberOfMachNumbers_; machIndex++ )
107  {
108  for (int angleOfAttackIndex = 0; angleOfAttackIndex < numberOfAnglesOfAttack_;
109  angleOfAttackIndex++ )
110  {
111  for (int coefficientIndex = 0; coefficientIndex < 6; coefficientIndex++ )
112  {
113  // The second index begins at entry 145
114  // The first section has 144 entries, the second 220 and the last 400. Therefore
115  // there is a repetition after 144+220+400=764 entries.
116  // every coefficients has 20 entries.
117  int dataVectorIndex_ = 145 + 764 * machIndex + angleOfAttackIndex
118  + 20 * coefficientIndex;
119  staticCoefficients_[ machIndex ][ angleOfAttackIndex ][ coefficientIndex ] =
120  datcomDataToBeConverted[ dataVectorIndex_ ];
121  }
122 
123  for ( int coefficientIndex = 6; coefficientIndex < 11; coefficientIndex++ )
124  {
125  // The second index begins at entry 145
126  // The first section has 144 entries, the second 220 and the last 400. Therefore
127  // there is a repetition after 144+220+400=764 entries.
128  // every coefficients has 20 entries.
129  // The coefficients are in per degree and needs to be converted to per radian
130  int dataVectorIndex_ = 145 + 764 * machIndex + angleOfAttackIndex
131  + 20 * coefficientIndex;
132  staticCoefficients_[ machIndex ][ angleOfAttackIndex ][ coefficientIndex ] =
133  datcomDataToBeConverted[ dataVectorIndex_ ]
134  / convertDegreesToRadians( 1.0 );
135  }
136  }
137  }
138 
139  // Loop over Mach, angle of attack and coefficient index to fill the dynamic coefficient array.
140  for ( int machIndex = 0; machIndex < numberOfMachNumbers_; machIndex++ )
141  {
142  for ( int angleOfAttackIndex = 0; angleOfAttackIndex < numberOfAnglesOfAttack_;
143  angleOfAttackIndex++ )
144  {
145  for ( int coefficientIndex = 0; coefficientIndex < 20; coefficientIndex++ )
146  {
147  // The third index begins at entry 365.
148  // The first section has 144 entries, the second 220 and the last 400. Therefore
149  // there is a repetition after 144+220+400=764 entries.
150  // every coefficients has 20 entries.
151  // WARNINNG: possible error in datcom user manual. It looks like to values are
152  // always in per radian (instead per degree, which is mentioned in the manual).
153  int dataVectorIndex_ = 365 + 764 * machIndex + angleOfAttackIndex
154  + 20 * coefficientIndex;
155  dynamicCoefficients_[ machIndex ][ angleOfAttackIndex ][ coefficientIndex ] =
156  datcomDataToBeConverted[ dataVectorIndex_ ];
157  }
158  }
159  }
160 }

177 void MissileDatcomData::writeCoefficientsToFile( const std::string& fileNameBase,
178  const int basePrecision,
179  const int exponentWidth )
180 {
181  // Write coefficients to a file spearately for each angle of attack.
182  for ( unsigned int i = 0; i < angleOfAttack_.size( ) ; i++ )
183  {
184  // Make filename for specific angle of attack.
185  std::ostringstream stringstream;
186  stringstream << fileNameBase << "_" << i;
187  std::string fileName = stringstream.str( );
188 
189  // Open output file.
190  std::ofstream outputFile;
191  outputFile.open( fileName.c_str( ) );
192  outputFile.precision( 10 );
193 
194  // Iterate over all Mach numbers in database.
195  for ( unsigned int j = 0; j < machNumber_.size( ) ; j++ )
196  {
197  // Write angle of attack and mach number.
198  outputFile << angleOfAttack_[ i ] << " " << machNumber_[ j ] << " ";
199 
200  // Write static coefficients.
201  for ( int k = 0; k < 11; k++ )
202  {
203  outputFile << printInFormattedScientificNotation(
204  staticCoefficients_[ j ][ i ][ k ], basePrecision,
205  exponentWidth ) << " ";
206  }
207 
208  // Write dynamic coefficients.
209  for ( int k = 0; k < 19; k++ )
210  {
211 
212  outputFile << printInFormattedScientificNotation(
213  dynamicCoefficients_[ j ][ i ][ k ], basePrecision,
214  exponentWidth ) << " ";
215  }
216 
217  outputFile << printInFormattedScientificNotation(
218  dynamicCoefficients_[ j ][ i ][ 19 ], basePrecision,
219  exponentWidth ) << std::endl;
220  }
221 
222  outputFile.close( );
223  }
224 }
