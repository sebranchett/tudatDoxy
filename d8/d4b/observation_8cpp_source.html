24  const std::shared_ptr< ObservationSettings >& observationSettings )
25 {
26  if ( !observationSettings )
27  {
28  return;
29  }
30  using namespace json_interface;
31  using K = Keys::Observation;
32 
33  const ObservableType observableType = observationSettings->observableType_;
34  jsonObject[ K::observableType ] = observableType;
35  assignIfNotEmpty( jsonObject, K::lightTimeCorrectionSettingsList, observationSettings->lightTimeCorrectionsList_ );
36  if( observationSettings->biasSettings_ != nullptr )
37  {
38  jsonObject[ K::biasSettings ] = observationSettings->biasSettings_;
39  }
40 
41 
42  switch ( observableType )
43  {
44  case one_way_range:
45  case angular_position:
46  case position_observable:
47  case one_way_doppler:
48  {
49  std::shared_ptr< OneWayDopplerObservationSettings > oneWayDopplerObservationSettings =
50  std::dynamic_pointer_cast< OneWayDopplerObservationSettings >( observationSettings );
51  if( !( oneWayDopplerObservationSettings == nullptr ) )
52  {
53  if( oneWayDopplerObservationSettings->transmitterProperTimeRateSettings_ != nullptr )
54  {
55  jsonObject[ K::transmitterProperTimeRateSettings ] = oneWayDopplerObservationSettings->transmitterProperTimeRateSettings_;
56  }
57 
58  if( oneWayDopplerObservationSettings->receiverProperTimeRateSettings_ != nullptr )
59  {
60  jsonObject[ K::receiverProperTimeRateSettings ] = oneWayDopplerObservationSettings->receiverProperTimeRateSettings_ ;
61  }
62  }
63  return;
64  }
65  case one_way_differenced_range:
66  {
67  std::shared_ptr< OneWayDifferencedRangeRateObservationSettings > oneWayDifferencedRangeObservationSettings =
68  std::dynamic_pointer_cast< OneWayDifferencedRangeRateObservationSettings >( observationSettings );
69  assertNonnullptrPointer( oneWayDifferencedRangeObservationSettings );
70 
71  std::cerr<<"Warning, assuming constant integration time of 1-way differenced range when parsing to JSON, correctness not ensured"<<std::endl;
72  jsonObject[ K::constantIntegrationTime ] = oneWayDifferencedRangeObservationSettings->integrationTimeFunction_( 0.0 );
73  return;
74  }
75  case n_way_range:
76  {
77  std::shared_ptr< NWayRangeObservationSettings > nWayRangeObservationSettings =
78  std::dynamic_pointer_cast< NWayRangeObservationSettings >( observationSettings );
79  if( !( nWayRangeObservationSettings == nullptr ) )
80  {
81  jsonObject[ K::oneWayRangeObsevationSettings ] = nWayRangeObservationSettings->oneWayRangeObsevationSettings_;
82 
83  if( !( nWayRangeObservationSettings->retransmissionTimesFunction_ ) )
84  {
85  std::vector< double > retransmissionTimes = nWayRangeObservationSettings->retransmissionTimesFunction_( 0.0 );
86  std::cerr<<"Warning, assuming constant delay times of n-way range when parsing to JSON, correctness not ensured"<<std::endl;

89  }
90  return;
91  }

110 void from_json( const nlohmann::json& jsonObject,
111  std::shared_ptr< ObservationSettings >& observationSettings )
112 {
113  using namespace json_interface;
114  using K = Keys::Observation;
115 
116  const ObservableType observableType =
117  getValue< ObservableType >( jsonObject, K::observableType );
118 
119  std::vector< std::shared_ptr< LightTimeCorrectionSettings > > lightTimeCorrectionsList =
120  getValue< std::vector< std::shared_ptr< LightTimeCorrectionSettings > > >(
121  jsonObject, K::lightTimeCorrectionSettingsList,
122  std::vector< std::shared_ptr< LightTimeCorrectionSettings > >( ) );
123  std::shared_ptr< ObservationBiasSettings > biasSettings =
124  getValue< std::shared_ptr< ObservationBiasSettings > >(
125  jsonObject, K::biasSettings, nullptr );
126 
127  switch ( observableType )
128  {
129  case one_way_doppler:
130  {
131  std::shared_ptr< DopplerProperTimeRateSettings > transmitterProperTimeRateSettings =
132  getValue< std::shared_ptr< DopplerProperTimeRateSettings > >(
133  jsonObject, K::transmitterProperTimeRateSettings, nullptr );
134  std::shared_ptr< DopplerProperTimeRateSettings > receiverProperTimeRateSettings =
135  getValue< std::shared_ptr< DopplerProperTimeRateSettings > >(
136  jsonObject, K::receiverProperTimeRateSettings, nullptr );
137 
138  observationSettings = std::make_shared< OneWayDopplerObservationSettings >(
139  lightTimeCorrectionsList,
140  transmitterProperTimeRateSettings,
141  receiverProperTimeRateSettings,
142  biasSettings );
143  return;
144  }
145  case one_way_differenced_range:
146  {
147  double constantIntegrationTime = getValue< double >(
148  jsonObject, K::constantIntegrationTime );
149 
150  observationSettings = std::make_shared< OneWayDifferencedRangeRateObservationSettings >(
151  [ = ]( const double ){ return constantIntegrationTime; },
152  lightTimeCorrectionsList,
153  biasSettings );
154  return;
155  }
156  case n_way_range:
157  {
158  std::vector< std::shared_ptr< ObservationSettings > > oneWayRangeObsevationSettings =
159  getValue< std::vector< std::shared_ptr< ObservationSettings > > >(
160  jsonObject, K::oneWayRangeObsevationSettings,
161  std::vector< std::shared_ptr< ObservationSettings > >( ) );
162  if( oneWayRangeObsevationSettings.size( ) == 0 )
163  {
164  observationSettings = std::make_shared< ObservationSettings >(
165  observableType,
166  lightTimeCorrectionsList,
167  biasSettings );
168  }
169  else
170  {
171  std::vector< double > retransmissionTimes =
172  getValue< std::vector< double > >(
173  jsonObject, K::retransmissionTimes,
174  std::vector< double >( ) );
175  observationSettings = std::make_shared< NWayRangeObservationSettings >(
176  oneWayRangeObsevationSettings, [ = ]( const double ){ return retransmissionTimes; },
177  biasSettings );
178  }
179  return;
180  }
181  case two_way_doppler:
182  {
183  std::shared_ptr< ObservationSettings > uplinkOneWayDopplerSettings =
184  getValue< std::shared_ptr< ObservationSettings > >(
185  jsonObject, K::uplinkOneWayDopplerSettings, nullptr );
186  std::shared_ptr< ObservationSettings > downlinkOneWayDopplerSettings =
187  getValue< std::shared_ptr< ObservationSettings > >(
188  jsonObject, K::downlinkOneWayDopplerSettings, nullptr );
189  if( ( uplinkOneWayDopplerSettings == nullptr ) && ( downlinkOneWayDopplerSettings == nullptr ) )
190  {
191  observationSettings = std::make_shared< ObservationSettings >(
192  observableType,
193  lightTimeCorrectionsList,
194  biasSettings );
195  }
196  else if( ( uplinkOneWayDopplerSettings != nullptr ) && ( downlinkOneWayDopplerSettings != nullptr ) )
197  {
198  if( std::dynamic_pointer_cast< OneWayDopplerObservationSettings >( uplinkOneWayDopplerSettings ) == nullptr ||
199  std::dynamic_pointer_cast< OneWayDopplerObservationSettings >( downlinkOneWayDopplerSettings ) == nullptr )
200  {
201  throw std::runtime_error( "Error when reading 2-way Doppler dettings from JSON file, link doppler settings are inconsistent" );
202  }
203  observationSettings = std::make_shared< TwoWayDopplerObservationSettings >(
204  std::dynamic_pointer_cast< OneWayDopplerObservationSettings >( uplinkOneWayDopplerSettings ),
205  std::dynamic_pointer_cast< OneWayDopplerObservationSettings >( downlinkOneWayDopplerSettings ),
206  biasSettings );
207  }
208  else
209  {
210  throw std::runtime_error( "Error when reading 2-way Doppler dettings from JSON file, settings are inconsistent" );
211  }
212  return;
213  }
214  default:
215  {
216  observationSettings = std::make_shared< ObservationSettings >(
217  observableType,
218  lightTimeCorrectionsList,
219  biasSettings );
220  return;
221  }
222  }
