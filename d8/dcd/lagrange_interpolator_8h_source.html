235  DependentVariableType interpolate( const IndependentVariableType targetIndependentVariableValue )
236  {
237  // Check whether boundary handling needs to be applied, if independent variable is beyond its defined range.
238  DependentVariableType interpolatedValue = zeroEntry_;
239  bool useValue = false;
240  this->checkBoundaryCase( interpolatedValue, useValue, targetIndependentVariableValue );
241  if( useValue )
242  {
243  return interpolatedValue;
244  }
245 
246  // Determine the lower entry in the table corresponding to the target independent variable
247  // value.
248  int lowerEntry = lookUpScheme_->findNearestLowerNeighbour(
249  targetIndependentVariableValue );
250 
251  // Check if requested interval is inside region in which centered lagrange interpolation
252  // can be used.
253  if( lowerEntry < offsetEntries_ )
254  {
255  if( lagrangeBoundaryHandling_ == lagrange_no_boundary_interpolation )
256  {
257  throw std::runtime_error(
258  "Error: Lagrange interpolator below allowed bounds." );
259  }
260  else if( numberOfStages_ > 2 )
261  {
262  interpolatedValue = beginInterpolator_->interpolate(
263  targetIndependentVariableValue );
264  }
265  }
266  else if( lowerEntry >= numberOfIndependentValues_ - offsetEntries_ - 1 )
267  {
268  if( lagrangeBoundaryHandling_ == lagrange_no_boundary_interpolation )
269  {
270  throw std::runtime_error(
271  "Error: Lagrange interpolator above allowed bounds." );
272  }
273  else if( numberOfStages_ > 2 )
274  {
275  interpolatedValue = endInterpolator_->interpolate(
276  targetIndependentVariableValue );
277  }
278  }
279  else
280  {
281  // Initialize repeated numerator to 1
282  ScalarType repeatedNumerator =
283  mathematical_constants::getFloatingInteger< ScalarType >( 1 );
284 
285  // Check if requested independent variable is equal to data point
286  if( independentValues_[ lowerEntry ] == targetIndependentVariableValue )
287  {
288  interpolatedValue = dependentValues_[ lowerEntry ];
289  }
290  else if( independentValues_[ lowerEntry + 1 ] == targetIndependentVariableValue )
291  {
292  interpolatedValue = dependentValues_[ lowerEntry + 1 ];
293  }
294  else if( independentValues_[ lowerEntry - 1 ] == targetIndependentVariableValue )
295  {
296  interpolatedValue = dependentValues_[ lowerEntry - 1 ];
297  }
298  else
299  {
300  // Set up repeated numerator and cache of independent variable values from which
301  // interpolant is created.
302  int j = 0;
303  for( int i = 0; i <= 2 * offsetEntries_ + 1; i++ )
304  {
305  j = i + lowerEntry - offsetEntries_;
306  independentVariableDifferenceCache[ i ] =
307  static_cast< ScalarType >(
308  targetIndependentVariableValue - independentValues_[ j ] );
309 
310  repeatedNumerator *= independentVariableDifferenceCache[ i ];
311 
312  }
313 
314  // Evaluate interpolating polynomial at requested data point.
315  for( int i = 0; i < numberOfStages_; i++ )
316  {
317  j = i + lowerEntry - offsetEntries_;
318  interpolatedValue += dependentValues_[ j ] *
319  ( repeatedNumerator /
320  ( independentVariableDifferenceCache[ i ] *
321  denominators[ lowerEntry ][ j - lowerEntry + offsetEntries_ ] ) );
322  }
323  }
324  }
325 
326  return interpolatedValue;
327  }

349  void initializeDenominators( )
350  {
351  // Check validity of requested number of stages"
352  if( numberOfStages_ % 2 != 0 )
353  {
354  throw std::runtime_error(
355  "Error, Lagrange interp. only implemented for even number of stages." );
356  }
357  if( numberOfStages_ < 2 )
358  {
359  throw std::runtime_error(
360  "Error, Lagrange interplator number of stages must be greater than 2." );
361  }
362 
363  // Determine offset from boundary of interpolation interval where interpolant is valid.
364  offsetEntries_ = numberOfStages_ / 2 - 1;
365 
366  // Iterate over all intervals and calculate denominators
367  int currentIterationStart;
368  denominators.resize( numberOfIndependentValues_ );
369  for( int i = offsetEntries_; i < numberOfIndependentValues_ - offsetEntries_ - 1 ; i++ )
370  {
371  // Determine start index in independent variables for current polynomial
372  currentIterationStart = i - offsetEntries_;
373 
374  denominators[ i ].resize( 2 * offsetEntries_ + 2 );
375 
376  // Calculate all denominators for single interval.
377  for( int j = 0; j < numberOfStages_; j++ )
378  {
379  denominators[ i ][ j ] =
380  mathematical_constants::getFloatingInteger< ScalarType >( 1 );
381 
382  for( int k = 0; k < numberOfStages_; k++ )
383  {
384  if( k != j )
385  {
386  denominators[ i ][ j ] *= static_cast< ScalarType >(
387  independentValues_[ j + currentIterationStart ] -
388  independentValues_[ k + currentIterationStart ] );
389  }
390  }
391  }
392  }

406  void initializeBoundaryInterpolators(
407  const AvailableLookupScheme selectedLookupScheme = huntingAlgorithm )
408  {
409  // Create interpolators
410  if( lagrangeBoundaryHandling_ == lagrange_cubic_spline_boundary_interpolation )
411  {
412  // Ensure sufficient data points for spline.
413  int cubicSplineInputSize = offsetEntries_;
414  if( cubicSplineInputSize < 3 )
415  {
416  cubicSplineInputSize = 3;
417  }
418 
419  // Set input maps for interpolators
420  std::map< IndependentVariableType, DependentVariableType > startMap;
421  for( int i = 0; i <= cubicSplineInputSize; i++ )
422  {
423  startMap[ independentValues_.at( i ) ] = dependentValues_.at( i );
424  }
425  std::map< IndependentVariableType, DependentVariableType > endMap;
426  for( int i = numberOfIndependentValues_ - cubicSplineInputSize - 1;
427  i < numberOfIndependentValues_; i++ )
428  {
429  endMap[ independentValues_.at( i ) ] = dependentValues_.at( i );
430  }
431 
432  // Create cubic spline interpolators
433  beginInterpolator_ = std::make_shared< CubicSplineInterpolator
434  < IndependentVariableType, DependentVariableType, ScalarType > >( startMap );
435  endInterpolator_ = std::make_shared< CubicSplineInterpolator
436  < IndependentVariableType, DependentVariableType, ScalarType > >( endMap );
437  }
