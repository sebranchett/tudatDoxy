52 template< typename ParameterType >
53 std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > createNWayRangePartials(
54  const observation_models::LinkEnds& nWayRangeLinkEnds,
55  const simulation_setup::NamedBodyMap& bodyMap,
56  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
57  const std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > lightTimeCorrections =
58  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > >( ) )
59 
60 {
61  // Define return partial list
62  SingleLinkObservationPartialList nWayRangePartialList;
63 
64  // Define list of constituent one-way partials.
65  typedef std::map< int, std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > >
66  OneWayRangePartialList;
67  OneWayRangePartialList constituentOneWayRangePartials;
68 
69  // Getr number of link ends,
70  observation_models::LinkEnds currentLinkEnds;
71  int numberOfLinkEnds = nWayRangeLinkEnds.size( );
72 
73  // Iterate over all links in the n-way range observable
74  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > currentLightTimeCorrections;
75  for( int i = 0; i < numberOfLinkEnds - 1; i++ )
76  {
77  currentLightTimeCorrections.clear( );
78  if( lightTimeCorrections.size( ) > 0 )
79  {
80  currentLightTimeCorrections = lightTimeCorrections.at( i );
81  }
82 
83  // Define links for current one-way range link
84  currentLinkEnds.clear( );
85  currentLinkEnds[ observation_models::transmitter ] = nWayRangeLinkEnds.at(
86  observation_models::getNWayLinkEnumFromIndex( i, numberOfLinkEnds ) );
87  currentLinkEnds[ observation_models::receiver ] = nWayRangeLinkEnds.at(
88  observation_models::getNWayLinkEnumFromIndex( i + 1, numberOfLinkEnds ) );
89 
90  // Create onw-way range partials for current link
91  constituentOneWayRangePartials[ i ] =
92  createOneWayRangePartials( currentLinkEnds, bodyMap, parametersToEstimate, currentLightTimeCorrections, false );
93  }
94 
95  // Retrieve sorted (by parameter index and link index) one-way range partials and (by link index) opne-way range partials
96  std::map< int, std::shared_ptr< OneWayRangeScaling > > oneWayRangeScalers;
97  std::map< std::pair< int, int >, std::map< int, std::shared_ptr< ObservationPartial< 1 > > > > sortedOneWayRangePartials;
98  std::map< std::pair< int, int >, estimatable_parameters::EstimatebleParameterIdentifier > parameterIdList;
99  for( OneWayRangePartialList::iterator oneWayPartialIterator = constituentOneWayRangePartials.begin( );
100  oneWayPartialIterator != constituentOneWayRangePartials.end( ); oneWayPartialIterator++ )
101  {
102  // Retrieve one-way range paritals
103  oneWayRangeScalers[ oneWayPartialIterator->first ] = std::dynamic_pointer_cast< OneWayRangeScaling >
104  ( oneWayPartialIterator->second.second );
105 
106  // Iterate over all one-way range partials of current link
107  for( SingleLinkObservationPartialList::iterator parameterIterator = oneWayPartialIterator->second.first.begin( );
108  parameterIterator != oneWayPartialIterator->second.first.end( ); parameterIterator++ )
109  {
110  sortedOneWayRangePartials[ parameterIterator->first ][ oneWayPartialIterator->first ] = parameterIterator->second;
111  if( parameterIdList.count( parameterIterator->first ) == 0 )
112  {
113  parameterIdList[ parameterIterator->first ] = parameterIterator->second->getParameterIdentifier( );
114  }
115 
116  else if( parameterIdList.at( parameterIterator->first ) != parameterIterator->second->getParameterIdentifier( ) )
117  {
118  throw std::runtime_error( "Error when making n way range partial, parameter indices are inconsistent" );
119  }
120  }
121  }
122 
123 
124  // Create n-way range scaling object
125  std::shared_ptr< NWayRangeScaling > nWayRangeScaler = std::make_shared< NWayRangeScaling >(
126  oneWayRangeScalers, nWayRangeLinkEnds.size( ) );
127 
128  // Create n-way range partial object
129  for( std::map< std::pair< int, int >, std::map< int, std::shared_ptr< ObservationPartial< 1 > > > >::iterator sortedPartialIterator =
130  sortedOneWayRangePartials.begin( ); sortedPartialIterator != sortedOneWayRangePartials.end( ); sortedPartialIterator++ )
131  {
132  nWayRangePartialList[ sortedPartialIterator->first ] = std::make_shared< NWayRangePartial >(
133  nWayRangeScaler, sortedPartialIterator->second, parameterIdList.at( sortedPartialIterator->first ),
134  numberOfLinkEnds );
135  }
136 
137  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
138  vectorParametersToEstimate = parametersToEstimate->getVectorParameters( );
139  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
140  parameterIterator =
141  vectorParametersToEstimate.begin( ); parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
142  {
143 
144  std::shared_ptr< ObservationPartial< 1 > > currentNWayRangePartial;
145  if( isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
146  {
147  currentNWayRangePartial = createObservationPartialWrtLinkProperty< 1 >(
148  nWayRangeLinkEnds, observation_models::n_way_range, parameterIterator->second );
149  }
150 
151  // Check if partial is non-nullptr
152  if( currentNWayRangePartial != nullptr )
153  {
154  // Add partial to the list.
155  std::pair< double, double > currentPair = std::pair< int, int >( parameterIterator->first,
156  parameterIterator->second->getParameterSize( ) );
157  nWayRangePartialList[ currentPair ] = currentNWayRangePartial;
158  }
159  }
160 
161  return std::make_pair( nWayRangePartialList, nWayRangeScaler );
162 }

180 template< typename ParameterType >
181 std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > >
182 createNWayRangePartials(
183  const std::vector< observation_models::LinkEnds >& linkEnds,
184  const simulation_setup::NamedBodyMap& bodyMap,
185  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
186  const std::map< observation_models::LinkEnds,
187  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >& lightTimeCorrections =
188  std::map< observation_models::LinkEnds,
189  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >( ) )
190 {
191  std::map< observation_models::LinkEnds,
192  std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > > partialMap;
193  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > currentLightTimeCorrections;
194 
195  // Iterate over all sets of link ends, and create associated n-way range partials
196  for( unsigned int i = 0; i < linkEnds.size( ); i++ )
197  {
198  // Retrieve light-time corrections
199  if( lightTimeCorrections.count( linkEnds.at( i ) ) > 0 )
200  {
201  currentLightTimeCorrections = lightTimeCorrections.at( linkEnds.at( i ) );
202  if( currentLightTimeCorrections.size( ) != linkEnds.at( i ).size( ) - 1 )
203  {
204  throw std::runtime_error(
205  "Error when making n-way range partials, found light time correction partials for " +
206  std::to_string( currentLightTimeCorrections.size( ) ) +
207  " links, with " + std::to_string( linkEnds.size( ) ) + " link ends" );
208  }
209  }
210  else
211  {
212  currentLightTimeCorrections.clear( );
213  }
214 
215  // Create n-way range partials for current LinkEnds
216  partialMap[ linkEnds.at( i ) ] = createNWayRangePartials< ParameterType >(
217  linkEnds.at( i ), bodyMap, parametersToEstimate, currentLightTimeCorrections );
218 
219  }
220 
221  return partialMap;
222 }
