20 void AerodynamicAccelerationPartial::update( const double currentTime )
21 {
22  Eigen::Vector6d nominalState = vehicleStateGetFunction_( );
23  Eigen::Vector6d perturbedState;
24 
25  // Compute state partial by numerical difference
26  Eigen::Vector3d upperturbedAcceleration, downperturbedAcceleration;
27  for( unsigned int i = 0; i < 6; i++ )
28  {
29  // Perturb state upwards
30  perturbedState = nominalState;
31  perturbedState( i ) += bodyStatePerturbations_( i );
32 
33  // Update environment/acceleration to perturbed state.
34  flightConditions_->resetCurrentTime( TUDAT_NAN );
35  aerodynamicAcceleration_->resetTime( TUDAT_NAN );
36  vehicleStateSetFunction_( perturbedState );
37  flightConditions_->updateConditions( currentTime );
38  aerodynamicAcceleration_->updateMembers( currentTime );
39 
40  // Retrieve perturbed acceleration.
41  upperturbedAcceleration = aerodynamicAcceleration_->getAcceleration( );
42 
43  // Perturb state downwards
44  perturbedState = nominalState;
45  perturbedState( i ) -= bodyStatePerturbations_( i );
46 
47  // Update environment/acceleration to perturbed state.
48  flightConditions_->resetCurrentTime( TUDAT_NAN );
49  aerodynamicAcceleration_->resetTime( TUDAT_NAN );
50  vehicleStateSetFunction_( perturbedState );
51  flightConditions_->updateConditions( currentTime );
52  aerodynamicAcceleration_->updateMembers( currentTime );
53 
54  // Retrieve perturbed acceleration.
55  downperturbedAcceleration = aerodynamicAcceleration_->getAcceleration( );
56 
57  // Compute partial
58  currentAccelerationStatePartials_.block( 0, i, 3, 1 ) =
59  ( upperturbedAcceleration - downperturbedAcceleration ) / ( 2.0 * bodyStatePerturbations_( i ) );
60  }
61 
62  // Reset environment/acceleration mode to nominal conditions
63  flightConditions_->resetCurrentTime( TUDAT_NAN );
64  aerodynamicAcceleration_->resetTime( TUDAT_NAN );
65 
66  vehicleStateSetFunction_( nominalState );
67  flightConditions_->updateConditions( currentTime );
68  aerodynamicAcceleration_->updateMembers( currentTime );
69 
70  currentTime_ = currentTime;
71 }
