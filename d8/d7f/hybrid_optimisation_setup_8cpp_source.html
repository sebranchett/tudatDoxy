68 std::pair< std::vector< double >, std::vector< double > > HybridMethodProblem::get_bounds() const {
69 
70  // Define lower bounds.
71  std::vector< double > lowerBounds;
72 
73  // Define upper bounds.
74  std::vector< double > upperBounds;
75 
76  if ( guessInitialAndFinalCostates_.size( ) != 0 )
77  {
78  if ( guessInitialAndFinalCostates_.size() != 10 )
79  {
80  throw std::runtime_error( "Error when providing an initial guess for hybrid method, size of the vector unconsistent"
81  "with the expected 5 initial and 5 final MEE costate values." );
82  }
83  else
84  {
85  for ( int i = 0 ; i < 10 ; i++ )
86  {
87  double lowerBoundsFromInitialGuess = ( 1.0 - relativeMarginWrtInitialGuess_ ) * guessInitialAndFinalCostates_[ i ];
88  lowerBounds.push_back( lowerBoundsFromInitialGuess );
89  }
90 
91  for ( int i = 0 ; i < 10 ; i++ )
92  {
93  double upperBoundsFromInitialGuess = ( 1.0 + relativeMarginWrtInitialGuess_ ) * guessInitialAndFinalCostates_[ i ];
94  upperBounds.push_back( upperBoundsFromInitialGuess );
95  }
96  }
97  }
98  else
99  {
100  for ( int i = 0 ; i < 10 ; i ++ )
101  {
102  lowerBounds.push_back( initialAndFinalMEEcostatesBounds_.first );
103  upperBounds.push_back( initialAndFinalMEEcostatesBounds_.second );
104  }
105  }
106 
107  return { lowerBounds, upperBounds };
108 }

111 std::vector< double > HybridMethodProblem::fitness( const std::vector< double > &designVariables ) const{
112 
113  // Re-initialise mass of the spacecraft.
114  bodyMap_[ bodyToPropagate_ ]->setConstantBodyMass( initialSpacecraftMass_ );
115 
116  // Transform vector of design variables into 3D vector of throttles.
117  Eigen::VectorXd initialCostates; initialCostates.resize( 5 );
118  Eigen::VectorXd finalCostates; finalCostates.resize( 5 );
119 
120  // Check consistency of the size of the design variables vector.
121  if ( designVariables.size( ) != 10 )
122  {
123  throw std::runtime_error( "Error, size of the design variables vector unconsistent with initial and final "
124  "MEE costates sizes." );
125  }
126 
127  for ( unsigned int i = 0 ; i < 5 ; i++ )
128  {
129  initialCostates[ i ] = designVariables[ i ];
130  finalCostates[ i ] = designVariables[ i + 5 ];
131  }
132 
133  std::vector< double > fitness;
134 
135  // Create hybrid method leg.
136  low_thrust_trajectories::HybridMethodModel currentLeg = low_thrust_trajectories::HybridMethodModel(
137  stateAtDeparture_, stateAtArrival_, initialCostates, finalCostates, maximumThrust_,
138  specificImpulse_, timeOfFlight_, bodyMap_, bodyToPropagate_, centralBody_, integratorSettings_ );
139 
140  // Propagate until time of flight is reached.
141  Eigen::Vector6d finalPropagatedState = currentLeg.propagateTrajectory( );
142 
143  // Convert final propagated state to MEE.
144  Eigen::Vector6d finalPropagatedMEEstate = orbital_element_conversions::convertCartesianToModifiedEquinoctialElements(
145  finalPropagatedState, bodyMap_[ centralBody_ ]->getGravityFieldModel()->getGravitationalParameter() );
146 
147  // Convert targeted final state to MEE.
148  Eigen::Vector6d finalTargetedMEEstate = orbital_element_conversions::convertCartesianToModifiedEquinoctialElements(
149  stateAtArrival_, bodyMap_[ centralBody_ ]->getGravityFieldModel()->getGravitationalParameter() );
150 
151 
152  // Fitness
153  double deltaV = currentLeg.getTotalDeltaV( );
154 
155  // Equality constraints (must be ... = 0 )
156  std::vector< double > equalityConstraints;
157 
158  // Differences in MEE at arrival.
159  equalityConstraints.push_back( std::fabs( finalPropagatedMEEstate[ orbital_element_conversions::semiParameterIndex ]
160  - finalTargetedMEEstate[ orbital_element_conversions::semiParameterIndex ] ) );
161  equalityConstraints.push_back( std::fabs( finalPropagatedMEEstate[ orbital_element_conversions::fElementIndex ]
162  - finalTargetedMEEstate[ orbital_element_conversions::fElementIndex ] ) );
163  equalityConstraints.push_back( std::fabs( finalPropagatedMEEstate[ orbital_element_conversions::gElementIndex ]
164  - finalTargetedMEEstate[ orbital_element_conversions::gElementIndex ] ) );
165  equalityConstraints.push_back( std::fabs( finalPropagatedMEEstate[ orbital_element_conversions::hElementIndex ]
166  - finalTargetedMEEstate[ orbital_element_conversions::hElementIndex ] ) );
167  equalityConstraints.push_back( std::fabs( finalPropagatedMEEstate[ orbital_element_conversions::kElementIndex ]
168  - finalTargetedMEEstate[ orbital_element_conversions::kElementIndex ] ) );
169 
170  // Compute auxiliary variables.
171  Eigen::Vector6d c;
172  Eigen::Vector6d r;
173  Eigen::Vector6d epsilon;
174  for ( int i = 0 ; i < 6 ; i++ )
175  {
176  c[ i ] = 1.0 / relativeToleranceConstraints_;
177  r[ i ] = 1.0 - relativeToleranceConstraints_ * c[ i ];
178  epsilon[ i ] = equalityConstraints[ i ] * c[ i ] + r[ i ];
179  }
180 
181  double weightDeltaV = 1.0;
182  double weightConstraints = 10.0;
183  double optimisationObjective = weightDeltaV * deltaV + weightConstraints * ( epsilon.norm( ) * epsilon.norm( ) );
184 
185 
186  // Output of the fitness function..
187  fitness.push_back( optimisationObjective );
188 
189 
190  return fitness;
191 }
