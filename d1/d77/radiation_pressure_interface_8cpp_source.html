31 void RadiationPressureInterface::updateInterfaceBase(
32  const double currentTime )
33 {
34  currentTime_ = currentTime;
35 
36  // Calculate current radiation pressure
37  currentSolarVector_ = sourcePositionFunction_( ) - targetPositionFunction_( );
38  double distanceFromSource = currentSolarVector_.norm( );
39  currentRadiationPressure_ = calculateRadiationPressure(
40  sourcePower_( ), distanceFromSource );
41 
42  // Calculate total shadowing due to occulting body; note that multiple concurrent
43  // occultations are not completely correctly (prints warning).
44  double shadowFunction = 1.0;
45  double currentShadowFunction = 1.0;
46  for( unsigned int i = 0; i < occultingBodyPositions_.size( ); i++ )
47  {
48  currentShadowFunction *= mission_geometry::computeShadowFunction(
49  sourcePositionFunction_( ), sourceRadius_, occultingBodyPositions_[ i ]( ),
50  occultingBodyRadii_[ i ], targetPositionFunction_( ) );
51 
52  if( currentShadowFunction != 1.0 && shadowFunction != 1.0 )
53  {
54  std::cerr << "Warning, multiple occultation occured in radiation pressure interface, results may be slightly in error" << std::endl;
55  }
56 
57  shadowFunction *= currentShadowFunction;
58  }
59 
60  currentRadiationPressure_ *= shadowFunction;
61 }
