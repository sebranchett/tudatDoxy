629  TabulatedAtmosphereSettings( const std::map< int, std::string >& atmosphereTableFile,
630  const std::vector< AtmosphereIndependentVariables >& independentVariablesNames,
631  const std::vector< AtmosphereDependentVariables >& dependentVariablesNames,
632  const std::vector< interpolators::BoundaryInterpolationType >& boundaryHandling,
633  const std::vector< double >& defaultExtrapolationValue ) :
634  AtmosphereSettings( tabulated_atmosphere ), atmosphereFile_( atmosphereTableFile ),
635  independentVariables_( independentVariablesNames ), dependentVariables_( dependentVariablesNames ),
636  specificGasConstant_( physical_constants::SPECIFIC_GAS_CONSTANT_AIR ), ratioOfSpecificHeats_( 1.4 ),
637  boundaryHandling_( boundaryHandling )
638  {
639  // Assign default values
640  defaultExtrapolationValue_.resize( dependentVariablesNames.size( ) );
641  for ( unsigned int i = 0; i < dependentVariablesNames.size( ); i++ )
642  {
643  for ( unsigned int j = 0; j < independentVariablesNames.size( ); j++ )
644  {
645  if ( boundaryHandling_.at( j ) == interpolators::use_default_value ||
646  boundaryHandling_.at( j ) == interpolators::use_default_value_with_warning )
647  {
648  defaultExtrapolationValue_.at( i ).push_back( std::make_pair( defaultExtrapolationValue.at( i ),
649  defaultExtrapolationValue.at( i ) ) );
650  }
651  else
652  {
653  defaultExtrapolationValue_.at( i ).push_back( std::make_pair( IdentityElement::getAdditionIdentity< double >( ),
654  IdentityElement::getAdditionIdentity< double >( ) ) );
655  }
656  }
657  }
658  }
