22 Eigen::Matrix< double, 3, 4 > getPartialDerivativeOfSecondDegreeGravitationalTorqueWrtQuaternion(
23  const double premultiplier,
24  const Eigen::Matrix3d& inertiaTensor,
25  const Eigen::Vector3d& bodyFixedRelativePosition,
26  const Eigen::Vector3d& inertialRelativePosition,
27  const std::vector< Eigen::Matrix3d > derivativeOfRotationMatrixWrtQuaternions )
28 {
29  Eigen::Matrix3d scalingMatrix = linear_algebra::getCrossProductMatrix(
30  bodyFixedRelativePosition ) * inertiaTensor - linear_algebra::getCrossProductMatrix(
31  inertiaTensor * bodyFixedRelativePosition );
32  Eigen::Matrix< double, 3, 4 > partialOfBodyFixedPositionWrtQuaternion = Eigen::Matrix< double, 3, 4 >::Zero( );
33  for( unsigned int i = 0; i < derivativeOfRotationMatrixWrtQuaternions.size( ); i++ )
34  {
35  partialOfBodyFixedPositionWrtQuaternion.block( 0, i, 3, 1 ) =
36  derivativeOfRotationMatrixWrtQuaternions.at( i ).transpose( ) * inertialRelativePosition;
37  }
38  return premultiplier * scalingMatrix * partialOfBodyFixedPositionWrtQuaternion;
39 }

66 std::pair< std::function< void( Eigen::MatrixXd& ) >, int > SecondDegreeGravitationalTorquePartial::getParameterPartialFunction(
67  std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameter )
68 {
69  using namespace estimatable_parameters;
70 
71  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunction = std::make_pair(
72  std::function< void( Eigen::MatrixXd& ) >( ), 0 );
73 
74  if( parameter->getParameterName( ).second.first == bodyUndergoingTorque_ )
75  {
76  switch( parameter->getParameterName( ).first )
77  {
78  case spherical_harmonics_cosine_coefficient_block:
79  {
80  // Cast parameter object to required type.
81  std::shared_ptr< SphericalHarmonicsCosineCoefficients > coefficientsParameter =
82  std::dynamic_pointer_cast< SphericalHarmonicsCosineCoefficients >( parameter );
83 
84  int c20Index, c21Index, c22Index;
85  coefficientsParameter->getDegreeTwoEntries( c20Index, c21Index, c22Index );
86 
87  if( c20Index >= 0 || c21Index >= 0 || c22Index >= 0 )
88  {
89  if( ( getInertiaTensorNormalizationFactor_ == nullptr ) )
90  {
91  throw std::runtime_error( "Error when getting partial of 2nd degree grac torque w.r.t. cosine sh parameters, inertia tensor normalization function not found." );
92  }
93  partialFunction = std::make_pair(
94  std::bind( &SecondDegreeGravitationalTorquePartial::
95  wrtCosineSphericalHarmonicCoefficientsOfCentralBody, this,
96  std::placeholders::_1, c20Index, c21Index, c22Index ), coefficientsParameter->getParameterSize( ) );
97  }
98 
99  break;
100  }
101  case spherical_harmonics_sine_coefficient_block:
102  {
103  // Cast parameter object to required type.
104 
105  std::shared_ptr< SphericalHarmonicsSineCoefficients > coefficientsParameter =
106  std::dynamic_pointer_cast< SphericalHarmonicsSineCoefficients >( parameter );
107 
108  int s21Index, s22Index;
109  coefficientsParameter->getDegreeTwoEntries( s21Index, s22Index );
110 
111  if( s21Index >= 0 || s22Index >= 0 )
112  {
113  if( getInertiaTensorNormalizationFactor_ == nullptr )
114  {
115  throw std::runtime_error( "Error when getting partial of 2nd degree grac torque w.r.t. sine sh parameters, inertia tensor normalization function not found." );
116  }
117  partialFunction = std::make_pair(
118  std::bind( &SecondDegreeGravitationalTorquePartial::
119  wrtSineSphericalHarmonicCoefficientsOfCentralBody, this,
120  std::placeholders::_1, s21Index, s22Index ), coefficientsParameter->getParameterSize( ) );
121  }
122 
123 
124  break;
125  }
126  default:
127  break;
128  }
129  }
130  return partialFunction;
131 }

134 void SecondDegreeGravitationalTorquePartial::wrtNonRotationalStateOfAdditionalBody(
135  Eigen::Block< Eigen::MatrixXd > partialMatrix,
136  const std::pair< std::string, std::string >& stateReferencePoint,
137  const propagators::IntegratedStateType integratedStateType )
138 {
139  if( ( stateReferencePoint.first == bodyExertingTorque_ ||
140  stateReferencePoint.first == bodyUndergoingTorque_ ) && integratedStateType == propagators::translational_state )
141  {
142  partialMatrix.block( 0, 0, 3, 3 ) +=
143  ( ( stateReferencePoint.first == bodyExertingTorque_ ) ? 1.0 : -1.0 ) *
144  ( torqueModel_->getCurrentTorqueMagnitudePremultiplier( ) *
145  ( linear_algebra::getCrossProductMatrix( currentBodyFixedRelativePosition_ ) *
146  torqueModel_->getCurrentInertiaTensorOfRotatingBody( ) -
147  linear_algebra::getCrossProductMatrix( torqueModel_->getCurrentInertiaTensorTimesRelativePositionOfBody( ) ) ) *
148  ( torqueModel_->getCurrentRotationToBodyFixedFrame( ) ).toRotationMatrix( ) -
149  5.0 * torqueModel_->getTorque( ) *
150  torqueModel_->getCurrentRelativePositionOfBodySubjectToTorque( ).normalized( ).transpose( ) /
151  currentBodyFixedRelativePosition_.norm( ) );
152  }

156 void SecondDegreeGravitationalTorquePartial::update( const double currentTime )
157 {
158  if( !( currentTime_ == currentTime ) )
159  {
160  torqueModel_->updateMembers( currentTime );
161 
162  currentQuaternionVector_ = linear_algebra::convertQuaternionToVectorFormat(

164  linear_algebra::computePartialDerivativeOfRotationMatrixWrtQuaternion(
165  currentQuaternionVector_, currentRotationMatrixDerivativesWrtQuaternion_ );
166  currentBodyFixedRelativePosition_ = torqueModel_->getCurrentRelativeBodyFixedPositionOfBodySubjectToTorque( );

169
