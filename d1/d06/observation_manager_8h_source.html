227  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, Eigen::MatrixXd >
228  computeObservationsWithPartials( const std::vector< TimeType >& times,
229  const LinkEnds linkEnds,
230  const LinkEndType linkEndAssociatedWithTime )
231  {
232  // Initialize return vectors.
233  std::map< TimeType, Eigen::Matrix< ObservationScalarType, ObservationSize, 1 > > observations;
234  std::map< TimeType, Eigen::Matrix< double, ObservationSize, Eigen::Dynamic > > observationMatrices;
235 
236  // Get observation model.
237  std::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > selectedObservationModel =
238  observationSimulator_->getObservationModel( linkEnds );
239 
240  // Initialize vectors of states and times of link ends to be used in calculations.
241  std::vector< Eigen::Vector6d > vectorOfStates;
242  std::vector< double > vectorOfTimes;
243 
244  Eigen::Matrix< ObservationScalarType, ObservationSize, 1 > currentObservation;
245 
246  // Iterate over all observation times
247  int currentObservationSize;
248  for( unsigned int i = 0; i < times.size( ); i++ )
249  {
250  vectorOfTimes.clear( );
251  vectorOfStates.clear( );
252 
253  // Compute observation
254  currentObservation = selectedObservationModel->computeObservationsWithLinkEndData(
255  times[ i ], linkEndAssociatedWithTime, vectorOfTimes, vectorOfStates );
256  observations[ times[ i ] ] = currentObservation;
257 
258  // Compute observation partial
259  currentObservationSize = currentObservation.rows( );
260  observationMatrices[ times[ i ] ] = determineObservationPartialMatrix(
261  currentObservationSize, vectorOfStates, vectorOfTimes, linkEnds, currentObservation,
262  linkEndAssociatedWithTime );
263 
264  }
265 
266  return std::make_pair( utilities::createConcatenatedEigenMatrixFromMapValues( observations ),
267  utilities::createConcatenatedEigenMatrixFromMapValues( observationMatrices ) );
268  }

333  Eigen::Matrix< double, ObservationSize, Eigen::Dynamic > determineObservationPartialMatrix(
334  const int observationSize,
335  const std::vector< Eigen::Vector6d >& states,
336  const std::vector< double >& times,
337  const LinkEnds& linkEnds,
338  const Eigen::Matrix< ObservationScalarType, ObservationSize, 1 > currentObservation,
339  const LinkEndType linkEndAssociatedWithTime )
340  {
341  // Initialize partial vector of observation w.r.t. all parameter.
342  int fullParameterVector = stateTransitionMatrixInterface_->getFullParameterVectorSize( );
343 
344  Eigen::Matrix< double, ObservationSize, Eigen::Dynamic > partialMatrix =
345  Eigen::MatrixXd::Zero( observationSize, fullParameterVector );
346 
347  // Initialize list of [Phi;S] matrices at times required by calculation (key)
348  std::map< double, Eigen::MatrixXd > combinedStateTransitionMatrices;
349 
350  // Perform updates of dependent variables used by (subset of) observation partials.
351  updatePartials( states, times, linkEnds, linkEndAssociatedWithTime, currentObservation );
352 
353  currentLinkEndPartials = observationPartials_[ linkEnds ];
354 
355  // Iterate over all observation partials associated with given link ends.
356  for( typename std::map< std::pair< int, int >, std::shared_ptr<
357  observation_partials::ObservationPartial< ObservationSize > > >::iterator
358  partialIterator = currentLinkEndPartials.begin( );
359  partialIterator != currentLinkEndPartials.end( ); partialIterator++ )
360  {
361  // Get Observation partial start and size indices in parameter veector.
362  std::pair< int, int > currentIndexInfo = partialIterator->first;
363 
364  // Calculate partials of observation w.r.t. parameters, with associated observation times (single partial
365  // can consist of multiple partial matrices, associated at different times)
366  std::vector< std::pair< Eigen::Matrix< double, ObservationSize, Eigen::Dynamic >, double > > singlePartialSet =
367  partialIterator->second->calculatePartial( states, times, linkEndAssociatedWithTime, currentObservation.template cast< double >( ) );
368 
369  // std::cout<<"Obs. "<<currentObservation.transpose( )<<std::endl;
370  // std::cout<<"Partial "<<singlePartialSet.at( 0 ).first<<std::endl<<std::endl;
371 
372  // If start index is smaller than size of state transition,
373  // current partial is w.r.t. to a body to be estimated current state.
374  if( currentIndexInfo.first < stateTransitionMatrixSize_ )
375  {
376  for( unsigned int i = 0; i < singlePartialSet.size( ); i++ )
377  {
378  // Evaluate [Phi;S] matrix at each time instant associated with partial, if not yet evaluated.
379  if( combinedStateTransitionMatrices.count( singlePartialSet[ i ].second ) == 0 )
380  {
381  combinedStateTransitionMatrices[ singlePartialSet[ i ].second ] =
382  this->getCombinedStateTransitionAndSensitivityMatrix( singlePartialSet[ i ].second );
383  }
384 
385  // Add partial of observation h w.r.t. initial state x_{0} (dh/dx_{0}=dh/dx*dx/dx_{0})
386  partialMatrix += ( singlePartialSet[ i ].first ) *
387  combinedStateTransitionMatrices[ singlePartialSet[ i ].second ].block
388  ( currentIndexInfo.first, 0, currentIndexInfo.second, fullParameterVector );
389  }
390  }
391  else
392  {
393  for( unsigned int i = 0; i < singlePartialSet.size( ); i++ )
394  {
395  // Add direct partial of observation w.r.t. parameter.
396  partialMatrix.block( 0, currentIndexInfo.first, observationSize, currentIndexInfo.second ) +=
397  singlePartialSet[ i ].first;
398  }
399  }
400  }
401  return partialMatrix;
402  }
