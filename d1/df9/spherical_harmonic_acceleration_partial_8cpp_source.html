68 std::pair< std::function< void( Eigen::MatrixXd& ) >, int > SphericalHarmonicsGravityPartial::getParameterPartialFunction(
69  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
70 {
71  // Declare return variables, default number of rows = 0 (i.e. no dependency)
72  std::function< void( Eigen::MatrixXd& ) > partialFunction;
73  int numberOfRows = 0;
74 
75 
76  // Check properties of body exerting acceleration.
77  if( parameter->getParameterName( ).first == estimatable_parameters::gravitational_parameter )
78  {
79  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionPair =
80  getGravitationalParameterPartialFunction( parameter->getParameterName( ) );
81  partialFunction = partialFunctionPair.first;
82  numberOfRows = partialFunctionPair.second;
83  }
84  else if( parameter->getParameterName( ).second.first == acceleratingBody_ )
85  {
86  // Check if partial is a rotational property of body exerting acceleration.
87  if( estimatable_parameters::isParameterRotationMatrixProperty( parameter->getParameterName( ).first ) )
88  {
89  // Check if required rotation matrix partial exists.
90  if( rotationMatrixPartials_.count( std::make_pair( parameter->getParameterName( ).first,
91  parameter->getSecondaryIdentifier( ) ) ) != 0 )
92  {
93  // Get partial function.
94  partialFunction = std::bind(
95  &SphericalHarmonicsGravityPartial::wrtRotationModelParameter,
96  this, std::placeholders::_1, parameter->getParameterName( ).first, parameter->getSecondaryIdentifier( ) );
97  numberOfRows = 1;
98  }
99  else
100  {
101  std::string errorMessage = "Error, not taking partial of sh acceleration wrt rotational parameter" +
102  std::to_string( parameter->getParameterName( ).first ) + " of " +
103  parameter->getParameterName( ).second.first;
104  throw std::runtime_error( errorMessage );
105  }
106  }
107 
108  // Check if partial is a tidal property of body exerting acceleration.
109  else if( estimatable_parameters::isParameterTidalProperty( parameter->getParameterName( ).first ) )
110  {
111  // Check input consistency
112  std::shared_ptr< estimatable_parameters::TidalLoveNumber< double > > tidalLoveNumber =
113  std::dynamic_pointer_cast< estimatable_parameters::TidalLoveNumber< double > >( parameter );
114  if( tidalLoveNumber == nullptr )
115  {
116  throw std::runtime_error( "Error when getting tidal Love number vector parameter, object is nullptr" );;
117  }
118 
119  // Get degree and order(s) of tidal variations
120  int degree = tidalLoveNumber->getDegree( );
121  std::vector< int > orders = tidalLoveNumber->getOrders( );
122  int sumOrders = tidalLoveNumber->getSumOrders( );
123 
124  std::pair< int, std::pair< int, int > > currentTidalPartialOutput;
125  for( unsigned int i = 0; i < tidalLoveNumberPartialInterfaces_.size( ); i++ )
126  {
127  // Check dependency on current partial object
128  currentTidalPartialOutput = tidalLoveNumberPartialInterfaces_.at( i )->setParameterPartialFunction(
129  parameter, maximumDegree_, maximumOrder_ );
130 
131  // Check consistency
132  if( numberOfRows != 0 && currentTidalPartialOutput.first > 0 )
133  {
134  throw std::runtime_error( "Error when getting double tidal parameter partial, multiple dependencies found " +
135  std::to_string( numberOfRows ) + ", " +
136  std::to_string( currentTidalPartialOutput.first ) );
137  }
138  else
139  {
140  // If tidal dependency esists, set partial function
141  if( currentTidalPartialOutput.first > 0 )
142  {
143  std::function< std::vector< Eigen::Matrix< double, 2, Eigen::Dynamic > >( ) > coefficientPartialFunction =
144  std::bind( &orbit_determination::TidalLoveNumberPartialInterface::getCurrentDoubleParameterPartial,
145  tidalLoveNumberPartialInterfaces_.at( i ),
146  parameter, currentTidalPartialOutput.second );
147  partialFunction = std::bind(
148  &SphericalHarmonicsGravityPartial::wrtTidalModelParameter, this, coefficientPartialFunction,
149  degree, orders, sumOrders, parameter->getParameterSize( ), std::placeholders::_1 );
150  numberOfRows = currentTidalPartialOutput.first;
151  }
152  }
153  }
154  }
155  }
156 
157  // Return partial function and partial size.
158  return std::make_pair( partialFunction, numberOfRows );
159 }

314 void SphericalHarmonicsGravityPartial::update( const double currentTime )
315 {
316  using namespace tudat::coordinate_conversions;
317 
318  if( !( currentTime_ == currentTime ) )
319  {
320  // Update acceleration model
321  updateFunction_( currentTime );
322 
323  // Calculate Cartesian position in frame fixed to body exerting acceleration
324  Eigen::Matrix3d currentRotationToBodyFixedFrame_ = fromBodyFixedToIntegrationFrameRotation_( ).inverse( );
325  bodyFixedPosition_ = currentRotationToBodyFixedFrame_ *
326  ( positionFunctionOfAcceleratedBody_( ) - positionFunctionOfAcceleratingBody_( ) );
327 
328  // Calculate spherical position in frame fixed to body exerting acceleration
329  bodyFixedSphericalPosition_ = convertCartesianToSpherical( bodyFixedPosition_ );
330  bodyFixedSphericalPosition_( 1 ) = mathematical_constants::PI / 2.0 - bodyFixedSphericalPosition_( 1 );
331 
332  // Get spherical harmonic coefficients
333  currentCosineCoefficients_ = cosineCoefficients_( );
334  currentSineCoefficients_ = sineCoefficients_( );
335 
336  // Update trogonometric functions of multiples of longitude.
337  sphericalHarmonicCache_->update(
338  bodyFixedSphericalPosition_( 0 ), std::sin( bodyFixedSphericalPosition_( 1 ) ),
339  bodyFixedSphericalPosition_( 2 ), bodyReferenceRadius_( ) );
340 
341  // Calculate partial of acceleration wrt position of body undergoing acceleration.
342  currentBodyFixedPartialWrtPosition_ = computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
343  bodyFixedPosition_, bodyReferenceRadius_( ), gravitationalParameterFunction_( ),
344  currentCosineCoefficients_, currentSineCoefficients_, sphericalHarmonicCache_ );
345 
346  currentPartialWrtVelocity_.setZero( );
347  currentPartialWrtPosition_.setZero( );
348 
349  // Compute partial w.r.t. position in inertial frame
350  currentPartialWrtPosition_ +=
351  currentRotationToBodyFixedFrame_.inverse( ) * currentBodyFixedPartialWrtPosition_ * currentRotationToBodyFixedFrame_;
352 
353  // If rotation matrix depends on translational state, add correction partials
354  if( rotationMatrixPartials_.count(
355  std::make_pair( estimatable_parameters::initial_body_state, "" ) ) > 0 )
356  {
357  // Compute the acceleration and body-fixed position partial, without the central term (to avoid numerical errors)
358  Eigen::Vector3d nonCentralAcceleration = accelerationFunction_( );
359  Eigen::Matrix3d nonCentralBodyFixedPartial = currentBodyFixedPartialWrtPosition_;
360  if( currentCosineCoefficients_( 0 ) > 0.0 )
361  {
362  nonCentralAcceleration -= gravitation::computeGravitationalAcceleration(
363  positionFunctionOfAcceleratedBody_( ), gravitationalParameterFunction_( ), positionFunctionOfAcceleratingBody_( ) );
364 
365  nonCentralBodyFixedPartial -=
366  currentRotationToBodyFixedFrame_ * calculatePartialOfPointMassGravityWrtPositionOfAcceleratedBody(
367  positionFunctionOfAcceleratedBody_( ), positionFunctionOfAcceleratingBody_( ), gravitationalParameterFunction_( ) ) *
368  currentRotationToBodyFixedFrame_.inverse( );
369  }
370 
371  // Compute rotation matrix partials
372  std::vector< Eigen::Matrix3d > rotationPositionPartials =
373  rotationMatrixPartials_.at(
374  std::make_pair( estimatable_parameters::initial_body_state, "" ) )->
375  calculatePartialOfRotationMatrixToBaseFrameWrParameter( currentTime );
376 
377  // Add correction terms to position and velocity partials
378  for( unsigned int i = 0; i < 3; i++ )
379  {
380  currentPartialWrtPosition_.block( 0, i, 3, 1 ) -=
381  rotationPositionPartials.at( i ) * ( currentRotationToBodyFixedFrame_ * nonCentralAcceleration );
382 
383  currentPartialWrtPosition_.block( 0, i, 3, 1 ) -=
384  currentRotationToBodyFixedFrame_.inverse( ) * nonCentralBodyFixedPartial *
385  ( rotationPositionPartials.at( i ).transpose( ) *
386  ( positionFunctionOfAcceleratedBody_( ) - positionFunctionOfAcceleratingBody_( ) ) );
387  currentPartialWrtVelocity_.block( 0, i, 3, 1 ) -=
388  rotationPositionPartials.at( i + 3 ) * ( currentRotationToBodyFixedFrame_ * nonCentralAcceleration );
389  }
390  }
391 
392  currentTime_ = currentTime;
393 
394  // Update tidal interfaces
395  for( unsigned int i = 0; i < tidalLoveNumberPartialInterfaces_.size( ); i++ )
396  {
397  tidalLoveNumberPartialInterfaces_.at( i )->update( currentTime );
398  }
399  }
400 }

429 void SphericalHarmonicsGravityPartial::wrtRotationModelParameter(
430  Eigen::MatrixXd& accelerationPartial,
431  const estimatable_parameters::EstimatebleParametersEnum parameterType,
432  const std::string& secondaryIdentifier )
433 {
434  // Calculate distance vector between bodies.
435  Eigen::Vector3d distanceVector = positionFunctionOfAcceleratedBody_( ) - positionFunctionOfAcceleratingBody_( );
436 
437  // Get rotation matrix partial(s) wrt requested parameter
438  std::vector< Eigen::Matrix3d > rotationMatrixPartials =
439  rotationMatrixPartials_.at( std::make_pair( parameterType, secondaryIdentifier ) )->
440  calculatePartialOfRotationMatrixToBaseFrameWrParameter( currentTime_ );
441 
442  // Iterate for each single parameter entry partial.
443  for( unsigned int i = 0; i < rotationMatrixPartials.size( ); i++ )
444  {
445  // Calculate acceleration partial for current parameter entry.
446  accelerationPartial.block( 0, i, 3, 1 ) = rotationMatrixPartials[ i ] *
447  ( fromBodyFixedToIntegrationFrameRotation_( ).inverse( ) ) * accelerationFunction_( ) +
448  fromBodyFixedToIntegrationFrameRotation_( ) * currentBodyFixedPartialWrtPosition_*
449  rotationMatrixPartials[ i ].transpose( ) * distanceVector;
450  }

454 void SphericalHarmonicsGravityPartial::wrtTidalModelParameter(
455  const std::function< std::vector< Eigen::Matrix< double, 2, Eigen::Dynamic > >( ) > coefficientPartialFunctions,
456  const int degree,
457  const std::vector< int >& orders,
458  const bool sumOrders,
459  const int parameterSize,
460  Eigen::MatrixXd& partialMatrix )
461 {
462  // Initialize partial matrix to zero values.
463  partialMatrix = Eigen::Matrix< double, 3, Eigen::Dynamic >::Zero( 3, parameterSize );
464 
465  // Calculate multiplicative term found in all partial terms (partial of C,S coefficients w.r.t. parameter).
466  std::vector< Eigen::Matrix< double, 2, Eigen::Dynamic > > coefficientPartialsPerOrder_ = coefficientPartialFunctions( );
467  int singleOrderPartialSize = coefficientPartialsPerOrder_.at( 0 ).cols( );
468 
469  Eigen::MatrixXd currentPartialContribution = Eigen::MatrixXd::Zero( 3, 1 );
470  std::vector< std::pair< int, int > > blockIndices;
471  blockIndices.resize( 1 );
472 
473  // Iterate over all required orders in current degree.
474  for( unsigned int i = 0; i < orders.size( ); i++ )
475  {
476  // Set coefficient degree/order for current partials
477  blockIndices[ 0 ] = std::make_pair( degree, orders.at( i ) );
478 
479  // Compute acceleration w.r.t. C and S coefficients, and multiply with partials of C,S coefficients w.r.t. parameter
480  if( sumOrders )
481  {
482  calculateSphericalHarmonicGravityWrtCCoefficients(
483  bodyFixedSphericalPosition_, bodyReferenceRadius_( ), gravitationalParameterFunction_( ),
484  sphericalHarmonicCache_,
485  blockIndices, coordinate_conversions::getSphericalToCartesianGradientMatrix(
486  bodyFixedPosition_ ), fromBodyFixedToIntegrationFrameRotation_( ), currentPartialContribution,
487  maximumDegree_, maximumOrder_ );
488 
489  partialMatrix.block( 0, 0, 3, singleOrderPartialSize ) +=
490  currentPartialContribution * coefficientPartialsPerOrder_.at( i ).block( 0, 0, 1, singleOrderPartialSize );
491 
492 
493  blockIndices[ 0 ] = std::make_pair( degree, orders.at( i ) );
494  calculateSphericalHarmonicGravityWrtSCoefficients(
495  bodyFixedSphericalPosition_, bodyReferenceRadius_( ), gravitationalParameterFunction_( ),
496  sphericalHarmonicCache_,
497  blockIndices, coordinate_conversions::getSphericalToCartesianGradientMatrix(
498  bodyFixedPosition_ ), fromBodyFixedToIntegrationFrameRotation_( ), currentPartialContribution,
499  maximumDegree_, maximumOrder_ );
500 
501  partialMatrix.block( 0, 0, 3, singleOrderPartialSize ) +=
502  currentPartialContribution * coefficientPartialsPerOrder_.at( i ).block( 1, 0, 1, singleOrderPartialSize );
503  }
504  else
505  {
506  calculateSphericalHarmonicGravityWrtCCoefficients(
507  bodyFixedSphericalPosition_, bodyReferenceRadius_( ), gravitationalParameterFunction_( ),
508  sphericalHarmonicCache_,
509  blockIndices, coordinate_conversions::getSphericalToCartesianGradientMatrix(
510  bodyFixedPosition_ ), fromBodyFixedToIntegrationFrameRotation_( ), currentPartialContribution,
511  maximumDegree_, maximumOrder_ );
512 
513  partialMatrix.block( 0, i * singleOrderPartialSize, 3, singleOrderPartialSize ) +=
514  currentPartialContribution * coefficientPartialsPerOrder_.at( i ).block( 0, 0, 1, singleOrderPartialSize );
515 
516  calculateSphericalHarmonicGravityWrtSCoefficients(
