23  const Eigen::Vector3d& centralBodyPosition,
24  const double ppnParameterGamma )

26  // Calculate Euclidean geometric distances between transmitter, receiver and gravitating body.
27  double distanceToReceiver = ( receiverPosition - centralBodyPosition ).norm( );
28  double distanceToTransmitter = ( transmitterPosition - centralBodyPosition ).norm( );
29  double linkEuclideanDistance = ( transmitterPosition - receiverPosition ).norm( );

31  // Calculate and return light time correction.

34  ( distanceToReceiver + distanceToTransmitter - linkEuclideanDistance ) );
