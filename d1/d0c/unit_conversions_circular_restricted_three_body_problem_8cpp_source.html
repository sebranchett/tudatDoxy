30 Eigen::VectorXd convertDimensionlessCartesianStateToDimensionalUnits(
31  const Eigen::Vector6d &dimensionlessCartesianState,
32  const double gravitationalParameterOfPrimaryBody,
33  const double gravitationalParameterOfSecondaryBody,
34  const double distanceBetweenPrimaries )
35 {
36  // Declare dimensional Cartesian state.
37  Eigen::Vector6d dimensionalCartesianState;
38 
39  // Convert position to dimensional units.
40  dimensionalCartesianState.segment( 0, 3 )
41  = dimensionlessCartesianState.segment( 0, 3 ) * distanceBetweenPrimaries;
42 
43  // Convert velocity to dimensional units.
44  dimensionalCartesianState.segment( 3, 3 )
45  = dimensionlessCartesianState.segment( 3, 3 )
46  * std::sqrt( ( gravitationalParameterOfPrimaryBody
47  + gravitationalParameterOfSecondaryBody ) / distanceBetweenPrimaries );
48 
49  // Return state in dimensional units.
50  return dimensionalCartesianState;
51 }

104 Eigen::Vector6d convertCorotatingNormalizedToCartesianCoordinates(
105  const double gravitationalParameterPrimary,
106  const double gravitationalParameterSecondary,
107  const double distancePrimarySecondary,
108  const Eigen::Vector6d& normalizedState,
109  const double normalizedTime )
110 {
111  Eigen::Vector3d normalizedPosition = normalizedState.segment( 0, 3 );
112  Eigen::Vector3d normalizedVelocity = normalizedState.segment( 3, 3 );
113 
114  Eigen::Matrix3d rotationMatrix;
115  rotationMatrix.setZero( );
116  Eigen::Matrix3d derivativeRotationMatrix;
117  derivativeRotationMatrix.setZero( );
118 
119  rotationMatrix( 0, 0 ) = std::cos( normalizedTime );
120  rotationMatrix( 0, 1 ) = - std::sin( normalizedTime );
121  rotationMatrix( 1, 0 ) = std::sin( normalizedTime );
122  rotationMatrix( 1, 1 ) = std::cos( normalizedTime );
123  rotationMatrix( 2, 2 ) = 1.0;
124 
125  derivativeRotationMatrix( 0, 0 ) = - std::sin( normalizedTime );
126  derivativeRotationMatrix( 0, 1 ) = - std::cos( normalizedTime );
127  derivativeRotationMatrix( 1, 0 ) = std::cos( normalizedTime );
128  derivativeRotationMatrix( 1, 1 ) = - std::sin( normalizedTime );
129 
130  Eigen::Vector6d inertialNormalizedState;
131  inertialNormalizedState.segment( 0, 3 ) = rotationMatrix * normalizedPosition;
132  inertialNormalizedState.segment( 3, 3 ) = derivativeRotationMatrix * normalizedPosition + rotationMatrix * normalizedVelocity;
133  Eigen::Vector6d cartesianState = circular_restricted_three_body_problem::convertDimensionlessCartesianStateToDimensionalUnits(
134  inertialNormalizedState, gravitationalParameterPrimary, gravitationalParameterSecondary, distancePrimarySecondary);
135 
136  return cartesianState;
137 }

142 Eigen::Vector6d convertCartesianToCorotatingNormalizedCoordinates(
143  const double gravitationalParameterPrimary,
144  const double gravitationalParameterSecondary,
145  const double distancePrimarySecondary,
146  const Eigen::Vector6d& cartesianState,
147  const double time )
148 {
149  Eigen::Vector3d cartesianPosition = cartesianState.segment( 0, 3 );
150  Eigen::Vector3d cartesianVelocity = cartesianState.segment( 3, 3 );
151 
152  double meanMotion = std::sqrt( ( gravitationalParameterPrimary + gravitationalParameterSecondary ) /
153  std::pow( distancePrimarySecondary, 3 ) );
154 
155  Eigen::Matrix3d rotationMatrix;
156  rotationMatrix.setZero( );
157  rotationMatrix( 0, 0 ) = std::cos( meanMotion * time );
158  rotationMatrix( 0, 1 ) = std::sin( meanMotion * time );
159  rotationMatrix( 1, 0 ) = -std::sin( meanMotion * time );
160  rotationMatrix( 1, 1 ) = std::cos( meanMotion * time );
161  rotationMatrix( 2, 2 ) = 1.0;
162 
163  Eigen::Matrix3d derivativeRotationMatrix;
164  derivativeRotationMatrix.setZero( );
165  derivativeRotationMatrix( 0, 0 ) = -std::sin( meanMotion * time );
166  derivativeRotationMatrix( 0, 1 ) = std::cos( meanMotion * time );
167  derivativeRotationMatrix( 1, 0 ) = -std::cos( meanMotion * time );
168  derivativeRotationMatrix( 1, 1 ) = -std::sin( meanMotion * time );
169  derivativeRotationMatrix = meanMotion * derivativeRotationMatrix;
170 
171  Eigen::Vector6d corotatingDimensionalState;
172  corotatingDimensionalState.segment( 0, 3 ) = rotationMatrix * cartesianPosition;
173  corotatingDimensionalState.segment( 3, 3 ) = derivativeRotationMatrix * cartesianPosition + rotationMatrix * cartesianVelocity;
174 
175  Eigen::Vector6d normalizedState = circular_restricted_three_body_problem::convertDimensionalCartesianStateToDimensionlessState(
176  corotatingDimensionalState, gravitationalParameterPrimary, gravitationalParameterSecondary, distancePrimarySecondary );
177 
178  return normalizedState;
179 }
