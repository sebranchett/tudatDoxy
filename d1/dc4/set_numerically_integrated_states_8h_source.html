69 template< typename TimeType, typename StateScalarType >
70 void resetIntegratedEphemerisOfBody(
71  const simulation_setup::NamedBodyMap& bodyMap,
72  const std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > >& ephemerisInput,
73  const std::string& bodyToIntegrate )
74 {
75  using namespace tudat::interpolators;
76  using namespace tudat::ephemerides;
77 
78  // If body does not have ephemeris, give error message.
79  if( bodyMap.at( bodyToIntegrate )->getEphemeris( ) == nullptr )
80  {
81  throw std::runtime_error( "Error when resetting integrated ephemeris of body " +
82  bodyToIntegrate + "no ephemeris found" );
83  }
84 
85  // If current ephemeris is not already a tabulated ephemeris, give error message.
86  else if( !isTabulatedEphemeris( bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) )
87  {
88  throw std::runtime_error( "Error when resetting integrated ephemeris of body " +
89  bodyToIntegrate + " no tabulated ephemeris found" );
90 
91  }
92  // Else, update existing tabulated ephemeris
93  else
94  {
95 // std::cerr << "Warning, tabulated ephemeris is being reset using data at different precision" << std::endl;
96  if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
97  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) != nullptr )
98  {
99  std::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > >
100  ephemerisInterpolator = createStateInterpolator( ephemerisInput );
101  std::shared_ptr< TabulatedCartesianEphemeris< StateScalarType, TimeType > > tabulatedEphemeris =
102  std::dynamic_pointer_cast< TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
103  bodyMap.at( bodyToIntegrate )->getEphemeris( ) );
104  tabulatedEphemeris->resetInterpolator( ephemerisInterpolator );
105  }
106  else
107  {
108  if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, double > >(
109  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) != nullptr )
110  {
111  resetIntegratedEphemerisOfBody(
112  ephemerisInput, std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, double > >(
113  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) );
114  }
115  else if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, double > >(
116  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) != nullptr )
117  {
118  resetIntegratedEphemerisOfBody(
119  ephemerisInput, std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, double > >(
120  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) );
121  }
122  else if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, Time > >(
123  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) != nullptr )
124  {
125  resetIntegratedEphemerisOfBody(
126  ephemerisInput, std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, Time > >(
127  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) );
128  }
129  else if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, Time > >(
130  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) != nullptr )
131  {
132  resetIntegratedEphemerisOfBody(
133  ephemerisInput, std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, Time > >(
134  bodyMap.at( bodyToIntegrate )->getEphemeris( ) ) );
135  }
136  else
137  {
138  throw std::runtime_error( "Error, no tabulated ephemeris found when resetting ephemeris" );
139  }
140  }

159 template< typename TimeType, typename StateScalarType >
160 void convertNumericalSolutionToEphemerisInput(
161  const int bodyIndex,
162  const int startIndex,
163  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >&
164  equationsOfMotionNumericalSolution,
165  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > >& ephemerisTable,
166  const std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) >
167  integrationToEphemerisFrameFunction = nullptr )
168 {
169  // If no integrationToEphemerisFrameFunction is provided, origin is already correct; only
170  // extract required indices.
171  if( integrationToEphemerisFrameFunction == 0 )
172  {
173  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >::const_iterator
174  bodyIterator = equationsOfMotionNumericalSolution.begin( );
175  bodyIterator != equationsOfMotionNumericalSolution.end( ); bodyIterator++ )
176  {
177  ephemerisTable[ bodyIterator->first ] = bodyIterator->second.block( startIndex + 6 * bodyIndex, 0, 6, 1 );
178  }
179  }
180  // Else, extract indices and add required translation from integrationToEphemerisFrameFunction
181  else
182  {
183  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >::const_iterator
184  bodyIterator = equationsOfMotionNumericalSolution.begin( );
185  bodyIterator != equationsOfMotionNumericalSolution.end( ); bodyIterator++ )
186  {
187 
188  ephemerisTable[ bodyIterator->first ] = bodyIterator->second.block( startIndex + 6 * bodyIndex, 0, 6, 1 ) -
189  integrationToEphemerisFrameFunction( bodyIterator->first );
190  }
191  }
192 }

209 template< typename TimeType, typename StateScalarType >
210 void getSingleBodyStateHistoryFromPropagationOutpiut(
211  const std::vector< std::string >& bodiesToIntegrate,
212  const int translationalStateStartIndex,
213  const std::string& bodyForWhichToRetrieveState,
214  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >& equationsOfMotionNumericalSolution,
215  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > >& ephemerisInput,
216  int& bodyIndex,
217  const std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >&
218  integrationToEphemerisFrameFunctions =
219  std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >( ) )
220 {
221  // Get index of current body to be updated in bodiesToIntegrate.
222  std::vector< std::string >::const_iterator bodyFindIterator = std::find(
223  bodiesToIntegrate.begin( ), bodiesToIntegrate.end( ), bodyForWhichToRetrieveState );
224  if( bodyFindIterator == bodiesToIntegrate.end( ) )
225  {
226  throw std::runtime_error( "Error when creating and setting ephemeris after integration, cannot find body " +

228  }

230 
231  // Get frame origin function if applicable
232  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > integrationToEphemerisFrameFunction = nullptr;
233  if( integrationToEphemerisFrameFunctions.count( bodiesToIntegrate.at( bodyIndex ) ) > 0 )
234  {
235  integrationToEphemerisFrameFunction =
236  integrationToEphemerisFrameFunctions.at( bodiesToIntegrate.at( bodyIndex ) );
237  }
238 
239  // Create and reset interpolator.
240  convertNumericalSolutionToEphemerisInput(
241  bodyIndex, translationalStateStartIndex, equationsOfMotionNumericalSolution, ephemerisInput, integrationToEphemerisFrameFunction );

301 template< typename TimeType, typename StateScalarType >
302 void resetIntegratedEphemerides(
303  const simulation_setup::NamedBodyMap& bodyMap,
304  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >& equationsOfMotionNumericalSolution,
305  const std::vector< std::string >& bodiesToIntegrate,
306  const std::pair< unsigned int, unsigned int > startIndexAndSize,
307  std::vector< std::string > ephemerisUpdateOrder = std::vector< std::string >( ),
308  const std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >&
309  integrationToEphemerisFrameFunctions =
310  std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >( ) )
311 {
312  // Set update order arbitrarily if no order is provided.
313  if( ephemerisUpdateOrder.size( ) == 0 )
314  {
315  ephemerisUpdateOrder = bodiesToIntegrate;
316  }
317  // Check input consistency
318  else if( ephemerisUpdateOrder.size( ) != bodiesToIntegrate.size( ) )
319  {
320  throw std::runtime_error( "Error when resetting ephemerides, input vectors have inconsistent size" );
321  }
322 
323  if( static_cast< unsigned int >( equationsOfMotionNumericalSolution.begin( )->second.rows( ) )
324  < startIndexAndSize.first + startIndexAndSize.second )
325  {
326  throw std::runtime_error( "Error when resetting ephemerides, input solution inconsistent with start index and size." );
327  }
328 
329  if( startIndexAndSize.second != 6 * bodiesToIntegrate.size( ) )
330  {
331  throw std::runtime_error( "Error when resetting ephemerides, number of bodies inconsistent with input size." );
332 
333  }
334 
335  // Create interpolators from numerical integration results (states) at discrete times.
336  createAndSetInterpolatorsForEphemerides(
337  bodyMap, bodiesToIntegrate, startIndexAndSize.first, ephemerisUpdateOrder,
338  equationsOfMotionNumericalSolution, integrationToEphemerisFrameFunctions );
339 }

357 template< typename TimeType, typename StateScalarType >
358 void resetMultiArcIntegratedEphemerides(
359  const simulation_setup::NamedBodyMap& bodyMap,
360  const std::vector< std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > >&
361  equationsOfMotionNumericalSolution,
362  const std::vector< double > arcStartTimes,
363  const std::vector< std::string >& bodiesToIntegrate,
364  const std::pair< unsigned int, unsigned int > startIndexAndSize,
365  std::vector< std::string > ephemerisUpdateOrder = std::vector< std::string >( ),
366  const std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >&
367  integrationToEphemerisFrameFunctions =
368  std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >( ) )
369 {
370  using namespace tudat::interpolators;
371  using namespace tudat::ephemerides;
372 
373  // Check input consistency
374  if( ephemerisUpdateOrder.size( ) == 0 )
375  {
376  ephemerisUpdateOrder = bodiesToIntegrate;
377  }
378 
379  // Update ephemerides
380  for( unsigned int i = 0; i < ephemerisUpdateOrder.size( ); i++ )
381  {
382  // Find index of current body in bodiesToIntegrate.
383  std::vector< std::string >::const_iterator bodyFindIterator = std::find(
384  bodiesToIntegrate.begin( ), bodiesToIntegrate.end( ), ephemerisUpdateOrder.at( i ) );
385  int bodyIndex = std::distance( bodiesToIntegrate.begin( ), bodyFindIterator );
386 
387 
388  // Check environment consistency
389  std::shared_ptr< MultiArcEphemeris > currentBodyEphemeris =
390  std::dynamic_pointer_cast< MultiArcEphemeris >(
391  bodyMap.at( bodiesToIntegrate.at( bodyIndex ) )->getEphemeris( ) );
392  if( currentBodyEphemeris == nullptr )
393  {
394  throw std::runtime_error( "Error when resetting ephemeris of body " + bodiesToIntegrate.at( bodyIndex ) +
395  ", original ephemeris is of incompatible type" );
396  }
397 
398  std::vector< std::shared_ptr< Ephemeris > > arcEphemerisList;
399  for( unsigned int j = 0; j < arcStartTimes.size( ); j++ )
400  {
401  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) >
402  integrationToEphemerisFrameFunction = nullptr;
403 
404  // Create transformation function, if needed.
405  if( integrationToEphemerisFrameFunctions.count( bodiesToIntegrate.at( bodyIndex ) ) > 0 )
406  {
407  integrationToEphemerisFrameFunction =
408  integrationToEphemerisFrameFunctions.at( bodiesToIntegrate.at( bodyIndex ) );
409  }
410 
411  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > currentArcSolution;
412  convertNumericalSolutionToEphemerisInput(
413  bodyIndex, startIndexAndSize.first,
414  equationsOfMotionNumericalSolution.at( j ), currentArcSolution, integrationToEphemerisFrameFunction );
415 
416  // Create interpolator.
417  std::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > >
418  ephemerisInterpolator = createStateInterpolator( currentArcSolution );
419 
420  arcEphemerisList.push_back( std::make_shared< TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
421  ephemerisInterpolator, currentBodyEphemeris->getReferenceFrameOrigin( ),
422  currentBodyEphemeris->getReferenceFrameOrientation( ) ) );
423  }
424  currentBodyEphemeris->resetSingleArcEphemerides( arcEphemerisList, arcStartTimes );
425  }
426 
427  // Having set new ephemerides, update body properties depending on ephemerides.
428  for( simulation_setup::NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
429  bodyIterator != bodyMap.end( ); bodyIterator++ )
430  {
431  bodyIterator->second->updateConstantEphemerisDependentMemberQuantities( );
432  }
433 }

941 template< typename TimeType, typename StateScalarType >
942 std::map< IntegratedStateType,
943 std::vector< std::shared_ptr< IntegratedStateProcessor< TimeType, StateScalarType > > > >
944 createIntegratedStateProcessors(
945  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
946  const simulation_setup::NamedBodyMap& bodyMap,
947  const std::shared_ptr< ephemerides::ReferenceFrameManager > frameManager,
948  const int startIndex = 0 )
949 {
950  std::map< IntegratedStateType, std::vector< std::shared_ptr< IntegratedStateProcessor
951  < TimeType, StateScalarType > > > > integratedStateProcessors;
952 
953  // Check dynamics type.
954  switch( propagatorSettings->getStateType( ) )
955  {
956  case hybrid:
957  {
958  // Check input consistency
959  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
960  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
961  if( multiTypePropagatorSettings == nullptr )
962  {
963  throw std::runtime_error( "Error, multi-type propagator settings are inconsistent when make state processors" );
964  }
965 
966  // Iterate over each propagated state type
967  std::map< IntegratedStateType, std::vector< std::shared_ptr< IntegratedStateProcessor< TimeType, StateScalarType > > > >
968  singleTypeIntegratedStateProcessors;
969  int currentStartIndex = 0;
970  for( typename std::map< IntegratedStateType,
971  std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::const_iterator
972  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
973  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
974  {
975  // Multi-type in multi-type not allowed (yet)
976  if( typeIterator->first != hybrid )
977  {
978  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
979  {
980  if( typeIterator->second.at( i ) == nullptr )
981  {
982  std::string errorMessage = "Error in when processing hybrid propagator settings, propagator entry " +
983  std::to_string( i ) + " is not defined.";
984  throw std::runtime_error( errorMessage );
985  }
986 
987  // Create state processor
988  singleTypeIntegratedStateProcessors = createIntegratedStateProcessors< TimeType, StateScalarType >(
989  typeIterator->second.at( i ), bodyMap, frameManager, currentStartIndex );
990 
991  if( singleTypeIntegratedStateProcessors.size( ) > 1 )
992  {
993  throw std::runtime_error( "Error when making hybrid integrated result processors, multiple types found" );
994  }
995  else if( ( singleTypeIntegratedStateProcessors.size( ) == 0 ) && ( typeIterator->first != custom_state ) )
996  {
997  throw std::runtime_error( "Error when making hybrid integrated result processors, no types found" );
998  }
999  else if( ( singleTypeIntegratedStateProcessors.size( ) == 1 ) )
1000  {
1001  if( singleTypeIntegratedStateProcessors.begin( )->second.size( ) != 1 )
1002  {
1003  throw std::runtime_error(
1004  "Error when making hybrid integrated result processors, multiple processors of single type found" );
1005  }
1006  else
1007  {
1008  integratedStateProcessors[ singleTypeIntegratedStateProcessors.begin( )->first ].push_back(
1009  singleTypeIntegratedStateProcessors.begin( )->second.at( 0 ) );
1010  }
1011  }
1012 
1013  currentStartIndex += typeIterator->second.at( i )->getConventionalStateSize( );
1014  }
1015  }
1016  else
1017  {
1018  throw std::runtime_error(
1019  "Error when making integrated state processors, cannot handle hybrid propagator inside hybrid propagator" );
1020  }
1021  }
1022  break;
1023  }
1024  case translational_state:
1025  {
1026  // Check input feasibility
1027  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
1028  translationalPropagatorSettings = std::dynamic_pointer_cast
1029  < TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
1030  if( translationalPropagatorSettings == nullptr )
1031  {
1032  throw std::runtime_error( "Error, input type is inconsistent in createIntegratedStateProcessors" );
1033  }
1034  checkTranslationalStatesFeasibility(
1035  translationalPropagatorSettings->bodiesToIntegrate_, bodyMap );
1036 
1037  // Create state processors
1038  integratedStateProcessors[ translational_state ].push_back(
1039  std::make_shared< TranslationalStateIntegratedStateProcessor< TimeType, StateScalarType > >(
1040  startIndex, bodyMap, translationalPropagatorSettings->bodiesToIntegrate_,
1041  translationalPropagatorSettings->centralBodies_, frameManager ) );
1042  break;
1043  }
1044  case rotational_state:
1045  {
1046  std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationalPropagatorSettings =
1047  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
1048 
1049  integratedStateProcessors[ rotational_state ].push_back(
1050  std::make_shared< RotationalStateIntegratedStateProcessor< TimeType, StateScalarType > >(
1051  startIndex, bodyMap, rotationalPropagatorSettings->bodiesToIntegrate_ ) );
1052  break;
1053  }
1054  case body_mass_state:
1055  {
1056  // Check input feasibility
1057  std::shared_ptr< MassPropagatorSettings< StateScalarType > >
1058  massPropagatorSettings = std::dynamic_pointer_cast
1059  < MassPropagatorSettings< StateScalarType > >( propagatorSettings );
1060  if( massPropagatorSettings == nullptr )
1061  {
1062  throw std::runtime_error( "Error, input type is inconsistent in createIntegratedStateProcessors" );
1063  }
1064 
1065  // Create mass processors.
1066  integratedStateProcessors[ body_mass_state ].push_back(
1067  std::make_shared< BodyMassIntegratedStateProcessor< TimeType, StateScalarType > >(
1068  startIndex, bodyMap, massPropagatorSettings->bodiesWithMassToPropagate_ ) );
1069  break;
1070  }
1071  case custom_state:
1072  {
1073  break;
1074  }
1075  default:
1076  throw std::runtime_error( "Error, could not process integrated state type " +
1077  std::to_string( propagatorSettings->getStateType( ) ) );
1078  }
1079 
1080  return integratedStateProcessors;
1081 }
