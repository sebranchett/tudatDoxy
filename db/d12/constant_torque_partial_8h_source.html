120  void update( const double currentTime = TUDAT_NAN )
121  {
122  if( !( currentTime_ == currentTime ) )
123  {
124  currentInertiaTensor_ = inertiaTensorFunction_( );
125  currentInverseInertiaTensor_ = currentInertiaTensor_.inverse( );
126 
127  currentInertiaTensorNormalizationFactor_ = getInertiaTensorNormalizationFactor_( );
128 
129  currentTotalTorque_.setZero( );
130  for( auto it = torqueVector_.begin( ); it != torqueVector_.end( ); it++ )
131  {
132  for( unsigned int i = 0; i < it->second.size( ); i++ )
133  {
134  it->second.at( i )->updateMembers( currentTime );
135  currentTotalTorque_ += it->second.at( i )->getTorque( );
136  }
137  }
138  }
