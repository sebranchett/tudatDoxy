25 SphericalShaping::SphericalShaping( const Eigen::Vector6d& initialState,
26  const Eigen::Vector6d& finalState,
27  const double requiredTimeOfFlight,
28  const double centralBodyGravitationalParameter,
29  const int numberOfRevolutions,
30  const double initialValueFreeCoefficient,
31  const std::shared_ptr< root_finders::RootFinderSettings > rootFinderSettings,
32  const double lowerBoundFreeCoefficient,
33  const double upperBoundFreeCoefficient,
34  const double initialBodyMass ):
35  ShapeBasedMethod( initialState, finalState, requiredTimeOfFlight, initialBodyMass ),
36  centralBodyGravitationalParameter_( centralBodyGravitationalParameter ),
37  numberOfRevolutions_( numberOfRevolutions ),
38  initialValueFreeCoefficient_( initialValueFreeCoefficient ),
39  rootFinderSettings_( rootFinderSettings ),
40  lowerBoundFreeCoefficient_( lowerBoundFreeCoefficient ),
41  upperBoundFreeCoefficient_( upperBoundFreeCoefficient )
42 {
43  // Normalize the initial state.
44  stateAtDeparture_.segment( 0, 3 ) = initialState.segment( 0, 3 ) / physical_constants::ASTRONOMICAL_UNIT;
45  stateAtDeparture_.segment( 3, 3 ) = initialState.segment( 3, 3 ) * physical_constants::JULIAN_YEAR / physical_constants::ASTRONOMICAL_UNIT;
46 
47  // Normalize the final state.
48  stateAtArrival_.segment( 0, 3 ) = finalState.segment( 0, 3 ) / physical_constants::ASTRONOMICAL_UNIT;
49  stateAtArrival_.segment( 3, 3 ) = finalState.segment( 3, 3 ) * physical_constants::JULIAN_YEAR / physical_constants::ASTRONOMICAL_UNIT;
50 
51  // Normalize the gravitational parameter of the central body.
52  centralBodyGravitationalParameter_ = centralBodyGravitationalParameter * std::pow( physical_constants::JULIAN_YEAR, 2.0 )
53  / std::pow( physical_constants::ASTRONOMICAL_UNIT, 3.0 );
54 
55 
56  // Compute initial state in spherical coordinates.
57  initialStateSphericalCoordinates_ = coordinate_conversions::convertCartesianToSphericalState( stateAtDeparture_ );
58 
59  // Compute final state in spherical coordinates.
60  finalStateSphericalCoordinates_ = coordinate_conversions::convertCartesianToSphericalState( stateAtArrival_ );
61 
62  if ( initialStateSphericalCoordinates_( 1 ) < 0.0 )
63  {
64  initialStateSphericalCoordinates_( 1 ) += 2.0 * mathematical_constants::PI;
65  }
66  if ( finalStateSphericalCoordinates_( 1 ) < 0.0 )
67  {
68  finalStateSphericalCoordinates_( 1 ) += 2.0 * mathematical_constants::PI;
69  }
70 
71  // Retrieve the initial value of the azimuth angle.
72  initialAzimuthAngle_ = initialStateSphericalCoordinates_[ 1 ];
73 
74  // Compute final value of the azimuth angle.
75  if ( ( finalStateSphericalCoordinates_( 1 ) - initialStateSphericalCoordinates_( 1 ) ) < 0.0 )
76  {
77  finalAzimuthAngle_ = finalStateSphericalCoordinates_( 1 ) + 2.0 * mathematical_constants::PI * ( numberOfRevolutions_ + 1.0 );
78  }
79  else
80  {
81  finalAzimuthAngle_ = finalStateSphericalCoordinates_( 1 ) + 2.0 * mathematical_constants::PI * ( numberOfRevolutions_ );
82  }
83 
84 
85  // Compute initial and final values of the derivative of the azimuth angle w.r.t. time.
86  double initialDerivativeAzimuthAngle = initialStateSphericalCoordinates_[ 4 ]
87  / ( initialStateSphericalCoordinates_[ 0 ] * std::cos( initialStateSphericalCoordinates_[ 2 ] ) );
88  double finalDerivativeAzimuthAngle = finalStateSphericalCoordinates_[ 4 ]
89  / ( finalStateSphericalCoordinates_[ 0 ] * std::cos( finalStateSphericalCoordinates_[ 2 ] ) );
90 
91  // Compute initial state parametrized by azimuth angle theta.
92  initialStateParametrizedByAzimuthAngle_ = ( Eigen::Vector6d() << initialStateSphericalCoordinates_[ 0 ],
93  initialStateSphericalCoordinates_[ 1 ],
94  initialStateSphericalCoordinates_[ 2 ],
95  initialStateSphericalCoordinates_[ 3 ] / initialDerivativeAzimuthAngle,
96  initialStateSphericalCoordinates_[ 4 ] / initialDerivativeAzimuthAngle,
97  initialStateSphericalCoordinates_[ 5 ] / initialDerivativeAzimuthAngle ).finished();
98 
99  // Compute final state parametrized by azimuth angle theta.
100  finalStateParametrizedByAzimuthAngle_ = ( Eigen::Vector6d() << finalStateSphericalCoordinates_[ 0 ],
101  finalStateSphericalCoordinates_[ 1 ],
102  finalStateSphericalCoordinates_[ 2 ],
103  finalStateSphericalCoordinates_[ 3 ] / finalDerivativeAzimuthAngle,
104  finalStateSphericalCoordinates_[ 4 ] / finalDerivativeAzimuthAngle,
105  finalStateSphericalCoordinates_[ 5 ] / finalDerivativeAzimuthAngle ).finished();
106 
107 
108  // Initialise coefficients for radial distance and elevation angle functions.
109  coefficientsRadialDistanceFunction_.resize( 7 );
110  for ( int i = 0 ; i < 7 ; i++ )
111  {
112  coefficientsRadialDistanceFunction_[ i ] = 1.0;
113  }
114  coefficientsElevationAngleFunction_.resize( 4 );
115  for ( int i = 0 ; i < 4 ; i++ )
116  {
117  coefficientsElevationAngleFunction_[ i ] = 1.0;
118  }
119 
120 
121  // Define coefficients for radial distance and elevation angle composite functions.
122  radialDistanceCompositeFunction_ = std::make_shared< CompositeRadialFunctionSphericalShaping >( coefficientsRadialDistanceFunction_ );
123  elevationAngleCompositeFunction_ = std::make_shared< CompositeElevationFunctionSphericalShaping >( coefficientsElevationAngleFunction_ );
124 
125 
126  // Define settings for numerical quadrature, to be used to compute time of flight and final deltaV.
127  quadratureSettings_ = std::make_shared< numerical_quadrature::GaussianQuadratureSettings < double > >( initialAzimuthAngle_, 16 );
128 
129  // Iterate on the free coefficient value until the time of flight matches its required value.
130  iterateToMatchRequiredTimeOfFlight( rootFinderSettings_, lowerBoundFreeCoefficient_, upperBoundFreeCoefficient_, initialValueFreeCoefficient_ );
131 
132 
133  // Retrieve initial step size.
134  double initialStepSize = 86400.0;//integratorSettings->initialTimeStep_;
135 
136  std::cerr<<"Warning, initialStepSize set to arbitrary value"<<std::endl;
137 
138  // Vector of azimuth angles at which the time should be computed.
139  Eigen::VectorXd azimuthAnglesToComputeAssociatedEpochs =
140  Eigen::VectorXd::LinSpaced( std::ceil( computeNormalizedTimeOfFlight() * physical_constants::JULIAN_YEAR / initialStepSize ),
141  initialAzimuthAngle_, finalAzimuthAngle_ );
142 
143  std::map< double, double > dataToInterpolate;
144  for ( int i = 0 ; i < azimuthAnglesToComputeAssociatedEpochs.size() ; i++ )
145  {
146  dataToInterpolate[ computeCurrentTimeFromAzimuthAngle( azimuthAnglesToComputeAssociatedEpochs[ i ] ) * physical_constants::JULIAN_YEAR ]
147  = azimuthAnglesToComputeAssociatedEpochs[ i ];
148  }
149 
150  // Create interpolator.
151  std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
152  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 10 );
153 
154  interpolator_ = interpolators::createOneDimensionalInterpolator( dataToInterpolate, interpolatorSettings );
155

167 double SphericalShaping::convertIndependentVariableToTime( const double independentVariable )
168 {
169  // Define the derivative of the time function w.r.t the azimuth angle theta.
170  std::function< double( const double ) > derivativeTimeFunction = [ = ] ( const double currentAzimuthAngle ){
171 
172  double scalarFunctionTimeEquation = computeScalarFunctionTimeEquation( currentAzimuthAngle );
173 
174  // Check that the trajectory is feasible, ie curved toward the central body.
175  if ( scalarFunctionTimeEquation < 0.0 )
176  {
177  throw std::runtime_error ( "Error, trajectory not curved toward the central body, and thus not feasible." );
178  }
179  else
180  {
181  return std::sqrt( computeScalarFunctionTimeEquation( currentAzimuthAngle ) *
182  std::pow( radialDistanceCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle ), 2.0 )
183  / centralBodyGravitationalParameter_ );
184  }
185  };
186 
187  // Create numerical quadrature from quadrature settings.
188  std::shared_ptr< numerical_quadrature::NumericalQuadrature< double, double > > quadrature =
189  numerical_quadrature::createQuadrature( derivativeTimeFunction, quadratureSettings_, independentVariable );
190 
191  double time = quadrature->getQuadrature( );
192 
193  return time;

197 Eigen::MatrixXd SphericalShaping::computeInverseMatrixBoundaryConditions( )
198 {
199  Eigen::MatrixXd matrixBoundaryConditions = Eigen::MatrixXd::Zero( 10, 10 );
200 
201  for ( int i = 0 ; i < 6 ; i++ )
202  {
203  int index = i;
204  if ( i >= 2 )
205  {
206  index = i + 1;
207  }
208  matrixBoundaryConditions( 0, i ) = radialDistanceCompositeFunction_->getComponentFunctionCurrentValue( index, initialAzimuthAngle_ );
209  matrixBoundaryConditions( 1, i ) = radialDistanceCompositeFunction_->getComponentFunctionCurrentValue( index, finalAzimuthAngle_ );
210  matrixBoundaryConditions( 2, i ) = radialDistanceCompositeFunction_->getComponentFunctionFirstDerivative( index, initialAzimuthAngle_ );
211  matrixBoundaryConditions( 3, i ) = radialDistanceCompositeFunction_->getComponentFunctionFirstDerivative( index, finalAzimuthAngle_ );
212  matrixBoundaryConditions( 4, i ) = - std::pow( initialStateSphericalCoordinates_[ 0 ], 2.0 )
213  * radialDistanceCompositeFunction_->getComponentFunctionSecondDerivative( index, initialAzimuthAngle_ );
214  matrixBoundaryConditions( 5, i ) = - std::pow( finalStateSphericalCoordinates_[ 0 ], 2.0 )
215  * radialDistanceCompositeFunction_->getComponentFunctionSecondDerivative( index, finalAzimuthAngle_ );
216  }
217 
218  // Compute value of variable alpha at initial time.
219  double initialValueAlpha = computeInitialAlphaValue();
220 
221  // Compute value of variable alpha at final time.
222  double finalValueAlpha = computeFinalAlphaValue();
223 
224  for ( int i = 0 ; i < 4 ; i++ )
225  {
226  matrixBoundaryConditions( 4, i + 6 ) =
227  initialValueAlpha * elevationAngleCompositeFunction_->getComponentFunctionSecondDerivative( i, initialAzimuthAngle_ );
228  matrixBoundaryConditions( 5, i + 6 ) =
229  finalValueAlpha * elevationAngleCompositeFunction_->getComponentFunctionSecondDerivative( i, finalAzimuthAngle_ );
230  matrixBoundaryConditions( 6, i + 6 ) = elevationAngleCompositeFunction_->getComponentFunctionCurrentValue( i, initialAzimuthAngle_ );
231  matrixBoundaryConditions( 7, i + 6 ) = elevationAngleCompositeFunction_->getComponentFunctionCurrentValue( i, finalAzimuthAngle_ );
232  matrixBoundaryConditions( 8, i + 6 ) = elevationAngleCompositeFunction_->getComponentFunctionFirstDerivative( i, initialAzimuthAngle_ );
233  matrixBoundaryConditions( 9, i + 6 ) = elevationAngleCompositeFunction_->getComponentFunctionFirstDerivative( i, finalAzimuthAngle_ );
234  }
235 
236  // Compute and return the inverse of the boundary conditions matrix.
237  return matrixBoundaryConditions.inverse();
238 }

254 double SphericalShaping::computeInitialValueBoundariesConstant( )
255 {
256  double radialDistance = initialStateParametrizedByAzimuthAngle_[ 0 ];
257  double elevationAngle = initialStateParametrizedByAzimuthAngle_[ 2 ];
258  double derivativeRadialDistance = initialStateParametrizedByAzimuthAngle_[ 3 ];
259  double derivativeElevationAngle = initialStateParametrizedByAzimuthAngle_[ 5 ] / initialStateParametrizedByAzimuthAngle_[ 0 ];
260  double derivativeOfTimeWrtAzimuthAngle = ( initialStateParametrizedByAzimuthAngle_[ 0 ] * std::cos( initialStateParametrizedByAzimuthAngle_[ 2 ] ) )
261  / initialStateSphericalCoordinates_[ 4 ];
262 
263  return - centralBodyGravitationalParameter_ * std::pow( derivativeOfTimeWrtAzimuthAngle, 2.0 ) / std::pow( radialDistance, 2.0 )
264  + 2.0 * std::pow( derivativeRadialDistance, 2.0 ) / radialDistance
265  + radialDistance * ( std::pow( derivativeElevationAngle, 2.0 ) + std::pow( std::cos( elevationAngle ), 2.0 ) )
266  - derivativeRadialDistance * derivativeElevationAngle * ( std::sin( elevationAngle ) * std::cos( elevationAngle ) )
267  / ( std::pow( derivativeElevationAngle, 2.0 ) + std::pow( std::cos( elevationAngle ), 2.0 ) );
268 }

270 double SphericalShaping::computeFinalValueBoundariesConstant( )
271 {
272  double radialDistance = finalStateParametrizedByAzimuthAngle_[ 0 ];
273  double elevationAngle = finalStateParametrizedByAzimuthAngle_[ 2 ];
274  double derivativeRadialDistance = finalStateParametrizedByAzimuthAngle_[ 3 ];
275  double derivativeElevationAngle = finalStateParametrizedByAzimuthAngle_[ 5 ] / finalStateParametrizedByAzimuthAngle_[ 0 ];
276  double derivativeOfTimeWrtAzimuthAngle = ( finalStateParametrizedByAzimuthAngle_[ 0 ] * std::cos( finalStateParametrizedByAzimuthAngle_[ 2 ] ) )
277  / finalStateSphericalCoordinates_[ 4 ];
278 
279  return - centralBodyGravitationalParameter_ * std::pow( derivativeOfTimeWrtAzimuthAngle, 2.0 ) / std::pow( radialDistance, 2.0 )
280  + 2.0 * std::pow( derivativeRadialDistance, 2.0 ) / radialDistance
281  + radialDistance * ( std::pow( derivativeElevationAngle, 2.0 ) + std::pow( std::cos( elevationAngle ), 2.0 ) )
282  - derivativeRadialDistance * derivativeElevationAngle * ( std::sin( elevationAngle ) * std::cos( elevationAngle ) )
283  / ( std::pow( derivativeElevationAngle, 2.0 ) + std::pow( std::cos( elevationAngle ), 2.0 ) );
284 }

286 void SphericalShaping::satisfyBoundaryConditions( )
287 {
288  Eigen::VectorXd vectorBoundaryValues;
289  vectorBoundaryValues.resize( 10 );
290 
291  vectorBoundaryValues[ 0 ] = 1.0 / initialStateParametrizedByAzimuthAngle_[ 0 ];
292  vectorBoundaryValues[ 1 ] = 1.0 / finalStateParametrizedByAzimuthAngle_[ 0 ];
293  vectorBoundaryValues[ 2 ] = - initialStateParametrizedByAzimuthAngle_[ 3 ] / std::pow( initialStateParametrizedByAzimuthAngle_[ 0 ], 2.0 );
294  vectorBoundaryValues[ 3 ] = - finalStateParametrizedByAzimuthAngle_[ 3 ] / std::pow( finalStateParametrizedByAzimuthAngle_[ 0 ], 2.0 );
295  vectorBoundaryValues[ 4 ] = computeInitialValueBoundariesConstant()
296  - 2.0 * std::pow( initialStateParametrizedByAzimuthAngle_[ 3 ], 2.0 ) / initialStateParametrizedByAzimuthAngle_[ 0 ];
297  vectorBoundaryValues[ 5 ] = computeFinalValueBoundariesConstant()
298  - 2.0 * std::pow( finalStateParametrizedByAzimuthAngle_[ 3 ], 2.0 ) / finalStateParametrizedByAzimuthAngle_[ 0 ];
299  vectorBoundaryValues[ 6 ] = initialStateParametrizedByAzimuthAngle_[ 2 ];
300  vectorBoundaryValues[ 7 ] = finalStateParametrizedByAzimuthAngle_[ 2 ];
301  vectorBoundaryValues[ 8 ] = initialStateParametrizedByAzimuthAngle_[ 5 ] / initialStateParametrizedByAzimuthAngle_[ 0 ];
302  vectorBoundaryValues[ 9 ] = finalStateParametrizedByAzimuthAngle_[ 5 ] / finalStateParametrizedByAzimuthAngle_[ 0 ];
303 
304  Eigen::VectorXd vectorSecondComponentContribution;
305  vectorSecondComponentContribution.resize( 10.0 );
306 
307  vectorSecondComponentContribution[ 0 ] = radialDistanceCompositeFunction_->getComponentFunctionCurrentValue( 2, initialAzimuthAngle_ );
308  vectorSecondComponentContribution[ 1 ] = radialDistanceCompositeFunction_->getComponentFunctionCurrentValue( 2, finalAzimuthAngle_ );
309  vectorSecondComponentContribution[ 2 ] = radialDistanceCompositeFunction_->getComponentFunctionFirstDerivative( 2, initialAzimuthAngle_ );
310  vectorSecondComponentContribution[ 3 ] = radialDistanceCompositeFunction_->getComponentFunctionFirstDerivative( 2, finalAzimuthAngle_ );
311  vectorSecondComponentContribution[ 4 ] = - std::pow( initialStateParametrizedByAzimuthAngle_[ 0 ], 2 )
312  * radialDistanceCompositeFunction_->getComponentFunctionSecondDerivative( 2, initialAzimuthAngle_ );
313  vectorSecondComponentContribution[ 5 ] = - std::pow( finalStateParametrizedByAzimuthAngle_[ 0 ], 2 )
314  * radialDistanceCompositeFunction_->getComponentFunctionSecondDerivative( 2, finalAzimuthAngle_ );
315  vectorSecondComponentContribution[ 6 ] = 0.0;
316  vectorSecondComponentContribution[ 7 ] = 0.0;
317  vectorSecondComponentContribution[ 8 ] = 0.0;
318  vectorSecondComponentContribution[ 9 ] = 0.0;
319 
320  vectorSecondComponentContribution *= initialValueFreeCoefficient_;
321 
322  Eigen::MatrixXd inverseMatrixBoundaryConditions_ = computeInverseMatrixBoundaryConditions();
323 
324  Eigen::MatrixXd compositeFunctionCoefficients = inverseMatrixBoundaryConditions_ * ( vectorBoundaryValues - vectorSecondComponentContribution );
325 
326  for ( int i = 0 ; i < 6 ; i++ )
327  {
328  if ( i < 2 )
329  {
330  coefficientsRadialDistanceFunction_( i ) = compositeFunctionCoefficients( i );
331  }
332  else
333  {
334  coefficientsRadialDistanceFunction_( i + 1 ) = compositeFunctionCoefficients( i );
335  }
336  }
337  coefficientsRadialDistanceFunction_( 2 ) = initialValueFreeCoefficient_;
338 
339  for ( int i = 0 ; i < 4 ; i++ )
340  {
341  coefficientsElevationAngleFunction_( i ) = compositeFunctionCoefficients( i + 6 );
342  }
343 
344  radialDistanceCompositeFunction_->resetCompositeFunctionCoefficients( coefficientsRadialDistanceFunction_ );
345  elevationAngleCompositeFunction_->resetCompositeFunctionCoefficients( coefficientsElevationAngleFunction_ );
346 
347 }

351 double SphericalShaping::computeScalarFunctionTimeEquation( double currentAzimuthAngle )
352 {
353  double radialFunctionValue = radialDistanceCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
354  double firstDerivativeRadialFunction = radialDistanceCompositeFunction_->evaluateCompositeFunctionFirstDerivative( currentAzimuthAngle );
355  double secondDerivativeRadialFunction = radialDistanceCompositeFunction_->evaluateCompositeFunctionSecondDerivative( currentAzimuthAngle );
356 
357  double elevationFunctionValue = elevationAngleCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
358  double firstDerivativeElevationFunction = elevationAngleCompositeFunction_->evaluateCompositeFunctionFirstDerivative( currentAzimuthAngle );
359  double secondDerivativeElevationFunction = elevationAngleCompositeFunction_->evaluateCompositeFunctionSecondDerivative( currentAzimuthAngle );
360 
361  return - secondDerivativeRadialFunction + 2.0 * std::pow( firstDerivativeRadialFunction, 2.0 ) / radialFunctionValue
362  + firstDerivativeRadialFunction * firstDerivativeElevationFunction
363  * ( secondDerivativeElevationFunction - std::sin( elevationFunctionValue ) * std::cos( elevationFunctionValue ) )
364  / ( std::pow( firstDerivativeElevationFunction, 2.0 ) + std::pow( std::cos( elevationFunctionValue ), 2.0 ) )
365  + radialFunctionValue * ( std::pow( firstDerivativeElevationFunction, 2.0 ) + std::pow( std::cos( elevationFunctionValue ), 2.0 ) );
366 }

368 double SphericalShaping::computeDerivativeScalarFunctionTimeEquation( double currentAzimuthAngle )
369 {
370  double radialFunctionValue = radialDistanceCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
371  double firstDerivativeRadialFunction = radialDistanceCompositeFunction_->evaluateCompositeFunctionFirstDerivative( currentAzimuthAngle );
372  double secondDerivativeRadialFunction = radialDistanceCompositeFunction_->evaluateCompositeFunctionSecondDerivative( currentAzimuthAngle );
373  double thirdDerivativeRadialFunction = radialDistanceCompositeFunction_->evaluateCompositeFunctionThirdDerivative( currentAzimuthAngle );
374 
375  double elevationFunctionValue = elevationAngleCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
376  double firstDerivativeElevationFunction = elevationAngleCompositeFunction_->evaluateCompositeFunctionFirstDerivative( currentAzimuthAngle );
377  double secondDerivativeElevationFunction = elevationAngleCompositeFunction_->evaluateCompositeFunctionSecondDerivative( currentAzimuthAngle );
378  double thirdDerivativeElevationFunction = elevationAngleCompositeFunction_->evaluateCompositeFunctionThirdDerivative( currentAzimuthAngle );
379 
380  // Define constants F1, F2, F3 and F4 as proposed in... (ADD REFERENCE).
381  double F1 = std::pow( firstDerivativeElevationFunction, 2.0 ) + std::pow( std::cos( elevationFunctionValue ), 2.0 );
382  double F2 = secondDerivativeElevationFunction - std::sin( 2.0 * elevationFunctionValue ) / 2.0;
383  double F3 = std::cos( 2.0 * elevationFunctionValue ) + 2.0 * std::pow( firstDerivativeElevationFunction, 2.0 ) + 1.0;
384  double F4 = 2.0 * secondDerivativeElevationFunction - std::sin( 2.0 * elevationFunctionValue );
385 
386  return F1 * firstDerivativeRadialFunction - thirdDerivativeRadialFunction
387  - 2.0 * std::pow( firstDerivativeRadialFunction, 3.0 ) / std::pow( radialFunctionValue, 2.0 )
388  + 4.0 * firstDerivativeRadialFunction * secondDerivativeRadialFunction / radialFunctionValue
389  + F4 * firstDerivativeElevationFunction * radialFunctionValue
390  + 2.0 * firstDerivativeElevationFunction * firstDerivativeRadialFunction
391  * ( thirdDerivativeElevationFunction - firstDerivativeElevationFunction * std::cos( 2.0 * elevationFunctionValue ) ) / F3
392  + F2 * firstDerivativeElevationFunction * secondDerivativeRadialFunction / F1
393  + F2 * firstDerivativeRadialFunction * secondDerivativeElevationFunction / F1
394  - 4.0 * F4 * F2 * std::pow( firstDerivativeElevationFunction, 2.0 ) * firstDerivativeRadialFunction / std::pow( F3, 2.0 );
395 }

454 void SphericalShaping::iterateToMatchRequiredTimeOfFlight( std::shared_ptr< root_finders::RootFinderSettings > rootFinderSettings,
455  const double lowerBound,
456  const double upperBound,
457  const double initialGuess )
458 {
459 
460  // Define the structure updating the time of flight from the free coefficient value, while still satisfying the boundary conditions.
461  std::function< void ( const double ) > resetFreeCoefficientFunction = std::bind( &SphericalShaping::resetValueFreeCoefficient, this, std::placeholders::_1 );
462  std::function< void( ) > satisfyBoundaryConditionsFunction = std::bind( &SphericalShaping::satisfyBoundaryConditions, this );
463  std::function< double ( ) > computeTOFfunction = std::bind( &SphericalShaping::computeNormalizedTimeOfFlight, this );
464  std::function< double ( ) > getRequiredTOFfunction = std::bind( &SphericalShaping::getNormalizedRequiredTimeOfFlight, this );
465 
466  std::shared_ptr< basic_mathematics::Function< double, double > > timeOfFlightFunction =
467  std::make_shared< SphericalShaping::TimeOfFlightFunction >( resetFreeCoefficientFunction, satisfyBoundaryConditionsFunction, computeTOFfunction, getRequiredTOFfunction );
468 
469  // Create root finder from root finder settings.
470  std::shared_ptr< root_finders::RootFinderCore< double > > rootFinder = root_finders::createRootFinder( rootFinderSettings, lowerBound, upperBound, initialGuess );

493 {
494  // Compute first derivative azimuth angle w.r.t. time.
495  double firstDerivativeAzimuthAngle = computeFirstDerivativeAzimuthAngleWrtTime( currentAzimuthAngle );
496 
497  // Compute scalar function of the time equation, and its derivative w.r.t. azimuth angle.
498  double scalarFunctionTimeEquation = computeScalarFunctionTimeEquation( currentAzimuthAngle );
499  double derivativeScalarFunctionTimeEquation = computeDerivativeScalarFunctionTimeEquation( currentAzimuthAngle );
500 
501  // Compute radial distance, and its derivative w.r.t. azimuth angle.

504

506  * ( derivativeScalarFunctionTimeEquation / ( 2.0 * scalarFunctionTimeEquation ) + firstDerivativeRadialDistance / radialDistance );

560 Eigen::Vector3d SphericalShaping::computeThrustAccelerationVectorParametrizedByAzimuthAngle( const double currentAzimuthAngle )
561 {
562  // Retrieve spherical coordinates and their derivatives w.r.t. to the azimuth angle.
563  double radialDistance = radialDistanceCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
564  double firstDerivativeRadialDistance = radialDistanceCompositeFunction_->evaluateCompositeFunctionFirstDerivative( currentAzimuthAngle );
565  double secondDerivativeRadialDistance = radialDistanceCompositeFunction_->evaluateCompositeFunctionSecondDerivative( currentAzimuthAngle );
566  double elevationAngle = elevationAngleCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
567  double firstDerivativeElevationAngle = elevationAngleCompositeFunction_->evaluateCompositeFunctionFirstDerivative( currentAzimuthAngle );
568  double secondDerivativeElevationAngle = elevationAngleCompositeFunction_->evaluateCompositeFunctionSecondDerivative( currentAzimuthAngle );
569 
570  // Compute and return acceleration vector parametrized by the azimuth angle theta.
571  Eigen::Vector3d accelerationParametrizedByAzimuthAngle;
572  accelerationParametrizedByAzimuthAngle[ 0 ] = secondDerivativeRadialDistance
573  - radialDistance * ( std::pow( firstDerivativeElevationAngle, 2.0 ) + std::pow( std::cos( elevationAngle ), 2.0 ) );
574  accelerationParametrizedByAzimuthAngle[ 1 ] = 2.0 * firstDerivativeRadialDistance * std::cos( elevationAngle )
575  - 2.0 * radialDistance * firstDerivativeElevationAngle * std::sin( elevationAngle );
576  accelerationParametrizedByAzimuthAngle[ 2 ] = 2.0 * firstDerivativeRadialDistance * firstDerivativeElevationAngle
577  + radialDistance * ( secondDerivativeElevationAngle + std::sin( elevationAngle ) * std::cos( elevationAngle ) );
578 
579  return accelerationParametrizedByAzimuthAngle;
580

585 {
586  // Compute current radial distance.
587  double radialDistance = radialDistanceCompositeFunction_->evaluateCompositeFunction( currentAzimuthAngle );
588 
589  // Compute first and second derivatives of the azimuth angle w.r.t. time.
590  double firstDerivativeAzimuthAngleWrtTime = computeFirstDerivativeAzimuthAngleWrtTime( currentAzimuthAngle );
591  double secondDerivativeAzimuthAngleWrtTime = computeSecondDerivativeAzimuthAngleWrtTime( currentAzimuthAngle );
592

595 
596  // Compute acceleration vector parametrized by azimuth angle theta.
597  Eigen::Vector3d accelerationParametrizedByAzimuthAngle = computeThrustAccelerationVectorParametrizedByAzimuthAngle( currentAzimuthAngle );

600  // Compute and return the current thrust acceleration vector in spherical coordinates.
601  return std::pow( firstDerivativeAzimuthAngleWrtTime, 2.0 ) * accelerationParametrizedByAzimuthAngle
602  + secondDerivativeAzimuthAngleWrtTime * velocityParametrizedByAzimuthAngle
603  + centralBodyGravitationalParameter_ / std::pow( radialDistance, 3.0 ) * ( Eigen::Vector3d() << radialDistance, 0.0, 0.0 ).finished();
