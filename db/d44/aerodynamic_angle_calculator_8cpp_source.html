101 void AerodynamicAngleCalculator::update( const double currentTime, const bool updateBodyOrientation )
102 {
103  // Clear all current rotation matrices.
104  currentRotationMatrices_.clear( );
105 
106  // Get current body-fixed state.
107  if( !( currentTime == currentTime_ ) )
108  {
109  currentBodyFixedGroundSpeedBasedState_ = bodyFixedStateFunction_( );
110  currentRotationFromCorotatingToInertialFrame_ = rotationFromCorotatingToInertialFrame_( );
111 
112  Eigen::Vector3d sphericalCoordinates = coordinate_conversions::convertCartesianToSpherical< double >(
113  currentBodyFixedGroundSpeedBasedState_.segment( 0, 3 ) );
114 
115  // Calculate latitude and longitude.
116  currentAerodynamicAngles_[ latitude_angle ] =
117  mathematical_constants::PI / 2.0 - sphericalCoordinates( 1 );
118  currentAerodynamicAngles_[ longitude_angle ] = sphericalCoordinates( 2 );
119 
120  // Compute wind velocity vector
121  Eigen::Vector3d localWindVelocity = Eigen::Vector3d::Zero( );
122  if( windModel_ != nullptr )
123  {
124  localWindVelocity = windModel_->getCurrentWindVelocity(
125  shapeModel_->getAltitude( currentBodyFixedGroundSpeedBasedState_.segment( 0, 3 ) ),
126  currentAerodynamicAngles_[ longitude_angle ],
127  currentAerodynamicAngles_[ latitude_angle ],
128  currentTime );
129  }
130 
131  // Compute airspeed-based velocity vector
132  currentBodyFixedAirspeedBasedState_ = currentBodyFixedGroundSpeedBasedState_;
133  currentBodyFixedAirspeedBasedState_.segment( 3, 3 ) += localWindVelocity;
134 
135  // Calculate vertical <-> aerodynamic <-> body-fixed angles if neede.
136  if( calculateVerticalToAerodynamicFrame_ )
137  {
138  Eigen::Vector3d verticalFrameVelocity =
139  getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
140  currentAerodynamicAngles_.at( longitude_angle ),
141  currentAerodynamicAngles_.at( latitude_angle ) ) *
142  currentBodyFixedAirspeedBasedState_.segment( 3, 3 );
143 
144  currentAerodynamicAngles_[ heading_angle ] = calculateHeadingAngle( verticalFrameVelocity );
145  currentAerodynamicAngles_[ flight_path_angle ] =
146  calculateFlightPathAngle( verticalFrameVelocity );
147  }
148 
149  currentTime_ = currentTime;
150  }
151 
152  if( updateBodyOrientation && !( currentBodyAngleTime_ == currentTime ) )
153  {
154  if( !( angleUpdateFunction_ == nullptr ) )
155  {
156  angleUpdateFunction_( currentTime );
157  }
158 
159  if( !( angleOfAttackFunction_ == nullptr ) )
160  {
161  currentAerodynamicAngles_[ angle_of_attack ] = angleOfAttackFunction_( );
162  }
163 
164  if( !( angleOfSideslipFunction_ == nullptr ) )
165  {
166  currentAerodynamicAngles_[ angle_of_sideslip ] = angleOfSideslipFunction_( );
167  }
168 
169  if( !( bankAngleFunction_ == nullptr ) )
170  {
171  currentAerodynamicAngles_[ bank_angle ] = bankAngleFunction_( );
172  }
173 
174  currentBodyAngleTime_ = currentTime;
175  }
176  else if( !( currentBodyAngleTime_ == currentTime ) )
177  {
178  currentAerodynamicAngles_[ angle_of_attack ] = 0.0;
179  currentAerodynamicAngles_[ angle_of_sideslip ] = 0.0;
180  currentAerodynamicAngles_[ bank_angle ] = 0.0;
181  }
182 }

185 Eigen::Quaterniond AerodynamicAngleCalculator::getRotationQuaternionBetweenFrames(
186  const AerodynamicsReferenceFrames originalFrame,
187  const AerodynamicsReferenceFrames targetFrame )
188 {
189  // Initialize rotation to identity matrix.
190  Eigen::Quaterniond rotationToFrame = Eigen::Quaterniond( Eigen::Matrix3d::Identity( ) );
191 
192  // Check if update settings are consistent with requested frames.
193  if( !calculateVerticalToAerodynamicFrame_ &&
194  ( originalFrame > vertical_frame || targetFrame > vertical_frame ) )
195  {
196  throw std::runtime_error( "Error in AerodynamicAngleCalculator, instance ends at vertical frame" );
197  }
198 
199  // Set current frame pair.
200  std::pair< AerodynamicsReferenceFrames, AerodynamicsReferenceFrames > currentRotationPair =
201  std::make_pair( originalFrame, targetFrame );
202 
203  // Calculate rotation matrix if current rotation is not yet calculated.
204  if( currentRotationMatrices_.count( currentRotationPair ) == 0 )
205  {
206  // Get indices of required frames.
207  int currentFrameIndex = static_cast< int >( originalFrame );
208  int targetFrameIndex = static_cast< int >( targetFrame );
209 
210  // Check if any rotation is needed.
211  if( currentFrameIndex != targetFrameIndex )
212  {
213  // Check 'direction' of transformation through AerodynamicsReferenceFrames list.
214  bool isTargetFrameUp;
215  if( targetFrameIndex > currentFrameIndex )
216  {
217  isTargetFrameUp = 1;
218  }
219  else if( targetFrameIndex < currentFrameIndex )
220  {
221  isTargetFrameUp = 0;
222  }
223  else
224  {
225  throw std::runtime_error(
226  "Error when identifying target frame direction in AerodynamicAngleCalculator." );
227  }
228 
229  // Add rotation sequence until final frame is reached.
230  while( currentFrameIndex != targetFrameIndex )
231  {
232  switch( currentFrameIndex )
233  {
234  case static_cast< int >( inertial_frame ):
235  if( isTargetFrameUp )
236  {
237  rotationToFrame = currentRotationFromCorotatingToInertialFrame_.inverse( ) *
238  rotationToFrame;
239  }
240  else
241  {
242  throw std::runtime_error(
243  "Error, inertial_frame is end frame in AerodynamicAngleCalculator" );
244  }
245  break;
246  case static_cast< int >( corotating_frame ):
247  if( isTargetFrameUp )
248  {
249  rotationToFrame =
250  getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
251  currentAerodynamicAngles_.at( longitude_angle ),
252  currentAerodynamicAngles_.at( latitude_angle ) ) *
253  rotationToFrame;
254  }
255  else
256  {
257  rotationToFrame = currentRotationFromCorotatingToInertialFrame_ *
258  rotationToFrame;
259  }
260  break;
261  case static_cast< int >( vertical_frame ):
262  if( isTargetFrameUp )
263  {
264 
265  rotationToFrame =
266  getLocalVerticalFrameToTrajectoryTransformationQuaternion(
267  currentAerodynamicAngles_.at( flight_path_angle ),
268  currentAerodynamicAngles_.at( heading_angle ) ) * rotationToFrame;
269  }
270  else
271  {
272  rotationToFrame =
273  getLocalVerticalToRotatingPlanetocentricFrameTransformationQuaternion(
274  currentAerodynamicAngles_.at( longitude_angle ),
275  currentAerodynamicAngles_.at( latitude_angle ) ) *
276  rotationToFrame;
277  }
278  break;
279  case static_cast< int >( trajectory_frame ):
280  if( isTargetFrameUp )
281  {
282  rotationToFrame =
283  getTrajectoryToAerodynamicFrameTransformationQuaternion(
284  currentAerodynamicAngles_.at( bank_angle ) ) *
285  rotationToFrame;
286  }
287  else
288  {
289  rotationToFrame =
290  getTrajectoryToLocalVerticalFrameTransformationQuaternion(
291  currentAerodynamicAngles_.at( flight_path_angle ),
292  currentAerodynamicAngles_.at( heading_angle ) ) *
293  rotationToFrame;
294  }
295  break;
296  case static_cast< int >( aerodynamic_frame ):
297  if( isTargetFrameUp )
298  {
299  rotationToFrame =
300  getAirspeedBasedAerodynamicToBodyFrameTransformationQuaternion(
301  currentAerodynamicAngles_.at( angle_of_attack ),
302  currentAerodynamicAngles_.at( angle_of_sideslip ) ) *
303  rotationToFrame;
304  }
305  else
306  {
307  rotationToFrame =
308  getAerodynamicToTrajectoryFrameTransformationQuaternion(
309  currentAerodynamicAngles_.at( bank_angle ) ) *
310  rotationToFrame;
311  }
312  break;
313  case static_cast< int >( body_frame ):
314  if( isTargetFrameUp )
315  {
316  throw std::runtime_error(
317  "Error, body frame is end frame in AerodynamicAngleCalculator." );
318  }
319  else
320  {
321  rotationToFrame =
322  getBodyToAirspeedBasedAerodynamicFrameTransformationQuaternion(
323  currentAerodynamicAngles_.at( angle_of_attack ),
324  currentAerodynamicAngles_.at( angle_of_sideslip ) ) *
325  rotationToFrame;
326  }
327  break;
328  default:
329  throw std::runtime_error(
330  "Error, index " + std::to_string( currentFrameIndex ) +
331  "not found in AerodynamicAngleCalculator." );
332  }
333 
334  // Increment/decrement current frame.
335  if( isTargetFrameUp )
336  {
337  currentFrameIndex++;
338  }
339  else
340  {
341  currentFrameIndex--;
342  }
343  }
344  }
345 
346  // Set current rotation (as well as inverse).
347  currentRotationMatrices_[ currentRotationPair ] = rotationToFrame;
348  currentRotationMatrices_[ std::make_pair( targetFrame, originalFrame ) ] =
349  rotationToFrame.inverse( );
350  }
351  else
352  {
353  rotationToFrame = currentRotationMatrices_.at( currentRotationPair );
354  }
355 
356  return rotationToFrame;
357 }

436 std::function< Eigen::Vector3d( const Eigen::Vector3d& ) >
437 getAerodynamicForceTransformationFunction(
438  const std::shared_ptr< AerodynamicAngleCalculator > aerodynamicAngleCalculator,
439  const AerodynamicsReferenceFrames accelerationFrame,
440  const std::function< Eigen::Quaterniond( ) > bodyFixedToInertialFrameFunction,
441  const AerodynamicsReferenceFrames propagationFrame )
442 {
443  std::function< Eigen::Vector3d( const Eigen::Vector3d& ) > transformationFunction;
444 
445  // If propagation frame is the inertial frame, use bodyFixedToInertialFrameFunction.
446  if( propagationFrame == inertial_frame )
447  {
448  std::vector< std::function< Eigen::Vector3d( const Eigen::Vector3d& ) > > rotationsList;
449 
450  // Get accelerationFrame to corotating frame transformation.
451  std::function< Eigen::Quaterniond( ) > firstRotation =
452  std::bind( &AerodynamicAngleCalculator::getRotationQuaternionBetweenFrames,
453  aerodynamicAngleCalculator, accelerationFrame, corotating_frame );
454  rotationsList.push_back(
455  std::bind( &transformVectorFromQuaternionFunction,
456  std::placeholders::_1, firstRotation ) );
457 
458  // Add corotating to inertial frame.
459  rotationsList.push_back(
460  std::bind( &transformVectorFromQuaternionFunction,
461  std::placeholders::_1, bodyFixedToInertialFrameFunction ) );
462 
463  // Create transformation function.
464  transformationFunction = std::bind( &transformVectorFromVectorFunctions,
465  std::placeholders::_1, rotationsList );
466  }
467  else
468  {
469  // Get accelerationFrame to propagationFrame frame transformation directly.
470  std::function< Eigen::Quaterniond( ) > rotationFunction =
471  std::bind( &AerodynamicAngleCalculator::getRotationQuaternionBetweenFrames,
472  aerodynamicAngleCalculator, accelerationFrame, propagationFrame );
473 
474  // Create transformation function.
475  transformationFunction = std::bind( &transformVectorFromQuaternionFunction, std::placeholders::_1,
476  rotationFunction );
477  }
478 
479  return transformationFunction;
480 }

483 void AerodynamicAnglesClosure::updateAngles( const double currentTime )
484 {
485  // Retrieve rotation matrix that is to be converted to orientation angles.
486  currentRotationFromBodyToTrajectoryFrame_ =
487  ( ( imposedRotationFromInertialToBodyFixedFrame_( currentTime ).toRotationMatrix( ) *
488  aerodynamicAngleCalculator_->getRotationQuaternionBetweenFrames(
489  trajectory_frame, inertial_frame ).toRotationMatrix( ) ) ).transpose( );
490 
491  // Compute associated Euler angles and set as orientation angles.
492  Eigen::Vector3d eulerAngles = basic_mathematics::get132EulerAnglesFromRotationMatrix(
493  currentRotationFromBodyToTrajectoryFrame_ );
494  currentBankAngle_ = eulerAngles( 0 );
495  currentAngleOfSideslip_ = eulerAngles( 1 );
496  currentAngleOfAttack_ = -eulerAngles( 2 );
497 }
