161  DependentVariableType interpolate( const IndependentVariableType targetIndependentVariableValue )
162  {
163  // Check whether boundary handling needs to be applied, if independent variable is beyond its defined range.
164  DependentVariableType targetValue;
165  bool useValue = false;
166  this->checkBoundaryCase( targetValue, useValue, targetIndependentVariableValue );
167  if( useValue )
168  {
169  return targetValue;
170  }
171 
172  // Determine the lower entry in the table corresponding to the target independent variable value.
173  int lowerEntry_ = lookUpScheme_->findNearestLowerNeighbour( targetIndependentVariableValue );
174 
175  // Compute Hermite spline
176  IndependentVariableType factor = ( targetIndependentVariableValue - independentValues_[ lowerEntry_ ] ) /
177  ( independentValues_[ lowerEntry_ + 1 ] - independentValues_[ lowerEntry_ ] );
178  targetValue =
179  coefficients_[ 0 ][ lowerEntry_ ] * factor * factor * factor +
180  coefficients_[ 1 ][ lowerEntry_ ] * factor * factor +
181  coefficients_[ 2 ][ lowerEntry_ ] * factor +
182  coefficients_[ 3 ][ lowerEntry_ ] ;
183 
184  return targetValue;
185  }
186 
187 protected:

190  void computeCoefficients( )
191  {
192  // Initialize vector
193  std::vector< DependentVariableType > zeroVect( independentValues_.size( ) - 1 );
194  for( int i = 0 ; i < 4 ; i++ )
195  {
196  coefficients_.push_back( zeroVect );
197  }
198 
199  // Compute coefficients for polynomials a, b, c and d so that interpolant p is:
200  // p(x) = a((x-x0)/(x1-x0))^3 + b((x-x0)/(x1-x0))^2 + c((x-x0)/(x1-x0)) + d.
201  for( unsigned int i = 0 ; i < ( independentValues_.size() - 1 ) ; i++ )
202  {
203  // Compute coefficient a
204  coefficients_[ 0 ][ i ] = 2.0 * dependentValues_[ i ] - 2.0 * dependentValues_[ i + 1 ] +
205  derivativeValues_[ i ] * ( independentValues_[ i + 1 ] - independentValues_[ i ] ) +
206  derivativeValues_[ i + 1 ] * ( independentValues_[ i + 1 ] - independentValues_[ i ] );
207 
208  // Compute coefficient b
209  coefficients_[ 1 ][ i ] = -3.0 * dependentValues_[ i ] + 3.0 * dependentValues_[ i + 1 ] -
210  2.0 * derivativeValues_[ i ] * ( independentValues_[ i + 1 ] - independentValues_[ i ] ) -
211  derivativeValues_[ i + 1 ] * ( independentValues_[ i + 1 ] - independentValues_[ i ] );
212 
213  // Compute coefficient c
214  coefficients_[ 2 ][ i ] = derivativeValues_[ i ] * ( independentValues_[ i + 1 ] - independentValues_[ i ] );
215 
216  // Compute coefficient d
217  coefficients_[ 3 ][ i ] = dependentValues_[ i ];
218  }
219  }
