102  const std::string& newFrame,

110

168 std::pair< Eigen::Quaterniond, Eigen::Matrix3d > computeRotationQuaternionAndRotationMatrixDerivativeBetweenFrames(
169  const std::string& originalFrame, const std::string& newFrame, const double ephemerisTime )
170 {
171  double stateTransition[ 6 ][ 6 ];
172 
173  sxform_c( originalFrame.c_str( ), newFrame.c_str( ), ephemerisTime, stateTransition );
174 
175  Eigen::Matrix3d matrixDerivative;
176  Eigen::Matrix3d rotationMatrix;
177 
178  for( unsigned int i = 0; i < 3; i++ )
179  {
180  for( unsigned int j = 0; j < 3; j++ )
181  {
182  rotationMatrix( i, j ) = stateTransition[ i ][ j ];
183  matrixDerivative( i, j ) = stateTransition[ i + 3 ][ j ];
184  }

186  return std::make_pair( Eigen::Quaterniond( rotationMatrix ), matrixDerivative );
