28 AvailableTorque getTorqueModelType(
29  std::shared_ptr< basic_astrodynamics::TorqueModel > torqueModel )
30 {
31  AvailableTorque torqueType = underfined_torque;
32  if( std::dynamic_pointer_cast< gravitation::SecondDegreeGravitationalTorqueModel >( torqueModel ) != nullptr )
33  {
34  torqueType = second_order_gravitational_torque;
35  }
36  else if( std::dynamic_pointer_cast< aerodynamics::AerodynamicTorque >( torqueModel ) != nullptr )
37  {
38  torqueType = aerodynamic_torque;
39  }
40  else if( std::dynamic_pointer_cast< gravitation::SphericalHarmonicGravitationalTorqueModel >( torqueModel ) != nullptr )
41  {
42  torqueType = spherical_harmonic_gravitational_torque;
43  }
44  else if( std::dynamic_pointer_cast< InertialTorqueModel >( torqueModel ) != nullptr )
45  {
46  torqueType = inertial_torque;
47  }
48  else if( std::dynamic_pointer_cast< basic_astrodynamics::DissipativeTorqueModel >( torqueModel ) != nullptr )
49  {
50  torqueType = dissipative_torque;
51  }
52  else
53  {
54  std::cerr << "Error, could not identify torque type" << std::endl;
55  }
56  return torqueType;

60 std::string getTorqueModelName( const AvailableTorque torqueType )
61 {
62  std::string torqueName;
63  switch( torqueType )
64  {
65  case second_order_gravitational_torque:
66  torqueName = "second-order gravitational torque ";
67  break;
68  case aerodynamic_torque:
69  torqueName = "aerodynamic torque ";
70  break;
71  case spherical_harmonic_gravitational_torque:
72  torqueName = "spherical harmonic gravitational torque ";
73  break;
74  case inertial_torque:
75  torqueName = "inertial torque ";
76  break;
77  case dissipative_torque:
78  torqueName = "dissipative torque ";
79  break;
80  default:
81  std::string errorMessage = "Error, torque type " +
82  std::to_string( torqueType ) +
83  "not found when retrieving torque name ";
84  throw std::runtime_error( errorMessage );
85  }
86  return torqueName;

90 std::vector< std::shared_ptr< TorqueModel > > getTorqueModelsOfType(
