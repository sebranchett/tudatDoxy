38 Eigen::Vector3d computeGeodesyNormalizedGravitationalAccelerationSum(
39  const Eigen::Vector3d& positionOfBodySubjectToAcceleration,
40  const double gravitationalParameter,
41  const double equatorialRadius,
42  const Eigen::MatrixXd& cosineHarmonicCoefficients,
43  const Eigen::MatrixXd& sineHarmonicCoefficients,
44  std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache,
45  std::map< std::pair< int, int >, Eigen::Vector3d >& accelerationPerTerm,
46  const bool saveSeparateTerms,
47  const Eigen::Matrix3d& accelerationRotation )
48 {
49  // Set highest degree and order.
50  const int highestDegree = cosineHarmonicCoefficients.rows( );
51  const int highestOrder = cosineHarmonicCoefficients.cols( );
52 
53  // Declare spherical position vector.
54  Eigen::Vector3d sphericalpositionOfBodySubjectToAcceleration = coordinate_conversions::
55  convertCartesianToSpherical( positionOfBodySubjectToAcceleration );
56  sphericalpositionOfBodySubjectToAcceleration( 1 ) = mathematical_constants::PI / 2.0 -
57  sphericalpositionOfBodySubjectToAcceleration( 1 );
58 
59  double sineOfAngle = std::sin( sphericalpositionOfBodySubjectToAcceleration( 1 ) );
60  sphericalHarmonicsCache->update( sphericalpositionOfBodySubjectToAcceleration( 0 ),
61  sineOfAngle,
62  sphericalpositionOfBodySubjectToAcceleration( 2 ),
63  equatorialRadius );
64 
65  std::shared_ptr< basic_mathematics::LegendreCache > legendreCacheReference =
66  sphericalHarmonicsCache->getLegendreCache( );
67 
68  // Compute gradient premultiplier.
69  const double preMultiplier = gravitationalParameter / equatorialRadius;
70 
71  // Initialize gradient vector.
72  Eigen::Vector3d sphericalGradient = Eigen::Vector3d::Zero( );
73 
74  Eigen::Matrix3d transformationToCartesianCoordinates = coordinate_conversions::getSphericalToCartesianGradientMatrix(
75  positionOfBodySubjectToAcceleration );
76 
77  // Loop through all degrees.
78  for ( int degree = 0; degree < highestDegree; degree++ )
79  {
80  // Loop through all orders.
81  for ( int order = 0; ( order <= degree ) && ( order < highestOrder ); order++ )
82  {
83  // Compute geodesy-normalized Legendre polynomials.
84  const double legendrePolynomial = legendreCacheReference->getLegendrePolynomial( degree, order );
85 
86  // Compute geodesy-normalized Legendre polynomial derivative.
87  const double legendrePolynomialDerivative = legendreCacheReference->getLegendrePolynomialDerivative(
88  degree, order );
89 
90  // Compute the potential gradient of a single spherical harmonic term.
91  if( saveSeparateTerms )
92  {
93  accelerationPerTerm[ std::make_pair( degree, order ) ] =
94  basic_mathematics::computePotentialGradient(
95  sphericalpositionOfBodySubjectToAcceleration,
96  preMultiplier,
97  degree,
98  order,
99  cosineHarmonicCoefficients( degree, order ),
100  sineHarmonicCoefficients( degree, order ),
101  legendrePolynomial,
102  legendrePolynomialDerivative, sphericalHarmonicsCache );
103  sphericalGradient += accelerationPerTerm[ std::make_pair( degree, order ) ];
104  accelerationPerTerm[ std::make_pair( degree, order ) ] =
105  accelerationRotation * (
106  transformationToCartesianCoordinates * accelerationPerTerm[ std::make_pair( degree, order ) ] );
107  }
108  else
109  {
110  // Compute the potential gradient of a single spherical harmonic term.
111  sphericalGradient += basic_mathematics::computePotentialGradient(
112  sphericalpositionOfBodySubjectToAcceleration,
113  preMultiplier,
114  degree,
115  order,
116  cosineHarmonicCoefficients( degree, order ),
117  sineHarmonicCoefficients( degree, order ),
118  legendrePolynomial,
119  legendrePolynomialDerivative, sphericalHarmonicsCache );
120  }
121  }
122  }
123 
124 
125  // Convert from spherical gradient to Cartesian gradient (which equals acceleration vector) and
126  // return the resulting acceleration vector.
127  return accelerationRotation * ( transformationToCartesianCoordinates * sphericalGradient );
128 }
