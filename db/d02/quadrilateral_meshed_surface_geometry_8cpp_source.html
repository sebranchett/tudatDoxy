35 void QuadrilateralMeshedSurfaceGeometry::performPanelCalculations( )
36 {
37  // Allocate memory for panel properties.
38  panelCentroids_.resize(boost::extents[ numberOfLines_ - 1 ][ numberOfPoints_ - 1 ]);
39  panelSurfaceNormals_.resize(boost::extents[ numberOfLines_ - 1 ][ numberOfPoints_ - 1 ]);
40  panelAreas_.resize(boost::extents[ numberOfLines_ - 1 ][ numberOfPoints_ - 1 ]);
41 
42  // Declare local variables for normal and area determination.
43  Eigen::Vector3d crossVector1;
44  Eigen::Vector3d crossVector2;
45 
46  // Reset total area.
47  totalArea_ = 0.0;
48 
49  // Loop over all panels to determine properties.
50  for ( int i = 0; i < numberOfLines_ - 1; i++ )
51  {
52  for ( int j = 0; j < numberOfPoints_ - 1; j++ )
53  {
54  // Set panel centroid.
55  panelCentroids_[ i ][ j ] = ( meshPoints_[ i ][ j ] +
56  meshPoints_[ i + 1 ][ j ] +
57  meshPoints_[ i ][ j + 1 ] +
58  meshPoints_[ i + 1 ][ j + 1 ] ) / 4;
59 
60  // Set panel cross vectors.
61  crossVector1 = meshPoints_[ i + 1 ][ j + 1 ] - meshPoints_[ i ][ j ];
62  crossVector2 = meshPoints_[ i + 1 ][ j ] - meshPoints_[ i ][ j + 1 ];
63 
64  // Set panel normal (not yet normalized).
65  panelSurfaceNormals_[ i ][ j ] = crossVector1.cross( crossVector2 );
66 
67  // Set panel area (not yet correct size).
68  panelAreas_[ i ][ j ] = panelSurfaceNormals_[ i ][ j ].norm( );
69  if ( panelAreas_[ i ][ j ] < std::numeric_limits< double >::epsilon( ) )
70  {
71  std::cerr << "Warning, panel area is zero in part at panel" << i
72  << ", " << j << std::endl;
73  }
74 
75  // Normalize panel normal and, if necessary, invert normal direction.
76  panelSurfaceNormals_[ i ][ j ] *= reversalOperator_;
77  panelSurfaceNormals_[ i ][ j ].normalize( );
78 
79  // Set panel area to correct size.
80  panelAreas_[ i ][ j ] *= 0.5;
81 
82  // Add panel area to total area.
83  totalArea_ += panelAreas_[ i ][ j ];
84  }
85  }
86 }
