132  DataType execute( const FunctionPointer rootFunction, const DataType initialGuess )
133  {
134  // Set the root function.
135  this->rootFunction = rootFunction;
136 
137  // Start at the two initial values that are used in the algorithm, and compute the function
138  // values.
139  DataType lastRootValue = TUDAT_NAN;
140  DataType currentRootValue = initialGuessOfRootOne_;
141  DataType nextRootValue = initialGuess;
142  DataType lastFunctionValue = TUDAT_NAN;
143  DataType currentFunctionValue = this->rootFunction->evaluate( currentRootValue );
144  DataType nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
145 
146  // Check if the next root value is the most accurate guess. If not, switch the values.
147  if( std::fabs( currentFunctionValue ) < std::fabs( nextFunctionValue ) )
148  {
149  // Switch the root value.
150  DataType temporaryRoot = currentRootValue;
151  currentRootValue = nextRootValue;
152  nextRootValue = temporaryRoot;
153 
154  // Switch the function value
155  DataType temporaryValue = currentFunctionValue;
156  currentFunctionValue = nextFunctionValue;
157  nextFunctionValue = temporaryValue;
158  }
159 
160  // Loop counter.
161  unsigned int counter = 1;
162 
163  // Loop until we have a solution with sufficient accuracy.
164  do
165  {
166  // Save the old values.
167  lastRootValue = currentRootValue;
168  lastFunctionValue = currentFunctionValue;
169  currentRootValue = nextRootValue;
170  currentFunctionValue = nextFunctionValue;
171 
172  // Compute next value of root using the following algorithm (see class documentation):
173  nextRootValue = currentRootValue - currentFunctionValue
174  * ( currentRootValue - lastRootValue )
175  / ( currentFunctionValue - lastFunctionValue );
176  nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
177 
178  // Update the counter.
179  counter++;
180  }
181  while( !this->terminationFunction( nextRootValue, currentRootValue, nextFunctionValue,
182  currentFunctionValue, counter ) );
183 
184  return nextRootValue;
185 
186  }
