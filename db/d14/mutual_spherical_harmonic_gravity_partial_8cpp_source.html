20 std::pair< std::function< void( Eigen::MatrixXd& ) >, int > MutualSphericalHarmonicsGravityPartial::getParameterPartialFunction(
21  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
22 {
23  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > parameterPartial;
24 
25  // Check if parameter is gravitational parameter of either
26  if( parameter->getParameterName( ).first == estimatable_parameters::gravitational_parameter )
27  {
28  // Check of acceleration depends on gravitational parameter of given body
29  if( parameter->getParameterName( ).second.first == acceleratingBody_ ||
30  ( parameter->getParameterName( ).second.first == acceleratedBody_ && accelerationUsesMutualAttraction_ ) )
31  {
32  parameterPartial = std::make_pair(
33  std::bind( &MutualSphericalHarmonicsGravityPartial::wrtGravitationalParameter, this, std::placeholders::_1 ), 1 );
34  }
35  else
36  {
37  parameterPartial = std::make_pair( std::function< void( Eigen::MatrixXd& ) >( ), 0 );
38  }
39  }
40  else
41  {
42  // Get partial functions for constituent partial objects
43  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromBodyExertingAcceleration =
44  accelerationPartialOfShExpansionOfBodyExertingAcceleration_->getParameterPartialFunction( parameter );
45  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromBodyUndergoingAcceleration =
46  accelerationPartialOfShExpansionOfBodyUndergoingAcceleration_->getParameterPartialFunction( parameter );
47 
48  // Combine partial functions
49  parameterPartial = orbit_determination::createMergedParameterPartialFunction(
50  partialFunctionFromBodyExertingAcceleration, partialFunctionFromBodyUndergoingAcceleration );
51  }
52  return parameterPartial;
53 }

69 int MutualSphericalHarmonicsGravityPartial::setParameterPartialUpdateFunction(
70  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
71 {
72  int partialSize = 0;
73 
74  // Check if parameter is gravitational parameter of either
75  if( parameter->getParameterName( ).first == estimatable_parameters::gravitational_parameter )
76  {
77  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunction =
78  getParameterPartialFunction( parameter );
79  partialSize = partialFunction.second;
80 
81  if( partialFunction.second > 0 )
82  {
83  parameterDoublePartialFunctions_[ parameter ] = partialFunction.first;
84  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
85  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
86  }
87  }
88  else
89  {
90  // Get partial function for acceleration from body exerting acceleration
91  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromExertingExpansion =
92  accelerationPartialOfShExpansionOfBodyExertingAcceleration_->getParameterPartialFunction( parameter );
93  if( partialFunctionFromExertingExpansion.second > 0 )
94  {
95  accelerationPartialOfShExpansionOfBodyExertingAcceleration_->setParameterPartialUpdateFunction( parameter );
96  }
97 
98  // Get partial function for acceleration from body undergoing acceleration
99  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromUndergoingGravity =
100  accelerationPartialOfShExpansionOfBodyUndergoingAcceleration_->getParameterPartialFunction( parameter );
101  if( partialFunctionFromUndergoingGravity.second > 0 )
102  {
103  accelerationPartialOfShExpansionOfBodyUndergoingAcceleration_->setParameterPartialUpdateFunction( parameter );
104  }
105 
106  // Update this partial object with new dependencies
107  if( partialFunctionFromUndergoingGravity.second > 0 || partialFunctionFromExertingExpansion.second > 0 )
108  {
109  parameterDoublePartialFunctions_[ parameter ] =
110  getCombinedCurrentDoubleParameterFunction(
111  accelerationPartialOfShExpansionOfBodyExertingAcceleration_,
112  accelerationPartialOfShExpansionOfBodyUndergoingAcceleration_,
113  parameter, partialFunctionFromExertingExpansion.second, partialFunctionFromUndergoingGravity.second, 1 );
114  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
115  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
116  }
117  partialSize = std::max( partialFunctionFromExertingExpansion.second, partialFunctionFromUndergoingGravity.second );
118  }
119  return partialSize;
120 }
