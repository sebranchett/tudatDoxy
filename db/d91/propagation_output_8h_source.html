227 template< typename StateScalarType, typename TimeType >
228 std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > >
229 getSphericalHarmonicAccelerationForDependentVariables(
230  const std::shared_ptr< SingleDependentVariableSaveSettings > dependentVariableSettings,
231  const std::unordered_map< IntegratedStateType,
232  std::vector< std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > > >& stateDerivativeModels,
233  const bool allowThirdBodyAcceleration = false )
234 {
235  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > selectedAccelerationModel;
236 
237  // Retrieve list of suitable acceleration models (size should be one to avoid ambiguities)s
238  std::vector< std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > >
239  listOfSuitableAccelerationModels = getAccelerationBetweenBodies(
240  dependentVariableSettings->associatedBody_,
241  dependentVariableSettings->secondaryBody_, stateDerivativeModels,
242  basic_astrodynamics::spherical_harmonic_gravity );
243 
244 
245  // Check if third-body counterpart of acceleration is found
246  if( listOfSuitableAccelerationModels.size( ) == 0 )
247  {
248  listOfSuitableAccelerationModels = getAccelerationBetweenBodies(
249  dependentVariableSettings->associatedBody_,
250  dependentVariableSettings->secondaryBody_,
251  stateDerivativeModels, basic_astrodynamics::getAssociatedThirdBodyAcceleration(
252  basic_astrodynamics::spherical_harmonic_gravity ) );
253  }
254 
255  if( listOfSuitableAccelerationModels.size( ) != 1 )
256  {
257  std::string errorMessage = "Error when getting spherical harmonic acceleration components between bodies " +
258  dependentVariableSettings->associatedBody_ + " and " +
259  dependentVariableSettings->secondaryBody_ + " of type " +
260  std::to_string(
261  basic_astrodynamics::spherical_harmonic_gravity ) +
262  ", no such acceleration found";
263  throw std::runtime_error( errorMessage );
264  }
265  else
266  {
267  std::shared_ptr< gravitation::SphericalHarmonicsGravitationalAccelerationModel > sphericalHarmonicAcceleration =
268  std::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravitationalAccelerationModel >(
269  listOfSuitableAccelerationModels.at( 0 ) );
270 
271 
272  if( sphericalHarmonicAcceleration != nullptr )
273  {
274  selectedAccelerationModel = sphericalHarmonicAcceleration;
275  }
276  else if( !allowThirdBodyAcceleration )
277  {
278  std::string errorMessage = "Error when getting spherical harmonic acceleration for dependent variable " +
279  dependentVariableSettings->associatedBody_ + " and " +
280  dependentVariableSettings->secondaryBody_ + " type is ionconsistent (third body not checked)";
281  throw std::runtime_error( errorMessage );
282  }
283  else
284  {
285  std::shared_ptr< gravitation::ThirdBodySphericalHarmonicsGravitationalAccelerationModel >
286  thirdBodySphericalHarmonicAcceleration =
287  std::dynamic_pointer_cast< gravitation::ThirdBodySphericalHarmonicsGravitationalAccelerationModel >(
288  listOfSuitableAccelerationModels.at( 0 ) );
289 
290  if( thirdBodySphericalHarmonicAcceleration == nullptr )
291  {
292  std::string errorMessage = "Error when getting spherical harmonic acceleration for dependent variable " +
293  dependentVariableSettings->associatedBody_ + " and " +
294  dependentVariableSettings->secondaryBody_ + " type is ionconsistent (third body checked)";
295  throw std::runtime_error( errorMessage );
296  }
297  else
298  {
299  selectedAccelerationModel = thirdBodySphericalHarmonicAcceleration;
300  }
301  }
302  }
303 
304  return selectedAccelerationModel;
305 }
