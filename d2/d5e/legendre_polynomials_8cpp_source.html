67 void LegendreCache::update( const double polynomialParameter )
68 {
69  // Check if cache needs update
70  if( !( polynomialParameter == currentPolynomialParameter_ ) )
71  {
72  currentPolynomialParameter_ = polynomialParameter;
73 
74  // Set complement of argument (assuming it to be sine of latitude) cosine of latitude is always positive.
75  currentPolynomialParameterComplement_ = std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
76 
77  LegendreCache& thisReference = *this;
78 
79  int jMax = -1;
80  for( int i = 0; i <= maximumDegree_; i++ )
81  {
82  jMax = std::min( i, maximumOrder_ );
83  for( int j = 0; j <= jMax ; j++ )
84  {
85  // Compute legendre polynomial
86  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ] = legendrePolynomialFunction_( i, j, thisReference );
87 
88  if( j != 0 )
89  {
90  // Compute legendre polynomial derivative
91  if( useGeodesyNormalization_ )
92  {
93  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
94  computeGeodesyLegendrePolynomialDerivative(
95  i, j - 1, currentPolynomialParameter_,
96  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
97  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ],
98  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] );
99  }
100  else
101  {
102  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
103  computeLegendrePolynomialDerivative(
104  j - 1, currentPolynomialParameter_,
105  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
106  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ] );
107  }
108 
109  }
110  }
111 
112  // Compute legendre polynomial derivative for i = j (if needed)
113  if( jMax == i )
114  {
115  if( useGeodesyNormalization_ )
116  {
117  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
118  computeGeodesyLegendrePolynomialDerivative(
119  i, jMax, currentPolynomialParameter_,
120  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
121  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + jMax ] );
122  }
123  else
124  {
125  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
126  computeLegendrePolynomialDerivative(
127  jMax, currentPolynomialParameter_,
128  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0 );
129  }
130  }
131  }
132 
133  // Compute second derivatives of Legendre polynomials if needed
134  if( computeSecondDerivatives_ )
135  {
136  for( int i = 0; i <= maximumDegree_; i++ )
137  {
138  jMax = std::min( i, maximumOrder_ );
139  for( int j = 0; j <= jMax ; j++ )
140  {
141  if( j != 0 )
142  {
143  // Compute legendre polynomial second derivatives
144  if( useGeodesyNormalization_ )
145  {
146  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
147  computeGeodesyLegendrePolynomialSecondDerivative(
148  i, j - 1, currentPolynomialParameter_,
149  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
150  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ],
151  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
152  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + j ],
153  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] );
154  }
155  else
156  {
157  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ] =
158  computeGeodesyLegendrePolynomialSecondDerivative(
159  i, j - 1, currentPolynomialParameter_,
160  legendreValues_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
161  legendreValues_[ i * ( maximumOrder_ + 1 ) + j ],
162  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + ( j - 1 ) ],
163  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + j ], 1.0 );
164  }
165 
166  }
167  }
168  // Compute legendre polynomial second derivative for i = j (if needed)
169  if( jMax == i )
170  {
171  if( useGeodesyNormalization_ )
172  {
173  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
174  computeGeodesyLegendrePolynomialSecondDerivative(
175  i, jMax, currentPolynomialParameter_,
176  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
177  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
178  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + jMax ] );
179  }
180  else
181  {
182  legendreSecondDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ] =
183  computeGeodesyLegendrePolynomialSecondDerivative(
184  i, jMax, currentPolynomialParameter_,
185  legendreValues_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
186  legendreDerivatives_[ i * ( maximumOrder_ + 1 ) + jMax ], 0.0,
187  1.0 );
188  }
189  }
190 
191  }
192  }

197 void LegendreCache::resetMaximumDegreeAndOrder( const int maximumDegree, const int maximumOrder )
198 {
199  maximumDegree_ = maximumDegree;
200  maximumOrder_ = maximumOrder;
201 
202  if( maximumOrder_ > maximumDegree_ )
203  {
204  maximumOrder_ = maximumDegree_;
205  }
206 
207  legendreValues_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
208  legendreDerivatives_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
209  legendreSecondDerivatives_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
210 
211  derivativeNormalizations_.resize( ( maximumDegree_ + 1 ) * ( maximumOrder_ + 1 ) );
212 
213  for( int i = 0; i <= maximumDegree_; i++ )
214  {
215  for( int j = 0; ( ( j <= i ) && ( j <= maximumOrder_ ) ) ; j++ )
216  {
217  // Compute normalization correction factor.
218  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + j ] = std::sqrt(
219  ( static_cast< double >( i + j + 1 ) )
220  * ( static_cast< double >( i - j ) ) );
221 
222  // If order is zero apply multiplication factor.
223  if ( j == 0 )
224  {
225  derivativeNormalizations_[ i * ( maximumOrder_ + 1 ) + j ] *= std::sqrt( 0.5 );
226  }
227  }
228  }
229 
230  currentPolynomialParameter_ = TUDAT_NAN;
231  currentPolynomialParameterComplement_ = TUDAT_NAN;
232 }

312 double computeLegendrePolynomialFromCache( const int degree,
313  const int order,
314  LegendreCache& legendreCache )
315 {
316  if( legendreCache.getUseGeodesyNormalization( ) )
317  {
318  throw std::runtime_error( "Error when computing Legendre polynomial, input uses normalization" );
319  }
320 
321  // If degree or order is negative...
322  if ( degree < 0 || order < 0 )
323  {
324  // Set error message.
325  std::stringstream errorMessage;
326  errorMessage << "Error: the Legendre polynomial of = " << degree << " and order = "
327  << order << " is undefined." << std::endl;
328 
329  // Throw a run-time error.
330  throw std::runtime_error( errorMessage.str( ) );
331  }
332 
333  // Else if order is greater than degree...
334  else if ( order > degree && degree >= 0 )
335  {
336  // Return zero.
337  return 0.0;
338  }
339 
340  // Else if order and degree are lower than 2...
341  else if ( degree <= 1 && order <= 1 )
342  {
343  // Compute polynomial explicitly.
344  return computeLegendrePolynomialExplicit( degree, order, legendreCache.getCurrentPolynomialParameter( ) );
345  }
346 
347  // Else if degree and order are sectoral...
348  else if ( degree == order )
349  {
350  // Obtain polynomial of degree one and order one.
351  const double degreeOneOrderOnePolynomial = legendreCache.getLegendrePolynomial(
352  1, 1 );
353 
354  // Obtain prior sectoral polynomial.
355  const double priorSectoralPolynomial = legendreCache.getLegendrePolynomial(
356  degree - 1, order - 1 );
357 
358  // Compute polynomial.
359  return computeLegendrePolynomialDiagonal(
360  degree, degreeOneOrderOnePolynomial, priorSectoralPolynomial );
361  }
362 
363  // Else degree and order are zonal/tessoral...
364  else
365  {
366  // Obtain prior degree polynomial.
367  const double oneDegreePriorPolynomial = legendreCache.getLegendrePolynomial(
368  degree - 1, order );
369 
370  // Obtain two degrees prior polynomial.
371  const double twoDegreesPriorPolynomial = legendreCache.getLegendrePolynomial(
372  degree - 2, order );
373 
374  // Compute polynomial.
375  return computeLegendrePolynomialVertical( degree,
376  order,
377  legendreCache.getCurrentPolynomialParameter( ),
378  oneDegreePriorPolynomial,
379  twoDegreesPriorPolynomial );
380  }
381 }

478 double computeLegendrePolynomialDerivative( const int order,
479  const double polynomialParameter,
480  const double currentLegendrePolynomial,
481  const double incrementedLegendrePolynomial )
482 {
483  // Return polynomial derivative.
484  return incrementedLegendrePolynomial
485  / std::sqrt( 1.0 - polynomialParameter * polynomialParameter )
486  - static_cast< double >( order ) * polynomialParameter
487  / ( 1.0 - polynomialParameter * polynomialParameter )
488  * currentLegendrePolynomial;
489 }

492 double computeGeodesyLegendrePolynomialDerivative( const int degree,
493  const int order,
494  const double polynomialParameter,
495  const double currentLegendrePolynomial,
496  const double incrementedLegendrePolynomial,
497  const double normalizationCorrection )
498 {
499  // Return polynomial derivative.
500  return normalizationCorrection * incrementedLegendrePolynomial
501  / std::sqrt( 1.0 - polynomialParameter * polynomialParameter )
502  - static_cast< double >( order ) * polynomialParameter
503  / ( 1.0 - polynomialParameter * polynomialParameter )
504  * currentLegendrePolynomial;
505 }

508 double computeGeodesyLegendrePolynomialDerivative( const int degree,
509  const int order,
510  const double polynomialParameter,
511  const double currentLegendrePolynomial,
512  const double incrementedLegendrePolynomial )
513 {
514  // Compute normalization correction factor.
515  double normalizationCorrection = std::sqrt(
516  ( static_cast< double >( degree + order + 1 ) )
517  * ( static_cast< double >( degree - order ) ) );
518 
519  // If order is zero apply multiplication factor.
520  if ( order == 0 )
521  {
522  normalizationCorrection *= std::sqrt( 0.5 );
523  }
524 
525  // Return polynomial derivative.
526  return computeGeodesyLegendrePolynomialDerivative(
527  degree, order, polynomialParameter, currentLegendrePolynomial,
528  incrementedLegendrePolynomial, normalizationCorrection );
529 }

532 double computeGeodesyLegendrePolynomialSecondDerivative( const int degree,
533  const int order,
534  const double polynomialParameter,
535  const double currentLegendrePolynomial,
536  const double incrementedLegendrePolynomial,
537  const double currentLegendrePolynomialDerivative,
538  const double incrementedLegendrePolynomialDerivative,
539  const double normalizationCorrection )
540 {
541  double polynomialParameterSquare = polynomialParameter * polynomialParameter;
542 
543  // Return polynomial derivative.
544  return normalizationCorrection * (
545  incrementedLegendrePolynomialDerivative / std::sqrt( 1.0 - polynomialParameterSquare ) +
546  polynomialParameter * std::pow( 1.0 - polynomialParameterSquare, -1.5 ) * incrementedLegendrePolynomial ) -
547  static_cast< double >( order ) *
548  ( polynomialParameter / ( 1.0 - polynomialParameter * polynomialParameter ) * currentLegendrePolynomialDerivative +
549  ( 1.0 + polynomialParameterSquare ) / ( ( 1.0 - polynomialParameterSquare ) * ( 1.0 - polynomialParameterSquare ) ) * currentLegendrePolynomial );
550 }

554 double computeLegendrePolynomialExplicit( const int degree,
555  const int order,
556  const double polynomialParameter )
557 {
558  // Check which order is required for Legendre polynomial.
559  switch( degree )
560  {
561  case 0:
562  switch( order )
563  {
564  case 0:
565  return 1.0;
566  default:
567  {
568  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
569  std::to_string( degree ) + ", " +
570  std::to_string( order );
571  throw std::runtime_error( errorMessage );
572  }
573  }
574  break;
575  case 1:
576  switch( order )
577  {
578  case 0:
579  return polynomialParameter;
580  case 1:
581  return std::sqrt( 1 - polynomialParameter * polynomialParameter );
582  default:
583  {
584  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
585  std::to_string( degree ) + ", " +
586  std::to_string( order );
587  throw std::runtime_error( errorMessage );
588  }
589  }
590  break;
591  case 2:
592  switch( order )
593  {
594  case 0:
595  return 0.5 * ( 3.0 * polynomialParameter * polynomialParameter - 1.0 );
596  case 1:
597  return 3.0 * polynomialParameter
598  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
599  case 2:
600  return 3.0 * ( 1.0 - polynomialParameter * polynomialParameter );
601  default:
602  {
603  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
604  std::to_string( degree ) + ", " +
605  std::to_string( order );
606  throw std::runtime_error( errorMessage );
607  }
608  }
609  break;
610  case 3:
611  switch( order )
612  {
613  case 0:
614  return 0.5 * polynomialParameter
615  * ( 5.0 * polynomialParameter * polynomialParameter - 3.0 );
616  case 1:
617  return 1.5 * ( 5.0 * polynomialParameter * polynomialParameter - 1.0 )
618  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
619  case 2:
620  return 15.0 * polynomialParameter * ( 1.0 - polynomialParameter * polynomialParameter );
621  case 3:
622  return 15.0 * ( 1.0 - polynomialParameter * polynomialParameter )
623  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
624  default:
625  {
626  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
627  std::to_string( degree ) + ", " +
628  std::to_string( order );
629  throw std::runtime_error( errorMessage );
630  }
631  }
632  break;
633  case 4:
634  switch( order )
635  {
636  case 0:
637  return ( 35.0 * polynomialParameter * polynomialParameter
638  * polynomialParameter * polynomialParameter
639  - 30.0 * polynomialParameter * polynomialParameter + 3.0 ) / 8.0;
640  case 1:
641  return -2.5 * ( 7.0 * polynomialParameter * polynomialParameter * polynomialParameter
642  - 3.0 * polynomialParameter )
643  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
644  case 2:
645  return 15.0 / 2.0 * ( - 1.0 + 7.0 * polynomialParameter * polynomialParameter )
646  * ( 1.0 - polynomialParameter * polynomialParameter );
647  case 3:
648  return -105.0 * polynomialParameter * ( 1.0 - polynomialParameter * polynomialParameter )
649  * std::sqrt( 1.0 - polynomialParameter * polynomialParameter );
650  case 4:
651  return 105.0 * ( 1.0 - polynomialParameter * polynomialParameter )
652  * ( 1.0 - polynomialParameter * polynomialParameter );
653 
654  default:
655  {
656  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
657  std::to_string( degree ) + ", " +
658  std::to_string( order );
659  throw std::runtime_error( errorMessage );
660  }
661  }
662  break;
663  default:
664  {
665  std::string errorMessage = "Error, explicit legendre polynomial not possible for " +
666  std::to_string( degree ) + ", " +
667  std::to_string( order );
668  throw std::runtime_error( errorMessage );
669  }
670  }
671  return TUDAT_NAN;
672 }

675 double computeGeodesyLegendrePolynomialExplicit( const int degree,
676  const int order,
677  const double polynomialParameter )
678 {
679  // If 0,0 term is requested return Legendre polynomial value.
680  if ( degree == 0 && order == 0 )
681  {
682  return 1.0;
683  }
684 
685  // Else if 1,0 term is requested return polynomial value.
686  else if ( degree == 1 && order == 0 )
687  {
688  return std::sqrt( 3.0 ) * polynomialParameter;
689  }
690 
691  // Else if 1,1 term is requested return polynomial value.
692  else if ( degree == 1 && order == 1 )
693  {
694  return std::sqrt( 3.0 - 3.0 * polynomialParameter * polynomialParameter );
695  }
696 
697  // Else the requested term cannot be computed; throw a run-time error.
698  else
699  {
700  // Set error message.
701  std::stringstream errorMessage;
702  errorMessage << "Error: computation of Legendre polynomial of = " << degree
703  << " and order = " << order << " is not supported." << std::endl;
704 
705  // Throw a run-time error.
706  throw std::runtime_error( errorMessage.str( ) );
707  }
708 }

763 double calculateLegendreGeodesyNormalizationFactor( const int degree, const int order )
764 {
765 
766  double deltaFunction = 0.0;
767  if( order == 0 )
768  {
769  deltaFunction = 1.0;
770  }
771 
772  double factor = std::sqrt(
773  boost::math::factorial< double >( static_cast< double >( degree + order ) )
774  / ( ( 2.0 - deltaFunction ) * ( 2.0 * static_cast< double >( degree ) + 1.0 )
775  * boost::math::factorial< double >( static_cast< double >( degree - order ) ) ) );
776  return 1.0 / factor;
