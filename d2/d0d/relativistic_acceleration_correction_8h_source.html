282  void updateMembers( const double currentTime = TUDAT_NAN )
283  {
284  if( !( this->currentTime_ == currentTime ) )
285  {
286  this->currentTime_ = currentTime;
287 
288  // Update common variables
289  stateOfAcceleratedBodyWrtCentralBody_ = stateFunctionOfAcceleratedBody_( ) - stateFunctionOfCentralBody_( );
290  gravitationalParameterOfCentralBody_ = gravitationalParameterFunctionOfCentralBody_( );
291 
292  ppnParameterGamma_ = ppnParameterGammaFunction_( );
293  ppnParameterBeta_ = ppnParameterBetaFunction_( );
294 
295  commonCorrectionTerm_ = calculateRelativisticAccelerationCorrectionsCommonterm(
296  gravitationalParameterOfCentralBody_,
297  stateOfAcceleratedBodyWrtCentralBody_.segment( 0, 3 ).norm( ) );
298 
299  currentAcceleration_.setZero( );
300 
301  double relativeDistance = stateOfAcceleratedBodyWrtCentralBody_.segment( 0, 3 ).norm( );
302 
303  // Compute Schwarzschild term (if requested)
304  if( calculateSchwarzschildCorrection_ )
305  {
306  currentAcceleration_ = calculateScharzschildGravitationalAccelerationCorrection(
307  gravitationalParameterOfCentralBody_,
308  stateOfAcceleratedBodyWrtCentralBody_.segment( 0, 3 ),
309  stateOfAcceleratedBodyWrtCentralBody_.segment( 3, 3 ),
310  relativeDistance, commonCorrectionTerm_, ppnParameterGamma_,
311  ppnParameterBeta_ );
312  }
313 
314  // Compute Lense-Thirring term (if requested)
315  if( calculateLenseThirringCorrection_ )
316  {
317  centalBodyAngularMomentum_ = centalBodyAngularMomentumFunction_( );
318  currentAcceleration_ += calculateLenseThirringCorrectionAcceleration(
319  stateOfAcceleratedBodyWrtCentralBody_.segment( 0, 3 ),
320  stateOfAcceleratedBodyWrtCentralBody_.segment( 3, 3 ),
321  relativeDistance, commonCorrectionTerm_, centalBodyAngularMomentum_,
322  ppnParameterGamma_ );
323 
324  }
325 
326  // Compute de Sitter term (if requested)
327  if( calculateDeSitterCorrection_ )
328  {
329  stateOfCentralBodyWrtPrimaryBody_ = stateFunctionOfCentralBody_( ) - stateFunctionOfPrimaryBody_( );
330  gravitationalParameterOfPrimaryBody_ = gravitationalParameterFunctionOfPrimaryBody_( );
331 
332  double primaryDistance = stateOfCentralBodyWrtPrimaryBody_.segment( 0, 3 ).norm( );
333 
334  stateOfCentralBodyWrtPrimaryBody_ = stateFunctionOfCentralBody_( ) - stateFunctionOfPrimaryBody_( );
335  gravitationalParameterOfPrimaryBody_ = gravitationalParameterFunctionOfPrimaryBody_( );
336 
337  double largerBodyCommonCorrectionTerm = gravitationalParameterOfPrimaryBody_ / (
338  primaryDistance * primaryDistance * primaryDistance *
339  physical_constants::SPEED_OF_LIGHT * physical_constants::SPEED_OF_LIGHT );
340 
341  currentAcceleration_ += calculateDeSitterCorrectionAcceleration(
342  stateOfAcceleratedBodyWrtCentralBody_.segment( 3, 3 ),
343  stateOfCentralBodyWrtPrimaryBody_.segment( 0, 3 ),
344  stateOfCentralBodyWrtPrimaryBody_.segment( 3, 3 ),
345  largerBodyCommonCorrectionTerm,
346  ppnParameterGamma_ );
347 
348  }
349 
350 
351  }
352  }
