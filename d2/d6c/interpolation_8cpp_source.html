23 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< InterpolatorSettings >& interpolatorSettings )
24 {
25  if ( ! interpolatorSettings )
26  {
27  return;
28  }
29  using namespace json_interface;
30  using K = Keys::Interpolation::Interpolator;
31 
32  const InterpolatorTypes interpolatorType = interpolatorSettings->getInterpolatorType( );
33  jsonObject[ K::type ] = interpolatorType;
34  jsonObject[ K::lookupScheme ] = interpolatorSettings->getSelectedLookupScheme( );
35  jsonObject[ K::useLongDoubleTimeStep ] = interpolatorSettings->getUseLongDoubleTimeStep( );
36  jsonObject[ K::boundaryHandling ] = interpolatorSettings->getBoundaryHandling( );
37 
38  switch ( interpolatorType )
39  {
40  case linear_interpolator:
41  case cubic_spline_interpolator:
42  case hermite_spline_interpolator:
43  case piecewise_constant_interpolator:
44  return;
45  case lagrange_interpolator:
46  {
47  std::shared_ptr< LagrangeInterpolatorSettings > lagrangeInterpolatorSettings =
48  std::dynamic_pointer_cast< LagrangeInterpolatorSettings >( interpolatorSettings );
49  assertNonnullptrPointer( lagrangeInterpolatorSettings );
50  jsonObject[ K::order ] = lagrangeInterpolatorSettings->getInterpolatorOrder( );
51  jsonObject[ K::lagrangeBoundaryHandling ] = lagrangeInterpolatorSettings->getLagrangeBoundaryHandling( );
52  return;
53  }
54  default:
55  handleUnimplementedEnumValue( interpolatorType, oneDimensionalInterpolatorTypes,

62 {

65 
66  // Get interpolator type

69 
70  switch ( interpolatorType )
71  {
72  case linear_interpolator:
73  case cubic_spline_interpolator:
74  case hermite_spline_interpolator:
75  case piecewise_constant_interpolator:
76  {
77  InterpolatorSettings defaults( linear_interpolator );
78  interpolatorSettings = std::make_shared< InterpolatorSettings >(
79  interpolatorType,

81  getValue( jsonObject, K::useLongDoubleTimeStep, defaults.getUseLongDoubleTimeStep( ) ),
82  getValue( jsonObject, K::boundaryHandling, defaults.getBoundaryHandling( ) ) );
83  return;
84  }
