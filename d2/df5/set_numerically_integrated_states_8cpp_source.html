22 void checkTranslationalStatesFeasibility(
23  const std::vector< std::string >& bodiesToIntegrate,
24  const simulation_setup::NamedBodyMap& bodyMap )
25 {
26  // Check feasibility of epheme ris origins.
27  for( simulation_setup::NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
28  bodyIterator != bodyMap.end( ); bodyIterator++ )
29  {
30  if( std::find( bodiesToIntegrate.begin( ), bodiesToIntegrate.end( ), bodyIterator->first ) ==
31  bodiesToIntegrate.end( ) )
32  {
33  std::string ephemerisOrigin
34  = bodyIterator->second->getEphemeris( )->getReferenceFrameOrigin( );
35 
36  if( std::find( bodiesToIntegrate.begin( ), bodiesToIntegrate.end( ), ephemerisOrigin )
37  != bodiesToIntegrate.end( ) )
38  {
39  std::cerr << "Warning, found non-integrated body with an integrated body as ephemeris origin" +
40  bodyIterator->second->getEphemeris( )->getReferenceFrameOrigin( ) + " " +
41  bodyIterator->first << std::endl;
42  }
43  }
44 
45  }
46 
47  // Check whether each integrated body exists, and whether it has a TabulatedEphemeris
48  for( unsigned int i = 0; i < bodiesToIntegrate.size( ); i++ )
49  {
50  std::string bodyToIntegrate = bodiesToIntegrate.at( i );
51 
52  if( bodyMap.count( bodyToIntegrate ) == 0 )
53  {
54  throw std::runtime_error( "Error when checking translational dynamics feasibility of body " +
55  bodyToIntegrate + " no such body found" );
56  }
57  else
58  {
59  if( bodyMap.at( bodyToIntegrate )->getEphemeris( ) == nullptr )
60  {
61  throw std::runtime_error( "Error when checking translational dynamics feasibility of body " +
62  bodyToIntegrate + " no ephemeris found" );
63  }
64  }
65  }
