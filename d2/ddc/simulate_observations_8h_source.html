133 template< typename ObservationScalarType = double, typename TimeType = double,
134  int ObservationSize = 1 >
135 std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,std::pair< std::vector< TimeType >, LinkEndType > >
136 simulateSingleObservationSet(
137  const std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > observationsToSimulate,
138  const std::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > observationModel,
139  const std::vector< std::shared_ptr< ObservationViabilityCalculator > > currentObservationViabilityCalculators =
140  std::vector< std::shared_ptr< ObservationViabilityCalculator > >( ) )
141 {
142  // Delcare return type.
143  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, std::pair< std::vector< TimeType >, LinkEndType > >
144  simulatedObservations;
145 
146  // Simulate observations from tabulated times.
147  if( std::dynamic_pointer_cast< TabulatedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate ) != nullptr )
148  {
149  std::shared_ptr< TabulatedObservationSimulationTimeSettings< TimeType > > tabulatedObservationSettings =
150  std::dynamic_pointer_cast< TabulatedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate );
151 
152  // Simulate observations at requested pre-defined time.
153  simulatedObservations = simulateObservationsWithCheckAndLinkEndIdOutput<
154  ObservationSize, ObservationScalarType, TimeType >(
155  tabulatedObservationSettings->simulationTimes_, observationModel, observationsToSimulate->linkEndType_,
156  currentObservationViabilityCalculators );
157 
158  }
159  // Simulate observations per arc from settings
160  else if( std::dynamic_pointer_cast< ArcLimitedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate ) != NULL )
161  {
162  std::shared_ptr< ArcLimitedObservationSimulationTimeSettings< TimeType > > arcLimitedObservationSettings =
163  std::dynamic_pointer_cast< ArcLimitedObservationSimulationTimeSettings< TimeType > >( observationsToSimulate );
164 
165  // Define constituents of return pair.
166  std::vector< TimeType > times;
167  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > observations;
168  int observableSize = observationModel->getObservationSize( );
169 
170  // Define vector to be used for storing single arc observations.
171  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > singleArcObservations =
172  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >::Zero(
173  observableSize * arcLimitedObservationSettings->observationLimitPerArc_ );
174 
175  // Set start and end times.
176  TimeType currentTime = arcLimitedObservationSettings->startTime_;
177  TimeType currentArcStartTime = arcLimitedObservationSettings->startTime_;
178 
179  int totalNumberOfObservations = 0;
180  int numberOfObservationsInCurrentArc = 0;
181 
182  // Simulate observations arcs until provided end time.
183  while( currentTime < arcLimitedObservationSettings->endTime_ - 0.1 )
184  {
185  // Reset variables for start of new arc.
186  numberOfObservationsInCurrentArc = 0;
187  singleArcObservations.setZero( );
188 
189  // Simulate observations for sinlge arc
190  while( ( currentTime < currentArcStartTime + arcLimitedObservationSettings->arcDuration_ ) &&
191  ( numberOfObservationsInCurrentArc < arcLimitedObservationSettings->observationLimitPerArc_ ) &&
192  ( currentTime < arcLimitedObservationSettings->endTime_ ) )
193  {
194  // Simulate single observation
195  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >, bool > currentObservation =
196  simulateObservationWithCheck<
197  ObservationSize, ObservationScalarType, TimeType >(
198  currentTime, observationModel, observationsToSimulate->linkEndType_,
199  currentObservationViabilityCalculators );
200 
201  // If observation is possible, set it in current arc observations.
202  if( currentObservation.second )
203  {
204  times.push_back( currentTime );
205  singleArcObservations.segment( numberOfObservationsInCurrentArc * observableSize, observableSize ) = currentObservation.first;
206  numberOfObservationsInCurrentArc += observableSize;
207  }
208  currentTime += arcLimitedObservationSettings->observationInterval_;
209  }
210 
211  // Add single arc observations to total observations.
212  observations.conservativeResize( totalNumberOfObservations + numberOfObservationsInCurrentArc );
213  observations.segment( totalNumberOfObservations, numberOfObservationsInCurrentArc ) = singleArcObservations.segment( 0, numberOfObservationsInCurrentArc );
214  totalNumberOfObservations += numberOfObservationsInCurrentArc;
215 
216  // Update times to next arc
217  currentArcStartTime += arcLimitedObservationSettings->arcDuration_;
218  currentTime = currentArcStartTime;
219 
220  }
221 
222  simulatedObservations = std::make_pair( observations, std::make_pair( times, arcLimitedObservationSettings->linkEndType_ ) );
223  }
224 
225  return simulatedObservations;
226 }

340 template< typename ObservationScalarType = double, typename TimeType = double >
341 std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
342 std::pair< std::vector< TimeType >, LinkEndType > > > >
343 simulateObservations(
344  const std::map< ObservableType, std::map< LinkEnds,
345  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >& observationsToSimulate,
346  const std::map< ObservableType,
347  std::shared_ptr< ObservationSimulatorBase< ObservationScalarType, TimeType > > >& observationSimulators,
348  const PerObservableObservationViabilityCalculatorList viabilityCalculatorList =
349  PerObservableObservationViabilityCalculatorList( ) )
350 {
351  // Declare return map.
352  std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
353  std::pair< std::vector< TimeType >, LinkEndType > > > > observations;
354 
355  // Iterate over all observables.
356  for( typename std::map< ObservableType, std::map< LinkEnds,
357  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >::const_iterator observationIterator =
358  observationsToSimulate.begin( ); observationIterator != observationsToSimulate.end( ); observationIterator++ )
359  {
360 
361  PerLinkEndsObservationViabilityCalculatorList perLinkViabilityCalculators;
362  if( viabilityCalculatorList.count( observationIterator->first ) > 0 )
363  {
364  perLinkViabilityCalculators = viabilityCalculatorList.at( observationIterator->first );
365  }
366  // Iterate over all link ends for current observable.
367  for( typename std::map< LinkEnds,
368  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > >::const_iterator linkEndIterator =
369  observationIterator->second.begin( ); linkEndIterator != observationIterator->second.end( ); linkEndIterator++ )
370  {
371 
372  std::vector< std::shared_ptr< ObservationViabilityCalculator > > currentObservationViabilityCalculators;
373  if( perLinkViabilityCalculators.count( linkEndIterator->first ) > 0 )
374  {
375  currentObservationViabilityCalculators = perLinkViabilityCalculators.at( linkEndIterator->first );
376  }
377 
378  int observationSize = observationSimulators.at( observationIterator->first )->getObservationSize(
379  linkEndIterator->first );
380 
381  switch( observationSize )
382  {
383  case 1:
384  {
385  std::shared_ptr< ObservationSimulator< 1, ObservationScalarType, TimeType > > derivedObservationSimulator =
386  std::dynamic_pointer_cast< ObservationSimulator< 1, ObservationScalarType, TimeType > >(
387  observationSimulators.at( observationIterator->first ) );
388 
389  if( derivedObservationSimulator == nullptr )
390  {
391  throw std::runtime_error( "Error when simulating observation: dynamic case to size 1 is nullptr" );
392  }
393 
394  // Simulate observations for current observable and link ends set.
395  observations[ observationIterator->first ][ linkEndIterator->first ] =
396  simulateSingleObservationSet< ObservationScalarType, TimeType, 1 >(
397  linkEndIterator->second, derivedObservationSimulator,
398  linkEndIterator->first, currentObservationViabilityCalculators );
399  break;
400  }
401  case 2:
402  {
403  std::shared_ptr< ObservationSimulator< 2, ObservationScalarType, TimeType > > derivedObservationSimulator =
404  std::dynamic_pointer_cast< ObservationSimulator< 2, ObservationScalarType, TimeType > >(
405  observationSimulators.at( observationIterator->first ) );
406 
407  if( derivedObservationSimulator == nullptr )
408  {
409  throw std::runtime_error( "Error when simulating observation: dynamic case to size 2 is nullptr" );
410  }
411 
412  // Simulate observations for current observable and link ends set.
413  observations[ observationIterator->first ][ linkEndIterator->first ] =
414  simulateSingleObservationSet< ObservationScalarType, TimeType, 2 >(
415  linkEndIterator->second, derivedObservationSimulator,
416  linkEndIterator->first, currentObservationViabilityCalculators );
417  break;
418  }
419  case 3:
420  {
421  std::shared_ptr< ObservationSimulator< 3, ObservationScalarType, TimeType > > derivedObservationSimulator =
422  std::dynamic_pointer_cast< ObservationSimulator< 3, ObservationScalarType, TimeType > >(
423  observationSimulators.at( observationIterator->first ) );
424 
425  if( derivedObservationSimulator == nullptr )
426  {
427  throw std::runtime_error( "Error when simulating observation: dynamic case to size 3 is nullptr" );
428  }
429 
430  // Simulate observations for current observable and link ends set.
431  observations[ observationIterator->first ][ linkEndIterator->first ] =
432  simulateSingleObservationSet< ObservationScalarType, TimeType, 3 >(
433  linkEndIterator->second, derivedObservationSimulator,
434  linkEndIterator->first, currentObservationViabilityCalculators );
435  break;
436  }
437  default:
438  throw std::runtime_error( "Error, simulation of observations not yet implemented for size " +
439  std::to_string( observationSize ) );
440 
441  }
442  }
443  }
444  return observations;
445 }

460 template< typename ObservationScalarType = double, typename TimeType = double >
461 std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
462 std::pair< std::vector< TimeType >, LinkEndType > > > >
463 simulateObservationsWithNoise(
464  const std::map< ObservableType, std::map< LinkEnds,
465  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >& observationsToSimulate,
466  const std::map< ObservableType,
467  std::shared_ptr< ObservationSimulatorBase< ObservationScalarType, TimeType > > >& observationSimulators,
468  const std::map< ObservableType, std::map< LinkEnds, std::function< Eigen::VectorXd( const double ) > > >& noiseFunctions,
469  const PerObservableObservationViabilityCalculatorList viabilityCalculatorList =
470  PerObservableObservationViabilityCalculatorList( ) )
471 {
472  typedef std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
473  std::pair< std::vector< TimeType >, LinkEndType > > > SingelTypeObservationsMap;
474  typedef std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
475  std::pair< std::vector< TimeType >, LinkEndType > > > > ObservationsMap;
476 
477  // Simulate noise-free observations
478  ObservationsMap noiseFreeObservationsList = simulateObservations(
479  observationsToSimulate, observationSimulators, viabilityCalculatorList );
480 
481  // Declare return map with noisy observations.
482  ObservationsMap noisyObservationsList;
483 
484  // Iterate over all observable types
485  for( typename ObservationsMap::iterator observationIterator = noiseFreeObservationsList.begin( );
486  observationIterator != noiseFreeObservationsList.end( ); observationIterator++ )
487  {
488  int currentObservableSize = getObservableSize( observationIterator->first );
489 
490  // Iterate over all link ends of current observable
491  SingelTypeObservationsMap singleObservableObservationsWithNoise;
492  for( typename SingelTypeObservationsMap::iterator linkIterator = observationIterator->second.begin( );
493  linkIterator != observationIterator->second.end( ); linkIterator++ )
494  {
495  // Retrieve noise-free observations/times
496  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
497  std::pair< std::vector< TimeType >, LinkEndType > > nominalObservationsAndTimes =
498  linkIterator->second;
499  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > noisyObservations = nominalObservationsAndTimes.first;
500  std::vector< TimeType > nominalTimes = nominalObservationsAndTimes.second.first;
501 
502  // Check data consistency
503  if( static_cast< int >( nominalTimes.size( ) ) != noisyObservations.rows( ) / currentObservableSize )
504  {
505  throw std::runtime_error( "Error when adding noise to observations, input data is inconsistent" );
506  }
507 
508  // Retrieve noise function
509  std::function< Eigen::VectorXd( const double ) > currentNoiseFunction =
510  noiseFunctions.at( observationIterator->first ).at( linkIterator->first );
511 
512  // Simulate noise for all observations and add to calculated values
513  for( unsigned int i = 0; i < nominalTimes.size( ); i++ )
514  {
515  // Check noise function consistency
516  if( i == 0 )
517  {
518  if( currentNoiseFunction( nominalTimes.at( i ) ).rows( ) != currentObservableSize )
519  {
520  throw std::runtime_error( "Error when adding noise to observations, noise size is inconsistent" );
521  }
522  }
523 
524  // Add noise to observation
525  noisyObservations.segment( i * currentObservableSize, currentObservableSize ) +=
526  currentNoiseFunction( nominalTimes.at( i ) );
527  }
528 
529  singleObservableObservationsWithNoise[ linkIterator->first ] =
530  std::make_pair( noisyObservations, nominalObservationsAndTimes.second );
531  }
532  noisyObservationsList[ observationIterator->first ] = singleObservableObservationsWithNoise;
533  }
534  return noisyObservationsList;
535 }
