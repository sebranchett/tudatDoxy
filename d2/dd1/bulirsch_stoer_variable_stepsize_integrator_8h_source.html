215  virtual StateType performIntegrationStep( const TimeStepType stepSize )
216  {
217  StateType stateAtFirstPoint_;
218  StateType stateAtCenterPoint_;
219  StateType stateAtLastPoint_;
220 
221  bool stepSuccessful = 0;
222 
223  // Compute sub steps to take.
224  for ( unsigned int p = 0; p < subSteps_.size( ); p++ )
225  {
226  subSteps_.at( p ) = stepSize / static_cast< double >(
227  sequence_.at( p ) );
228  }
229 
230  double errorScaleTerm = TUDAT_NAN;
231  for ( unsigned int i = 0; i <= maximumStepIndex_; i++ )
232  {
233  // Compute Euler step and set as state at center point for use with mid-point method.
234  stateAtCenterPoint_ = currentState_ + subSteps_.at( i )
235  * this->stateDerivativeFunction_( currentIndependentVariable_, currentState_ );
236 
237  // Apply modified mid-point rule.
238  stateAtFirstPoint_ = currentState_;
239  IndependentVariableType independentVariableAtFirstPoint_ = currentIndependentVariable_;
240  for ( unsigned int j = 0; j < sequence_.at( i ) - 1; j++ )
241  {
242  stateAtLastPoint_ = executeMidPointMethod(
243  stateAtFirstPoint_, stateAtCenterPoint_,
244  independentVariableAtFirstPoint_, subSteps_.at( i ) );
245 
246  if ( j < sequence_.at( i ) - 2 )
247  {
248  stateAtFirstPoint_ = stateAtCenterPoint_;
249  stateAtCenterPoint_ = stateAtLastPoint_;
250  independentVariableAtFirstPoint_ += subSteps_.at( i );
251  }
252  }
253 
254  // Apply end-point correction.
255  integratedStates_[ i ][ 0 ]
256  = 0.5 * ( stateAtLastPoint_ + stateAtCenterPoint_+ subSteps_.at( i ) * this->stateDerivativeFunction_(
257  currentIndependentVariable_ + stepSize, stateAtLastPoint_ ) );
258 
259  for ( unsigned int k = 1; k < i + 1; k++ )
260  {
261  integratedStates_[ i ][ k ] =
262  integratedStates_[ i ][ k - 1 ] + 1.0 /
263  ( pow( subSteps_.at( i - k ), 2.0 ) / std::pow( subSteps_.at( i ), 2.0 ) - 1.0 )
264  * ( integratedStates_[ i ][ k - 1 ] - integratedStates_[ i - 1 ][ k - 1 ] );
265  }
266 
267  if( i == maximumStepIndex_ )
268  {
269  const StateType errorTolerance_ =
270  ( integratedStates_.at( i ).at( i ).array( ).abs( ) * relativeErrorTolerance_.array( ) ).matrix( )
271  + absoluteErrorTolerance_;
272 
273  double maximumAllowableErrorValue = errorTolerance_.array( ).maxCoeff( );
274  double maximumErrorValue = ( integratedStates_.at( i ).at( i ) - integratedStates_.at( i ).at( i - 1 ) ).array( ).abs( ).maxCoeff( );
275 
276  errorScaleTerm = safetyFactorForNextStepSize_ * std::pow( maximumAllowableErrorValue / maximumErrorValue,
277  ( 1.0 / static_cast< double >( 2 * i - 1 ) ) );
278 
279  if( maximumErrorValue < maximumAllowableErrorValue )
280  {
281  // Accept the current step.
282  lastIndependentVariable_ = currentIndependentVariable_;
283  lastState_ = currentState_;
284  currentIndependentVariable_ += stepSize;
285  currentState_ = integratedStates_[ i ][ i ];
286  stepSize_ = stepSize;
287  stepSuccessful = true;
288  }
289  else
290  {
291  stepSuccessful = false;
292  }
293  }
294  }
295 
296  if( !stepSuccessful )
297  {
298  if( safetyFactorForNextStepSize_ * errorScaleTerm < minimumFactorDecreaseForNextStepSize_ )
299  {
300  this->stepSize_ = stepSize * minimumFactorDecreaseForNextStepSize_ * errorScaleTerm;
301  }
302  else
303  {
304  this->stepSize_ = stepSize * safetyFactorForNextStepSize_ * errorScaleTerm;
305  }
306 
307  if( std::fabs( stepSize_ ) < std::fabs( minimumStepSize_ ) )
308  {
309  isMinimumStepSizeViolated_ = true;
310  throw std::runtime_error( "Error in BS integrator, minimum step size exceeded" );
311  }
312  performIntegrationStep( stepSize_ );
313  }
314  else
315  {
316  if( errorScaleTerm > maximumFactorIncreaseForNextStepSize_ )
317  {
318  this->stepSize_ = stepSize * maximumFactorIncreaseForNextStepSize_;
319  }
320  else
321  {
322  this->stepSize_ = stepSize * errorScaleTerm;
323  }
324 
325  if( std::fabs( this->stepSize_ ) >= std::fabs( maximumStepSize_ ) )
326  {
327  this->stepSize_ = stepSize / ( std::fabs( stepSize ) ) * maximumStepSize_ ;
328  }
329 
330  }
331 
332 
333  return currentState_;
334  }
