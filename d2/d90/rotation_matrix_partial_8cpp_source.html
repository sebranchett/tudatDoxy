20 Eigen::Matrix3d calculatePartialOfRotationMatrixFromLocalFrameWrtConstantRotationRate(
21  const Eigen::Quaterniond initialBodyFixedToIntegrationFrame,
22  const double rotationRate, const double timeSinceEpoch )
23 {
24  double currentRotationAngle = rotationRate * timeSinceEpoch;
25 
26  // Compute partial of rotation term containing rotation rate
27  Eigen::Matrix3d rotationMatrixDerivative;
28  double sineOfAngle = sin( currentRotationAngle );
29  double cosineOfAngle = cos( currentRotationAngle );
30  rotationMatrixDerivative << -sineOfAngle, -cosineOfAngle, 0.0, cosineOfAngle, - sineOfAngle, 0.0, 0.0, 0.0, 0.0;
31 
32  return timeSinceEpoch * ( initialBodyFixedToIntegrationFrame.toRotationMatrix( ) ) * rotationMatrixDerivative;

48 std::vector< Eigen::Matrix3d > calculatePartialOfRotationMatrixFromLocalFrameWrtPoleOrientation(
49  const Eigen::Vector3d initialOrientationAngles,
50  const double rotationRate, const double timeSinceEpoch )
51 {
52  Eigen::Matrix3d commonTerm = Eigen::AngleAxisd(
53  -1.0 * ( -initialOrientationAngles.z( ) - rotationRate * timeSinceEpoch ),
54  Eigen::Vector3d::UnitZ( ) ).toRotationMatrix( );
55 
56 
57  double rightAscension = initialOrientationAngles.x( );
58  double declination = initialOrientationAngles.y( );
59 
60  // Compute partial of rotation term containing right ascension.
61  Eigen::Matrix3d rightAscensionPartial = -reference_frames::getDerivativeOfZAxisRotationWrtAngle(
62  -( rightAscension + mathematical_constants::PI / 2.0 ) );
63 
64 
65  // Compute partial of rotation term containing declination.
66  Eigen::Matrix3d declinationPartial = reference_frames::getDerivativeOfXAxisRotationWrtAngle(
67  - ( mathematical_constants::PI / 2.0 - declination ) );
68 
69  // Compute partials.
70  std::vector< Eigen::Matrix3d > rotationMatrixPartials;
71  rotationMatrixPartials.push_back(
72  rightAscensionPartial * Eigen::Matrix3d(
73  Eigen::AngleAxisd( -( declination - mathematical_constants::PI / 2.0 ),
74  Eigen::Vector3d::UnitX( ) ) ) * commonTerm );
75  rotationMatrixPartials.push_back(
76  Eigen::AngleAxisd( ( rightAscension + mathematical_constants::PI / 2.0 ),
77  Eigen::Vector3d::UnitZ( ) ).toRotationMatrix( ) *
78  declinationPartial * commonTerm );
79 
80  return rotationMatrixPartials;
81 }
