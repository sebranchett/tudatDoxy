36 Eigen::Vector3d calculateInertialToPlanetFixedRotationAnglesFromMatrix(
37  const Eigen::Matrix3d& rotationMatrixFromInertialToPlanetFixedFrame )
38 {
39  Eigen::Vector3d rotationAngles;
40  rotationAngles.x( ) = basic_mathematics::computeModulo(
41  std::atan2( rotationMatrixFromInertialToPlanetFixedFrame( 2, 0 ),
42  -rotationMatrixFromInertialToPlanetFixedFrame( 2, 1 ) ) - mathematical_constants::PI / 2.0,
43  2.0 * mathematical_constants::PI );//right ascension
44  rotationAngles.y( ) = -std::acos( rotationMatrixFromInertialToPlanetFixedFrame( 2, 2 ) ) + mathematical_constants::PI / 2.0 ; // declination
45  rotationAngles.z( ) = std::atan2( rotationMatrixFromInertialToPlanetFixedFrame( 0, 2 ),
46  rotationMatrixFromInertialToPlanetFixedFrame( 1, 2 ) ); // longitude of prime meridian
47  return rotationAngles;
48 }

68 Eigen::Vector3d transformVectorFromVectorFunctions(
69  const Eigen::Vector3d& originalVector,
70  const std::vector< std::function< Eigen::Vector3d( const Eigen::Vector3d& ) > >& rotationsList )
71 {
72  Eigen::Vector3d currentVector = originalVector;
73  Eigen::Vector3d newVector;
74 
75  // Apply each of the required tranformations.
76  for( unsigned int i = 0; i < rotationsList.size( ); i++ )
77  {
78  newVector = rotationsList.at( i )( currentVector );
79  currentVector = newVector;
80  }
81  return currentVector;

114 Eigen::Quaterniond getRotatingPlanetocentricToInertialFrameTransformationQuaternion(
115  const double declinationOfPole,
116  const double rightAscensionOfPole,
117  const double longitudeOfPrimeMeridian )
118 {
119  // Compute transformation quaternion.
120  // Note the sign change, because how angleAxisd is defined.
121  Eigen::AngleAxisd firstRotationAroundZaxis =
122  Eigen::AngleAxisd( longitudeOfPrimeMeridian, Eigen::Vector3d::UnitZ( ) );
123  Eigen::AngleAxisd rotationAroundXaxis =
124  Eigen::AngleAxisd(
125  ( mathematical_constants::PI / 2.0 - declinationOfPole ),
126  Eigen::Vector3d::UnitX( ) );
127  Eigen::AngleAxisd secondRotationAroundZaxis = Eigen::AngleAxisd(
128  rightAscensionOfPole
129  + mathematical_constants::PI / 2.0, Eigen::Vector3d::UnitZ( ) );
130  Eigen::Quaterniond frameTransformationQuaternion = Eigen::Quaterniond(
131  ( secondRotationAroundZaxis * rotationAroundXaxis * firstRotationAroundZaxis ) );
132  return frameTransformationQuaternion;

149 Eigen::Matrix3d getVelocityBasedLvlhToInertialRotation(
150  const Eigen::Vector6d& vehicleState,
151  const Eigen::Vector6d& centralBodyState,
152  const bool doesNaxisPointAwayFromCentralBody )
153 {
154  Eigen::Vector3d vehicleVelocity, vehicleRadius;
155  vehicleRadius = vehicleState.segment( 0, 3 ) - centralBodyState.segment( 0, 3 );
156  vehicleVelocity = vehicleState.segment( 3, 3 ) - centralBodyState.segment( 3, 3 );
157 
158  Eigen::Vector3d unitT = vehicleVelocity.normalized( );// / vehicleVelocity.norm( );
159  if ( vehicleRadius.cross( vehicleVelocity ).norm( ) == 0.0 )
160  {
161  std::string errorMessage = "Division by zero: radius and velocity are in the same direction.";
162  throw std::runtime_error( errorMessage );
163  }
164 
165  Eigen::Vector3d unitW = ( ( ( doesNaxisPointAwayFromCentralBody ) ? -1.0 : 1.0 ) *
166  ( vehicleRadius.cross( vehicleVelocity ) ).normalized( ) );
167 
168  Eigen::Vector3d unitN = ( unitW.cross( unitT ) ).normalized( );
169 
170  Eigen::Matrix3d transformationMatrix;
171  transformationMatrix << unitT( 0 ), unitN( 0 ), unitW( 0 ),
172  unitT( 1 ), unitN( 1 ), unitW( 1 ),
173  unitT( 2 ), unitN( 2 ), unitW( 2 );
174 
175  return transformationMatrix;
176 }

189 Eigen::Quaterniond getVelocityBasedLvlhToPlanetocentricRotationKeplerian(
190  const Eigen::Matrix< double, 6, 1 > spacecraftKeplerianState )
191 {
192  double eccentricity = spacecraftKeplerianState( 1 );
193  double inclination = spacecraftKeplerianState( 2 );
194  double argumentOfPeriapsis = spacecraftKeplerianState( 3 );
195  double rightAscensionOfAscendingNode = spacecraftKeplerianState( 4 );
196  double trueAnomaly = spacecraftKeplerianState( 5 );
197 
198  double flightPathAngle = std::atan( ( eccentricity * std::sin( trueAnomaly ) ) /
199  ( 1.0 + eccentricity * std::cos( trueAnomaly ) ) );
200 
201  // Compute first rotation around Z axis.
202  Eigen::AngleAxisd firstRotationAroundZaxis(
203  -( -mathematical_constants::PI * 0.5 + flightPathAngle - ( trueAnomaly + argumentOfPeriapsis ) ),
204  Eigen::Vector3d::UnitZ( ) );
205 
206  // Compute rotation around X axis.
207  Eigen::AngleAxisd rotationAroundXaxis( inclination, Eigen::Vector3d::UnitX( ) );
208 
209  // Compute second rotation around Z axis.
210  Eigen::AngleAxisd secondRotationAroundZaxis( rightAscensionOfAscendingNode, Eigen::Vector3d::UnitZ( ) );
211 
212  Eigen::Quaterniond frameTransformationQuaternion = Eigen::Quaterniond(
213  ( secondRotationAroundZaxis * rotationAroundXaxis * firstRotationAroundZaxis) );
214 
215  // Return transformation quaternion.
216  return frameTransformationQuaternion;
217 }

220 Eigen::Matrix3d getInertialToRswSatelliteCenteredFrameRotationMatrix(
221  const Eigen::Vector6d bodyState )
222 {
223  Eigen::Vector3d vehicleVelocity, vehicleRadius;
224  vehicleRadius = bodyState.segment( 0, 3 );
225  vehicleVelocity = bodyState.segment( 3, 3 );
226 
227  Eigen::Vector3d unitR = vehicleRadius.normalized( );// / vehicleVelocity.norm( );
228  if ( vehicleRadius.cross( vehicleVelocity ).norm( ) == 0.0 )
229  {
230  std::string errorMessage = "Division by zero: radius and velocity are in the same direction in RSW frame.";
231  throw std::runtime_error( errorMessage );
232  }
233 
234  Eigen::Vector3d unitW = ( vehicleRadius.cross( vehicleVelocity ) ).normalized( );
235 
236  Eigen::Vector3d unitS = ( unitW.cross( unitR ) ).normalized( );
237 
238  Eigen::Matrix3d transformationMatrix;
239  transformationMatrix << unitR( 0 ), unitR( 1 ), unitR( 2 ),
240  unitS( 0 ), unitS( 1 ), unitS( 2 ),
241  unitW( 0 ), unitW( 1 ), unitW( 2 );
242  return transformationMatrix;
243 }

441 Eigen::Quaterniond getEnuLocalVerticalToRotatingPlanetocentricFrameTransformationQuaternion(
442  double longitude, double latitude )
443 {
444  // Compute transformation quaternion.
445  // source: http://www.navipedia.net/index.php/Transformations_between_ECEF_and_ENU_coordinates
446  // Note the sign change (-1.0), because how angleAxisd is defined.
447  Eigen::AngleAxisd RotationAroundZaxis = Eigen::AngleAxisd(
448  longitude + mathematical_constants::PI / 2.0, Eigen::Vector3d::UnitZ( ) );
449  Eigen::AngleAxisd RotationAroundXaxis =
450  Eigen::AngleAxisd( ( mathematical_constants::PI / 2.0 - latitude ),
451  Eigen::Vector3d::UnitX( ) );
452  Eigen::Quaterniond frameTransformationQuaternion = Eigen::Quaterniond(
453  ( RotationAroundZaxis * RotationAroundXaxis ) );
454 
455  // Return transformation quaternion.
456  return frameTransformationQuaternion;
