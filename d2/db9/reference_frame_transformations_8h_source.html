273 template< typename AngleScalarType = double >
274 Eigen::Quaternion< AngleScalarType > getRotatingPlanetocentricToLocalVerticalFrameTransformationQuaternion(
275  const AngleScalarType longitude, const AngleScalarType latitude )
276 {
277  // Compute transformation quaternion.
278  // Note the sign change, because how angleAxisd is defined.
279  Eigen::AngleAxis< AngleScalarType > RotationAroundZaxis = Eigen::AngleAxis< AngleScalarType >(
280  -longitude, Eigen::Matrix< AngleScalarType, 3, 1 >::UnitZ( ) );
281  Eigen::AngleAxis< AngleScalarType > RotationAroundYaxis = Eigen::AngleAxis< AngleScalarType >(
282  -( -latitude - mathematical_constants::getPi< AngleScalarType >( ) /
283  mathematical_constants::getFloatingInteger< AngleScalarType >( 2 ) ),
284  Eigen::Matrix< AngleScalarType, 3, 1 >::UnitY( ) );
285  Eigen::Quaternion< AngleScalarType > frameTransformationQuaternion = Eigen::Quaternion< AngleScalarType >(
286  ( RotationAroundYaxis * RotationAroundZaxis ) );
287 
288  // Return transformation quaternion.
289  return frameTransformationQuaternion;
290 }
