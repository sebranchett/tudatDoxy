40 template< typename S, typename T >
41 std::map< S, T > linearlyScaleKeyOfMap(
42  const std::map< S, T >& originalMap, S offset, S scale, bool isOffsetAppliedFirst = true )
43 {
44  // Declare new map
45  std::map< S, T > scaledMap;
46  double newKey;
47 
48  // Iterate over old map and calculate new key values.
49  for( typename std::map< S, T >::const_iterator mapIterator = originalMap.begin( ); mapIterator != originalMap.end( ); mapIterator++ )
50  {
51  // Determine order in which modifications are to be applied
52  if( isOffsetAppliedFirst )
53  {
54  newKey = ( mapIterator->first - offset ) * scale;
55  }
56  else
57  {
58  newKey = mapIterator->first * scale - offset;
59  }
60 
61  // Set scalted map key with corresponding value in new map
62  scaledMap[ newKey ] = mapIterator->second;
63  }
64  return scaledMap;
65 }

225 Eigen::Matrix< ScalarType, Eigen::Dynamic, NumberOfColumns > createConcatenatedEigenMatrixFromMapValues(
226  const std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >& inputMap )
227 {
228  // Create and size return vector.
229  Eigen::Matrix< ScalarType, Eigen::Dynamic, NumberOfColumns > outputVector;
230 
231  int columns;
232  if( NumberOfColumns != Eigen::Dynamic )
233  {
234  columns = NumberOfColumns;
235  }
236  else
237  {
238  columns = inputMap.begin( )->second.cols( );
239  }
240 
241  if( NumberOfRows != Eigen::Dynamic )
242  {
243  int counter = 0;
244 
245  outputVector.resize( inputMap.size( ) * NumberOfRows, columns );
246  for( typename std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >::const_iterator mapIterator =
247  inputMap.begin( ); mapIterator != inputMap.end( ); mapIterator++ )
248  {
249  outputVector.block( counter, 0, NumberOfRows, columns ) = mapIterator->second;
250  counter += NumberOfRows;
251  }
252  }
253  else
254  {
255  int concatenatedSize = 0;
256 
257  for( typename std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >::const_iterator mapIterator =
258  inputMap.begin( ); mapIterator != inputMap.end( ); mapIterator++ )
259  {
260  concatenatedSize += mapIterator->second.rows( );
261  }
262 
263  outputVector.resize( concatenatedSize, columns );
264  int currentSize;
265  int counter = 0;
266 
267  for( typename std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >::const_iterator mapIterator =
268  inputMap.begin( ); mapIterator != inputMap.end( ); mapIterator++ )
269  {
270  currentSize = mapIterator->second.rows( );
271  outputVector.block( counter, 0, currentSize, columns ) = mapIterator->second;
272  counter += currentSize;
273  }
274  }

276  return outputVector;

591 std::vector< T > sliceStlVector( const std::vector< T >& vectorToBeSliced, unsigned int startIndex = 0,
592  unsigned int endIndex = std::numeric_limits< unsigned int >::signaling_NaN( ) )
593 {
594  // Declare output vector
595  std::vector< T > slicedVector;
596 
597  // Give value to end index
598  if ( endIndex == std::numeric_limits< unsigned int >::signaling_NaN( ) )
599  {
600  endIndex = vectorToBeSliced.size( ) - 1;
601  }
602 
603  // Check that boundaries make sense
604  if ( startIndex > endIndex )
605  {
606  // Warn user of inconsistency
607  std::cerr << "Warning in slicing of std::vector. The starting index is greater than the end index. "
608  "The indices will be swapped." << std::endl;
609 
610  // Swap indices
611  unsigned int temporaryIndex = startIndex;
612  startIndex = endIndex;
613  endIndex = temporaryIndex;
614  }
615 
616  // Transfer values to sliced array
617  for ( unsigned int i = startIndex; i < ( endIndex + 1 ); i++ )
618  {
619  slicedVector.push_back( vectorToBeSliced.at( i ) );
620  }
621  return slicedVector;
622 }
