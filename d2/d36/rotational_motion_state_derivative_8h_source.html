81  RotationalMotionStateDerivative(
82  const basic_astrodynamics::TorqueModelMap& torqueModelsPerBody,
83  const RotationalPropagatorType propagatorType,
84  const std::vector< std::string >& bodiesToPropagate,
85  std::vector< std::function< Eigen::Matrix3d( ) > > bodyInertiaTensorFunctions,
86  std::vector< std::function< Eigen::Matrix3d( ) > > bodyInertiaTensorTimeDerivativeFunctions =
87  std::vector< std::function< Eigen::Matrix3d( ) > >( ) ):
88  propagators::SingleStateTypeDerivative< StateScalarType, TimeType >(
89  propagators::rotational_state ),
90  torqueModelsPerBody_( torqueModelsPerBody ),
91  propagatorType_( propagatorType ),
92  bodiesToPropagate_( bodiesToPropagate ),
93  bodyInertiaTensorFunctions_( bodyInertiaTensorFunctions ),
94  bodyInertiaTensorTimeDerivativeFunctions_( bodyInertiaTensorTimeDerivativeFunctions )
95  {
96  // Check input consistency
97  if( bodiesToPropagate_.size( ) != bodyInertiaTensorFunctions_.size( ) )
98  {
99  throw std::runtime_error(
100  "Error when making rotational state derivative model, inertia tensor list is of incompatible size" );
101  }
102 
103  if( bodyInertiaTensorTimeDerivativeFunctions_.size( ) == 0 )
104  {
105  for( unsigned int i = 0; i < bodiesToPropagate.size( ); i++ )
106  {
107  bodyInertiaTensorTimeDerivativeFunctions_.push_back( [ ]( ){ return Eigen::Matrix3d::Zero( ); } );
108  }
109  }
110  else if( bodiesToPropagate_.size( ) != bodyInertiaTensorTimeDerivativeFunctions_.size( ) )
111  {
112  throw std::runtime_error(
113  "Error when making rotational state derivative model, inertia tensor time derivative list is of incompatible size" );
114  }
115 
116  for( unsigned int i = 0; i < bodiesToPropagate.size( ); i++ )
117  {
118  if( torqueModelsPerBody_.count( bodiesToPropagate.at( i ) ) == 0 )
119  {
120  torqueModelsPerBody_[ bodiesToPropagate.at( i ) ][ bodiesToPropagate.at( i ) ] =
121  std::vector< std::shared_ptr< basic_astrodynamics::TorqueModel > >( );
122  }
123  }

236  Eigen::Vector3d getTotalTorqueForBody( const std::string& bodyName )
237  {
238  // Check if body is propagated.
239  Eigen::Vector3d totalTorque = Eigen::Vector3d::Zero( );
240  if( std::find( bodiesToPropagate_.begin( ),
241  bodiesToPropagate_.end( ),
242  bodyName ) == bodiesToPropagate_.end( ) )
243  {
244  std::string errorMessage = "Error when getting total torque for body " + bodyName +
245  ", no such torque is found";
246  throw std::runtime_error( errorMessage );
247  }
248  else
249  {
250  if( torqueModelsPerBody_.count( bodyName ) != 0 )
251  {
252  basic_astrodynamics::SingleBodyTorqueModelMap torquesOnBody =
253  torqueModelsPerBody_.at( bodyName );
254 
255  // Iterate over all torques acting on body
256  for( innerTorqueIterator = torquesOnBody.begin( );
257  innerTorqueIterator != torquesOnBody.end( );
258  innerTorqueIterator++ )
259  {
260  for( unsigned int j = 0; j < innerTorqueIterator->second.size( ); j++ )
261  {
262  // Calculate torque and add to state derivative.
263  totalTorque += innerTorqueIterator->second[ j ]->getTorque( );
264  }
265  }
266  }
267  }
268  return totalTorque;
269  }
