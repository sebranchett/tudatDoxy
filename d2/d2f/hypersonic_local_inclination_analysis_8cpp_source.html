170  const std::vector< std::vector< double > >& dataPointsOfIndependentVariables,
171  const std::shared_ptr< SurfaceGeometry > inputVehicleSurface,
172  const std::vector< int >& numberOfLines,
173  const std::vector< int >& numberOfPoints,
174  const std::vector< bool >& invertOrders,
175  const std::vector< std::vector< int > >& selectedMethods,
176  const double referenceArea,
177  const double referenceLength,
178  const Eigen::Vector3d& momentReferencePoint,
179  const bool savePressureCoefficients )

183  stagnationPressureCoefficient( 2.0 ),

185  selectedMethods_( selectedMethods ),
186  savePressureCoefficients_( savePressureCoefficients )
187 {
188  // Set geometry if it is a single surface.
189  if ( std::dynamic_pointer_cast< SingleSurfaceGeometry > ( inputVehicleSurface ) !=
190  std::shared_ptr< SingleSurfaceGeometry >( ) )
191  {
192  // Set number of geometries and allocate memory.
193  vehicleParts_.resize( 1 );
194 
195  vehicleParts_[ 0 ] = std::make_shared< LawgsPartGeometry >( );
196  vehicleParts_[ 0 ]->setReversalOperator( invertOrders[ 0 ] );
197 
198  // If part is not already a LaWGS part, convert it.
199  if ( std::dynamic_pointer_cast< LawgsPartGeometry >
200  ( inputVehicleSurface ) ==
201  std::shared_ptr< LawgsPartGeometry >( ) )

209  // Else, set geometry directly.
210  else
211  {
212  vehicleParts_[ 0 ] = std::dynamic_pointer_cast< LawgsPartGeometry >
213  ( inputVehicleSurface );
214  }
215  }
216 
217  // Set geometry if it is a composite surface.

219  std::shared_ptr< CompositeSurfaceGeometry >( ) )

242  vehicleParts_[ i ]->setMesh(
243  compositeSurfaceGeometry_->getSingleSurfaceGeometry( i ),
244  numberOfLines[ i ], numberOfPoints[ i ] );
245  }
246 
247  // Else, set geometry directly.
248  else
249  {
250  vehicleParts_[ i ] = std::dynamic_pointer_cast< LawgsPartGeometry >(
251  compositeSurfaceGeometry_->getSingleSurfaceGeometry( i ) );
252  }
253  }
254  }
255 
256  // Allocate memory for panel inclinations and pressureCoefficient_.
257  inclination_.resize( vehicleParts_.size( ) );
258  pressureCoefficient_.resize( vehicleParts_.size( ) );
259  for ( unsigned int i = 0 ; i < vehicleParts_.size( ); i++ )
260  {
261  inclination_[ i ].resize( vehicleParts_[ i ]->getNumberOfLines( ) );

263  for ( int j = 0 ; j < vehicleParts_[ i ]->getNumberOfLines( ) ; j++ )
264  {
265  inclination_[ i ][ j ].resize( vehicleParts_[ i ]->getNumberOfPoints( ) );
266  pressureCoefficient_[ i ][ j ].resize( vehicleParts_[ i ]->getNumberOfPoints( ) );
267  }
268  }
269

274  dataPointsOfIndependentVariables_[ i ].size( );

279  std::fill( isCoefficientGenerated_.origin( ),

281 
282  generateCoefficients( );
283  createInterpolator( );

349 Vector6d HypersonicLocalInclinationAnalysis::determinePartCoefficients(
350  const int partNumber, const boost::array< int, 3 > independentVariableIndices )
351 {
352  // Declare and determine angles of attack and sideslip for analysis.
353  double angleOfAttack = dataPointsOfIndependentVariables_[ 1 ]
354  [ independentVariableIndices[ 1 ] ];
355 
356  double angleOfSideslip = dataPointsOfIndependentVariables_[ 2 ]
357  [ independentVariableIndices[ 2 ] ];
358 
359  // Declare partCoefficient vector.
360  Vector6d partCoefficients = Vector6d::Zero( );
361 
362  // Check whether the inclinations of the vehicle part have already been computed.
363  if ( previouslyComputedInclinations_.count( std::pair< double, double >(
364  angleOfAttack, angleOfSideslip ) ) == 0 )
365  {
366  // Determine panel inclinations for part.
367  determineInclinations( angleOfAttack, angleOfSideslip );
368 
369  // Add panel inclinations to container
370  previouslyComputedInclinations_[ std::pair< double, double >(
371  angleOfAttack, angleOfSideslip ) ] = inclination_;
372  }
373 
374  else
375  {
376  // Fetch inclinations from container
377  inclination_ = previouslyComputedInclinations_[ std::pair< double, double >(
378  angleOfAttack, angleOfSideslip ) ];
379  }
380 
381  // Set pressureCoefficient_ array for given independent variables.
382  determinePressureCoefficients( partNumber, independentVariableIndices );
383 
384  // Calculate force coefficients from pressure coefficients.
385  partCoefficients.segment( 0, 3 ) = calculateForceCoefficients( partNumber );
386 
387  // Calculate moment coefficients from pressure coefficients.
388  partCoefficients.segment( 3, 3 ) = calculateMomentCoefficients( partNumber );
389 
390  return partCoefficients;
391 }

411 Eigen::Vector3d HypersonicLocalInclinationAnalysis::calculateForceCoefficients(
412  const int partNumber )
413 {
414  // Declare force coefficient vector and intialize to zeros.
415  Eigen::Vector3d forceCoefficients = Eigen::Vector3d::Zero( );
416 
417  // Loop over all panels and add pressures, scaled by panel area, to force
418  // coefficients.
419  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
420  {
421  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++)
422  {
423  forceCoefficients -=
424  pressureCoefficient_[ partNumber ][ i ][ j ] *
425  vehicleParts_[ partNumber ]->getPanelArea( i, j ) *
426  vehicleParts_[ partNumber ]->getPanelSurfaceNormal( i, j );
427  }
428  }
429 
430  // Normalize result by reference area.
431  forceCoefficients /= referenceArea_;
432 
433  return forceCoefficients;
434 }

437 Eigen::Vector3d HypersonicLocalInclinationAnalysis::calculateMomentCoefficients(
438  const int partNumber )
439 {
440  // Declare moment coefficient vector and intialize to zeros.
441  Eigen::Vector3d momentCoefficients = Eigen::Vector3d::Zero( );
442 
443  // Declare moment arm for panel moment determination.
444  Eigen::Vector3d referenceDistance;
445 
446  // Loop over all panels and add moments due pressures.
447  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
448  {
449  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
450  {
451  // Determine moment arm for given panel centroid.
452  referenceDistance = ( vehicleParts_[ partNumber ]->getPanelCentroid( i, j ) -
453  momentReferencePoint_ );
454 
455  momentCoefficients -=
456  pressureCoefficient_[ partNumber ][ i ][ j ] *
457  vehicleParts_[ partNumber ]->getPanelArea( i, j ) *
458  ( referenceDistance.cross( vehicleParts_[ partNumber ]->
459  getPanelSurfaceNormal( i, j ) ) );
460  }
461  }
462 
463  // Scale result by reference length and area.
464  momentCoefficients /= ( referenceLength_ * referenceArea_ );
465 
466  return momentCoefficients;
467 }

470 void HypersonicLocalInclinationAnalysis::determineInclinations( const double angleOfAttack,
471  const double angleOfSideslip )
472 {
473  // Declare free-stream velocity vector.
474  Eigen::Vector3d freestreamVelocityDirection;
475 
476  // Set freestream velocity vector in body frame.
477  double freestreamVelocityDirectionX = cos( angleOfAttack )* cos( angleOfSideslip );
478  double freestreamVelocityDirectionY = sin( angleOfSideslip );
479  double freestreamVelocityDirectionZ = sin( angleOfAttack ) * cos( angleOfSideslip );
480  freestreamVelocityDirection( 0 ) = freestreamVelocityDirectionX;
481  freestreamVelocityDirection( 1 ) = freestreamVelocityDirectionY;
482  freestreamVelocityDirection( 2 ) = freestreamVelocityDirectionZ;
483 
484  // Declare cosine of inclination angle.
485  double cosineOfInclination;
486 
487  // Loop over all panels of given vehicle part and set inclination angles.
488  for( unsigned int k = 0; k < vehicleParts_.size( ); k++ )
489  {
490  for ( int i = 0 ; i < vehicleParts_[ k ]->getNumberOfLines( ) - 1 ; i++ )
491  {
492  for ( int j = 0 ; j < vehicleParts_[ k ]->getNumberOfPoints( ) - 1 ; j++ )
493  {
494 
495  // Determine cosine of inclination angle from inner product between
496  // surface normal and free-stream direction.
497  cosineOfInclination = vehicleParts_[ k ]->
498  getPanelSurfaceNormal( i, j ).
499  dot( freestreamVelocityDirection );
500 
501  // Set inclination angle.
502  inclination_[ k ][ i ][ j ] = PI / 2.0 - acos( cosineOfInclination );
503  }
504  }
505  }
506 }

509 void HypersonicLocalInclinationAnalysis::updateCompressionPressures( const double machNumber,
510  const int partNumber )
511 {

513 
514  std::function< double( double ) > pressureFunction;
515

517  switch( method )

522  break;
523 
524  case 1:
525  pressureFunction =
526  std::bind( aerodynamics::computeModifiedNewtonianPressureCoefficient, std::placeholders::_1,
527  stagnationPressureCoefficient );
528  break;
529 
530  case 2:
531  // Method currently disabled.
532  break;
533 
534  case 3:
535  // Method currently disabled.
536  break;
537 
538  case 4:
539  pressureFunction =
540  std::bind( aerodynamics::computeEmpiricalTangentWedgePressureCoefficient, std::placeholders::_1,
541  machNumber );
542  break;
543 
544  case 5:
545  pressureFunction =
546  std::bind( aerodynamics::computeEmpiricalTangentConePressureCoefficient, std::placeholders::_1,

548  break;
549 
550  case 6:
551  pressureFunction =
552  std::bind( aerodynamics::computeModifiedDahlemBuckPressureCoefficient, std::placeholders::_1,
553  machNumber );
554  break;
555 
556  case 7:
557  pressureFunction =
558  std::bind( aerodynamics::computeVanDykeUnifiedPressureCoefficient, std::placeholders::_1,
559  machNumber, ratioOfSpecificHeats, 1 );

562  case 8:

564  std::bind( aerodynamics::computeSmythDeltaWingPressureCoefficient, std::placeholders::_1,

566  break;
567 
568  case 9:

578  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1; i++ )
579  {
580  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
581  {
582  if ( inclination_[ partNumber ][ i ][ j ] > 0 )

584  // If panel inclination is positive, calculate pressure coefficient.

593 void HypersonicLocalInclinationAnalysis::updateExpansionPressures( const double machNumber,
594  const int partNumber )
595 {
596  // Get analysis method of part to analyze.
597  int method = selectedMethods_[ 1 ][ partNumber ];
598 
599  if ( method == 0 || method == 1 || method == 4 )
600  {
601  std::function< double( ) > pressureFunction;
602  switch( method )
603  {
604  case 0:
605  pressureFunction = std::bind( &aerodynamics::computeVacuumPressureCoefficient,
606  machNumber, ratioOfSpecificHeats );
607  break;
608 
609  case 1:
610  pressureFunction = [ ]( ){ return 0.0; };
611  break;
612 
613  case 4:
614  pressureFunction = std::bind( &aerodynamics::computeHighMachBasePressure,
615  machNumber );
616  break;
617 
618  }
619 
620  // Iterate over all panels on part.
621  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
622  {
623  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
624  {
625  if ( inclination_[ partNumber ][ i ][ j ] <= 0 )
626  {
627  // If panel inclination is negative, calculate pressure using
628  // Van Dyke unified method.
629  pressureCoefficient_[ partNumber ][ i ][ j ] =
630  pressureFunction( );
631  }
632  }
633  }
634  }
635 
636  else if( method == 3 || method == 5 || method == 6 )
637  {
638 
639  std::function< double( double ) > pressureFunction;
640 
641  // Declare local variable.
642  double freestreamPrandtlMeyerFunction;
643 
644  // Switch to analyze part using correct method.
645  switch( method )
646  {
647  case 3:
648  // Calculate freestream Prandtl-Meyer function.
649  freestreamPrandtlMeyerFunction = aerodynamics::computePrandtlMeyerFunction(
650  machNumber, ratioOfSpecificHeats );
651  pressureFunction =
652  std::bind( &aerodynamics::computePrandtlMeyerFreestreamPressureCoefficient,
653  std::placeholders::_1, machNumber, ratioOfSpecificHeats,
654  freestreamPrandtlMeyerFunction );
655  break;
656 
657  case 5:
658  pressureFunction =
659  std::bind( &aerodynamics::computePrandtlMeyerFreestreamPressureCoefficient,
660  std::placeholders::_1, machNumber, ratioOfSpecificHeats, -1 );
661  break;
662 
663  case 6:
664  pressureFunction = std::bind( &aerodynamics::computeAcmEmpiricalPressureCoefficient,
665  std::placeholders::_1, machNumber );
666  break;
667  }
668 
669  // Iterate over all panels on part.
670  for ( int i = 0 ; i < vehicleParts_[ partNumber ]->getNumberOfLines( ) - 1 ; i++ )
671  {
672  for ( int j = 0 ; j < vehicleParts_[ partNumber ]->getNumberOfPoints( ) - 1 ; j++ )
673  {
674  if ( inclination_[ partNumber ][ i ][ j ] <= 0 )
675  {
676  // If panel inclination is negative, calculate pressure using
