42 Eigen::Vector6d ApproximatePlanetPositions::getKeplerianStateFromEphemeris(
43  const double secondsSinceEpoch )
44 {
45  using std::pow;
46  using std::sin;
47  using std::cos;
48  using namespace orbital_element_conversions;
49 
50  // Set Julian date.
51  julianDate_ = basic_astrodynamics::convertSecondsSinceEpochToJulianDay(
52  secondsSinceEpoch, basic_astrodynamics::JULIAN_DAY_ON_J2000 );
53 
54  // Compute number of centuries past J2000.
55  numberOfCenturiesPastJ2000_ = ( julianDate_ - 2451545.0 ) / 36525.0;
56 
57  // Compute and set semi-major axis of planet at given Julian date.
58  planetKeplerianElementsAtGivenJulianDate_( semiMajorAxisIndex )
59  = approximatePlanetPositionsDataContainer_.semiMajorAxis_
60  + ( approximatePlanetPositionsDataContainer_
61  .rateOfChangeOfSemiMajorAxis_ * numberOfCenturiesPastJ2000_ );
62 
63  // Compute and set eccentricity of planet at given Julian date.
64  planetKeplerianElementsAtGivenJulianDate_( eccentricityIndex )
65  = approximatePlanetPositionsDataContainer_.eccentricity_
66  + ( approximatePlanetPositionsDataContainer_
67  .rateOfChangeOfEccentricity_ * numberOfCenturiesPastJ2000_ );
68 
69  // Compute and set inclination of planet at given Julian date.
70  planetKeplerianElementsAtGivenJulianDate_( inclinationIndex )
71  = approximatePlanetPositionsDataContainer_.inclination_
72  + ( approximatePlanetPositionsDataContainer_
73  .rateOfChangeOfInclination_ * numberOfCenturiesPastJ2000_ );
74 
75  // Compute and set longitude of ascending node of planet at given Julian date.
76  planetKeplerianElementsAtGivenJulianDate_( longitudeOfAscendingNodeIndex )
77  = approximatePlanetPositionsDataContainer_.longitudeOfAscendingNode_
78  + ( approximatePlanetPositionsDataContainer_
79  .rateOfChangeOfLongitudeOfAscendingNode_ * numberOfCenturiesPastJ2000_ );
80 
81  // Compute longitude of perihelion of planet at given Julian date.
82  longitudeOfPerihelionAtGivenJulianDate_
83  = approximatePlanetPositionsDataContainer_.longitudeOfPerihelion_
84  + ( approximatePlanetPositionsDataContainer_.rateOfChangeOfLongitudeOfPerihelion_
85  * numberOfCenturiesPastJ2000_ );
86 
87  // Compute and set argument of periapsis of planet at given Julian date.
88  planetKeplerianElementsAtGivenJulianDate_( argumentOfPeriapsisIndex )
89  = longitudeOfPerihelionAtGivenJulianDate_
90  - planetKeplerianElementsAtGivenJulianDate_( longitudeOfAscendingNodeIndex );
91 
92  // Compute mean longitude of planet at given Julian date.
93  meanLongitudeAtGivenJulianDate_ = approximatePlanetPositionsDataContainer_.meanLongitude_
94  + ( approximatePlanetPositionsDataContainer_.rateOfChangeOfMeanLongitude_
95  * numberOfCenturiesPastJ2000_ );
96 
97  // Compute mean anomaly of planet at given Julian date.
98  meanAnomalyAtGivenJulianDate_ = meanLongitudeAtGivenJulianDate_
99  - longitudeOfPerihelionAtGivenJulianDate_
100  + ( approximatePlanetPositionsDataContainer_.additionalTermB_
101  * pow( numberOfCenturiesPastJ2000_, 2.0 ) )
102  + ( approximatePlanetPositionsDataContainer_.additionalTermC_
103  * cos( unit_conversions::convertDegreesToRadians(approximatePlanetPositionsDataContainer_.additionalTermF_ *
104  numberOfCenturiesPastJ2000_ )) )
105  + ( approximatePlanetPositionsDataContainer_.additionalTermS_
106  * sin( unit_conversions::convertDegreesToRadians(approximatePlanetPositionsDataContainer_.additionalTermF_ *
107  numberOfCenturiesPastJ2000_) ) );
108 
109  // Compute modulo of mean anomaly for interval :
110  // 0 <= meanAnomalyAtGivenJulianDate_ < 360.
111  meanAnomalyAtGivenJulianDate_ = basic_mathematics::computeModulo(
112  meanAnomalyAtGivenJulianDate_, 360.0 );
113 
114  // Translate mean anomaly to:
115  // -180 < meanAnomalyAtGivenJulianDate_ <= 180 bounds.
116  if ( meanAnomalyAtGivenJulianDate_ > 180.0 )
117  {
118  meanAnomalyAtGivenJulianDate_ -= 360.0;
119  }
120 
121  // Convert mean anomaly to eccentric anomaly.
122  eccentricAnomalyAtGivenJulianDate_ = convertMeanAnomalyToEccentricAnomaly(
123  planetKeplerianElementsAtGivenJulianDate_( eccentricityIndex ),
124  unit_conversions::convertDegreesToRadians(
125  meanAnomalyAtGivenJulianDate_ ) );
126 
127  // Convert eccentric anomaly to true anomaly and set in planet elements.
128  trueAnomalyAtGivenJulianData_
129  = orbital_element_conversions::convertEccentricAnomalyToTrueAnomaly(
130  eccentricAnomalyAtGivenJulianDate_,
131  planetKeplerianElementsAtGivenJulianDate_( eccentricityIndex ) );
132 
133  planetKeplerianElementsAtGivenJulianDate_( trueAnomalyIndex )
134  = trueAnomalyAtGivenJulianData_;
135 
136  // Convert Keplerian elements to standard units.
137  // Convert semi-major axis from AU to meters.
138  planetKeplerianElementsAtGivenJulianDate_( semiMajorAxisIndex )
139  = unit_conversions::convertAstronomicalUnitsToMeters(
140  planetKeplerianElementsAtGivenJulianDate_( semiMajorAxisIndex ) );
141 
142  // Convert inclination from degrees to radians.
143  planetKeplerianElementsAtGivenJulianDate_( inclinationIndex )
144  = unit_conversions::convertDegreesToRadians(
145  planetKeplerianElementsAtGivenJulianDate_( inclinationIndex ) );
146 
147  // Convert longitude of ascending node from degrees to radians.
148  planetKeplerianElementsAtGivenJulianDate_( longitudeOfAscendingNodeIndex )
149  = unit_conversions::convertDegreesToRadians(
150  planetKeplerianElementsAtGivenJulianDate_( longitudeOfAscendingNodeIndex ) );
151 
152  // Convert argument of periapsis from degrees to radians.
153  planetKeplerianElementsAtGivenJulianDate_( argumentOfPeriapsisIndex )
154  = unit_conversions::convertDegreesToRadians(
155  planetKeplerianElementsAtGivenJulianDate_( argumentOfPeriapsisIndex ) );
156 
157  return planetKeplerianElementsAtGivenJulianDate_;
158 }
