42 double ImprovedInversePolynomialWall::evaluate( const double anAzimuthalAngle )
43 {
44  // Inverse polynomial function (improved version from Wall et al. [2010]).
45  const double radialDistance = 1.0 /
46  ( boundaryParameters_( ).first( 0 )
47  + boundaryParameters_( ).first( 1 )
48  * std::cos( anAzimuthalAngle + boundaryParameters_( ).first( 2 ) )
49  + timeDependentParameter_( )
50  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
51  + boundaryParameters_( ).second( 0 ) * anAzimuthalAngle * anAzimuthalAngle
52  * anAzimuthalAngle * anAzimuthalAngle
53  + boundaryParameters_( ).second( 1 ) * anAzimuthalAngle * anAzimuthalAngle
54  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
55  + boundaryParameters_( ).second( 2 ) * anAzimuthalAngle * anAzimuthalAngle
56  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle );
57 
58  return radialDistance;
59 }

62 double ImprovedInversePolynomialWall::computeDerivative(
63  const unsigned int order,
64  const double anAzimuthalAngle )
65 {
66  // Function itself.
67  if ( order == 0 )
68  {
69  return evaluate( anAzimuthalAngle );
70  }
71  // First derivative.
72  else if ( order == 1 )
73  {
74  // Radial distance.
75  double radialDistance = evaluate( anAzimuthalAngle );
76 
77  // First derivative of the inverse polynomial function with respect to the (in-plane)
78  // azimuthal angle.
79  const double firstDerivative = -1.0 * radialDistance * radialDistance
80  * ( -1.0 * boundaryParameters_( ).first( 1 )
81  * std::sin( anAzimuthalAngle + boundaryParameters_( ).first( 2 ) )
82  + 3.0 * timeDependentParameter_( ) * anAzimuthalAngle * anAzimuthalAngle
83  + 4.0 * boundaryParameters_( ).second( 0 )
84  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
85  + 5.0 * boundaryParameters_( ).second( 1 )
86  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
87  + 6.0 * boundaryParameters_( ).second( 2 )
88  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
89  * anAzimuthalAngle );
90 
91  return firstDerivative;
92  }
93  // Second derivative.
94  else if ( order == 2 )
95  {
96  // Radial distance.
97  double radialDistance = evaluate( anAzimuthalAngle );
98 
99  // First derivative.
100  double firstDerivative = computeDerivative( 1 , anAzimuthalAngle );
101 
102  // Second derivative of the inverse polynomial function with respect to the (in-plane)
103  // azimuthal angle.
104  const double secondDerivative = ( 2.0 / radialDistance )
105  * firstDerivative * firstDerivative
106  - radialDistance * radialDistance
107  * ( -1.0 * boundaryParameters_( ).first( 1 )
108  * std::cos( anAzimuthalAngle + boundaryParameters_( ).first( 2 ) )
109  + 6.0 * timeDependentParameter_( ) * anAzimuthalAngle
110  + 12.0 * boundaryParameters_( ).second( 0 )
111  * anAzimuthalAngle * anAzimuthalAngle
112  + 20.0 * boundaryParameters_( ).second( 1 )
113  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
114  + 30.0 * boundaryParameters_( ).second( 2 )
115  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle
116  * anAzimuthalAngle );
117 
118  return secondDerivative;
119  }
120  // Third derivative.
121  else if ( order == 3 )
122  {
123  // Radial distance.
124  double radialDistance = evaluate( anAzimuthalAngle );
125 
126  // First derivative.
127  double firstDerivative = computeDerivative( 1 , anAzimuthalAngle );
128 
129  // Second derivative.
130  double secondDerivative = computeDerivative( 2 , anAzimuthalAngle );
131 
132  // Third derivative of the inverse polynomial function with respect to the (in-plane)
133  // azimuthal angle.
134  const double thirdDerivative = 6.0 *
135  ( ( firstDerivative * firstDerivative * firstDerivative ) /
136  ( radialDistance * radialDistance ) )
137  - 6.0 * ( firstDerivative / radialDistance ) * (
138  ( ( 2.0 * firstDerivative * firstDerivative ) / radialDistance )
139  - secondDerivative )
140  - radialDistance * radialDistance * (
141  boundaryParameters_( ).first( 1 )
142  * std::sin( anAzimuthalAngle + boundaryParameters_( ).first( 2 ) )
143  + 6.0 * timeDependentParameter_( )
144  + 24.0 * boundaryParameters_( ).second( 0 ) * anAzimuthalAngle
145  + 60.0 * boundaryParameters_( ).second( 1 )
146  * anAzimuthalAngle * anAzimuthalAngle
147  + 120.0 * boundaryParameters_( ).second( 2 )
148  * anAzimuthalAngle * anAzimuthalAngle * anAzimuthalAngle );
149 
150  return thirdDerivative;
151  }
152  // Throw runtime error, when order is higher that 3.
153  else
154  {
155  throw std::runtime_error( "Derivatives of order higher than 3 are not supported." );
156  }
157 }
