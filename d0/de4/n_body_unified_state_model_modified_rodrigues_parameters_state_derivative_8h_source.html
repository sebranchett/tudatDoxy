143  void calculateSystemStateDerivative(

145  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
146  {

149  currentAccelerationInIntertialFrame.resizeLike( currentCartesianLocalSolution_ );
150  this->sumStateDerivativeContributions( stateOfSystemToBeIntegrated, currentAccelerationInIntertialFrame, false );
151 
152  // Compute RSW accelerations for each body, and evaluate equations of motion for USM6 elements.
153  stateDerivative.setZero( );
154  Eigen::Vector3d currentAccelerationInRswFrame;
155  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )

158  currentCartesianLocalSolution_.segment( i * 6, 6 ).template cast< double >( ) ) *

249  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
250  {
251  // Loop over each body
252  Eigen::Matrix< StateScalarType, 3, 1 > modifiedRodriguesParametersVector;
253  StateScalarType modifiedRodriguesParametersMagnitude;
254  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
255  {
256  // Convert to/from shadow modifed Rodrigues parameters (SMRP) (transformation is the same either way)
257  modifiedRodriguesParametersVector = unprocessedState.block( i * 7 + 3, 0, 3, 1 );
258  modifiedRodriguesParametersMagnitude = modifiedRodriguesParametersVector.norm( );
259  if ( modifiedRodriguesParametersMagnitude >= 1.0 )
260  {
261  // Invert flag
262  unprocessedState.segment( i * 7 + 6, 1 ) = ( unprocessedState.block( i * 7 + 6, 0, 1, 1 ) -
263  Eigen::Matrix< StateScalarType, 1, 1 >::Ones( ) ).cwiseAbs( );
264 
265  // Convert to MRP/SMRP
266  modifiedRodriguesParametersVector /= - modifiedRodriguesParametersMagnitude *
267  modifiedRodriguesParametersMagnitude;
268 
269  // Replace MRP with SMPR, or vice-versa
270  unprocessedState.segment( i * 7 + 3, 3 ) = modifiedRodriguesParametersVector;
271  }
272  }
273  }
