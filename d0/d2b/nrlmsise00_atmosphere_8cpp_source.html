21 void NRLMSISE00Atmosphere::computeProperties(
22  const double altitude, const double longitude,
23  const double latitude, const double time )
24 {
25  // Compute the hash key
26  size_t hashKey = hashFunc( altitude, longitude, latitude, time );
27 
28  // If hash key is same do nothing
29  if (hashKey == hashKey_)
30  {
31  return;
32  }
33  hashKey_ = hashKey;
34 
35  // Retrieve input data.
36  inputData_ = nrlmsise00InputFunction_(
37  altitude, longitude, latitude, time );
38  std::copy( inputData_.apVector.begin( ), inputData_.apVector.end( ), aph_.a );
39  std::copy( inputData_.switches.begin( ), inputData_.switches.end( ), flags_.switches);
40 
41  input_.g_lat = latitude * 180.0 / mathematical_constants::PI; // rad to deg
42  input_.g_long = longitude * 180.0 / mathematical_constants::PI; // rad to deg
43  input_.alt = altitude * 1.0E-3; // m to km
44  input_.year = inputData_.year;
45  input_.doy = inputData_.dayOfTheYear;
46  input_.sec = inputData_.secondOfTheDay;
47  input_.lst = inputData_.localSolarTime;
48  input_.f107 = inputData_.f107;
49  input_.f107A = inputData_.f107a;
50  input_.ap = inputData_.apDaily;
51  input_.ap_a = &aph_;
52 
53  // Call NRLMSISE00
54  gtd7(&input_, &flags_, &output_);
55 
56  // Retrieve density and temperature
57  density_ = output_.d[ 5 ] * 1000.0; // GM/CM3 to kg/M3
58  temperature_ = output_.t[1];
59 
60  // Get number densities
61  numberDensities_.resize(8);
62  numberDensities_[0] = output_.d[0] * 1.0E6 ; // HE NUMBER DENSITY (M-3)
63  numberDensities_[1] = output_.d[1] * 1.0E6 ; // O NUMBER DENSITY (M-3)
64  numberDensities_[2] = output_.d[2] * 1.0E6 ; // N2 NUMBER DENSITY (M-3)
65  numberDensities_[3] = output_.d[3] * 1.0E6 ; // O2 NUMBER DENSITY (M-3)
66  numberDensities_[4] = output_.d[4] * 1.0E6 ; // AR NUMBER DENSITY (M-3)
67  numberDensities_[5] = output_.d[6] * 1.0E6 ; // H NUMBER DENSITY (M-3)
68  numberDensities_[6] = output_.d[7] * 1.0E6 ; // N NUMBER DENSITY (M-3)
69  numberDensities_[7] = output_.d[8] * 1.0E6 ; // Anomalous oxygen NUMBER DENSITY (M-3)
70 
71  // Get average number density
72  double sumOfNumberDensity = 0.0 ;
73  for( unsigned int i = 0 ; i < numberDensities_.size( ) ; i++)
74  {
75  sumOfNumberDensity += numberDensities_[ i ];
76  }
77  averageNumberDensity_ = sumOfNumberDensity / double( numberDensities_.size( ) );
78 
79  // Mean molar mass (Thermodynamics an Engineering Approach, Michael A. Boles)
80  meanMolarMass_ = numberDensities_[0] * gasComponentProperties_.molarMassHelium;
81  meanMolarMass_ += numberDensities_[1] * gasComponentProperties_.molarMassAtomicOxygen;
82  meanMolarMass_ += numberDensities_[2] * gasComponentProperties_.molarMassNitrogen;
83  meanMolarMass_ += numberDensities_[3] * gasComponentProperties_.molarMassOxygen;
84  meanMolarMass_ += numberDensities_[4] * gasComponentProperties_.molarMassArgon;
85  meanMolarMass_ += numberDensities_[5] * gasComponentProperties_.molarMassAtomicHydrogen;
86  meanMolarMass_ += numberDensities_[6] * gasComponentProperties_.molarMassAtomicNitrogen;
87  meanMolarMass_ += numberDensities_[7] * gasComponentProperties_.molarMassOxygen;
88  meanMolarMass_ = meanMolarMass_ / sumOfNumberDensity ;
89 
90  // Speed of sound
91  speedOfSound_ = aerodynamics::computeSpeedOfSound(
92  temperature_, specificHeatRatio_, molarGasConstant_ / meanMolarMass_ );
93 
94  // Collision diameter
95  weightedAverageCollisionDiameter_ = numberDensities_[0]* gasComponentProperties_.diameterHelium ;
96  weightedAverageCollisionDiameter_ += numberDensities_[1]* gasComponentProperties_.diameterAtomicOxygen ;
97  weightedAverageCollisionDiameter_ += numberDensities_[2]* gasComponentProperties_.diameterNitrogen ;
98  weightedAverageCollisionDiameter_ += numberDensities_[3]* gasComponentProperties_.diameterOxygen ;
99  weightedAverageCollisionDiameter_ += numberDensities_[4]* gasComponentProperties_.diameterArgon ;
100  weightedAverageCollisionDiameter_ += numberDensities_[5]* gasComponentProperties_.diameterAtomicHydrogen ;
101  weightedAverageCollisionDiameter_ += numberDensities_[6]* gasComponentProperties_.diameterAtomicNitrogen ;
102  weightedAverageCollisionDiameter_ += numberDensities_[7]* gasComponentProperties_.diameterAtomicOxygen ;
103  weightedAverageCollisionDiameter_ = weightedAverageCollisionDiameter_ / sumOfNumberDensity;
104 
105  // Mean free path.
106  meanFreePath_ = aerodynamics::computeMeanFreePath( weightedAverageCollisionDiameter_, averageNumberDensity_ );
107 
108  // Calculate pressure using ideal gas law (Thermodynamics an Engineering Approach, Michael A. Boles)
109  if( useIdealGasLaw_ )
110  {
111  pressure_ = density_ * molarGasConstant_ * temperature_ / meanMolarMass_ ;
112  }
113  else
114  {
115  pressure_ = TUDAT_NAN;
116  }
117 }
