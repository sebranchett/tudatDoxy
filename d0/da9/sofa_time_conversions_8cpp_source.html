55 double getDeltaAtFromTai( const double taiInJulianDays )
56 {
57  // Declare return variables (by reference) from sofa calendar function.
58  int year, month, day;
59  double fractionOfDay;
60 
61  // Get calendar date and check feasibility of input.
62  if( iauJd2cal( basic_astrodynamics::JULIAN_DAY_ON_J2000, taiInJulianDays, &year, &month, &day, &fractionOfDay ) != 0 )
63  {
64  throw std::runtime_error( "Provided julian date too small to convert to calendar date" );
65  }
66 
67  // Estimate number of leap seconds (by assuming TAI = UTC for Sofa input) and check feasibility of calculation
68  double deltaAt;
69  int deltaAtReturn = iauDat( year, month, day, fractionOfDay, &deltaAt);
70  if( deltaAtReturn != 0 )
71  {
72  throw std::runtime_error( "Provided caledar date cannot properly give Delta AT" );
73  }
74 
75  // Reperform calculation with converted utc time and check consistency with previous calculation.
76  double utcInJulianDays = taiInJulianDays - deltaAt / physical_constants::JULIAN_DAY;
77  double deltaAtCheck = getDeltaAtFromUtc( utcInJulianDays );
78  if( deltaAt != deltaAtCheck )
79  {
80  deltaAt--;
81  throw std::runtime_error( "Warning, Delta TAI calculation encountered error, iteration not yet implemented" );
82  }
83 
84 
85  return deltaAt;
86 }

110 double getTDBminusTT( const double ttOrTdbSinceJ2000, const double stationLongitude, const double distanceFromSpinAxis,
111  const double distanceFromEquatorialPlane )
112 {
113  // Calculate current TAI (approximately if input is in TDB)
114  double tai = basic_astrodynamics::convertTTtoTAI< double >( ttOrTdbSinceJ2000 );
115 
116  // Calculate current UT1 (by assuming it equal to UTC)
117  double ut1 = static_cast< double >( convertTAItoUTC< double >( tai ) );
118  double ut1FractionOfDay = std::fmod( ( ut1 / physical_constants::JULIAN_DAY ) -
119  static_cast< double >( std::floor( ut1 / physical_constants::JULIAN_DAY ) ) + 0.5, 1.0 );
120 
121  // Calculate and return difference (introducing addition approximation if input is in TT, by assuming TDB is equal to TT)
122  return getTDBminusTT( ttOrTdbSinceJ2000, ut1FractionOfDay, stationLongitude, distanceFromSpinAxis,
123  distanceFromEquatorialPlane );
124 }
