140 Eigen::VectorXd evaluateListOfVectorFunctions(
141  const std::vector< std::pair< std::function< Eigen::VectorXd( ) >, int > > vectorFunctionList,
142  const int totalSize )
143 {
144  Eigen::VectorXd variableList = Eigen::VectorXd::Zero( totalSize );
145  int currentIndex = 0;
146 
147  for( std::pair< std::function< Eigen::VectorXd( ) >, int > vectorFunction: vectorFunctionList )
148  {
149  variableList.segment( currentIndex, vectorFunction.second ) = vectorFunction.first( );
150  currentIndex += vectorFunction.second;
151  }
152 
153  // Check consistency with input
154  if( currentIndex != totalSize )
155  {
156  std::string errorMessage = "Error when evaluating lists of functions, sizes are inconsistent: " +
157  std::to_string( currentIndex ) + " and " +
158  std::to_string( totalSize );
159  throw std::runtime_error( errorMessage );
160  }
161 
162  return variableList;
163 }

165 Eigen::VectorXd getNormsOfAccelerationDifferencesFromLists(
166  const std::function< Eigen::VectorXd( ) > firstAccelerationFunction,
167  const std::function< Eigen::VectorXd( ) > secondAccelerationFunction )
168 {
169  Eigen::VectorXd firstAcceleration = firstAccelerationFunction( );
170  Eigen::VectorXd secondAcceleration = secondAccelerationFunction( );
171 
172  if( firstAcceleration.rows( ) != secondAcceleration.rows( ) )
173  {
174  throw std::runtime_error( "Error when computing acceleration difference norms, inputs are inconsistent." );
175  }
176 
177  if( firstAcceleration.rows( ) % 3 != 0 )
178  {
179  throw std::runtime_error( "Error when computing acceleration difference norms, input size is inconsistent." );
180  }
181 
182  Eigen::VectorXd accelerationDifference = Eigen::VectorXd::Zero(
183  firstAcceleration.rows( ) / 3 );
184  for( int i = 0; i < accelerationDifference.rows( ); i++ )
185  {
186  accelerationDifference( i ) =
187  ( firstAcceleration.segment( i * 3, 3 ) - secondAcceleration.segment( i * 3, 3 ) ).norm( );
188  }
189 
190  return accelerationDifference;
191 }
