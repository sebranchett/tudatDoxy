24 std::pair< Eigen::MatrixXd, Eigen::MatrixXd > readAmplitudesAndFundamentalArgumentMultipliers(
25  const std::string amplitudesFile,
26  const std::string fundamentalArgumentMultipliersFile,
27  const double minimumAmplitude )
28 {
29 
30  // Read amplitudes and fundamental argument multipliers into matrices.
31  Eigen::MatrixXd amplitudesRaw = input_output::readMatrixFromFile( amplitudesFile );
32  Eigen::MatrixXd fundamentalArgumentMultipliersRaw = input_output::readMatrixFromFile( fundamentalArgumentMultipliersFile );
33 
34  // Check whether amplitudes and fundamental argument multipliers matrices have same number of rows
35  if( amplitudesRaw.rows( ) != fundamentalArgumentMultipliersRaw.rows( ) )
36  {
37  throw std::runtime_error( "Amplitude and argument multipler files contain unequal set of entries" );
38  }
39 
40  // Check whether number of columns in fundamental argument multiplier matrix is equal to 6.
41  if( fundamentalArgumentMultipliersRaw.cols( ) != 6 )
42  {
43  throw std::runtime_error( "Number of columns in fundamental argument multipler matrix not equal to 6" );
44  }
45 
46 
47  // If acceptance amplitude is larger than 0.0 check RSS of amplitudes for each entry.
48  Eigen::MatrixXd fundamentalArgumentMultipliers, amplitudes;
49  if( minimumAmplitude > 0.0 )
50  {
51  // Filter raw data and remove entries with RSS amplitude that is too low
52  std::pair< Eigen::MatrixXd, Eigen::MatrixXd > filteredData =
53  filterRawDataForAmplitudes( fundamentalArgumentMultipliersRaw, amplitudesRaw, minimumAmplitude );
54  amplitudes = filteredData.first;
55  fundamentalArgumentMultipliers = filteredData.second;
56  }
57  // If all amplitudes are to be expected, raw data is same as return data.
58  else
59  {
60  amplitudes = amplitudesRaw;
61  fundamentalArgumentMultipliers = fundamentalArgumentMultipliersRaw;
62  }
63 
64  // return amplitudes and fundamental argument multipliers.
65  return std::pair< Eigen::MatrixXd, Eigen::MatrixXd >( amplitudes, fundamentalArgumentMultipliers );
66 }

69 std::pair< Eigen::MatrixXd, Eigen::MatrixXd > filterRawDataForAmplitudes(
70  const Eigen::MatrixXd rawAmplitudes,
71  const Eigen::MatrixXd rawFundamentalArgumentMultipliers,
72  const double minimumAmplitude )
73 {
74  // Get number of amplitudes per entry.
75  int numberOfAmplitudes = rawAmplitudes.cols( );
76 
77  // Initialize return matrices
78  int numberOfAcceptedAmplitudes = 0;
79  Eigen::MatrixXd filteredAmplitudes;
80  filteredAmplitudes.resize( numberOfAcceptedAmplitudes, numberOfAmplitudes );
81  Eigen::MatrixXd filteredFundamentalArgumentMultipliers;
82  filteredFundamentalArgumentMultipliers.resize( numberOfAcceptedAmplitudes, 6 );
83 
84  // Iterate over all entries, calculate RSS amplitude and accept or reject entry.
85  double rssAmplitude = 0.0;
86  for( int i = 0; i < rawAmplitudes.rows( ); i++ )
87  {
88  // Calculate RSS of amplitude.
89  rssAmplitude = 0.0;
90  for( int j =0; j < numberOfAmplitudes; j++ )
91  {
92  rssAmplitude += rawAmplitudes( i, j ) * rawAmplitudes( i, j );
93  }
94  rssAmplitude = std::sqrt( rssAmplitude );
95 
96  // If RSS amplitude is sufficiently large, accept value.
97  if( rssAmplitude > minimumAmplitude )
98  {
99  if( numberOfAcceptedAmplitudes == 0 )
100  {
101  numberOfAcceptedAmplitudes++;
102  filteredAmplitudes.resize( 1, numberOfAmplitudes );
103  filteredAmplitudes = rawAmplitudes.block( i, 0, 1, numberOfAmplitudes );
104 
105  filteredFundamentalArgumentMultipliers.resize( 1, 6 );
106  filteredFundamentalArgumentMultipliers = rawFundamentalArgumentMultipliers.block( i, 0, 1, 6 );
107  }
108  else
109  {
110  numberOfAcceptedAmplitudes++;
111  filteredAmplitudes.conservativeResize( numberOfAcceptedAmplitudes, numberOfAmplitudes );
112  filteredAmplitudes.block( numberOfAcceptedAmplitudes - 1, 0, 1, numberOfAmplitudes ) =
113  rawAmplitudes.block( i, 0, 1, numberOfAmplitudes );
114 
115  filteredFundamentalArgumentMultipliers.conservativeResize( numberOfAcceptedAmplitudes, 6 );
116  filteredFundamentalArgumentMultipliers.block( numberOfAcceptedAmplitudes - 1, 0, 1, 6 ) =
117  rawFundamentalArgumentMultipliers.block( i, 0, 1, 6 );
118  }
119  }
120  }
121 
122  return std::pair< Eigen::MatrixXd, Eigen::MatrixXd >( filteredAmplitudes, filteredFundamentalArgumentMultipliers );
123 }
