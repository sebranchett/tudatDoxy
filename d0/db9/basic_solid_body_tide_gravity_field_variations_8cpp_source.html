22 std::vector< std::vector< std::complex< double > > > getFullLoveNumbersVector(
23  const std::complex< double > constantLoveNumber, const int maximumDegree, const int maximumOrder )
24 {
25  // Define list of Love numbers
26  std::vector< std::vector< std::complex< double > > > fullLoveNumbersVector;
27  fullLoveNumbersVector.resize( maximumDegree - 1 );
28 
29  // Iterate over all degrees and orders.
30  for( unsigned int i = 0; i <= static_cast< unsigned int >( maximumDegree ) - 2;i++ )
31  {
32  for( unsigned int j = 0; j <= ( i + 2 ); j++ )
33  {
34  // Set current Love numbers.
35  if( static_cast< int >( j ) <= maximumOrder )
36  {
37  fullLoveNumbersVector[ i ].push_back( constantLoveNumber );
38  }
39  else
40  {
41  fullLoveNumbersVector[ i ].push_back( std::complex< double >( 0.0, 0.0 ) );
42  }
43  }
44  }
45 
46  return fullLoveNumbersVector;
47 }

58 std::complex< double > calculateSolidBodyTideSingleCoefficientSetCorrectionFromAmplitude(
59  const std::complex< double > loveNumber, const double massRatio,
60  const double radiusRatioPowerN,
61  const double amplitude, const std::complex< double > tideArgument,
62  const int degree, const int order )
63 {
64  // Calculate and return corrections.
65  return loveNumber / ( 2.0 * static_cast< double >( degree ) + 1.0 ) *
66  massRatio * radiusRatioPowerN *
67  amplitude * basic_mathematics::calculateLegendreGeodesyNormalizationFactor(
68  degree, order ) * std::exp( -tideArgument );
69 }

73 std::complex< double > calculateSolidBodyTideSingleCoefficientSetCorrectionFromAmplitude(
74  const std::complex< double > loveNumber, const double massRatio,
75  const double referenceRadius, const Eigen::Vector3d& relativeBodyFixedPosition, const int degree, const int order )
76 {
77  // Calculate spherical position of perturbing body.
78  Eigen::Vector3d sphericalPosition = coordinate_conversions::convertCartesianToSpherical(
79  relativeBodyFixedPosition );
80 
81  // Calculate amplitude and argument of tide
82  double tideAmplitude = basic_mathematics::computeLegendrePolynomialExplicit(
83  degree, order, std::sin( mathematical_constants::PI / 2.0 - sphericalPosition.y( ) ) ) *
84  basic_mathematics::calculateLegendreGeodesyNormalizationFactor(
85  degree, order );
86  std::complex< double > tideArgument = static_cast< double >( order ) * std::complex< double >( 0.0, sphericalPosition( 2 ) );
87 
88  // Calculate tidal corrections.
89  return loveNumber / ( 2.0 * static_cast< double >( degree ) + 1.0 ) *
90  massRatio * std::pow( referenceRadius / sphericalPosition( 0 ), degree + 1 ) *
91  tideAmplitude * std::exp( -tideArgument );
92 }

96 std::pair< Eigen::MatrixXd, Eigen::MatrixXd > calculateSolidBodyTideSingleCoefficientSetCorrectionFromAmplitude(
97  const std::vector< std::vector< std::complex< double > > > loveNumbers, const double massRatio,
98  const double referenceRadius, const Eigen::Vector3d& relativeBodyFixedPosition,
99  const int maximumDegree, const int maximumOrder )
100 {
101  // Initialize results.
102  Eigen::MatrixXd cosineCorrections = Eigen::MatrixXd::Zero( maximumDegree + 1, maximumOrder + 1 );
103  Eigen::MatrixXd sineCorrections = Eigen::MatrixXd::Zero( maximumDegree + 1, maximumOrder + 1 );
104 
105  std::complex< double > currentCorrections;
106 
107  // Iterate over all requested degrees and orders
108  for( int n = 2; n <= maximumDegree; n++ )
109  {
110  for( int m = 0; ( m <= maximumOrder && m <= n ); m++ )
111  {
112  // Calculate and set corrections at current degree and order.
113  currentCorrections = calculateSolidBodyTideSingleCoefficientSetCorrectionFromAmplitude(
114  loveNumbers.at( n - 2 ).at( m ), massRatio, referenceRadius, relativeBodyFixedPosition, n, m );
115  cosineCorrections( n, m ) = currentCorrections.real( );
116  if( m > 0 )
117  {
118  sineCorrections( n, m ) = -currentCorrections.imag( );
119  }
120  }
121  }
122 
123  return std::make_pair( cosineCorrections, sineCorrections );
124 }

128 void BasicSolidBodyTideGravityFieldVariations::setBodyGeometryParameters(
129  const int bodyIndex, const double evaluationTime )
130 {
131  // Calculate current state and orientation of deformed body.
132  if( bodyIndex == 0 )
133  {
134  deformedBodyPosition = std::move( deformedBodyStateFunction_( evaluationTime ) ).segment( 0, 3 );
135  toDeformedBodyFrameRotation = deformedBodyOrientationFunction_( evaluationTime );
136  }
137 
138  // Calculate current state of body causing deformation.
139  Eigen::Vector3d relativeDeformingBodyPosition = toDeformedBodyFrameRotation * (
140  std::move( deformingBodyStateFunctions_[ bodyIndex ]( evaluationTime ) ).segment( 0, 3 ) -
141  deformedBodyPosition );
142  Eigen::Vector3d relativeDeformingBodySphericalPosition =coordinate_conversions::
143  convertCartesianToSpherical( relativeDeformingBodyPosition );
144 
145  // Set geometric parameters of body causing deformation.
146  radiusRatio = deformedBodyReferenceRadius_ / relativeDeformingBodySphericalPosition.x( );
147  iLongitude = mathematical_constants::COMPLEX_I * relativeDeformingBodySphericalPosition.z( );
148  sineOfLatitude = std::sin( mathematical_constants::PI / 2.0 -
149  relativeDeformingBodySphericalPosition.y( ) );

153 std::pair< Eigen::MatrixXd, Eigen::MatrixXd > BasicSolidBodyTideGravityFieldVariations::
154 calculateBasicSphericalHarmonicsCorrections(
155  const double time )
156 {
157  // Initialize corrections to zero.
158  Eigen::MatrixXd cTermCorrections = Eigen::MatrixXd::Zero( numberOfDegrees_, numberOfOrders_ );
159  Eigen::MatrixXd sTermCorrections = Eigen::MatrixXd::Zero( numberOfDegrees_, numberOfOrders_ );
160 
161  // Iterate over all bodies causing deformation and calculate and add associated corrections
162  for( unsigned int i = 0; i < deformingBodyStateFunctions_.size( ); i++ )
163  {
164  setBodyGeometryParameters( i, time );
165 
166  // Calculate properties of currently considered body
167  massRatio = deformingBodyMasses_[ i ]( ) / deformedBodyMass_( );
168 
169  // Calculate all correction functions.
170  for( unsigned int j = 0; j < correctionFunctions.size( ); j++ )
171  {
172  correctionFunctions[ j ]( cTermCorrections, sTermCorrections );
173  }
174  }
175 
176  return std::make_pair( cTermCorrections, sTermCorrections );
177 }

180 void BasicSolidBodyTideGravityFieldVariations::addBasicSolidBodyTideCorrections(
181  Eigen::MatrixXd& cTermCorrections,
182  Eigen::MatrixXd& sTermCorrections )
183 {
184 // // Initialize power of radiusRatio^(N+1) (calculation starts at N=2)
185  radiusRatioPower = radiusRatio * radiusRatio * radiusRatio;
186 
187  currentCosineCorrections_.setZero( );
188  currentSineCorrections_.setZero( );
189 
190  // // Iterate over all love
191  std::complex< double > stokesCoefficientCorrection( 0.0, 0.0 );
192 
193  for( unsigned int n = 2; n < loveNumbers_.size( ) + 2; n++ )
194  {
195  for( unsigned int m = 0; ( m <= n && m < loveNumbers_.at( n - 2 ).size( ) ); m++ )
196  {
197  updateTidalAmplitudeAndArgument( n, m );
198 
199  // Calculate and add coefficients.
200  stokesCoefficientCorrection =
201  calculateSolidBodyTideSingleCoefficientSetCorrectionFromAmplitude(
202  loveNumbers_[ n - 2 ][ m ], massRatio, radiusRatioPower, tideAmplitude,
203  tideArgument, n, m );
204 
205 
206  currentCosineCorrections_( n - 2, m ) += stokesCoefficientCorrection.real( );
207  if( m != 0 )
208  {
209  currentSineCorrections_( n - 2, m ) -= stokesCoefficientCorrection.imag( );
210  }
211  }
212 
213  // Increment radius ratio power.
214  radiusRatioPower *= radiusRatio;
215  }
216 
217  cTermCorrections.block( 0, 0, maximumDegree_ - minimumDegree_ + 1, maximumOrder_ - minimumOrder_ + 1 ) +=
218  currentCosineCorrections_;
219  sTermCorrections.block( 0, 0, maximumDegree_ - minimumDegree_ + 1, maximumOrder_ - minimumOrder_ + 1 ) +=
220  currentSineCorrections_;
221 
222 }
223 
224 }
