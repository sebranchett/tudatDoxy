27 template< typename VariableType >
28 void checkVariableUniqueness( std::vector< VariableType > variables )
29 {
30  // Sort variables
31  std::sort( variables.begin( ), variables.end( ) );
32 
33  // Check uniqueness
34  unsigned int numberOfUniqueElements = std::distance( variables.begin( ),
35  std::unique( variables.begin( ), variables.end( ) ) );
36 
37  // Give error in case of non-unique variables
38  if ( numberOfUniqueElements != variables.size( ) )
39  {
40  throw std::runtime_error( "Error, in tabulated atmosphere. Duplicate entry in (in)dependent variables." );
41  }
42 }

45 void TabulatedAtmosphere::createAtmosphereInterpolators( )
46 {
47  // Check uniqueness
48  checkVariableUniqueness< AtmosphereDependentVariables >( dependentVariables_ );
49  checkVariableUniqueness< AtmosphereIndependentVariables >( independentVariables_ );
50 
51  // Retrieve number of dependent variables from user.
52  unsigned int numberOfDependentVariables = dependentVariables_.size( );
53 
54  // Check that number of dependent variables does not exceed limit
55  if ( numberOfDependentVariables > dependentVariablesDependency_.size( ) )
56  {
57  throw std::runtime_error( "Error, in tabulated atmosphere. Number of dependent variables exceeds current limit." );
58  }
59 
60  // Check input consistency
61  if ( independentVariables_.size( ) != 1 )
62  {
63  if ( atmosphereTableFile_.size( ) != numberOfDependentVariables )
64  {
65  throw std::runtime_error( "Error when creating tabulated atmosphere from file, "
66  "number of specified dependent variables differs from file." );
67  }
68 
69  // Retrieve number of independent variables from file.
70  numberOfIndependentVariables_ = input_output::getNumberOfIndependentVariablesInCoefficientFile(
71  atmosphereTableFile_.at( 0 ) );
72 
73  // Check that number of independent variables does not exceed limit
74  if ( ( numberOfIndependentVariables_ < 1 ) || ( numberOfIndependentVariables_ > 4 ) )
75  {
76  throw std::runtime_error( "Error when reading tabulated atmosphere from file, found " +
77  std::to_string( numberOfIndependentVariables_ ) +
78  " independent variables, up to 4 currently supported." );
79  }
80 
81  // Check input consistency
82  if ( independentVariables_.size( ) != numberOfIndependentVariables_ )
83  {
84  throw std::runtime_error( "Error when creating tabulated atmosphere from file, "
85  "number of specified independent variables differs from file." );
86  }
87  }
88  else
89  {
90  numberOfIndependentVariables_ = 1; // if only one independent variable is specified, only one file will
91  // be provided, and it cannot be opened with the same function
92  }
93 
94  // Get order of dependent variables
95  for ( unsigned int i = 0; i < numberOfDependentVariables; i++ )
96  {
97  dependentVariablesDependency_.at( dependentVariables_.at( i ) ) = true;
98  dependentVariableIndices_.at( dependentVariables_.at( i ) ) = i;
99  }
100 
101  // Check that density, pressure and temperature are present
102  if ( !( dependentVariablesDependency_.at( 0 ) || dependentVariablesDependency_.at( 1 ) ||
103  dependentVariablesDependency_.at( 2 ) ) )
104  {
105  throw std::runtime_error( "Error, tabulated atmosphere must be initialized with at least "
106  "density, pressure and temperature." );
107  }
108 
109  // Assign values to default boundary handling methods
110  if ( boundaryHandling_.empty( ) )
111  {
112  boundaryHandling_ = std::vector< BoundaryInterpolationType >( numberOfIndependentVariables_,
113  use_boundary_value );
114  }
115  else
116  {
117  if ( boundaryHandling_.size( ) != numberOfIndependentVariables_ )
118  {
119  throw std::runtime_error( "Error, in tabulated atmosphere. Number of boundary handling methods provided does "
120  "not match number of independent variables." );
121  }
122  }
123 
124  // Assign values to default extrapolation values
125  if ( defaultExtrapolationValue_.empty( ) )
126  {
127  defaultExtrapolationValue_ = std::vector< std::vector< std::pair< double, double > > >(
128  numberOfDependentVariables, std::vector< std::pair< double, double > >(
129  numberOfIndependentVariables_, std::make_pair(
130  IdentityElement::getAdditionIdentity< double >( ),
131  IdentityElement::getAdditionIdentity< double >( ) ) ) );
132  }
133  else
134  {
135  if ( defaultExtrapolationValue_.size( ) != numberOfDependentVariables )
136  {
137  throw std::runtime_error( "Error, in tabulated atmosphere. Number of default extrapolation values provided "
138  "does not match number of dependent variables." );
139  }
140  }
141 
142  // Create interpolators for variables requested by users, depending on the number of variables
143  switch ( numberOfIndependentVariables_ )
144  {
145  case 1:
146  {
147  // Call approriate file reading function for 1 independent variables
148  Eigen::MatrixXd tabulatedAtmosphereData = input_output::readMatrixFromFile( atmosphereTableFile_.at( 0 ), " \t", "%" );
149 
150  // Extract information on file size
151  unsigned int numberOfColumnsInFile = tabulatedAtmosphereData.cols( );
152  unsigned int numberOfRowsInFile = tabulatedAtmosphereData.rows( );
153 
154 
155  // Check whether data is present in the file.
156  if ( numberOfRowsInFile < 1 || numberOfColumnsInFile < 1 )
157  {
158  std::string errorMessage = "Error, in tabulated atmosphere. The atmosphere table file " +
159  atmosphereTableFile_.at( 0 ) + " is empty";
160  throw std::runtime_error( errorMessage );
161  }
162 
163  // Check whether number of dependent variables matches number of columns
164  if ( numberOfDependentVariables != ( numberOfColumnsInFile - 1 ) )
165  {
166  throw std::runtime_error( "Error, in tabulated atmosphere. "
167  "Number of specified dependent variables does not match file." );
168  }
169 
170  // Assign sizes to vectors
171  independentVariablesData_.resize( numberOfIndependentVariables_ );
172  std::vector< std::vector< double > > dependentVariablesData;
173  dependentVariablesData.resize( numberOfDependentVariables );
174 
175  // Extract variables from file
176  for ( unsigned int i = 0; i < numberOfRowsInFile; i++ )
177  {
178  independentVariablesData_.at( 0 ).push_back( tabulatedAtmosphereData( i, 0 ) );
179  for ( unsigned int j = 0; j < dependentVariablesDependency_.size( ); j++ )
180  {
181  if ( dependentVariablesDependency_.at( j ) )
182  {
183  dependentVariablesData.at( dependentVariableIndices_.at( j ) ).push_back(
184  tabulatedAtmosphereData( i, dependentVariableIndices_.at( j ) + 1 ) );
185  }
186  }
187  }
188 
189  // Create interpolators for density, pressure and temperature
190  interpolatorForDensity_ = std::make_shared< CubicSplineInterpolatorDouble >(
191  independentVariablesData_.at( 0 ), dependentVariablesData.at( dependentVariableIndices_.at( 0 ) ),
192  huntingAlgorithm, boundaryHandling_.at( 0 ), defaultExtrapolationValue_.at( dependentVariableIndices_.at( 0 ) ).at( 0 ) );
193  interpolatorForPressure_ = std::make_shared< CubicSplineInterpolatorDouble >(
194  independentVariablesData_.at( 0 ), dependentVariablesData.at( dependentVariableIndices_.at( 1 ) ),
195  huntingAlgorithm, boundaryHandling_.at( 0 ), defaultExtrapolationValue_.at( dependentVariableIndices_.at( 1 ) ).at( 0 ) );
196  interpolatorForTemperature_ = std::make_shared< CubicSplineInterpolatorDouble >(
197  independentVariablesData_.at( 0 ), dependentVariablesData.at( dependentVariableIndices_.at( 2 ) ),
198  huntingAlgorithm, boundaryHandling_.at( 0 ), defaultExtrapolationValue_.at( dependentVariableIndices_.at( 2 ) ).at( 0 ) );
199 
200  // Create remaining interpolators, if requested by user
201  if ( dependentVariablesDependency_.at( 3 ) )
202  {
203  interpolatorForGasConstant_ = std::make_shared< CubicSplineInterpolatorDouble >(
204  independentVariablesData_.at( 0 ), dependentVariablesData.at( dependentVariableIndices_.at( 3 ) ),
205  huntingAlgorithm, boundaryHandling_.at( 0 ), defaultExtrapolationValue_.at( dependentVariableIndices_.at( 3 ) ).at( 0 ) );
206  }
207  if ( dependentVariablesDependency_.at( 4 ) )
208  {
209  interpolatorForSpecificHeatRatio_ = std::make_shared< CubicSplineInterpolatorDouble >(
210  independentVariablesData_.at( 0 ), dependentVariablesData.at( dependentVariableIndices_.at( 4 ) ),
211  huntingAlgorithm, boundaryHandling_.at( 0 ), defaultExtrapolationValue_.at( dependentVariableIndices_.at( 4 ) ).at( 0 ) );
212  }
213  if ( dependentVariablesDependency_.at( 5 ) )
214  {
215  interpolatorForMolarMass_ = std::make_shared< CubicSplineInterpolatorDouble >(
216  independentVariablesData_.at( 0 ), dependentVariablesData.at( dependentVariableIndices_.at( 5 ) ),
217  huntingAlgorithm, boundaryHandling_.at( 0 ), defaultExtrapolationValue_.at( dependentVariableIndices_.at( 5 ) ).at( 0 ) );
218  }
219  break;
220  }
221  case 2:
222  {
223  createMultiDimensionalAtmosphereInterpolators< 2 >( );
224  break;
225  }
226  case 3:
227  {
228  createMultiDimensionalAtmosphereInterpolators< 3 >( );
229  break;
230  }
231  case 4:
232  {
233  createMultiDimensionalAtmosphereInterpolators< 4 >( );
234  break;
235  }
236  default:
237  throw std::runtime_error( "Error, multi-dimensional atmosphere with N>4 not yet implemented" );
238  }

247  std::pair< std::vector< boost::multi_array< double, static_cast< size_t >( NumberOfIndependentVariables ) > >,

249 
250  // Extract data
251  tabulatedAtmosphereData = input_output::readTabulatedAtmosphere< NumberOfIndependentVariables >( atmosphereTableFile_ );
252 
253  // Assign independent variables
254  independentVariablesData_ = tabulatedAtmosphereData.second;
255 
256  // Create interpolators for density, pressure and temperature
257  interpolatorForDensity_ =
258  std::make_shared< MultiLinearInterpolator< double, double, NumberOfIndependentVariables > >(
