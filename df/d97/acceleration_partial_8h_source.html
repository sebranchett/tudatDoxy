66  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
67  getDerivativeFunctionWrtStateOfIntegratedBody(
68  const std::pair< std::string, std::string >& stateReferencePoint,
69  const propagators::IntegratedStateType integratedStateType )
70  {
71  // Initialize to empty function; 0 parameter size.
72  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int >
73  partialFunction = std::make_pair( std::function< void( Eigen::Block< Eigen::MatrixXd > ) >( ), 0 );
74 
75  // Check if state dependency exists
76  switch( integratedStateType )
77  {
78  case propagators::translational_state:
79  {
80  // Check if reference id is consistent.
81  if( stateReferencePoint.second != "" )
82  {
83  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for dynamics" );
84  }
85  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
86  else if( stateReferencePoint.first == acceleratedBody_ )
87  {
88  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAcceleratedBody, this, std::placeholders::_1 ), 3 );
89  }
90  else if( stateReferencePoint.first == acceleratingBody_ )
91  {
92  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAcceleratingBody, this, std::placeholders::_1 ), 3 );
93  }
94  else if( isAccelerationPartialWrtAdditionalBodyNonnullptr( stateReferencePoint.first ) )
95  {
96  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtStateOfAdditionalBody,
97  this, std::placeholders::_1, stateReferencePoint.first ), 3 );
98  }
99  break;
100  }
101  case propagators::rotational_state:
102  {
103  // Check if reference id is consistent.
104  if( stateReferencePoint.second != "" )
105  {
106  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for body mass" );
107  }
108  else if( isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType ) )
109  {
110  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtNonTranslationalStateOfAdditionalBody,
111  this, std::placeholders::_1, stateReferencePoint, integratedStateType, true ), 1 );
112  }
113  break;
114  }
115  case propagators::body_mass_state:
116  {
117  // Check if reference id is consistent.
118  if( stateReferencePoint.second != "" )
119  {
120  throw std::runtime_error( "Error when getting state derivative partial acceleration model, cannot have reference point on body for body mass" );
121  }
122  else if( isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType ) )
123  {
124  partialFunction = std::make_pair( std::bind( &AccelerationPartial::wrtNonTranslationalStateOfAdditionalBody,
125  this, std::placeholders::_1, stateReferencePoint, integratedStateType, true ), 1 );
126  }
127  break;
128  }
129  case propagators::custom_state:
130  {
131  break;
132  }
133  default:
134  std::string errorMessage =
135  "Error when getting state derivative partial acceleration model, dynamics type " +
136  std::to_string( integratedStateType ) + "not recognized" ;
137  throw std::runtime_error( errorMessage );
138  break;
139  }
140 
141 
142  return partialFunction;
143  }

168  bool isStateDerivativeDependentOnIntegratedState(
169  const std::pair< std::string, std::string >& stateReferencePoint,
170  const propagators::IntegratedStateType integratedStateType )
171  {
172  bool isDependent = 0;
173 
174  // Check if state is translational.
175  switch( integratedStateType )
176  {
177  case propagators::translational_state:
178  {
179  // Check if reference id is consistent.
180  if( stateReferencePoint.second != "" )
181  {
182  throw std::runtime_error( "Error when checking state derivative partial dependency of acceleration model, cannot have reference point on body for dynamics" );
183  }
184 
185  // Check if propagated body corresponds to accelerated, accelerating, ro relevant third body.
186  else if( stateReferencePoint.first == acceleratedBody_ || stateReferencePoint.first == acceleratingBody_ ||
187  isAccelerationPartialWrtAdditionalBodyNonnullptr( stateReferencePoint.first ) )
188  {
189  isDependent = 1;
190  }
191  break;
192  }
193  default:
194  isDependent = isStateDerivativeDependentOnIntegratedAdditionalStateTypes( stateReferencePoint, integratedStateType );
195  break;
196  }
197  return isDependent;
198  }
