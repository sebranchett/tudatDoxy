54 template< typename StateScalarType, typename TimeType >
55 std::shared_ptr< CentralBodyData< StateScalarType, TimeType > > createCentralBodyData(
56  const std::vector< std::string >& centralBodies,
57  const std::vector< std::string >& bodiesToIntegrate,
58  const simulation_setup::NamedBodyMap& bodyMap )
59 {
60 
61  // Check whether the bodies that are to e integrated exist in bodyMap
62  for( unsigned int i = 0; i < bodiesToIntegrate.size( ); i++ )
63  {
64  if( bodyMap.count( bodiesToIntegrate.at( i ) ) == 0 )
65  {
66  throw std::runtime_error(
67  "Warning when creating CentralBodyData, body " + bodiesToIntegrate.at( i )
68  + " not present in provided body map." );
69  }
70  }
71 
72  std::vector< std::string > centralBodiesToUse;
73  // Generate central body data for requested settings; if no
74  // central bodies provided, get inertial SSB as central body
75  if( centralBodies.size( ) == 0 )
76  {
77  for( unsigned int i = 0; i < bodiesToIntegrate.size( ); i++ )
78  {
79  centralBodiesToUse.push_back( "SSB" );
80  }
81  }
82  else
83  {
84  centralBodiesToUse = centralBodies;
85  }
86 
87  std::map< std::string, std::function< Eigen::Matrix< StateScalarType, 6, 1 >
88  ( const TimeType ) > > bodyStateFunctions;
89 
90  // Retrieve frame origin state functions
91  for( unsigned int i = 0; i < centralBodiesToUse.size( ); i++ )
92  {
93  if( centralBodiesToUse.at( i ) != "SSB" )
94  {
95  bodyStateFunctions[ centralBodiesToUse.at( i ) ] =
96  std::bind( &simulation_setup::Body::getStateInBaseFrameFromEphemeris
97  < StateScalarType, TimeType >,
98  bodyMap.at( centralBodiesToUse.at( i ) ), std::placeholders::_1 );
99  }
100  else
101  {
102  bodyStateFunctions[ centralBodiesToUse.at( i ) ] = [ ]( const TimeType ){ return
103  Eigen::Matrix< StateScalarType, 6, 1 >::Zero( ); };
104  }
105  }
106 
107  // Get state function of global frame origin w.r.t. barycenter
108  std::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > globalFrameOriginBarycentricFunction;
109  std::string globalFrameOrigin = simulation_setup::getGlobalFrameOrigin( bodyMap );
110  if( globalFrameOrigin == "SSB" )
111  {
112  globalFrameOriginBarycentricFunction = [ ]( const TimeType ){ return
113  Eigen::Matrix< StateScalarType, 6, 1 >::Zero( ); };
114  }
115  else
116  {
117  globalFrameOriginBarycentricFunction =
118  std::bind( &simulation_setup::Body::getGlobalFrameOriginBarycentricStateFromEphemeris< StateScalarType, TimeType >,
119  bodyMap.at( globalFrameOrigin ), std::placeholders::_1 );
120  }
121 
122  return std::make_shared< CentralBodyData< StateScalarType, TimeType > >(
123  centralBodiesToUse, bodiesToIntegrate, bodyStateFunctions,
124  globalFrameOriginBarycentricFunction, globalFrameOrigin );
125 }

136 template< typename StateScalarType = double, typename TimeType = double >
137 std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > >
138 createTranslationalStateDerivativeModel(
139  const std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
140  translationPropagatorSettings,
141  const simulation_setup::NamedBodyMap& bodyMap,
142  const TimeType propagationStartTime )
143 {
144 
145  // Create object for frame origin transformations.
146  std::shared_ptr< CentralBodyData< StateScalarType, TimeType > > centralBodyData =
147  createCentralBodyData< StateScalarType, TimeType >(
148  translationPropagatorSettings->centralBodies_,
149  translationPropagatorSettings->bodiesToIntegrate_,
150  bodyMap );
151 
152  std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > stateDerivativeModel;
153 
154  // Check propagator type and create corresponding state derivative object.
155  switch( translationPropagatorSettings->propagator_ )
156  {
157  case cowell:
158  {
159  stateDerivativeModel = std::make_shared<
160  NBodyCowellStateDerivative< StateScalarType, TimeType > >
161  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData,
162  translationPropagatorSettings->bodiesToIntegrate_ );
163  break;
164  }
165  case encke:
166  {
167  // Calculate initial Kepler elements for Encke propagator
168  std::vector< Eigen::Matrix< StateScalarType, 6, 1 > > initialKeplerElements;
169  initialKeplerElements.resize( translationPropagatorSettings->bodiesToIntegrate_.size( ) );
170  std::vector< std::string > centralBodies = translationPropagatorSettings->centralBodies_;
171 
172  for( unsigned int i = 0; i < translationPropagatorSettings->bodiesToIntegrate_.size( ); i++ )
173  {
174  if( bodyMap.count( centralBodies[ i ] ) == 0 )
175  {
176  std::string errorMessage = "Error when creating Encke propagator, did not find central body " + centralBodies[ i ];
177  throw std::runtime_error( errorMessage );
178  }
179  initialKeplerElements[ i ] = orbital_element_conversions::convertCartesianToKeplerianElements< StateScalarType >(
180  translationPropagatorSettings->getInitialStates( ).segment( i * 6, 6 ), static_cast< StateScalarType >(
181  bodyMap.at( centralBodies[ i ] )->getGravityFieldModel( )->getGravitationalParameter( ) ) );
182  }
183 
184  // Create Encke state derivative object.
185  stateDerivativeModel = std::make_shared< NBodyEnckeStateDerivative< StateScalarType, TimeType > >
186  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData, translationPropagatorSettings->bodiesToIntegrate_,
187  initialKeplerElements, propagationStartTime );
188 
189  break;
190  }
191  case gauss_keplerian:
192  {
193  // Create Keplerian state derivative object.
194  stateDerivativeModel = std::make_shared< NBodyGaussKeplerStateDerivative< StateScalarType, TimeType > >
195  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData,
196  translationPropagatorSettings->bodiesToIntegrate_ );
197  break;
198  }
199  case gauss_modified_equinoctial:
200  {
201  std::vector< Eigen::Matrix< StateScalarType, 6, 1 > > initialKeplerElements;
202  std::vector< std::string > centralBodies = translationPropagatorSettings->centralBodies_;
203 
204  for( unsigned int i = 0; i < translationPropagatorSettings->bodiesToIntegrate_.size( ); i++ )
205  {
206  if( bodyMap.count( centralBodies[ i ] ) == 0 )
207  {
208  std::string errorMessage = "Error when creating modified equinoctial propagator, did not find central body " + centralBodies[ i ];
209  throw std::runtime_error( errorMessage );
210  }
211  initialKeplerElements.push_back( orbital_element_conversions::convertCartesianToKeplerianElements< StateScalarType >(
212  translationPropagatorSettings->getInitialStates( ).segment( i * 6, 6 ), static_cast< StateScalarType >(
213  bodyMap.at( centralBodies[ i ] )->getGravityFieldModel( )->getGravitationalParameter( ) ) ) );
214  }
215 
216  // Create modified equinoctial state derivative object.:
217  stateDerivativeModel = std::make_shared< NBodyGaussModifiedEquinictialStateDerivative< StateScalarType, TimeType > >
218  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData,
219  translationPropagatorSettings->bodiesToIntegrate_, initialKeplerElements );
220 
221  break;
222  }
223  case unified_state_model_quaternions:
224  {
225  // Create USM7 state derivative object.
226  stateDerivativeModel = std::make_shared< NBodyUnifiedStateModelQuaternionsStateDerivative< StateScalarType, TimeType > >
227  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData,
228  translationPropagatorSettings->bodiesToIntegrate_ );
229  break;
230  }
231  case unified_state_model_modified_rodrigues_parameters:
232  {
233  // Create USM6 state derivative object.
234  stateDerivativeModel = std::make_shared< NBodyUnifiedStateModelModifiedRodriguesParametersStateDerivative< StateScalarType, TimeType > >
235  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData,
236  translationPropagatorSettings->bodiesToIntegrate_ );
237  break;
238  }
239  case unified_state_model_exponential_map:
240  {
241  // Create USMEM state derivative object.
242  stateDerivativeModel = std::make_shared< NBodyUnifiedStateModelExponentialMapStateDerivative< StateScalarType, TimeType > >
243  ( translationPropagatorSettings->getAccelerationsMap( ), centralBodyData,
244  translationPropagatorSettings->bodiesToIntegrate_ );
245  break;
246  }
247  default:
248  throw std::runtime_error( "Error, did not recognize translational state propagation type: " +
249  std::to_string( translationPropagatorSettings->propagator_ ) );
250  }
251  return stateDerivativeModel;
252 }

262 template< typename StateScalarType = double, typename TimeType = double >
263 std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > createRotationalStateDerivativeModel(
264  const std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationPropagatorSettings,
265  const simulation_setup::NamedBodyMap& bodyMap, const TimeType startTime )
266 {
267  std::vector< std::function< Eigen::Matrix3d( ) > > momentOfInertiaFunctions;
268  for( unsigned int i = 0; i < rotationPropagatorSettings->bodiesToIntegrate_.size( ); i++ )
269  {
270  momentOfInertiaFunctions.push_back(
271  std::bind( &simulation_setup::Body::getBodyInertiaTensor,
272  bodyMap.at( rotationPropagatorSettings->bodiesToIntegrate_.at( i ) ) ) );
273  }
274 
275  // Check propagator type and create corresponding state derivative object.
276  std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > stateDerivativeModel;
277  switch( rotationPropagatorSettings->propagator_ )
278  {
279  case quaternions:
280  {
281  stateDerivativeModel = std::make_shared< RotationalMotionQuaternionsStateDerivative< StateScalarType, TimeType > >(
282  rotationPropagatorSettings->getTorqueModelsMap( ), rotationPropagatorSettings->bodiesToIntegrate_,
283  momentOfInertiaFunctions );
284  break;
285  }
286  case modified_rodrigues_parameters:
287  {
288  stateDerivativeModel = std::make_shared< RotationalMotionModifiedRodriguesParametersStateDerivative< StateScalarType, TimeType > >(
289  rotationPropagatorSettings->getTorqueModelsMap( ), rotationPropagatorSettings->bodiesToIntegrate_,
290  momentOfInertiaFunctions );
291  break;
292  }
293  case exponential_map:
294  {
295  stateDerivativeModel = std::make_shared< RotationalMotionExponentialMapStateDerivative< StateScalarType, TimeType > >(
296  rotationPropagatorSettings->getTorqueModelsMap( ), rotationPropagatorSettings->bodiesToIntegrate_,
297  momentOfInertiaFunctions );
298  break;
299  }
300  default:
301  throw std::runtime_error( "Error, did not recognize rotational state propagation type: " +
302  std::to_string( rotationPropagatorSettings->propagator_ ) );
303  }
304 
305  return stateDerivativeModel;

333 template< typename StateScalarType = double, typename TimeType = double >
334 std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > >
335 createStateDerivativeModel(
336  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
337  const simulation_setup::NamedBodyMap& bodyMap,
338  const TimeType propagationStartTime )
339 {
340  std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > stateDerivativeModel;
341 
342  // Check dynamics type and call associated function to create
343  // specific type of state derivative model.
344  switch( propagatorSettings->getStateType( ) )
345  {
346  case translational_state:
347  {
348  // Check input consistency.
349  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
350  translationPropagatorSettings =
351  std::dynamic_pointer_cast<
352  TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
353  if( translationPropagatorSettings == nullptr )
354  {
355  throw std::runtime_error(
356  "Error, expected translational state propagation settings when making state derivative model" );
357  }
358  else
359  {
360  stateDerivativeModel = createTranslationalStateDerivativeModel< StateScalarType, TimeType >(
361  translationPropagatorSettings, bodyMap, propagationStartTime );
362  }
363  break;
364  }
365  case rotational_state:
366  {
367  std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationPropagatorSettings =
368  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
369  if( rotationPropagatorSettings == nullptr )
370  {
371  throw std::runtime_error(
372  "Error, expected rotation state propagation settings when making state derivative model" );
373  }
374  else
375  {
376  stateDerivativeModel = createRotationalStateDerivativeModel< StateScalarType, TimeType >(
377  rotationPropagatorSettings, bodyMap, propagationStartTime );
378  }
379  break;
380  }
381  case body_mass_state:
382  {
383  // Check input consistency.
384  std::shared_ptr< MassPropagatorSettings< StateScalarType > > massPropagatorSettings =
385  std::dynamic_pointer_cast< MassPropagatorSettings< StateScalarType > >( propagatorSettings );
386  if( massPropagatorSettings == nullptr )
387  {
388  throw std::runtime_error(
389  "Error, expected mass propagation settings when making state derivative model" );
390  }
391  else
392  {
393  stateDerivativeModel = createBodyMassStateDerivativeModel< StateScalarType, TimeType >(
394  massPropagatorSettings, bodyMap );
395  }
396  break;
397  }
398  case custom_state:
399  {
400  // Check input consistency.
401  std::shared_ptr< CustomStatePropagatorSettings< StateScalarType, TimeType > > customPropagatorSettings =
402  std::dynamic_pointer_cast< CustomStatePropagatorSettings< StateScalarType, TimeType > >(
403  propagatorSettings );
404  if( customPropagatorSettings == nullptr )
405  {
406  throw std::runtime_error(
407  "Error, expected custom propagation settings when making state derivative model" );
408  }
409  else
410  {
411  stateDerivativeModel = std::make_shared< CustomStateDerivative< StateScalarType, TimeType > >(
412  customPropagatorSettings->stateDerivativeFunction_, customPropagatorSettings->stateSize_ );
413  }
414  break;
415  }
416  default:
417  throw std::runtime_error(
418  "Error, could not process state type "
419  + std::to_string( propagatorSettings->getStateType( ) )
420  + " when making state derivative model" );
421  }
422  return stateDerivativeModel;
423 }

431 template< typename StateScalarType = double >
432 void setMultiTypePropagationClosure(
433  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
434  const simulation_setup::NamedBodyMap& bodyMap )
435 {
436  // Cast to multi-type settings, and perform closure if
437  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
438  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
439 
440  if( multiTypePropagatorSettings != nullptr )
441  {
442  // Perform closure for the case where both translational and rotational states are propagated
443  if( multiTypePropagatorSettings->propagatorSettingsMap_.count( translational_state ) > 0 &&
444  multiTypePropagatorSettings->propagatorSettingsMap_.count( rotational_state ) > 0 )
445  {
446  std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > translationalStateSettings =
447  multiTypePropagatorSettings->propagatorSettingsMap_.at( translational_state );
448  std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > rotationalStateSettings =
449  multiTypePropagatorSettings->propagatorSettingsMap_.at( rotational_state );
450 
451  // Iterate over all accelerations, and identify those bodies for which an aerodynamic acceleration is applied
452  std::vector< std::string > bodiesWithAerodynamicAcceleration;
453  for( unsigned int i = 0; i < translationalStateSettings.size( ); i++ )
454  {
455  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > currentTranslationalState =
456  std::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >(
457  translationalStateSettings.at( i ) );
458  basic_astrodynamics::AccelerationMap currentAccelerationsMap = currentTranslationalState->getAccelerationsMap( );
459 
460  for( basic_astrodynamics::AccelerationMap::const_iterator accelerationIterator = currentAccelerationsMap.begin( );
461  accelerationIterator != currentAccelerationsMap.end( ); accelerationIterator++ )
462  {
463  for( basic_astrodynamics::SingleBodyAccelerationMap::const_iterator singleBodyIterator =
464  accelerationIterator->second.begin( ); singleBodyIterator != accelerationIterator->second.end( );
465  singleBodyIterator++ )
466  {
467  for( unsigned int j = 0; j < singleBodyIterator->second.size( ); j++ )
468  {
469  if( std::find( bodiesWithAerodynamicAcceleration.begin( ), bodiesWithAerodynamicAcceleration.end( ),
470  accelerationIterator->first ) == bodiesWithAerodynamicAcceleration.end( ) )
471  {
472  if( basic_astrodynamics::getAccelerationModelType( singleBodyIterator->second.at( j ) ) ==
473  basic_astrodynamics::aerodynamic )
474  {
475  bodiesWithAerodynamicAcceleration.push_back( accelerationIterator->first );
476  }
477  }
478  }
479  }
480  }
481  }
482 
483  // Iterate over all settings and identify those bodies for which rotational dynamics is propagated.
484  std::vector< std::string > bodiesWithPropagatedRotation;
485  for( unsigned int i = 0; i < rotationalStateSettings.size( ); i++ )
486  {
487  std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > currentTranslationalState =
488  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >(
489  rotationalStateSettings.at( i ) );
490  std::vector< std::string > currentBodiesWithPropagatedRotation = currentTranslationalState->bodiesToIntegrate_;
491  for( unsigned int j = 0; j < currentBodiesWithPropagatedRotation.size( ); j++ )
492  {
493  if( std::find( bodiesWithPropagatedRotation.begin( ), bodiesWithPropagatedRotation.end( ),
494  currentBodiesWithPropagatedRotation.at( j ) ) == bodiesWithPropagatedRotation.end( ) )
495  {
496  bodiesWithPropagatedRotation.push_back( currentBodiesWithPropagatedRotation.at( j ) );
497  }
498  }
499  }
500 
501  // Find bodies for which both aerodynamic acceleration is used and rotational propagation is performed.
502  std::vector< std::string > bodiesWithAerodynamicRotationalClosure;
503  for( unsigned int i = 0; i < bodiesWithPropagatedRotation.size( ); i++ )
504  {
505  if( std::find( bodiesWithAerodynamicAcceleration.begin( ), bodiesWithAerodynamicAcceleration.end( ),
506  bodiesWithPropagatedRotation.at( i ) ) != bodiesWithAerodynamicAcceleration.end( ) )
507  {
508  bodiesWithAerodynamicRotationalClosure.push_back( bodiesWithPropagatedRotation.at( i ) );
509  }
510  }
511 
512  // Ensure that vehicle orientation is correctly set for aerodynamic acceleration/torque
513  for( unsigned int i = 0; i < bodiesWithAerodynamicRotationalClosure.size( ); i++ )
514  {
515  std::shared_ptr< aerodynamics::FlightConditions > currentFlightConditions =
516  bodyMap.at( bodiesWithAerodynamicRotationalClosure.at( i ) )->getFlightConditions( );
517  reference_frames::setAerodynamicDependentOrientationCalculatorClosure(
518  std::bind( &simulation_setup::Body::getCurrentRotationToLocalFrame,
519  bodyMap.at( bodiesWithAerodynamicRotationalClosure.at( i ) ) ),
520  currentFlightConditions->getAerodynamicAngleCalculator( ) );
521  }
522  }
523  }
524 }

537 template< typename StateScalarType = double, typename TimeType = double >
538 std::vector< std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > >
539 createStateDerivativeModels(
540  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
541  const simulation_setup::NamedBodyMap& bodyMap,
542  const TimeType propagationStartTime )
543 {
544  std::vector< std::shared_ptr< SingleStateTypeDerivative< StateScalarType, TimeType > > >
545  stateDerivativeModels;
546 
547  // Check type of state derivative model and call associated create function.
548  switch( propagatorSettings->getStateType( ) )
549  {
550  // If hybrid, call create function separately for each entry.
551  case hybrid:
552  {
553  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
554  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
555 
556  // Iterate over all propagation settings
557  for( typename std::map< IntegratedStateType,
558  std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::iterator
559  propagatorIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
560  propagatorIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); propagatorIterator++ )
561  {
562  for( unsigned int i = 0; i < propagatorIterator->second.size( ); i++ )
563  {
564  // Call create function for current propagation settings.
565  if( propagatorIterator->first != hybrid )
566  {
567  stateDerivativeModels.push_back( createStateDerivativeModel< StateScalarType, TimeType >(
568  propagatorIterator->second.at( i ), bodyMap, propagationStartTime ) );
569  }
570  else
571  {
572  throw std::runtime_error(
573  "Error when making state derivative model, cannot process nested hybrid propagators" );
574  }
575  }
576  }
577 
578  setMultiTypePropagationClosure( propagatorSettings, bodyMap );
579 
580  break;
581  }
582  // If not hybrid, call create function for single object directly.
583  default:
584  stateDerivativeModels.push_back( createStateDerivativeModel< StateScalarType, TimeType >(
585  propagatorSettings, bodyMap, propagationStartTime ) );
586  }
587 
588  return stateDerivativeModels;
589 }
