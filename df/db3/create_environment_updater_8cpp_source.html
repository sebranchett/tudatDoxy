23 void checkValidityOfRequiredEnvironmentUpdates(
24  const std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >&
25  requestedUpdates, const simulation_setup::NamedBodyMap& bodyMap )
26 {
27  using namespace propagators;
28 
29  // Iterate over all environment update types.
30  for( std::map< propagators::EnvironmentModelsToUpdate,
31  std::vector< std::string > >::const_iterator
32  updateIterator = requestedUpdates.begin( );
33  updateIterator != requestedUpdates.end( ); updateIterator++ )
34  {
35  // Iterate over all updated bodies.
36  for( unsigned int i = 0; i < updateIterator->second.size( ); i++ )
37  {
38  // Ignore global required updates.
39  if( updateIterator->second.at( i ) != "" )
40  {
41  // Check if body exists.
42  if( bodyMap.count( updateIterator->second.at( i ) ) == 0 )
43  {
44  throw std::runtime_error(
45  "Error when making environment model update settings, could not find body "
46  + updateIterator->second.at( i ) );
47  }
48 
49  // Check if requested environment model exists.
50  switch( updateIterator->first )
51  {
52  case body_translational_state_update:
53  {
54  if( bodyMap.at( updateIterator->second.at( i ) )->getEphemeris( ) == nullptr )
55  {
56  throw std::runtime_error(
57  "Error when making environment model update settings, could not find ephemeris of body "
58  + updateIterator->second.at( i ) );
59  }
60  break;
61  }
62  case body_rotational_state_update:
63  {
64  if( ( bodyMap.at( updateIterator->second.at( i ) )->
65  getRotationalEphemeris( ) == nullptr ) &&
66  ( bodyMap.at( updateIterator->second.at( i ) )->getDependentOrientationCalculator( ) == nullptr ) )
67  {
68  throw std::runtime_error(
69  "Error when making environment model update settings, could not find rotational ephemeris or dependent orientation calculator of body "
70  + updateIterator->second.at( i ) );
71  }
72  break;
73  }
74  case spherical_harmonic_gravity_field_update:
75  {
76  std::shared_ptr< gravitation::SphericalHarmonicsGravityField >
77  gravityFieldModel =
78  std::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravityField >(
79  bodyMap.at( updateIterator->second.at( i ) )->getGravityFieldModel( ) );
80  if( gravityFieldModel == nullptr )
81  {
82  throw std::runtime_error(
83  "Error when making environment model update settings, could not find spherical harmonic gravity field of body "
84  + updateIterator->second.at( i ) );
85  }
86  break;
87  }
88  case vehicle_flight_conditions_update:
89  {
90  std::shared_ptr< aerodynamics::FlightConditions > flightConditions = bodyMap.at(
91  updateIterator->second.at( i ) )->getFlightConditions( );
92  if( flightConditions == nullptr )
93  {
94  throw std::runtime_error(
95  "Error when making environment model update settings, could not find flight conditions of body "
96  + updateIterator->second.at( i ) );
97  }
98  break;
99  }
100  case radiation_pressure_interface_update:
101  {
102  std::map< std::string, std::shared_ptr< electro_magnetism::RadiationPressureInterface > >
103  radiationPressureInterfaces = bodyMap.at(
104  updateIterator->second.at( i ) )->getRadiationPressureInterfaces( );
105  if( radiationPressureInterfaces.size( ) == 0 )
106  {
107  throw std::runtime_error(
108  "Error when making environment model update settings, could not find radiation pressure interface of body "
109  + updateIterator->second.at( i ) );
110  }
111  break;
112  }
113  case body_mass_update:
114  if( bodyMap.at( updateIterator->second.at( i ) )->getBodyMassFunction( ) == nullptr )
115  {
116  throw std::runtime_error(
117  "Error when making environment model update settings, no body mass function of body "
118  + updateIterator->second.at( i ) );
119  }
120 
121  break;
122  }
123  }
124  }
125  }

129 void removePropagatedStatesFomEnvironmentUpdates(
130  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >& environmentModelsToUpdate,
131  const std::map< IntegratedStateType, std::vector< std::pair< std::string, std::string > > >& integratedStateList )
132 {
133  // Iterate over all propagated states
134  for( auto it = integratedStateList.begin( ); it != integratedStateList.end( ); it++ )
135  {
136  // Iterate over all propagated bodies for current state
137  for( unsigned int i = 0; i < it->second.size( ); i++ )
138  {
139  switch( it->first )
140  {
141  // Check for propagated translational states in update list, and remove if necessary
142  case translational_state:
143  if( environmentModelsToUpdate.count( body_translational_state_update ) > 0 )
144  {
145  std::vector< std::string > bodiesToUpdate = environmentModelsToUpdate.at( body_translational_state_update );
146  std::vector< std::string >::iterator findIterator =
147  std::find( bodiesToUpdate.begin( ), bodiesToUpdate.end( ), it->second.at( i ).first );
148 
149  if( findIterator != bodiesToUpdate.end( ) )
150  {
151  bodiesToUpdate.erase( findIterator );
152  environmentModelsToUpdate[ body_translational_state_update ] = bodiesToUpdate;
153 
154  }
155  }
156  break;
157  // Check for propagated rotational states in update list, and remove if necessary
158  case rotational_state:
159  if( environmentModelsToUpdate.count( body_rotational_state_update ) > 0 )
160  {
161  std::vector< std::string > bodiesToUpdate = environmentModelsToUpdate.at( body_rotational_state_update );
162  std::vector< std::string >::iterator findIterator =
163  std::find( bodiesToUpdate.begin( ), bodiesToUpdate.end( ), it->second.at( i ).first );
164 
165  if( findIterator != bodiesToUpdate.end( ) )
166  {
167  bodiesToUpdate.erase( findIterator );
168  environmentModelsToUpdate[ body_rotational_state_update ] = bodiesToUpdate;
169 
170  }
171  }
172  break;
173  // Check for propagated mass states in update list, and remove if necessary
174  case body_mass_state:
175  if( environmentModelsToUpdate.count( body_mass_update ) > 0 )
176  {
177  std::vector< std::string > bodiesToUpdate = environmentModelsToUpdate.at( body_mass_update );
178  std::vector< std::string >::iterator findIterator =
179  std::find( bodiesToUpdate.begin( ), bodiesToUpdate.end( ), it->second.at( i ).first );
180 
181  if( findIterator != bodiesToUpdate.end( ) )
182  {
183  bodiesToUpdate.erase( findIterator );
184  environmentModelsToUpdate[ body_mass_update ] = bodiesToUpdate;
185 
186  }
187  }
188  break;
189  case custom_state:
190  break;
191  default:
192  throw std::runtime_error( "Error when removing propagated states from environment updates, state type " +
193  std::to_string( it->first ) + " not recognized." );
194  }
195  }
196  }

200 std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >
201 createRotationalEquationsOfMotionEnvironmentUpdaterSettings(
202  const basic_astrodynamics::TorqueModelMap& torqueModels, const simulation_setup::NamedBodyMap& bodyMap )
203 {
204  using namespace basic_astrodynamics;
205 
206  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > environmentModelsToUpdate;
207  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleTorqueUpdateNeeds;
208 
209  // Iterate over all bodies on which torques are being exerting
210  for( TorqueModelMap::const_iterator acceleratedBodyIterator = torqueModels.begin( );
211  acceleratedBodyIterator != torqueModels.end( ); acceleratedBodyIterator++ )
212  {
213  // Iterate over all bodies exerting on current body
214  for( SingleBodyTorqueModelMap::const_iterator torqueModelIterator = acceleratedBodyIterator->second.begin( );
215  torqueModelIterator != acceleratedBodyIterator->second.end( ); torqueModelIterator++ )
216  {
217  singleTorqueUpdateNeeds.clear( );
218  for( unsigned int i = 0; i < torqueModelIterator->second.size( ); i++ )
219  {
220  AvailableTorque currentTorqueModelType =
221  getTorqueModelType( torqueModelIterator->second.at( i ) );
222 
223  switch( currentTorqueModelType )
224  {
225  case second_order_gravitational_torque:
226  singleTorqueUpdateNeeds[ body_translational_state_update ].push_back(
227  torqueModelIterator->first );
228  singleTorqueUpdateNeeds[ body_translational_state_update ].push_back(
229  acceleratedBodyIterator->first );
230  break;
231  case spherical_harmonic_gravitational_torque:
232  singleTorqueUpdateNeeds[ body_translational_state_update ].push_back(
233  torqueModelIterator->first );
234  singleTorqueUpdateNeeds[ body_translational_state_update ].push_back(
235  acceleratedBodyIterator->first );
236  singleTorqueUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
237  acceleratedBodyIterator->first );
238  break;
239 
240  case aerodynamic_torque:
241  singleTorqueUpdateNeeds[ body_translational_state_update ].push_back(
242  torqueModelIterator->first );
243  singleTorqueUpdateNeeds[ body_translational_state_update ].push_back(
244  acceleratedBodyIterator->first );
245  singleTorqueUpdateNeeds[ body_rotational_state_update ].push_back(
246  torqueModelIterator->first );
247  singleTorqueUpdateNeeds[ vehicle_flight_conditions_update ].push_back(
248  acceleratedBodyIterator->first );
249  break;
250  case inertial_torque:
251  break;
252  default:
253  std::cerr << "Error, update information not found for torque model " << currentTorqueModelType << std::endl;
254  break;
255  }
256  }
257 
258  addEnvironmentUpdates( environmentModelsToUpdate, singleTorqueUpdateNeeds );
259  }
260  }
261 
262  return environmentModelsToUpdate;
263 }

267 std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >
268 createTranslationalEquationsOfMotionEnvironmentUpdaterSettings(
269  const basic_astrodynamics::AccelerationMap& translationalAccelerationModels,
270  const simulation_setup::NamedBodyMap& bodyMap )
271 {
272  using namespace basic_astrodynamics;
273  using namespace propagators;
274 
275  std::map< propagators::EnvironmentModelsToUpdate,
276  std::vector< std::string > > environmentModelsToUpdate;
277  std::map< propagators::EnvironmentModelsToUpdate,
278  std::vector< std::string > > singleAccelerationUpdateNeeds;
279 
280  // Iterate over all bodies being accelerated
281  for( AccelerationMap::const_iterator acceleratedBodyIterator
282  = translationalAccelerationModels.begin( );
283  acceleratedBodyIterator != translationalAccelerationModels.end( );
284  acceleratedBodyIterator++ )
285  {
286  // Iterate over all bodies exerting acceleration.
287  for( SingleBodyAccelerationMap::const_iterator accelerationModelIterator
288  = acceleratedBodyIterator->second.begin( );
289  accelerationModelIterator != acceleratedBodyIterator->second.end( );
290  accelerationModelIterator++ )
291  {
292  singleAccelerationUpdateNeeds.clear( );
293  for( unsigned int i = 0; i < accelerationModelIterator->second.size( ); i++ )
294  {
295  AvailableAcceleration currentAccelerationModelType =
296  getAccelerationModelType( accelerationModelIterator->second.at( i ) );
297 
298  // Add translational state of both bodies to update list for current acceleration model.
299  if( translationalAccelerationModels.count( accelerationModelIterator->first ) == 0 )
300  {
301  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
302  accelerationModelIterator->first );
303  }
304 
305  // Check acceleration model type and change environment update list accordingly.
306  switch( currentAccelerationModelType )
307  {
308  case central_gravity:
309  break;
310  case third_body_central_gravity:
311  {
312  std::shared_ptr< gravitation::ThirdBodyCentralGravityAcceleration >
313  thirdBodyAcceleration = std::dynamic_pointer_cast<
314  gravitation::ThirdBodyCentralGravityAcceleration >(
315  accelerationModelIterator->second.at( i ) );
316  if( thirdBodyAcceleration != nullptr && translationalAccelerationModels.count(
317  thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
318  {
319  if( translationalAccelerationModels.count( thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
320  {
321  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
322  thirdBodyAcceleration->getCentralBodyName( ) );
323  }
324  }
325  else if( thirdBodyAcceleration == nullptr )
326  {
327  throw std::runtime_error(
328  std::string( "Error, incompatible input (ThirdBodyCentralGravityAcceleration) to" )
329  + std::string( "createTranslationalEquationsOfMotionEnvironmentUpdaterSettings" ) );
330  }
331  break;
332  }
333  case aerodynamic:
334  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
335  accelerationModelIterator->first );
336  singleAccelerationUpdateNeeds[ vehicle_flight_conditions_update ].push_back(
337  acceleratedBodyIterator->first );
338  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
339  acceleratedBodyIterator->first );
340  break;
341  case cannon_ball_radiation_pressure:
342  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].push_back(
343  acceleratedBodyIterator->first );
344  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
345  acceleratedBodyIterator->first );
346  break;
347  case panelled_radiation_pressure_acceleration:
348  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].push_back(
349  acceleratedBodyIterator->first );
350  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
351  acceleratedBodyIterator->first );
352  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
353  acceleratedBodyIterator->first );
354  break;
355  case spherical_harmonic_gravity:
356  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
357  accelerationModelIterator->first );
358  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
359  push_back( accelerationModelIterator->first );
360  break;
361  case mutual_spherical_harmonic_gravity:
362  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
363  accelerationModelIterator->first );
364  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
365  accelerationModelIterator->first );
366  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
367  acceleratedBodyIterator->first );
368  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
369  acceleratedBodyIterator->first );
370  break;
371  case third_body_spherical_harmonic_gravity:
372  {
373  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
374  accelerationModelIterator->first );
375  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
376  push_back( accelerationModelIterator->first );
377 
378  std::shared_ptr<
379  gravitation::ThirdBodySphericalHarmonicsGravitationalAccelerationModel >
380  thirdBodyAcceleration = std::dynamic_pointer_cast<
381  gravitation::ThirdBodySphericalHarmonicsGravitationalAccelerationModel >(
382  accelerationModelIterator->second.at( i ) );;
383  if( thirdBodyAcceleration != nullptr && translationalAccelerationModels.count(
384  thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
385  {
386  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
387  thirdBodyAcceleration->getCentralBodyName( ) );
388  }
389  else if( thirdBodyAcceleration == nullptr )
390  {
391  throw std::runtime_error(
392  std::string( "Error, incompatible input (ThirdBodySphericalHarmonicsGravitational" )
393  + std::string( "AccelerationModel) to createTranslationalEquationsOfMotion ")
394  + std::string( "EnvironmentUpdaterSettings" ) );
395  }
396  break;
397  }
398  case third_body_mutual_spherical_harmonic_gravity:
399  {
400  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
401  accelerationModelIterator->first );
402  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
403  accelerationModelIterator->first );
404  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
405  acceleratedBodyIterator->first );
406  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
407  acceleratedBodyIterator->first );
408 
409  std::shared_ptr< gravitation::ThirdBodyMutualSphericalHarmonicsGravitationalAccelerationModel >
410  thirdBodyAcceleration = std::dynamic_pointer_cast<
411  gravitation::ThirdBodyMutualSphericalHarmonicsGravitationalAccelerationModel >(
412  accelerationModelIterator->second.at( i ) );
413  if( thirdBodyAcceleration != nullptr && translationalAccelerationModels.count(
414  thirdBodyAcceleration->getCentralBodyName( ) ) == 0 )
415  {
416  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
417  thirdBodyAcceleration->getCentralBodyName( ) );
418  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
419  thirdBodyAcceleration->getCentralBodyName( ) );
420  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].push_back(
421  thirdBodyAcceleration->getCentralBodyName( ) );
422  }
423  else if( thirdBodyAcceleration == nullptr )
424  {
425  throw std::runtime_error(
426  std::string( "Error, incompatible input (ThirdBodyMutualSphericalHarmonicsGravitational" ) +
427  std::string( "AccelerationModel) to createTranslationalEquationsOfMotion ") +
428  std::string( "EnvironmentUpdaterSettings" ) );
429  }
430  break;
431  }
432  case thrust_acceleration:
433  {
434  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > thrustModelUpdates =
435  std::dynamic_pointer_cast< propulsion::ThrustAcceleration >(
436  accelerationModelIterator->second.at( i ) )->getRequiredModelUpdates( );
437  addEnvironmentUpdates( singleAccelerationUpdateNeeds, thrustModelUpdates );
438 
439  singleAccelerationUpdateNeeds[ body_mass_update ].push_back(
440  acceleratedBodyIterator->first );
441 
442  break;
443  }
444  case relativistic_correction_acceleration:
445  {
446  std::shared_ptr< relativity::RelativisticAccelerationCorrection >
447  accelerationCorrection = std::dynamic_pointer_cast< relativity::RelativisticAccelerationCorrection >(
448  accelerationModelIterator->second.at( i ) );
449  if( accelerationCorrection->getCalculateDeSitterCorrection( ) )
450  {
451  std::string primaryBody = accelerationCorrection->getPrimaryBodyName( );
452  if( translationalAccelerationModels.count(primaryBody ) == 0 )
453  {
454  singleAccelerationUpdateNeeds[ body_translational_state_update ].push_back(
455  primaryBody );
456  }
457  }
458  break;
459  }
460  case direct_tidal_dissipation_in_central_body_acceleration:
461  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
462  accelerationModelIterator->first );
463  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
464  push_back( accelerationModelIterator->first );
465  break;
466  case direct_tidal_dissipation_in_orbiting_body_acceleration:
467  singleAccelerationUpdateNeeds[ body_rotational_state_update ].push_back(
468  accelerationModelIterator->first );
469  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
470  push_back( accelerationModelIterator->first );
471  break;
472  case empirical_acceleration:
473  break;
474  case momentum_wheel_desaturation_acceleration:
475  break;
476  case solar_sail_acceleration:
477  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].push_back( acceleratedBodyIterator->first );
478  singleAccelerationUpdateNeeds[ body_mass_update ].push_back( acceleratedBodyIterator->first );
479  break;
480  default:
481  throw std::runtime_error( std::string( "Error when setting acceleration model update needs, model type not recognized: " ) +
482  std::to_string( currentAccelerationModelType ) );
483  break;
484  }
485 
486  }
487 
488  // Add requested updates of current acceleration model to
489  // full list of environment updates.
490  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
491  }
492  }
493 
494  return environmentModelsToUpdate;
495 }

581 std::map< propagators::EnvironmentModelsToUpdate,

599  variablesToUpdate[ body_translational_state_update ].push_back( dependentVariableSaveSettings->secondaryBody_ );

895  // Iterate over all bodies.

916  // Add each interface update function to update list.

918  iterator = radiationPressureInterfaces.begin( );

921  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].
922  push_back( bodyIterator->first );
923  }

927  ( bodyIterator->second->getDependentOrientationCalculator( ) != nullptr ) )
928  {

935  std::dynamic_pointer_cast< TimeDependentSphericalHarmonicsGravityField >

940  push_back( bodyIterator->first );
