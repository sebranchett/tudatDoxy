30 HodographicShaping::HodographicShaping(
31  const Eigen::Vector6d& initialState,
32  const Eigen::Vector6d& finalState,
33  const double timeOfFlight,
34  const double centralBodyGravitationalParameter,
35  const int numberOfRevolutions,
36  const std::vector< std::shared_ptr< shape_based_methods::BaseFunctionHodographicShaping > >& radialVelocityFunctionComponents,
37  const std::vector< std::shared_ptr< shape_based_methods::BaseFunctionHodographicShaping > >& normalVelocityFunctionComponents,
38  const std::vector< std::shared_ptr< shape_based_methods::BaseFunctionHodographicShaping > >& axialVelocityFunctionComponents,
39  const Eigen::VectorXd& freeCoefficientsRadialVelocityFunction,
40  const Eigen::VectorXd& freeCoefficientsNormalVelocityFunction,
41  const Eigen::VectorXd& freeCoefficientsAxialVelocityFunction,
42  const double initialBodyMass ) :
43  ShapeBasedMethod( initialState, finalState, timeOfFlight, initialBodyMass ),
44  centralBodyGravitationalParameter_( centralBodyGravitationalParameter ),
45  numberOfRevolutions_( numberOfRevolutions ),
46  freeCoefficientsRadialVelocityFunction_( freeCoefficientsRadialVelocityFunction ),
47  freeCoefficientsNormalVelocityFunction_( freeCoefficientsNormalVelocityFunction ),
48  freeCoefficientsAxialVelocityFunction_( freeCoefficientsAxialVelocityFunction )
49 {
50  // Define composite function in radial direction.
51  Eigen::VectorXd radialVelocityCoefficients;
52  radialVelocityCoefficients.resize( 3 + freeCoefficientsRadialVelocityFunction_.size() );
53  radialVelocityCoefficients.segment( 0, 3 ) = Eigen::Vector3d::Zero();
54  radialVelocityCoefficients.segment( 3, freeCoefficientsRadialVelocityFunction_.size() ) = freeCoefficientsRadialVelocityFunction_;
55 
56  radialVelocityFunction_ = std::make_shared< shape_based_methods::CompositeFunctionHodographicShaping >(
57  radialVelocityFunctionComponents, radialVelocityCoefficients );
58 
59  // Define composite function in normal direction.
60  Eigen::VectorXd normalVelocityCoefficients;
61  normalVelocityCoefficients.resize( 3 + freeCoefficientsNormalVelocityFunction_.size() );
62  normalVelocityCoefficients.segment( 0, 3 ) = Eigen::Vector3d::Zero();
63  normalVelocityCoefficients.segment( 3, freeCoefficientsNormalVelocityFunction_.size() ) = freeCoefficientsNormalVelocityFunction_;
64 
65  normalVelocityFunction_ = std::make_shared< shape_based_methods::CompositeFunctionHodographicShaping >(
66  normalVelocityFunctionComponents, normalVelocityCoefficients );
67 
68  // Define composite function in axial direction.
69  Eigen::VectorXd axialVelocityCoefficients;
70  axialVelocityCoefficients.resize( 3 + freeCoefficientsAxialVelocityFunction_.size() );
71  axialVelocityCoefficients.segment( 0, 3 ) = Eigen::Vector3d::Zero();
72  axialVelocityCoefficients.segment( 3, freeCoefficientsAxialVelocityFunction_.size() ) = freeCoefficientsAxialVelocityFunction_;
73 
74  axialVelocityFunction_ = std::make_shared< shape_based_methods::CompositeFunctionHodographicShaping >(
75  axialVelocityFunctionComponents, axialVelocityCoefficients );
76 
77 
78  // Compute initial cylindrical state.
79  Eigen::Vector6d initialCylindricalState = coordinate_conversions::convertCartesianToCylindricalState( stateAtDeparture_ );
80 
81  // Compute final cylindrical state.
82  Eigen::Vector6d finalCylindricalState = coordinate_conversions::convertCartesianToCylindricalState( stateAtArrival_ );
83 
84  // Compute number of free coefficients for each velocity function.
85  numberOfFreeCoefficientsRadialVelocityFunction_ = radialVelocityFunction_->getNumberOfCompositeFunctionComponents() - 3;
86  numberOfFreeCoefficientsNormalVelocityFunction_ = normalVelocityFunction_->getNumberOfCompositeFunctionComponents() - 3;
87  numberOfFreeCoefficientsAxialVelocityFunction = axialVelocityFunction_->getNumberOfCompositeFunctionComponents() - 3;
88  if ( numberOfFreeCoefficientsRadialVelocityFunction_ < 0 || numberOfFreeCoefficientsNormalVelocityFunction_ < 0 ||
89  numberOfFreeCoefficientsAxialVelocityFunction < 0 )
90  {
91  std::cerr << "The number of base functions in one of the velocity functions is smaller than "
92  << "the number of corresponding boundary conditions. The boundary conditions cannot be set!\n";
93  }
94  else
95  {
96  // Define numerical quadrature settings, required to compute the current polar angle and final deltaV.
97  quadratureSettings_ = std::make_shared< numerical_quadrature::GaussianQuadratureSettings< double > >( 0.0, 64 );
98 
99  // Set boundary conditions in the radial direction.
100  radialBoundaryConditions_.push_back( initialCylindricalState[ 0 ] );
101  radialBoundaryConditions_.push_back( finalCylindricalState[ 0 ] );
102  radialBoundaryConditions_.push_back( initialCylindricalState[ 3 ] );
103  radialBoundaryConditions_.push_back( finalCylindricalState[ 3 ] );
104 
105  // Set boundary conditions in the normal direction.
106  normalBoundaryConditions_.push_back( initialCylindricalState[ 4 ] );
107  normalBoundaryConditions_.push_back( finalCylindricalState[ 4 ] );
108 
109  // Final value polar angle
110  normalBoundaryConditions_.push_back(
111  numberOfRevolutions_ * 2.0 * mathematical_constants::PI
112  + basic_mathematics::computeModulo( ( finalCylindricalState[ 1 ] - initialCylindricalState[ 1 ] ),
113  2.0 * mathematical_constants::PI ) );
114 
115  // Set boundary conditions in the axial direction.
116  axialBoundaryConditions_.push_back( initialCylindricalState[ 2 ] );
117  axialBoundaryConditions_.push_back( finalCylindricalState[ 2 ] );
118  axialBoundaryConditions_.push_back( initialCylindricalState[ 5 ] );
119  axialBoundaryConditions_.push_back( finalCylindricalState[ 5 ] );
120 
121  // Compute inverse of matrices containing boundary values.
122  inverseMatrixRadialBoundaryValues_ = computeInverseMatrixRadialOrAxialBoundaries( radialVelocityFunction_ );
123  inverseMatrixNormalBoundaryValues_ = computeInverseMatrixNormalBoundaries( normalVelocityFunction_ );
124  inverseAxialMatrixBoundaryValues_ = computeInverseMatrixRadialOrAxialBoundaries( axialVelocityFunction_ );
125 
126  // Satisfy boundary conditions.
127  satisfyRadialBoundaryConditions( freeCoefficientsRadialVelocityFunction_ );
128  satisfyNormalBoundaryConditions( freeCoefficientsNormalVelocityFunction_ );
129  satisfyAxialBoundaryConditions( freeCoefficientsAxialVelocityFunction_ );
130  }

182 void HodographicShaping::satisfyRadialBoundaryConditions( Eigen::VectorXd freeCoefficients ){
183 
184  // Vector containing boundary conditions on radial distance and initial and final radial velocity.
185  Eigen::Vector3d vectorBoundaryConditionsRadial;
186  vectorBoundaryConditionsRadial[ 0 ] = radialBoundaryConditions_[ 1 ] - radialBoundaryConditions_[ 0 ];
187  vectorBoundaryConditionsRadial[ 1 ] = radialBoundaryConditions_[ 2 ];
188  vectorBoundaryConditionsRadial[ 2 ] = radialBoundaryConditions_[ 3 ];
189 
190  // Subtract boundary values of free components of velocity function from corresponding boundary conditions.
191  for ( int i = 0 ; i < numberOfFreeCoefficientsRadialVelocityFunction_ ; i++ )
192  {
193  vectorBoundaryConditionsRadial[ 0 ] -=
194  freeCoefficients( i ) * ( radialVelocityFunction_->getComponentFunctionIntegralCurrentTime( i + 3, timeOfFlight_ )
195  - radialVelocityFunction_->getComponentFunctionIntegralCurrentTime( i + 3, 0.0 ) );
196  vectorBoundaryConditionsRadial[ 1 ] -= freeCoefficients( i ) *
197  radialVelocityFunction_->getComponentFunctionCurrentValue( i + 3, 0.0 );
198  vectorBoundaryConditionsRadial[ 2 ] -= freeCoefficients( i ) *
199  radialVelocityFunction_->getComponentFunctionCurrentValue( i + 3, timeOfFlight_ );
200  }
201 
202  // Compute fixed coefficients.
203  Eigen::Vector3d fixedCoefficientsRadial = inverseMatrixRadialBoundaryValues_ * vectorBoundaryConditionsRadial;
204 
205  // Create vector containing all radial velocity function coefficients.
206  Eigen::VectorXd radialVelocityFunctionCoefficients =
207  Eigen::VectorXd::Zero( fixedCoefficientsRadial.size() + numberOfFreeCoefficientsRadialVelocityFunction_ );
208 
209  // Check whether the radial velocity function has free coefficients.
210  if ( numberOfFreeCoefficientsRadialVelocityFunction_ == 0 )
211  {
212  radialVelocityFunctionCoefficients << fixedCoefficientsRadial;
213  }
214  else
215  {
216  radialVelocityFunctionCoefficients << fixedCoefficientsRadial,
217  freeCoefficients.segment( 0, numberOfFreeCoefficientsRadialVelocityFunction_);
218  }
219 
220  // Set the coefficients of the radial velocity function.
221  radialVelocityFunction_->resetCompositeFunctionCoefficients( radialVelocityFunctionCoefficients );
222 
223 }

267 void HodographicShaping::satisfyNormalBoundaryConditions( Eigen::VectorXd freeCoefficients ){
268 
269  // Compute coefficient of the third component of the composite function, from the required value of the final polar angle.
270  double C3 = computeThirdFixedCoefficientAxialVelocityFromFinalPolarAngle( freeCoefficients );
271 
272  // Vector containing boundary conditions on initial and final normal velocity.
273  Eigen::Vector2d vectorBoundaryConditionsNormal;
274  vectorBoundaryConditionsNormal( 0 ) = normalBoundaryConditions_[ 0 ];
275  vectorBoundaryConditionsNormal( 1 ) = normalBoundaryConditions_[ 1 ];
276 
277  // Subtract boundary values of velocity function component used to solve for final polar angle from corresponding boundary conditions.
278  vectorBoundaryConditionsNormal[ 0 ] -= C3 * normalVelocityFunction_->getComponentFunctionCurrentValue( 2, 0.0 );
279  vectorBoundaryConditionsNormal[ 1 ] -= C3 * normalVelocityFunction_->getComponentFunctionCurrentValue( 2, timeOfFlight_ );
280 
281  // Subtract boundary values of free velocity function components from corresponding boundary conditions.
282  for ( int i = 0 ; i < numberOfFreeCoefficientsNormalVelocityFunction_ ; i++ )
283  {
284  vectorBoundaryConditionsNormal[ 0 ] -= freeCoefficients[ i ]
285  * normalVelocityFunction_->getComponentFunctionCurrentValue( i + 3, 0.0 );
286  vectorBoundaryConditionsNormal[ 1 ] -= freeCoefficients[ i ]
287  * normalVelocityFunction_->getComponentFunctionCurrentValue( i + 3, timeOfFlight_ );
288  }
289 
290  // Compute fixed coefficients by satisfying the boundary conditions.
291  Eigen::Vector2d fixedCoefficientsNormal = inverseMatrixNormalBoundaryValues_ * vectorBoundaryConditionsNormal;
292 
293  // Create vector containing all normal velocity function coefficients.
294  Eigen::VectorXd normalVelocityFunctionCoefficients =
295  Eigen::VectorXd::Zero( fixedCoefficientsNormal.size() + 1 + numberOfFreeCoefficientsNormalVelocityFunction_ );
296 
297  // Check whether the normal velocity function has free coefficients.
298  if ( numberOfFreeCoefficientsNormalVelocityFunction_ == 0 )
299  {
300  normalVelocityFunctionCoefficients << fixedCoefficientsNormal, C3;
301  }
302  else
303  {
304  normalVelocityFunctionCoefficients << fixedCoefficientsNormal, C3,
305  freeCoefficients.segment( 0, numberOfFreeCoefficientsNormalVelocityFunction_);
306  }
307 
308  // Set the coefficients of the normal velocity function.
309  normalVelocityFunction_->resetCompositeFunctionCoefficients( normalVelocityFunctionCoefficients );
310 
311 }

313 double HodographicShaping::computeThirdFixedCoefficientAxialVelocityFromFinalPolarAngle( Eigen::VectorXd freeCoefficients ){
314 
315  // Compute the third fixed coefficient of the normal velocity composite function, so that the condition on the final
316  // polar angle is fulfilled.
317  // The calculation is based on Equation (16) in Gondelach D., and Noomen R.
318  // "Hodographic-shaping method for low-thrust interplanetary trajectory design." Journal of Spacecraft and Rockets 52.3 (2015): 728-738.
319 
320  // Vector containing boundary conditions on initial and final normal velocity.
321  Eigen::Vector2d vectorBoundaryConditionsNormal;
322  vectorBoundaryConditionsNormal[ 0 ] = normalBoundaryConditions_[ 0 ];
323  vectorBoundaryConditionsNormal[ 1 ] = normalBoundaryConditions_[ 1 ];
324 
325  // Subtract boundary values of free velocity function components from corresponding boundary conditions.
326  for ( int i = 0 ; i < numberOfFreeCoefficientsNormalVelocityFunction_; i++ )
327  {
328  vectorBoundaryConditionsNormal[ 0 ] -= freeCoefficients[ i ]
329  * normalVelocityFunction_->getComponentFunctionCurrentValue( i + 3, 0.0 );
330  vectorBoundaryConditionsNormal[ 1 ] -= freeCoefficients[ i ]
331  * normalVelocityFunction_->getComponentFunctionCurrentValue( i + 3, timeOfFlight_ );
332  }
333 
334  // Define matrix L, as proposed in ... (ADD PROPER REFERENCE AND EQUATION NUMBER)
335  Eigen::Vector2d matrixL;
336  matrixL = inverseMatrixNormalBoundaryValues_ * vectorBoundaryConditionsNormal;
337 
338  Eigen::Vector2d initialAndFinalValuesThirdComponentFunction(
339  - normalVelocityFunction_->getComponentFunctionCurrentValue( 2, 0.0 ),
340  - normalVelocityFunction_->getComponentFunctionCurrentValue( 2, timeOfFlight_ ) );
341 
342  // Define matrix K, as proposed in ... (ADD PROPER REFERENCE AND EQUATION NUMBER)
343  Eigen::Vector2d matrixK;
344  matrixK = inverseMatrixNormalBoundaryValues_ * initialAndFinalValuesThirdComponentFunction;
345 
346 
347  // Define angular velocity due to the third component of the composite function only.
348  std::function< double( const double ) > derivativePolarAngleDueToThirdComponent = [ = ] ( const double currentTime ){
349 
350  double angularVelocityDueToThirdComponent = (
351  matrixK( 0 ) * normalVelocityFunction_->getComponentFunctionCurrentValue( 0, currentTime )
352  + matrixK( 1 ) * normalVelocityFunction_->getComponentFunctionCurrentValue( 1, currentTime )
353  + normalVelocityFunction_->getComponentFunctionCurrentValue( 2, currentTime ) ) /
354  computeCurrentRadialDistance( currentTime );
355 
356  return angularVelocityDueToThirdComponent;
357 
358  };
359 
360 
361  // Define the angular velocity due to all the other components of the composite function, once combined.
362  std::function< double( const double ) > derivativePolarAngleDueToOtherComponents = [ = ] ( const double currentTime ){
363 
364  double angularVelocityDueToFreeCoefficients = 0.0;
365  for( int j = 0 ; j < numberOfFreeCoefficientsNormalVelocityFunction_ ; j++ )
366  {
367  angularVelocityDueToFreeCoefficients += freeCoefficients( j )
368  * normalVelocityFunction_->getComponentFunctionCurrentValue( j + 3 , currentTime );
369  }
370 
371  double angularVelocityDueToOtherComponents = ( matrixL( 0 ) * normalVelocityFunction_->getComponentFunctionCurrentValue( 0 , currentTime )
372  + matrixL( 1 ) * normalVelocityFunction_->getComponentFunctionCurrentValue( 1 , currentTime )
373  + angularVelocityDueToFreeCoefficients )
374  / computeCurrentRadialDistance( currentTime );
375 
376  return angularVelocityDueToOtherComponents;
377 
378  };
379 
380  // Define numerical quadratures.
381  std::shared_ptr< numerical_quadrature::NumericalQuadrature< double, double > > integratorPolarAngleDueToThirdComponentTest =
382  numerical_quadrature::createQuadrature( derivativePolarAngleDueToThirdComponent, quadratureSettings_, timeOfFlight_ );
383 
384  std::shared_ptr< numerical_quadrature::NumericalQuadrature< double, double > > integratorPolarAngleDueToOtherComponentsTest =
385  numerical_quadrature::createQuadrature( derivativePolarAngleDueToOtherComponents, quadratureSettings_, timeOfFlight_ );
386 
387  return ( normalBoundaryConditions_[ 2 ] - integratorPolarAngleDueToOtherComponentsTest->getQuadrature() )
388  / integratorPolarAngleDueToThirdComponentTest->getQuadrature();
389 
390 }

481 Eigen::Vector3d HodographicShaping::computeThrustAccelerationInCylindricalCoordinates( double currentTime )
482 {
483 
484  Eigen::Vector3d thrustAccelerationComponents;
485 
486  // Compute radial distance from the central body.
487  double radialDistance = radialVelocityFunction_->evaluateCompositeFunctionIntegralCurrentTime( currentTime )
488  - radialVelocityFunction_->evaluateCompositeFunctionIntegralCurrentTime( 0.0 ) + radialBoundaryConditions_[0];
489 
490  // Compute axial distance from the central body.
491  double axialDistance = axialVelocityFunction_->evaluateCompositeFunctionIntegralCurrentTime( currentTime )
492  - axialVelocityFunction_->evaluateCompositeFunctionIntegralCurrentTime( 0.0 ) + axialBoundaryConditions_[0];
493 
494  // Compute distance from the central body.
495  double distanceFromCentralBody = sqrt( pow( radialDistance, 2.0 ) + pow( axialDistance, 2.0 ) );
496 
497  // Computation of normal velocity.
498  double normalVelocity = normalVelocityFunction_->evaluateCompositeFunctionCurrentTime( currentTime );
499 
500  // Compute angular velocity.
501  double angularVelocity = normalVelocity / radialDistance;
502 
503  // Compute radial thrust acceleration.
504  thrustAccelerationComponents[ 0 ] = radialVelocityFunction_->evaluateCompositeFunctionDerivativeCurrentTime( currentTime )
505  - angularVelocity * normalVelocity + centralBodyGravitationalParameter_ / std::pow( distanceFromCentralBody, 3.0 ) * radialDistance;
506 
507  // Compute normal thrust acceleration.
508  thrustAccelerationComponents[ 1 ] = normalVelocityFunction_->evaluateCompositeFunctionDerivativeCurrentTime( currentTime )
509  + angularVelocity * radialVelocityFunction_->evaluateCompositeFunctionCurrentTime( currentTime );
510 
511  // Compute axial thrust acceleration.
512  thrustAccelerationComponents[ 2 ] = axialVelocityFunction_->evaluateCompositeFunctionDerivativeCurrentTime( currentTime )
513  + centralBodyGravitationalParameter_ / std::pow( distanceFromCentralBody, 3.0 ) * axialDistance;
514 
515  // Return total thrust acceleration.
516  return thrustAccelerationComponents;
517 }

530 Eigen::Vector3d HodographicShaping::computeThrustAccelerationVector( double currentTime )
531 {
532  if( thrustAccelerationVectorCache_.count( currentTime ) == 0 )
533  {
534  Eigen::Vector3d cylindricalAcceleration = computeThrustAccelerationInCylindricalCoordinates( currentTime );
535  Eigen::Vector3d cylindricalState = computeStateVectorInCylindricalCoordinates( currentTime ).segment(0,3);
