71 template< typename StateScalarType >
72 Eigen::Matrix< StateScalarType, 6, 1 > transformStateToFrameFromRotations(
73  const Eigen::Matrix< StateScalarType, 6, 1 >& stateInBaseFrame,
74  const Eigen::Quaterniond& rotationToFrame,
75  const Eigen::Matrix3d& rotationMatrixToFrameDerivative )
76 {
77  Eigen::Matrix< StateScalarType, 6, 1 >stateInTargetFrame;
78  stateInTargetFrame.segment( 0, 3 ) =
79  rotationToFrame.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 0, 3 );
80  stateInTargetFrame.segment( 3, 3 ) =
81  rotationMatrixToFrameDerivative.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 0, 3 ) +
82  rotationToFrame.template cast< StateScalarType >( ) * stateInBaseFrame.segment( 3, 3 );
83  return stateInTargetFrame;
84 }
