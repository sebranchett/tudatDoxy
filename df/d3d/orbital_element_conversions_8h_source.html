49 template< typename ScalarType = double >
50 ScalarType computeSemiLatusRectum(
51  const ScalarType eccentricity,
52  const ScalarType semiMajorAxis,
53  const ScalarType tolerance )
54 {
55  // Declare semi-latus rectum.
56  ScalarType semiLatusRectum;
57 
58  // Compute semi-latus rectum in the case it is not a parabola.
59  if ( std::fabs( eccentricity - mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ) >
60  tolerance )
61  {
62  semiLatusRectum = semiMajorAxis * (
63  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
64  eccentricity * eccentricity );
65  }
66 
67  // Else set the semi-latus rectum given for a parabola as the first element in the vector
68  // of Keplerian elements..
69  else
70  {
71  semiLatusRectum = semiMajorAxis;
72  }
73  return semiLatusRectum;
74 }

125 template< typename ScalarType = double >
126 Eigen::Matrix< ScalarType, 6, 1 > convertKeplerianToCartesianElements(
127  const Eigen::Matrix< ScalarType, 6, 1 >& keplerianElements,
128  const ScalarType centralBodyGravitationalParameter )
129 {
130 
131  // Set tolerance.
132  const ScalarType tolerance_ = std::numeric_limits< ScalarType >::epsilon( );
133 
134  // Set local keplerian elements.
135  ScalarType semiMajorAxis_ = keplerianElements( semiMajorAxisIndex );
136  ScalarType eccentricity_ = keplerianElements( eccentricityIndex );
137  ScalarType inclination_ = keplerianElements( inclinationIndex );
138  ScalarType argumentOfPeriapsis_ = keplerianElements( argumentOfPeriapsisIndex );
139  ScalarType longitudeOfAscendingNode_ = keplerianElements( longitudeOfAscendingNodeIndex );
140  ScalarType trueAnomaly_ = keplerianElements( trueAnomalyIndex );
141 
142  // Pre-compute sines and cosines of involved angles for efficient computation.
143  ScalarType cosineOfInclination_ = std::cos( inclination_ );
144  ScalarType sineOfInclination_ = std::sin( inclination_ );
145  ScalarType cosineOfArgumentOfPeriapsis_ = std::cos( argumentOfPeriapsis_ );
146  ScalarType sineOfArgumentOfPeriapsis_ = std::sin( argumentOfPeriapsis_ );
147  ScalarType cosineOfLongitudeOfAscendingNode_ = std::cos( longitudeOfAscendingNode_ );
148  ScalarType sineOfLongitudeOfAscendingNode_ = std::sin( longitudeOfAscendingNode_ );
149  ScalarType cosineOfTrueAnomaly_ = std::cos( trueAnomaly_ );
150  ScalarType sineOfTrueAnomaly_ = std::sin( trueAnomaly_ );
151 
152  // Declare semi-latus rectum.
153  ScalarType semiLatusRectum_ = computeSemiLatusRectum< ScalarType >( eccentricity_, semiMajorAxis_, tolerance_ );
154 
155  // Definition of position in the perifocal coordinate system.
156  Eigen::Matrix< ScalarType, 2, 1 > positionPerifocal_;
157  positionPerifocal_.x( ) = semiLatusRectum_ * cosineOfTrueAnomaly_
158  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
159  eccentricity_ * cosineOfTrueAnomaly_ );
160  positionPerifocal_.y( ) = semiLatusRectum_ * sineOfTrueAnomaly_
161  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
162  eccentricity_ * cosineOfTrueAnomaly_ );
163 
164  // Definition of velocity in the perifocal coordinate system.
165  Eigen::Matrix< ScalarType, 2, 1 > velocityPerifocal_(
166  -std::sqrt( centralBodyGravitationalParameter / semiLatusRectum_ ) * sineOfTrueAnomaly_,
167  std::sqrt( centralBodyGravitationalParameter / semiLatusRectum_ )
168  * ( eccentricity_ + cosineOfTrueAnomaly_ ) );
169 
170  // Definition of the transformation matrix.
171  Eigen::Matrix< ScalarType, 3, 2 > transformationMatrix_;
172 
173  // Compute the transformation matrix.
174  transformationMatrix_( 0, 0 ) = cosineOfLongitudeOfAscendingNode_
175  * cosineOfArgumentOfPeriapsis_ -sineOfLongitudeOfAscendingNode_
176  * sineOfArgumentOfPeriapsis_ * cosineOfInclination_;
177  transformationMatrix_( 0, 1 ) = -cosineOfLongitudeOfAscendingNode_
178  * sineOfArgumentOfPeriapsis_ -sineOfLongitudeOfAscendingNode_
179  * cosineOfArgumentOfPeriapsis_ * cosineOfInclination_;
180  transformationMatrix_( 1, 0 ) = sineOfLongitudeOfAscendingNode_
181  * cosineOfArgumentOfPeriapsis_ + cosineOfLongitudeOfAscendingNode_
182  * sineOfArgumentOfPeriapsis_ * cosineOfInclination_;
183  transformationMatrix_( 1, 1 ) = -sineOfLongitudeOfAscendingNode_
184  * sineOfArgumentOfPeriapsis_ + cosineOfLongitudeOfAscendingNode_
185  * cosineOfArgumentOfPeriapsis_ * cosineOfInclination_;
186  transformationMatrix_( 2, 0 ) = sineOfArgumentOfPeriapsis_ * sineOfInclination_;
187  transformationMatrix_( 2, 1 ) = cosineOfArgumentOfPeriapsis_ * sineOfInclination_;
188 
189  // Declare converted Cartesian elements.
190  Eigen::Matrix< ScalarType, 6, 1 > convertedCartesianElements_;
191 
192  // Compute value of position in Cartesian coordinates.
193  Eigen::Matrix< ScalarType, 3, 1 > position_ = transformationMatrix_ * positionPerifocal_;
194  convertedCartesianElements_.segment( 0, 3 ) = position_;
195 
196  // Compute value of velocity in Cartesian coordinates.
197  Eigen::Matrix< ScalarType, 3, 1 > velocity_ = transformationMatrix_ * velocityPerifocal_;
198  convertedCartesianElements_.segment( 3, 3 ) = velocity_;
199 
200  // Return Cartesian elements.
201  return convertedCartesianElements_;
202 }

240 template< typename ScalarType = double >
241 Eigen::Matrix< ScalarType, 6, 1 > convertCartesianToKeplerianElements(
242  const Eigen::Matrix< ScalarType, 6, 1 >& cartesianElements,
243  const ScalarType centralBodyGravitationalParameter )
244 {
245  // Set tolerance.
246  const ScalarType tolerance = 20.0 * std::numeric_limits< ScalarType >::epsilon( );
247 
248  // Declare converted Keplerian elements.
249  Eigen::Matrix< ScalarType, 6, 1 > computedKeplerianElements_;
250 
251  // Set position and velocity vectors.
252  const Eigen::Matrix< ScalarType, 3, 1 > position_( cartesianElements.segment( 0, 3 ) );
253  const Eigen::Matrix< ScalarType, 3, 1 > velocity_( cartesianElements.segment( 3, 3 ) );
254 
255  // Compute orbital angular momentum vector.
256  const Eigen::Matrix< ScalarType, 3, 1 > angularMomentum_( position_.cross( velocity_ ) );
257 
258  // Compute semi-latus rectum.
259  const ScalarType semiLatusRectum_ = angularMomentum_.squaredNorm( )
260  / centralBodyGravitationalParameter;
261 
262  // Compute unit vector to ascending node.
263  Eigen::Matrix< ScalarType, 3, 1 > unitAscendingNodeVector_(
264  ( Eigen::Matrix< ScalarType, 3, 1 >::UnitZ( ).cross(
265  angularMomentum_.normalized( ) ) ).normalized( ) );
266 
267  // Compute eccentricity vector.
268  Eigen::Matrix< ScalarType, 3, 1 > eccentricityVector_(
269  velocity_.cross( angularMomentum_ ) / centralBodyGravitationalParameter
270  - position_.normalized( ) );
271 
272  // Store eccentricity.
273  computedKeplerianElements_( eccentricityIndex ) = eccentricityVector_.norm( );
274 
275  // Compute and store semi-major axis.
276  // Check if orbit is parabolic. If it is, store the semi-latus rectum instead of the
277  // semi-major axis.
278  if ( std::fabs( computedKeplerianElements_( eccentricityIndex ) -
279  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ) < tolerance )
280  {
281  computedKeplerianElements_( semiLatusRectumIndex ) = semiLatusRectum_;
282  }
283 
284  // Else the orbit is either elliptical or hyperbolic, so store the semi-major axis.
285  else
286  {
287  computedKeplerianElements_( semiMajorAxisIndex ) = semiLatusRectum_
288  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
289  computedKeplerianElements_( eccentricityIndex )
290  * computedKeplerianElements_( eccentricityIndex ) );
291  }
292 
293  // Compute and store inclination.
294  computedKeplerianElements_( inclinationIndex ) = std::acos( angularMomentum_.z( )
295  / angularMomentum_.norm( ) );
296 
297  // Compute and store longitude of ascending node.
298  // Define the quadrant condition for the argument of perigee.
299  ScalarType argumentOfPeriapsisQuandrantCondition = eccentricityVector_.z( );
300 
301  // Check if the orbit is equatorial. If it is, set the vector to the line of nodes to the
302  // x-axis.
303  if ( std::fabs( computedKeplerianElements_( inclinationIndex ) ) < tolerance )
304  {
305  unitAscendingNodeVector_ = Eigen::Matrix< ScalarType, 3, 1 >::UnitX( );
306 
307  // If the orbit is equatorial, eccentricityVector_.z( ) is zero, therefore the quadrant
308  // condition is taken to be the y-component, eccentricityVector_.y( ).
309  argumentOfPeriapsisQuandrantCondition = eccentricityVector_.y( );
310  }
311 
312  // Compute and store the resulting longitude of ascending node.
313  computedKeplerianElements_( longitudeOfAscendingNodeIndex )
314  = std::acos( unitAscendingNodeVector_.x( ) );
315 
316  // Check if the quandrant is correct.
317  if ( unitAscendingNodeVector_.y( ) <
318  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
319  {
320  computedKeplerianElements_( longitudeOfAscendingNodeIndex ) =
321  mathematical_constants::getFloatingInteger< ScalarType >( 2 ) *
322  mathematical_constants::getPi< ScalarType >( ) -
323  computedKeplerianElements_( longitudeOfAscendingNodeIndex );
324  }
325 
326  // Compute and store argument of periapsis.
327  // Define the quadrant condition for the true anomaly.
328  ScalarType trueAnomalyQuandrantCondition = position_.dot( velocity_ );
329 
330  // Check if the orbit is circular. If it is, set the eccentricity vector to unit vector
331  // pointing to the ascending node, i.e. set the argument of periapsis to zero.
332  if ( std::fabs( computedKeplerianElements_( eccentricityIndex ) ) < tolerance )
333  {
334  eccentricityVector_ = unitAscendingNodeVector_;
335 
336  computedKeplerianElements_( argumentOfPeriapsisIndex ) =
337  mathematical_constants::getFloatingInteger< ScalarType >( 0 );
338 
339  // Check if orbit is also equatorial and set true anomaly quandrant check condition
340  // accordingly.
341  if ( unitAscendingNodeVector_ == Eigen::Matrix< ScalarType, 3, 1 >::UnitX( ) )
342  {
343  // If the orbit is circular, position_.dot( velocity_ ) = 0, therefore this value
344  // cannot be used as a quadrant condition. Moreover, if the orbit is equatorial,
345  // position_.z( ) is also zero and therefore the quadrant condition is taken to be the
346  // y-component, position_.y( ).
347  trueAnomalyQuandrantCondition = position_.y( );
348  }
349 
350  else
351  {
352  // If the orbit is circular, position_.dot( velocity_ ) = 0, therefore the quadrant
353  // condition is taken to be the z-component of the position, position_.z( ).
354  trueAnomalyQuandrantCondition = position_.z( );
355  }
356  }
357 
358  // Else, compute the argument of periapsis as the angle between the eccentricity vector and
359  // the unit vector to the ascending node.
360  else
361  {
362  ScalarType eccentricityAscendingNodeDotProduct = eccentricityVector_.normalized( ).dot( unitAscendingNodeVector_ );
363 
364  // Check whether dot product is in bounds (might be out of bounds due to numerical noise).
365  if( eccentricityAscendingNodeDotProduct < mathematical_constants::getFloatingInteger< ScalarType >( -1 ) )
366  {
367  computedKeplerianElements_( argumentOfPeriapsisIndex ) = mathematical_constants::getPi< ScalarType >( );
368  }
369  else if( eccentricityAscendingNodeDotProduct > mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
370  {
371  computedKeplerianElements_( argumentOfPeriapsisIndex ) =
372  mathematical_constants::getFloatingInteger< ScalarType >( 0 );
373  }
374  else
375  {
376  computedKeplerianElements_( argumentOfPeriapsisIndex ) = std::acos( eccentricityAscendingNodeDotProduct );
377  }
378 
379  // Check if the quadrant is correct.
380  if ( argumentOfPeriapsisQuandrantCondition <
381  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
382  {
383  computedKeplerianElements_( argumentOfPeriapsisIndex ) =
384  mathematical_constants::getFloatingInteger< ScalarType >( 2 ) *
385  mathematical_constants::getPi< ScalarType >( ) -
386  computedKeplerianElements_( argumentOfPeriapsisIndex );
387  }
388  }
389 
390  // Compute dot-product of position and eccentricity vectors.
391  ScalarType dotProductPositionAndEccentricityVectors
392  = position_.normalized( ).dot( eccentricityVector_.normalized( ) );
393 
394  // Check if the dot-product is one of the limiting cases: 0.0, -1.0 or 1.0
395  // (within prescribed tolerance).
396  if ( std::fabs( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
397  dotProductPositionAndEccentricityVectors ) < tolerance )
398  {
399  dotProductPositionAndEccentricityVectors =
400  mathematical_constants::getFloatingInteger< ScalarType >( 1 );
401  }
402 
403  if ( std::fabs( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
404  dotProductPositionAndEccentricityVectors ) < tolerance )
405  {
406  dotProductPositionAndEccentricityVectors =
407  -mathematical_constants::getFloatingInteger< ScalarType >( 1 );
408  }
409 
410  if ( std::fabs( dotProductPositionAndEccentricityVectors ) < tolerance )
411  {
412  dotProductPositionAndEccentricityVectors =
413  mathematical_constants::getFloatingInteger< ScalarType >( 0 );
414  }
415 
416  // Compute and store true anomaly.
417  computedKeplerianElements_( trueAnomalyIndex )
418  = std::acos( dotProductPositionAndEccentricityVectors );
419 
420  // Check if the quandrant is correct.
421  if ( trueAnomalyQuandrantCondition <
422  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
423  {
424  computedKeplerianElements_( trueAnomalyIndex ) =
425  mathematical_constants::getFloatingInteger< ScalarType >( 2 ) *
426  mathematical_constants::getPi< ScalarType >( ) -
427  computedKeplerianElements_( trueAnomalyIndex );
428  }
429 
430  // Return converted Keplerian elements.
431  return computedKeplerianElements_;
432 }

460 template< typename ScalarType = double >
461 ScalarType convertTrueAnomalyToEllipticalEccentricAnomaly(
462  const ScalarType trueAnomaly, const ScalarType eccentricity )
463 
464 {
465  if ( eccentricity >= mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ||
466  eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
467  {
468  throw std::runtime_error( "Eccentricity is invalid when converting true to elliptical eccentric anomaly." );
469  }
470  else
471  {
472  // Declare and compute sine and cosine of eccentric anomaly.
473  ScalarType sineOfEccentricAnomaly_ =
474  std::sqrt( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
475  eccentricity * eccentricity ) * std::sin( trueAnomaly ) /
476  ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
477  eccentricity * std::cos( trueAnomaly ) );
478  ScalarType cosineOfEccentricAnomaly_ = ( eccentricity + std::cos( trueAnomaly ) )
479  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
480  eccentricity * std::cos( trueAnomaly ) );
481 
482  // Return elliptic eccentric anomaly.
483  return std::atan2( sineOfEccentricAnomaly_, cosineOfEccentricAnomaly_ );
484  }
485 }

495 template< typename ScalarType = double >
496 ScalarType convertTrueAnomalyToHyperbolicEccentricAnomaly( const ScalarType trueAnomaly,
497  const ScalarType eccentricity )
498 {
499  if ( eccentricity <= mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
500  {
501  throw std::runtime_error( "Eccentricity is invalid when converting true to hyperbolic eccentric anomaly." );
502  }
503 
504  else
505  {
506  // Compute hyperbolic sine and hyperbolic cosine of hyperbolic eccentric anomaly.
507  ScalarType hyperbolicSineOfHyperbolicEccentricAnomaly_
508  = std::sqrt( eccentricity * eccentricity -
509  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
510  * std::sin( trueAnomaly ) /
511  ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
512  std::cos( trueAnomaly ) );
513 
514  ScalarType hyperboliccosineOfHyperbolicEccentricAnomaly_
515  = ( std::cos( trueAnomaly ) + eccentricity ) /
516  ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) +
517  std::cos( trueAnomaly ) );
518 
519  // Return hyperbolic eccentric anomaly.
520  return boost::math::atanh( hyperbolicSineOfHyperbolicEccentricAnomaly_
521  / hyperboliccosineOfHyperbolicEccentricAnomaly_ );
522  }

538 template< typename ScalarType = double >
539 ScalarType convertTrueAnomalyToEccentricAnomaly( const ScalarType trueAnomaly,
540  const ScalarType eccentricity )
541 {
542  // Declare computed eccentric anomaly.
543  ScalarType eccentricAnomaly_ = 0.0;
544 
545  // Check if eccentricity is invalid and throw an error if true.
546  if ( eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
547  {
548  throw std::runtime_error( "Eccentricity is invalid when converting true to eccentric anomaly." );
549  }
550 
551  // Check if orbit is parabolic and throw an error if true.
552  else if ( std::fabs( eccentricity -
553  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ) <
554  std::numeric_limits< ScalarType >::epsilon( ) )
555  {
556  throw std::runtime_error( "Parabolic orbits have not yet been implemented when converting true to eccentric anomaly." );
557  }
558 
559  // Check if orbit is elliptical and compute eccentric anomaly.
560  else if ( eccentricity >= mathematical_constants::getFloatingInteger< ScalarType >( 0 ) &&
561  eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
562  {
563  eccentricAnomaly_ = convertTrueAnomalyToEllipticalEccentricAnomaly< ScalarType >(
564  trueAnomaly, eccentricity );
565  }
566 
567  else if ( eccentricity > mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
568  {
569  eccentricAnomaly_ = convertTrueAnomalyToHyperbolicEccentricAnomaly< ScalarType >(
570  trueAnomaly, eccentricity );
571  }
572 
573  // Return computed eccentric anomaly.
574  return eccentricAnomaly_;
575 }

585 template< typename ScalarType = double >
586 ScalarType convertEllipticalEccentricAnomalyToTrueAnomaly(
587  const ScalarType ellipticEccentricAnomaly,
588  const ScalarType eccentricity )
589 {
590  if ( eccentricity >= mathematical_constants::getFloatingInteger< ScalarType >( 1 ) ||
591  eccentricity < mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
592  {
593  throw std::runtime_error( "Eccentricity is invalid when converting elliptical eccentric to true anomaly." );
594  }
595 
596  else
597  {
598 
599  // Compute sine and cosine of true anomaly.
600  ScalarType sineOfTrueAnomaly_ =
601  std::sqrt( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
602  eccentricity * eccentricity ) *
603  std::sin( ellipticEccentricAnomaly )
604  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
605  eccentricity * std::cos( ellipticEccentricAnomaly ) );
606 
607  ScalarType cosineOfTrueAnomaly_ = ( std::cos( ellipticEccentricAnomaly ) - eccentricity )
608  / ( mathematical_constants::getFloatingInteger< ScalarType >( 1 ) -
609  eccentricity * std::cos( ellipticEccentricAnomaly ) );
610 
611  // Return true anomaly.
612  return std::atan2( sineOfTrueAnomaly_, cosineOfTrueAnomaly_ );
613  }
