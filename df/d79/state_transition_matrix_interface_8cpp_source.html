35 Eigen::MatrixXd SingleArcCombinedStateTransitionAndSensitivityMatrixInterface::getCombinedStateTransitionAndSensitivityMatrix(
36  const double evaluationTime )
37 {
38  combinedStateTransitionMatrix_.setZero( );
39 
40 
41  // Set Phi and S matrices.
42  combinedStateTransitionMatrix_.block( 0, 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_ ) =
43  stateTransitionMatrixInterpolator_->interpolate( evaluationTime );
44 
45  if( sensitivityMatrixSize_ > 0 )
46  {
47  combinedStateTransitionMatrix_.block( 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_, sensitivityMatrixSize_ ) =
48  sensitivityMatrixInterpolator_->interpolate( evaluationTime );
49  }
50 
51  for( unsigned int i = 0; i < statePartialAdditionIndices_.size( ); i++ )
52  {
53  combinedStateTransitionMatrix_.block( statePartialAdditionIndices_.at( i ).first, 0, 6, stateTransitionMatrixSize_ + sensitivityMatrixSize_ ) +=
54  combinedStateTransitionMatrix_.block( statePartialAdditionIndices_.at( i ).second, 0, 6, stateTransitionMatrixSize_ + sensitivityMatrixSize_ );
55  }
56 
57 
58  return combinedStateTransitionMatrix_;
59 }

63  const std::vector< std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::MatrixXd > > >
64  stateTransitionMatrixInterpolators,
65  const std::vector< std::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::MatrixXd > > >
66  sensitivityMatrixInterpolators,
67  const std::vector< double >& arcStartTimes,
68  const std::vector< double >& arcEndTimes,
69  const int numberOfInitialDynamicalParameters,

88  stateTransitionMatrixInterpolators_.size( ) != static_cast< unsigned int >( numberOfStateArcs_ ) )
89  {
90  throw std::runtime_error(

92  }
93 
94  std::vector< double > arcSplitTimes = arcStartTimes_;
95  arcSplitTimes.push_back( std::numeric_limits< double >::max( ));
96  lookUpscheme_ = std::make_shared< interpolators::HuntingAlgorithmLookupScheme< double > >(

131 Eigen::MatrixXd MultiArcCombinedStateTransitionAndSensitivityMatrixInterface::getCombinedStateTransitionAndSensitivityMatrix(
132  const double evaluationTime,
133  const bool addCentralBodySensitivity )
134 {
135  Eigen::MatrixXd combinedStateTransitionMatrix = Eigen::MatrixXd::Zero(
136  stateTransitionMatrixSize_, stateTransitionMatrixSize_ + sensitivityMatrixSize_ );
137 
138  int currentArc = getCurrentArc( evaluationTime ).first;
139 
140  // Set Phi and S matrices.
141  if( currentArc >= 0 )
142  {
143  combinedStateTransitionMatrix.block( 0, 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_ ) =
144  stateTransitionMatrixInterpolators_.at( currentArc )->interpolate( evaluationTime );
145  combinedStateTransitionMatrix.block( 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_, sensitivityMatrixSize_ ) =
146  sensitivityMatrixInterpolators_.at( currentArc )->interpolate( evaluationTime );
147 
148  for( unsigned int i = 0; i < statePartialAdditionIndices_.at( currentArc ).size( ); i++ )
149  {
150  int indicesToAdd = addCentralBodySensitivity ?
151  ( stateTransitionMatrixSize_ + sensitivityMatrixSize_ ) : stateTransitionMatrixSize_;
152  combinedStateTransitionMatrix.block(
153  statePartialAdditionIndices_.at( currentArc ).at( i ).first, 0, 6, indicesToAdd ) +=
154  combinedStateTransitionMatrix.block(
155  statePartialAdditionIndices_.at( currentArc ).at( i ).second, 0, 6, indicesToAdd );
156  }
157  }
158  return combinedStateTransitionMatrix;
159 }

170  const double evaluationTime,

174  evaluationTime, addCentralBodySensitivity );

176  stateTransitionMatrixSize_, numberOfStateArcs_ * stateTransitionMatrixSize_ + sensitivityMatrixSize_ );

179 
180  // Set Phi and S matrices of current arc.

185  stateTransitionMatrixInterpolators_.at( currentArc )->interpolate( evaluationTime );
186  fullCombinedStateTransitionMatrix.block(
187  0, numberOfStateArcs_ * stateTransitionMatrixSize_, stateTransitionMatrixSize_, sensitivityMatrixSize_ ) =
188  combinedStateTransitionMatrix.block( 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_, sensitivityMatrixSize_ );
189  }

216 Eigen::MatrixXd HybridArcCombinedStateTransitionAndSensitivityMatrixInterface::getCombinedStateTransitionAndSensitivityMatrix(
217  const double evaluationTime )
218 {
219  Eigen::MatrixXd combinedStateTransitionMatrix = Eigen::MatrixXd::Zero(
220  stateTransitionMatrixSize_, stateTransitionMatrixSize_ + sensitivityMatrixSize_ );
221 
222  // Get single-arc matrices
223  Eigen::MatrixXd singleArcStateTransition = singleArcInterface_->getCombinedStateTransitionAndSensitivityMatrix(
224  evaluationTime );
225 
226  // Get multi-arc matrices
227  Eigen::MatrixXd multiArcStateTransition = multiArcInterface_->getCombinedStateTransitionAndSensitivityMatrix(
228  evaluationTime, false );
229  std::pair< int, double > currentArc = multiArcInterface_->getCurrentArc( evaluationTime );
230 
231  // Set single-arc block
232  combinedStateTransitionMatrix.block( 0, 0, singleArcStateSize_, singleArcStateSize_ ) =
233  singleArcStateTransition.block( 0, 0, singleArcStateSize_, singleArcStateSize_ );
234 
235  // Set single-arc sensitivity block
236  combinedStateTransitionMatrix.block( 0, multiArcStateSize_, singleArcStateSize_, sensitivityMatrixSize_ ) =
237  singleArcStateTransition.block( 0, singleArcStateSize_, singleArcStateSize_, sensitivityMatrixSize_ );
238 
239  if( !( currentArc.first < 0 ) )
240  {
241  // Set multi-arc block
242  combinedStateTransitionMatrix.block(
243  singleArcStateSize_, singleArcStateSize_, originalMultiArcStateSize_, originalMultiArcStateSize_ ) =
244  multiArcStateTransition.block(
245  singleArcStateSize_, singleArcStateSize_, originalMultiArcStateSize_, originalMultiArcStateSize_ );
246 
247  // Get single-arc matrices at current arc start
248  Eigen::MatrixXd singleArcStateTransitionAtArcStart = singleArcInterface_->getCombinedStateTransitionAndSensitivityMatrix(
249  currentArc.second );
250 
251  // Set coupled block
252  combinedStateTransitionMatrix.block(
253  singleArcStateSize_, 0, originalMultiArcStateSize_, singleArcStateSize_ ) =
254  multiArcStateTransition.block(
255  singleArcStateSize_, 0, originalMultiArcStateSize_, singleArcStateSize_ ) *
256  singleArcStateTransitionAtArcStart.block(
257  0, 0, singleArcStateSize_, singleArcStateSize_ );
258 
259  // Set multi-arc sensitivity block
260  combinedStateTransitionMatrix.block(
261  singleArcStateSize_, multiArcStateSize_, originalMultiArcStateSize_, sensitivityMatrixSize_ ) =
262  multiArcStateTransition.block(
263  singleArcStateSize_, multiArcStateSize_, originalMultiArcStateSize_, sensitivityMatrixSize_ );
264 
265  std::vector< std::pair< int, int > > statePartialAdditionIndices =
266  multiArcInterface_->getStatePartialAdditionIndices( currentArc.first );
267 
268  for( unsigned int i = 0; i < statePartialAdditionIndices.size( ); i++ )
269  {
270  combinedStateTransitionMatrix.block(
271  statePartialAdditionIndices.at( i ).first, multiArcStateSize_,
272  6, sensitivityMatrixSize_ ) +=
273  combinedStateTransitionMatrix.block(
274  statePartialAdditionIndices.at( i ).second, multiArcStateSize_,
275  6, sensitivityMatrixSize_ );
276 
277  }
278  }
279 
280  return combinedStateTransitionMatrix;
281 
282 }

285 Eigen::MatrixXd HybridArcCombinedStateTransitionAndSensitivityMatrixInterface::getFullCombinedStateTransitionAndSensitivityMatrix(
286  const double evaluationTime )
287 {
288  Eigen::MatrixXd combinedStateTransitionMatrix = getCombinedStateTransitionAndSensitivityMatrix( evaluationTime );
289  Eigen::MatrixXd fullCombinedStateTransitionMatrix = Eigen::MatrixXd::Zero(
290  stateTransitionMatrixSize_, singleArcStateSize_ + numberOfMultiArcs_ * originalMultiArcStateSize_ + sensitivityMatrixSize_ );
291 
292  fullCombinedStateTransitionMatrix.block(
293  0, 0, singleArcStateSize_, singleArcStateSize_ ) =
294  combinedStateTransitionMatrix.block(
295  0, 0, singleArcStateSize_, singleArcStateSize_ );
296 
297  fullCombinedStateTransitionMatrix.block(
298  0, singleArcStateSize_ + numberOfMultiArcs_ * originalMultiArcStateSize_, singleArcStateSize_, sensitivityMatrixSize_ ) =
299  combinedStateTransitionMatrix.block(
300  0, multiArcStateSize_, singleArcStateSize_, sensitivityMatrixSize_ );
301 
302  int currentArc = multiArcInterface_->getCurrentArc( evaluationTime ).first;
303 
304  // Set Phi and S matrices of current arc.
305  if( currentArc >= 0 )
306  {
307 
308  // Set multi-arc block
309  fullCombinedStateTransitionMatrix.block(
310  singleArcStateSize_, singleArcStateSize_ + currentArc * originalMultiArcStateSize_,
311  originalMultiArcStateSize_, originalMultiArcStateSize_ ) =
312  combinedStateTransitionMatrix.block(
313  singleArcStateSize_, singleArcStateSize_, originalMultiArcStateSize_, originalMultiArcStateSize_ );
314 
315 
316  // Set coupled block
317  fullCombinedStateTransitionMatrix.block(
318  singleArcStateSize_, 0, originalMultiArcStateSize_, singleArcStateSize_ ) =
319  combinedStateTransitionMatrix.block(
320  singleArcStateSize_, 0, originalMultiArcStateSize_, singleArcStateSize_ );
321 
322  // Set multi-arc sensitivity block
323  fullCombinedStateTransitionMatrix.block(
324  singleArcStateSize_, singleArcStateSize_ + numberOfMultiArcs_ * originalMultiArcStateSize_,
325  originalMultiArcStateSize_, sensitivityMatrixSize_ ) =
326  combinedStateTransitionMatrix.block(
327  singleArcStateSize_, multiArcStateSize_, originalMultiArcStateSize_, sensitivityMatrixSize_ );
328 
329  }
330 
331  return fullCombinedStateTransitionMatrix;
332 }
