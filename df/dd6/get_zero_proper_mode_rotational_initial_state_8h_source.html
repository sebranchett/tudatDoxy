52 template< typename StateScalarType = double, typename TimeType = double >
53 void integrateForwardWithDissipationAndBackwardsWithout(
54  const std::shared_ptr< SingleArcDynamicsSimulator< StateScalarType, TimeType > > dynamicsSimulator,
55  const std::shared_ptr< basic_astrodynamics::DissipativeTorqueModel > dissipativeTorque,
56  std::pair< std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1> >,
57  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > >& propagatedStates,
58  std::pair< std::map< TimeType, Eigen::Matrix< double, Eigen::Dynamic, 1> >,
59  std::map< TimeType, Eigen::Matrix< double, Eigen::Dynamic, 1 > > >& dependentVariables )
60 {
61  using namespace tudat::numerical_integrators;
62 
63  std::shared_ptr< SingleArcPropagatorSettings< TimeType > > propagatorSettings = dynamicsSimulator->getPropagatorSettings( );
64  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1> initialState = propagatorSettings->getInitialStates( );
65  std::shared_ptr< numerical_integrators::IntegratorSettings< TimeType > > integratorSettings =
66  dynamicsSimulator->getIntegratorSettings( );
67 
68  // Integrate forward with dissipation and retrieve results
69  dynamicsSimulator->integrateEquationsOfMotion( initialState );
70  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > forwardIntegrated =
71  dynamicsSimulator->getEquationsOfMotionNumericalSolution( );
72  std::map< TimeType, Eigen::Matrix< double, Eigen::Dynamic, 1 > > forwardIntegratedDependent =
73  dynamicsSimulator->getDependentVariableHistory( );
74 
75  // Save original integration/propagation settings
76  TimeType originalStartTime = integratorSettings->initialTime_;
77  TimeType originalEndTime = std::dynamic_pointer_cast< PropagationTimeTerminationSettings >(
78  propagatorSettings->getTerminationSettings( ) )->terminationTime_;
79  double originalTimeStep = integratorSettings->initialTimeStep_;
80 
81  // Turn off dissipation
82  dissipativeTorque->setDampingMatrixFunction( Eigen::Matrix3d::Zero( ) );
83 
84  // Reset propagation/integration settings for backwards propagation
85  auto outputMapIterator = forwardIntegrated.rbegin( );
86  integratorSettings->initialTime_ = outputMapIterator->first;
87  dynamicsSimulator->resetInitialPropagationTime( outputMapIterator->first );
88  propagatorSettings->resetTerminationSettings(
89  std::make_shared< PropagationTimeTerminationSettings >( originalStartTime ) );
90  integratorSettings->initialTimeStep_ = -originalTimeStep;
91 
92  // Integrate backward without dissipation and retrieve results
93  dynamicsSimulator->integrateEquationsOfMotion(
94  outputMapIterator->second );
95  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1> > backwardIntegrated =
96  dynamicsSimulator->getEquationsOfMotionNumericalSolution( );
97  std::map< TimeType, Eigen::Matrix< double, Eigen::Dynamic, 1 > > backwardIntegratedDependent =
98  dynamicsSimulator->getDependentVariableHistory( );
99 
100  // Reset original integration/propagation settings
101  integratorSettings->initialTime_ = originalStartTime;
102  dynamicsSimulator->resetInitialPropagationTime( originalStartTime );
103  propagatorSettings->resetTerminationSettings(
104  std::make_shared< PropagationTimeTerminationSettings >( originalEndTime ) );
105  integratorSettings->initialTimeStep_ = originalTimeStep;
106 
107  // Save states and dependent variables for both forward and backward integration
108  propagatedStates = std::make_pair( forwardIntegrated, backwardIntegrated );
109  dependentVariables = std::make_pair( forwardIntegratedDependent, backwardIntegratedDependent );
110 }

135 template< typename TimeType, typename StateScalarType >
136 Eigen::VectorXd getZeroProperModeRotationalState(
137  const simulation_setup::NamedBodyMap& bodyMap,
138  const std::shared_ptr< numerical_integrators::IntegratorSettings< TimeType > > integratorSettings,
139  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
140  const double bodyMeanRotationRate,
141  const std::vector< double > dissipationTimes,
142  std::vector< std::pair< std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >,
143  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > > >& propagatedStates,
144  std::vector< std::pair< std::map< TimeType, Eigen::Matrix< double, Eigen::Dynamic, 1 > >,
145  std::map< TimeType, Eigen::Matrix< double, Eigen::Dynamic, 1 > > > >& dependentVariables,
146  const bool propagateNominal = true,
147  const bool writeToFileInLoop = false,
148  const std::string baseFileName = "",
149  const std::string outputFolder = input_output::getTudatRootPath( ) )
150 {
151  propagatedStates.resize( dissipationTimes.size( ) + 1 );
152  dependentVariables.resize( dissipationTimes.size( ) + 1 );
153 
154  basic_astrodynamics::TorqueModelMap torqueModelMap;
155 
156  // Find rotational propagator settings
157  std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationPropagationSettings_ =
158  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >(
159  propagatorSettings );
160  if( rotationPropagationSettings_ == nullptr )
161  {
162  // Retrieve rotational propagator settings from multi-type settings
163  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
164  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
165  if( multiTypePropagatorSettings != nullptr )
166  {
167  // Retrieve rotational propagator settings
168  std::map< IntegratedStateType, std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >
169  propagatorSettingsMap = multiTypePropagatorSettings->propagatorSettingsMap_;
170  if( propagatorSettingsMap.count( rotational_state ) == 0 )
171  {
172  throw std::runtime_error( "Error when finding initial rotational state, no rotational dynamics in multi-type list" );
173  }
174  else if( propagatorSettingsMap.at( rotational_state ).size( ) != 1 )
175  {
176  throw std::runtime_error( "Error when finding initial rotational state, multiple rotational dynamics in multi-type list" );
177  }
178  else
179  {
180  rotationPropagationSettings_ = std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >(
181  propagatorSettingsMap.at( rotational_state ).at( 0 ) );
182  }
183  }
184  else
185  {
186  throw std::runtime_error( "Error when finding initial rotational state, no propagator settings found" );
187  }
188  }
189 
190  // Get torque map
191  torqueModelMap = rotationPropagationSettings_->getTorqueModelsMap( );
192  if( torqueModelMap.size( ) != 1 )
193  {
194  std::cerr<<"Error when finding initial rotational state, "<<torqueModelMap.size( )<<" bodies are propagated."<<std::endl;
195  }
196 
197  //Retrieve body inertia tensor and create synthetic dissipation model
198  Eigen::Matrix3d inertiaTensor = bodyMap.at( torqueModelMap.begin( )->first )->getBodyInertiaTensor( );
199  std::shared_ptr< basic_astrodynamics::DissipativeTorqueModel > dissipativeTorque =
200  std::make_shared< basic_astrodynamics::DissipativeTorqueModel >(
201  std::bind( &simulation_setup::Body::getCurrentAngularVelocityVectorInLocalFrame,
202  bodyMap.at( torqueModelMap.begin( )->first ) ),
203  [ = ]( ){ return Eigen::Matrix3d::Zero( ); },
204  bodyMeanRotationRate );
205  torqueModelMap[ torqueModelMap.begin( )->first ][ torqueModelMap.begin( )->first ].push_back(
206  dissipativeTorque );
207  rotationPropagationSettings_->resetTorqueModelsMap( torqueModelMap );
208 
209  // Create object to propagate dynamics
210  std::shared_ptr< SingleArcDynamicsSimulator< StateScalarType, TimeType > > dynamicsSimulator =
211  std::make_shared< SingleArcDynamicsSimulator< StateScalarType, TimeType > >(
212  bodyMap, integratorSettings, propagatorSettings, 0, 0, 0 );
213 
214  // Propagate nominal dynamics forward and backward (without dissipation) if required
215  if( propagateNominal )
216  {
217  integrateForwardWithDissipationAndBackwardsWithout< StateScalarType, TimeType >(
218  dynamicsSimulator, dissipativeTorque, propagatedStates.at( 0 ), dependentVariables.at( 0 ) );
219 
220  // Write data to files if required
221  int i = 0;
222  if( writeToFileInLoop )
223  {
224  input_output::writeDataMapToTextFile(
225  dependentVariables.at( i ).second,
226  baseFileName + "_dependent_damped_" + std::to_string( i ) + ".dat", outputFolder );
227  input_output::writeDataMapToTextFile(
228  dependentVariables.at( i ).first,
229  baseFileName + "_dependent_damped_forward_" + std::to_string( i ) + ".dat", outputFolder );
230  input_output::writeDataMapToTextFile(
231  propagatedStates.at( i ).second,
232  baseFileName + "_states_damped_" + std::to_string( i ) + ".dat", outputFolder );
233  input_output::writeDataMapToTextFile(
234  propagatedStates.at( i ).first,
235  baseFileName + "_states_damped_forward_" + std::to_string( i ) + ".dat", outputFolder );
236  propagatedStates[ 0 ].first.clear( );
237  dependentVariables[ 0 ].first.clear( );
238 
239  propagatedStates[ 0 ].second.clear( );
240  dependentVariables[ 0 ].second.clear( );
241  }
242 
243  }
244 
245  // Retrieve original initial state
246  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > currentInitialState =
247  propagatorSettings->getInitialStates( );
248 
249  double newFinalTime;
250  for( unsigned int i = 0; i < dissipationTimes.size( ); i++ )
251  {
252  std::cout<<"Getting zero proper mode, iteration "<<i<<std::endl;
253 
254  // Set damping for current iteration
255  dissipativeTorque->setDampingMatrixFunction(
256  getDissipationMatrix( dissipationTimes.at( i ), inertiaTensor ) );
257 
258  // Reset final time (propagate for 10 times the dissipation time)
259  newFinalTime = integratorSettings->initialTime_ + 10.0 * dissipationTimes.at( i );
260  propagatorSettings->resetTerminationSettings(
261  std::make_shared< PropagationTimeTerminationSettings >( newFinalTime ) );
262 
263  // Propagate forward (with dissipation) and backward (without dissipation)
264  integrateForwardWithDissipationAndBackwardsWithout< StateScalarType, TimeType >(
265  dynamicsSimulator, dissipativeTorque, propagatedStates.at( i + 1 ), dependentVariables.at( i + 1 ) );
266 
267  // Update initial state to current damped result
268  currentInitialState = propagatedStates.at( i + 1 ).second.begin( )->second;
269  propagatorSettings->resetInitialStates( currentInitialState );
270 
271  // Write data to files if required
272  if( writeToFileInLoop )
273  {
274  input_output::writeDataMapToTextFile(
275  dependentVariables.at(i + 1).second,
276  baseFileName + "_dependent_damped_" + std::to_string(i + 1) + ".dat", outputFolder );
277  input_output::writeDataMapToTextFile(
278  dependentVariables.at(i + 1).first,
279  baseFileName + "_dependent_damped_forward_" + std::to_string(i + 1) + ".dat", outputFolder );
280  input_output::writeDataMapToTextFile(
281  propagatedStates.at(i + 1).second,
282  baseFileName + "_states_damped_" + std::to_string(i + 1) + ".dat", outputFolder );
283  input_output::writeDataMapToTextFile(
284  propagatedStates.at(i + 1).first,
285  baseFileName + "_states_damped_forward_" + std::to_string(i + 1) + ".dat", outputFolder );
286  propagatedStates[ i + 1 ].first.clear( );
287  dependentVariables[ i + 1 ].first.clear( );
288 
289  propagatedStates[ i + 1 ].second.clear( );
290  dependentVariables[ i + 1 ].second.clear( );
291  }
292 
293 
294  }
295 
296  // Return damped initial state that is computed
297  return currentInitialState;
298 }
