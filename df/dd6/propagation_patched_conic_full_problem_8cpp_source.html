569  double halvedTimeOfFlight = timeOfFlight / 2.0;

575  Eigen::Vector3d cartesianPositionAtDepartureForPatchedConics;
576  if( cartesianPositionAtDeparture != cartesianPositionAtDeparture )

586  bodyMap.at( departureAndArrivalBodies.at( 0 ) )->getEphemeris( )->getCartesianState( initialTime );

593  }

604  cartesianStateAtDeparture, gravitationalParameterCentralBody );
605 
606  // Propagate the keplerian elements until half of the time of flight.

608  halvedTimeOfFlight, gravitationalParameterCentralBody );
609

612  keplerianStateAtHalvedTimeOfFlight, gravitationalParameterCentralBody );

652  gravitationalParameterCentralBody ), gravitationalParameterCentralBody );

654  fullProblemResult[ itr->first ] = itr->second;
655  dependentVariables[ itr->first ] = dependentVariableHistoryFullProblemForwardPropagation[ itr->first ];
656  }

665  dynamicsSimulatorIntegrationBackwards.getEquationsOfMotionNumericalSolution( );

667  dynamicsSimulatorIntegrationBackwards.getDependentVariableHistory( );

670  for( std::map< double, Eigen::VectorXd >::iterator itr = stateHistoryFullProblemBackwardPropagation.begin( );
671  itr != stateHistoryFullProblemBackwardPropagation.end( ); itr++ )
672  {
673  cartesianStateKeplerianOrbit = orbital_element_conversions::convertKeplerianToCartesianElements(
674  orbital_element_conversions::propagateKeplerOrbit(
675  orbital_element_conversions::convertCartesianToKeplerianElements(
676  initialStatePropagation, gravitationalParameterCentralBody ),
677  - ( initialTime + halvedTimeOfFlight) + itr->first,
678  gravitationalParameterCentralBody ), gravitationalParameterCentralBody );
679 
680  keplerianOrbitResult[ itr->first ] = cartesianStateKeplerianOrbit;
681  fullProblemResult[ itr->first ] = itr->second;
682  dependentVariables[ itr->first ] = dependentVariableHistoryFullProblemBackwardsPropagation[ itr->first ];
683 
684  }

687  integratorSettings->initialTimeStep_ = -integratorSettings->initialTimeStep_;
