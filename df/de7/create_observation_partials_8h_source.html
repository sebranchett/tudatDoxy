43 template< typename ObservationScalarType, typename TimeType, int ObservationSize >

46  ObservationSize, ObservationScalarType, TimeType> > > observationModels )

48  PerLinkEndPerLightTimeSolutionCorrections lightTimeCorrectionsList;
49  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > currentLightTimeCorrections;
50 
51  // Retrieve type of observable
52  observation_models::ObservableType observableType = observationModels.begin( )->second->getObservableType( );
53 
54  // Iterate over link ends
55  for( typename std::map< observation_models::LinkEnds, std::shared_ptr< observation_models::ObservationModel<
56  ObservationSize, ObservationScalarType, TimeType> > >::const_iterator
57  observationModelIterator = observationModels.begin( );
58  observationModelIterator != observationModels.end( ); observationModelIterator++ )
59  {
60  // Clear list, for current link ends.
61  currentLightTimeCorrections.clear( );
62 
63  if( observationModelIterator->second->getObservableType( ) != observableType )
64  {
65  throw std::runtime_error( "Error when making grouped light time correction list, observable type is not constant" );
66  }
67  else
68  {
69  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > singleObservableCorrectionList;
70 
71  // Check type of observable
72  switch( observableType )
73  {
74  case observation_models::one_way_range:
75  {
76  std::shared_ptr< observation_models::OneWayRangeObservationModel
77  < ObservationScalarType, TimeType> > oneWayRangeModel =
78  std::dynamic_pointer_cast< observation_models::OneWayRangeObservationModel
79  < ObservationScalarType, TimeType> >
80  ( observationModelIterator->second );
81  singleObservableCorrectionList = (
82  oneWayRangeModel->getLightTimeCalculator( )->getLightTimeCorrection( ) );
83  break;
84  }
85  case observation_models::one_way_doppler:
86  {
87  std::shared_ptr< observation_models::OneWayDopplerObservationModel
88  < ObservationScalarType, TimeType> > oneWayRangeModel =
89  std::dynamic_pointer_cast< observation_models::OneWayDopplerObservationModel
90  < ObservationScalarType, TimeType> >
91  ( observationModelIterator->second );
92  singleObservableCorrectionList = (
93  oneWayRangeModel->getLightTimeCalculator( )->getLightTimeCorrection( ) );
94  break;
95  }
96  case observation_models::two_way_doppler:
97  {
98  std::shared_ptr< observation_models::TwoWayDopplerObservationModel
99  < ObservationScalarType, TimeType> > twoWaDopplerModel =
100  std::dynamic_pointer_cast< observation_models::TwoWayDopplerObservationModel
101  < ObservationScalarType, TimeType> >
102  ( observationModelIterator->second );
103  currentLightTimeCorrections.push_back(
104  twoWaDopplerModel->getUplinkDopplerCalculator( )->getLightTimeCalculator( )->getLightTimeCorrection( ) );
105  currentLightTimeCorrections.push_back(
106  twoWaDopplerModel->getDownlinkDopplerCalculator( )->getLightTimeCalculator( )->getLightTimeCorrection( ) );
107  break;
108  }
109  case observation_models::angular_position:

113  std::dynamic_pointer_cast< observation_models::AngularPositionObservationModel
114  < ObservationScalarType, TimeType> >

116  singleObservableCorrectionList = (
117  angularPositionModel->getLightTimeCalculator( )->getLightTimeCorrection( ) );
118  break;
119  }
120  case observation_models::one_way_differenced_range:
121  {
122  std::shared_ptr< observation_models::OneWayDifferencedRangeObservationModel

124  std::dynamic_pointer_cast< observation_models::OneWayDifferencedRangeObservationModel
125  < ObservationScalarType, TimeType> >
126  ( observationModelIterator->second );
127  currentLightTimeCorrections.push_back(
128  oneWayDifferencedRangeObservationModel->getArcStartLightTimeCalculator( )->
129  getLightTimeCorrection( ) );
130  currentLightTimeCorrections.push_back(
131  oneWayDifferencedRangeObservationModel->getArcEndLightTimeCalculator( )->
132  getLightTimeCorrection( ) );
133 
134  break;
135  }
136  case observation_models::n_way_range:

139  std::dynamic_pointer_cast< observation_models::NWayRangeObservationModel< ObservationScalarType, TimeType > >
140  ( observationModelIterator->second );
141  std::vector< std::shared_ptr< observation_models::LightTimeCalculator< ObservationScalarType, TimeType > > > lightTimeCalculatorList =
142  nWayRangeObservationModel->getLightTimeCalculators( );
143  for( unsigned int i = 0; i < lightTimeCalculatorList.size( ); i++ )
144  {
145  currentLightTimeCorrections.push_back( lightTimeCalculatorList.at( i )->getLightTimeCorrection( ) );
146  }
147  break;
148  }
149  case observation_models::position_observable:
150  {
151  break;
152  }
153  case observation_models::euler_angle_313_observable:
154  {
155  break;
156  }
157  case observation_models::velocity_observable:
158  {
159  break;
160  }
161  default:
162  std::string errorMessage =
163  "Error in light time correction list creation, observable type " +
164  std::to_string( observableType ) + " not recognized.";
165  throw std::runtime_error( errorMessage );
166  }
167 
168  if( singleObservableCorrectionList.size( ) > 0 )
169  {
170  currentLightTimeCorrections.push_back( singleObservableCorrectionList );
171  }
172 
173  // Add light-time correctionsfor current link ends.
174  if( currentLightTimeCorrections.size( ) > 0 )
175  {
176  lightTimeCorrectionsList[ observationModelIterator->first ] = currentLightTimeCorrections;
177  }
178  }

180  }
