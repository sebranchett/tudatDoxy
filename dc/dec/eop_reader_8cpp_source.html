22 EOPReader::EOPReader( const std::string& eopFile,
23  const std::string& format,
24  const basic_astrodynamics::IAUConventions nutationTheory )
25 {

28  throw std::runtime_error( "Error, only C04 EOP file format currently supported by reader." );

38 void EOPReader::readEopFile( const std::string& fileName )
39 {
40  using namespace tudat::unit_conversions;
41 
42  // Open file and create file stream.
43  std::fstream stream( fileName.c_str( ), std::ios::in );
44 
45  // Check if file opened correctly.
46  if ( stream.fail( ) )
47  {
48  throw std::runtime_error( "Data file could not be opened." + fileName );
49  }
50 
51  // Initialize boolean that gets set to true once the file header is passed.
52  bool isHeaderPassed = 0;
53 
54  // Line based parsing
55  std::string line;
56  std::vector< std::string > vectorOfIndividualStrings;
57  while ( !stream.fail( ) && !stream.eof( ) )
58  {
59  // Get line from stream
60  std::getline( stream, line );
61 
62  // Trim input string (removes all leading and trailing whitespaces).
63  boost::algorithm::trim( line );
64 
65  // Split string into multiple strings, each containing one element from a line from the
66  // data file.
67  boost::algorithm::split( vectorOfIndividualStrings,
68  line,
69  boost::algorithm::is_any_of( " " ),
70  boost::algorithm::token_compress_on );
71 
72  // If first line before data is found, check entry names and set isHeaderPassed to true.
73  if( !isHeaderPassed && vectorOfIndividualStrings[ 0 ] == "Date" )
74  {
75  if( vectorOfIndividualStrings[ 6 ] == "dPsi" )
76  {
77  throw std::runtime_error( "Warning, found dPsi, expected dX as CIP offset in GCRS, wrong nutation format requested" );
78  }
79  else if( vectorOfIndividualStrings[ 7 ] == "dEps" )
80  {
81  throw std::runtime_error( "Warning, found dEps, expected dY as CIP offset in GCRS, wrong nutation format requested" );
82  }
83 
84  isHeaderPassed = 1;
85  }
86 
87  // Read line of data and set values.
88  if( isHeaderPassed && vectorOfIndividualStrings.size( ) == 16 )
89  {
90 
91  // Read pole positions.
92  cipInItrs[ std::stod( vectorOfIndividualStrings[ 3 ] ) ].x( ) =
93  convertArcSecondsToRadians< double >( std::stod( vectorOfIndividualStrings[ 4 ] ) );
94  cipInItrs[ std::stod( vectorOfIndividualStrings[ 3 ] ) ].y( ) =
95  convertArcSecondsToRadians< double >( std::stod( vectorOfIndividualStrings[ 5 ] ) );
96 
97  // Read UTC-UT1 and LOD corrections
98  ut1MinusUtc[ std::stod( vectorOfIndividualStrings[ 3 ] ) ] =
99  std::stod( vectorOfIndividualStrings[ 6 ] );
100  lengthOfDayOffset[ std::stod( vectorOfIndividualStrings[ 3 ] ) ] =
101  std::stod( vectorOfIndividualStrings[ 7 ] );
102 
103  // Read precession-nutation corrections.
104  cipInGcrsCorrection[ std::stod( vectorOfIndividualStrings[ 3 ] ) ].x( ) =
105  convertArcSecondsToRadians< double >( std::stod( vectorOfIndividualStrings[ 8 ] ) );
106  cipInGcrsCorrection[ std::stod( vectorOfIndividualStrings[ 3 ] ) ].y( ) =
107  convertArcSecondsToRadians< double >( std::stod( vectorOfIndividualStrings[ 9 ] ) );
108 
109  }
110  }
111 }
