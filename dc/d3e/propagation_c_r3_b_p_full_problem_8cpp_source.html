23 std::map< std::string, std::shared_ptr< simulation_setup::BodySettings > > setupBodySettingsCR3BP(
24  const double distancePrimarySecondary,
25  const std::string& namePrimaryBody,
26  const std::string& nameSecondaryBody,
27  const std::string& frameOrientation,
28  const double primaryGravitationalParameter,
29  const double secondaryGravitationalParameter )
30 {
31  spice_interface::loadStandardSpiceKernels( );
32 
33  double gravitationalParameterPrimary = ( primaryGravitationalParameter == primaryGravitationalParameter ) ?
34  primaryGravitationalParameter :
35  simulation_setup::createGravityFieldModel(
36  simulation_setup::getDefaultGravityFieldSettings( namePrimaryBody, TUDAT_NAN, TUDAT_NAN ),
37  namePrimaryBody )->getGravitationalParameter( );
38  double gravitationalParameterSecondary =( secondaryGravitationalParameter == secondaryGravitationalParameter ) ?
39  secondaryGravitationalParameter :
40  simulation_setup::createGravityFieldModel(
41  simulation_setup::getDefaultGravityFieldSettings( namePrimaryBody, TUDAT_NAN, TUDAT_NAN ),
42  nameSecondaryBody )->getGravitationalParameter( );
43  double massParameter = circular_restricted_three_body_problem::computeMassParameter(
44  gravitationalParameterPrimary, gravitationalParameterSecondary );
45 
46  // Initial state for the primary
47  Eigen::Vector6d initialStateInKeplerianElementsPrimary = Eigen::Vector6d::Zero( );
48  initialStateInKeplerianElementsPrimary( orbital_element_conversions::semiMajorAxisIndex ) =
49  massParameter * distancePrimarySecondary;
50  initialStateInKeplerianElementsPrimary( orbital_element_conversions::trueAnomalyIndex ) = mathematical_constants::PI;
51 
52  // Initial state for the secondary
53  Eigen::Vector6d initialStateInKeplerianElementsSecondary = Eigen::Vector6d::Zero( );
54  initialStateInKeplerianElementsSecondary( orbital_element_conversions::semiMajorAxisIndex ) =
55  ( 1.0 - massParameter) * distancePrimarySecondary;
56 
57  // Create body objects.
58  std::vector< std::string > bodiesToCreate;
59  bodiesToCreate.push_back( namePrimaryBody );
60  bodiesToCreate.push_back( nameSecondaryBody );
61  std::map< std::string, std::shared_ptr< simulation_setup::BodySettings > > bodySettings =
62  simulation_setup::getDefaultBodySettings( bodiesToCreate );
63 
64  // Compute effective gravitational parameters
65  double sumGravitationalParameter = gravitationalParameterPrimary + gravitationalParameterSecondary;
66  double distanceBarycenterPrimary = gravitationalParameterSecondary * distancePrimarySecondary / ( sumGravitationalParameter );
67  double distanceBarycenterSecondary = distancePrimarySecondary - distanceBarycenterPrimary;
68  double gravitationalParameterPrimaryTwoBodyProblem =
69  std::pow( distanceBarycenterPrimary, 3 ) * sumGravitationalParameter /
70  std::pow( distanceBarycenterPrimary + distanceBarycenterSecondary, 3 );
71  double gravitationalParameterSecondaryTwoBodyProblem =
72  std::pow( distanceBarycenterSecondary, 3 ) * sumGravitationalParameter /
73  std::pow( distanceBarycenterPrimary + distanceBarycenterSecondary, 3 );
74 
75 
76  // Define body ephemeris settings
77  bodySettings[ namePrimaryBody ]->ephemerisSettings = std::make_shared< simulation_setup::KeplerEphemerisSettings >(
78  initialStateInKeplerianElementsPrimary, 0.0, gravitationalParameterPrimaryTwoBodyProblem,
79  "SSB", frameOrientation );
80  bodySettings[ nameSecondaryBody ]->ephemerisSettings = std::make_shared< simulation_setup::KeplerEphemerisSettings >(
81  initialStateInKeplerianElementsSecondary, 0.0, gravitationalParameterSecondaryTwoBodyProblem,
82  "SSB", frameOrientation );
83  for( unsigned int j = 0; j < bodiesToCreate.size( ); j++ )
84  {
85  bodySettings[ bodiesToCreate.at( j ) ]->ephemerisSettings->resetFrameOrientation( frameOrientation );
86  bodySettings[ bodiesToCreate.at( j ) ]->rotationModelSettings->resetOriginalFrame( frameOrientation );
87  }
88 
89  return bodySettings;
90 }

144 void propagateCR3BPFromEnvironment(
145  const double initialTime,
146  const double finalPropagationTime,
147  const Eigen::Vector6d& initialState,
148  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
149  const simulation_setup::NamedBodyMap& bodyMap,
150  const std::vector < std::string >& bodiesCR3BP,
151  std::map< double, Eigen::Vector6d >& stateHistory,
152  const bool outputInNormalizedCoordinates )
153 {
154  // CR3BP definition
155  double gravitationalParameterPrimary;
156  double gravitationalParameterSecondary;
157  Eigen::Vector6d initialStatePrimary;
158  Eigen::Vector6d initialStateSecondary;
159 
160  if( bodyMap.at( bodiesCR3BP.at( 0 ) )->getBodyMass( ) > bodyMap.at( bodiesCR3BP.at( 1 ) )->getBodyMass( ) )
161  {
162  gravitationalParameterPrimary = bodyMap.at( bodiesCR3BP.at( 0 ) )->getGravityFieldModel( )->getGravitationalParameter( );
163  initialStatePrimary = bodyMap.at( bodiesCR3BP.at( 0 ) )->getEphemeris( )->getCartesianState( initialTime );
164  gravitationalParameterSecondary = bodyMap.at( bodiesCR3BP.at( 1 ) )->getGravityFieldModel( )->getGravitationalParameter( );
165  initialStateSecondary = bodyMap.at( bodiesCR3BP.at( 1 ) )->getEphemeris( )->getCartesianState( initialTime );
166  }
167  else
168  {
169  gravitationalParameterPrimary = bodyMap.at( bodiesCR3BP.at( 1 ) )->getGravityFieldModel( )->getGravitationalParameter( );
170  initialStatePrimary = bodyMap.at( bodiesCR3BP.at( 1 ) )->getEphemeris( )->getCartesianState( initialTime );
171  gravitationalParameterSecondary = bodyMap.at( bodiesCR3BP.at( 0 ) )->getGravityFieldModel( )->getGravitationalParameter( );
172  initialStateSecondary = bodyMap.at( bodiesCR3BP.at( 0 ) )->getEphemeris( )->getCartesianState( initialTime );
173  }
174 
175 
176  double massParameter = circular_restricted_three_body_problem::computeMassParameter(
177  gravitationalParameterPrimary, gravitationalParameterSecondary );
178  double distanceBetweenPrimaries = ( initialStateSecondary.segment(0,3) - initialStatePrimary.segment(0,3) ).norm( );
179 
180  double normalizedInitialTime = circular_restricted_three_body_problem::convertDimensionalTimeToDimensionlessTime(
181  initialTime, gravitationalParameterPrimary, gravitationalParameterSecondary, distanceBetweenPrimaries);
182  double normalizedFinalPropagationTime = circular_restricted_three_body_problem::convertDimensionalTimeToDimensionlessTime(
183  finalPropagationTime, gravitationalParameterPrimary, gravitationalParameterSecondary, distanceBetweenPrimaries);
184  Eigen::Vector6d normalizedInitialState =
185  circular_restricted_three_body_problem::convertCartesianToCorotatingNormalizedCoordinates(
186  gravitationalParameterSecondary, gravitationalParameterPrimary,
187  distanceBetweenPrimaries, initialState, initialTime );
188 
189 
190 
191  // CR3BP propagation
192  std::shared_ptr< numerical_integrators::IntegratorSettings< double > > CR3BPintegratorSettings = integratorSettings;
193  double originalInitialTime = CR3BPintegratorSettings->initialTime_;
194  double originalInitialTimeStep = CR3BPintegratorSettings->initialTimeStep_;
195 
196  CR3BPintegratorSettings->initialTime_ = normalizedInitialTime;
197  CR3BPintegratorSettings->initialTimeStep_ = circular_restricted_three_body_problem::convertDimensionalTimeToDimensionlessTime(
198  integratorSettings->initialTimeStep_, gravitationalParameterPrimary, gravitationalParameterSecondary,
199  distanceBetweenPrimaries );
200 
201  stateHistory.clear( );
202  if( outputInNormalizedCoordinates )
203  {
204  stateHistory = performCR3BPIntegration(
205  CR3BPintegratorSettings, massParameter, normalizedInitialState, normalizedFinalPropagationTime, true );
206  }
207  else
208  {
209  std::map< double, Eigen::Vector6d > normalizedStateHistory = performCR3BPIntegration(
210  CR3BPintegratorSettings, massParameter, normalizedInitialState, normalizedFinalPropagationTime, true );
211  for( auto it = normalizedStateHistory.begin( ); it != normalizedStateHistory.end( ); it++ )
212  {
213  stateHistory[ circular_restricted_three_body_problem::convertDimensionlessTimeToDimensionalTime(
214  it->first, gravitationalParameterPrimary, gravitationalParameterSecondary, distanceBetweenPrimaries ) ] =
215  circular_restricted_three_body_problem::convertCorotatingNormalizedToCartesianCoordinates(
216  gravitationalParameterPrimary, gravitationalParameterSecondary, distanceBetweenPrimaries,
217  it->second, it->first );
218  }
219  }
220 
221  CR3BPintegratorSettings->initialTime_ = originalInitialTime;
222  CR3BPintegratorSettings->initialTimeStep_ = originalInitialTimeStep;
223 }

225 void propagateCR3BPAndFullDynamicsProblem(
226  const double initialTime,
227  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
228  const std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > propagatorSettings,
229  const simulation_setup::NamedBodyMap& bodyMap,
230  const std::vector < std::string >& bodiesCR3BP,
231  std::map< double, Eigen::Vector6d >& directPropagationResult,
232  std::map< double, Eigen::Vector6d >& cr3bpPropagationResult,
233  std::map< double, Eigen::VectorXd >& dependentVariableValues )
234 {
235  // Propagate the full problem
236  SingleArcDynamicsSimulator< > dynamicsSimulator( bodyMap, integratorSettings, propagatorSettings );
237 
238 
239  std::map< double, Eigen::VectorXd > stateHistory = dynamicsSimulator.getEquationsOfMotionNumericalSolution( );
240  utilities::castDynamicToFixedSizeEigenVectorMap< double, double, 6 >(
241  stateHistory, directPropagationResult );
242  dependentVariableValues = dynamicsSimulator.getDependentVariableHistory( );
243 
244  double finalPropagationTime = directPropagationResult.rbegin( )->first;
245 
246  cr3bpPropagationResult.clear( );
247  propagateCR3BPFromEnvironment(
248  initialTime, finalPropagationTime, propagatorSettings->getInitialStates( ), integratorSettings, bodyMap,
249  bodiesCR3BP, cr3bpPropagationResult, false );
250 }

280  const double initialTime,
281  const double finalTime,
282  const Eigen::Vector6d& initialState,
283  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
284  const basic_astrodynamics::AccelerationMap& accelerationModelMap,
285  const std::vector< std::string >& bodiesToPropagate,
286  const std::vector< std::string >& centralBodies,
287  const simulation_setup::NamedBodyMap& bodyMap,
288  const std::vector < std::string >& bodiesCR3BP )
289 {

295 
296  Eigen::Vector6d finalPropagatedStateFullProblem = directPropagationResult.rbegin( )->second;
297  Eigen::Vector6d finalPropagatedStateCR3BP = cr3bpPropagationResult.rbegin( )->second;
298 
299  // Difference between the two propagated states at finalTime
300  return finalPropagatedStateCR3BP - finalPropagatedStateFullProblem;
