138 template< typename ParameterType >
139 std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > createOneWayRangePartials(
140  const observation_models::LinkEnds oneWayRangeLinkEnds,
141  const simulation_setup::NamedBodyMap& bodyMap,
142  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
143  const std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >& lightTimeCorrections =
144  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >( ),
145  const bool useBiasPartials = true )
146 {
147 
148  std::vector< std::shared_ptr< observation_partials::LightTimeCorrectionPartial > > lightTimeCorrectionPartialObjects;
149  if( lightTimeCorrections.size( ) > 0 )
150  {
151  lightTimeCorrectionPartialObjects = observation_partials::createLightTimeCorrectionPartials( lightTimeCorrections );
152  }
153 
154  // Create scaling object, to be used for all one-way range partials in current link end.
155  std::shared_ptr< OneWayRangeScaling > oneWayRangeScaling = std::make_shared< OneWayRangeScaling >( );
156 
157  // Initialize vector index variables.
158  int currentIndex = 0;
159  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
160 
161  SingleLinkObservationPartialList rangePartials;
162 
163  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
164  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
165  parametersToEstimate->getEstimatedInitialStateParameters( );
166 
167  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
168  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
169  {
170  std::string acceleratedBody;
171  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state ||
172  initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
173  {
174  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
175 
176  // Create position one-way range partial for current body
177  std::shared_ptr< OneWayRangePartial > currentRangePartial = createOneWayRangePartialWrtBodyPosition(
178  oneWayRangeLinkEnds, bodyMap, acceleratedBody, oneWayRangeScaling, lightTimeCorrectionPartialObjects );
179 
180  // Check if partial is non-null (i.e. whether dependency exists between current range and current body)
181  if( currentRangePartial != nullptr )
182  {
183  // Add partial to the list.
184  currentPair = std::pair< int, int >( currentIndex, 6 );
185  rangePartials[ currentPair ] = currentRangePartial;
186  }
187 
188  // Increment current index by size of body initial state (6).
189  currentIndex += 6;
190  }
191  else if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_rotational_body_state )
192  {
193  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
194 
195  // Create position one-way range partial for current body
196  std::shared_ptr< OneWayRangePartial > currentRangePartial = createOneWayRangePartialWrtBodyRotationalState(
197  oneWayRangeLinkEnds, bodyMap, acceleratedBody, oneWayRangeScaling, lightTimeCorrectionPartialObjects );
198 
199  // Check if partial is non-null (i.e. whether dependency exists between current range and current body)
200  if( currentRangePartial != nullptr )
201  {
202  // Add partial to the list.
203  currentPair = std::pair< int, int >( currentIndex, 7 );
204  rangePartials[ currentPair ] = currentRangePartial;
205  }
206 
207  // Increment current index by size of body initial state (6).
208  currentIndex += 7;
209  }
210  else
211  {
212  throw std::runtime_error( "Error when making one way range partials, could not identify parameter " +
213  std::to_string( initialDynamicalParameters.at( i )->getParameterName( ).first ) );
214  }
215  }
216 
217  // Iterate over all double parameters that are to be estimated.
218  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParametersToEstimate =
219  parametersToEstimate->getDoubleParameters( );
220  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
221  parameterIterator =
222  doubleParametersToEstimate.begin( ); parameterIterator != doubleParametersToEstimate.end( ); parameterIterator++ )
223  {
224  // Create position one-way range partial for current parameter
225  std::shared_ptr< ObservationPartial< 1 > > currentRangePartial = createOneWayRangePartialWrtParameter(
226  oneWayRangeLinkEnds, bodyMap, parameterIterator->second,
227  oneWayRangeScaling, lightTimeCorrectionPartialObjects );
228 
229  // Check if partial is non-nullptr (i.e. whether dependency exists between current range and current parameter)
230  if( currentRangePartial != nullptr )
231  {
232  // Add partial to the list.
233  currentPair = std::pair< int, int >( parameterIterator->first, 1 );
234  rangePartials[ currentPair ] = currentRangePartial;
235  }
236  }
237 
238  // Iterate over all vector parameters that are to be estimated.
239  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
240  vectorParametersToEstimate =
241  parametersToEstimate->getVectorParameters( );
242  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
243  parameterIterator =
244  vectorParametersToEstimate.begin( ); parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
245  {
246  // Create position one-way range partial for current parameter
247  std::shared_ptr< ObservationPartial< 1 > > currentRangePartial;
248 
249  if( !isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
250  {
251  currentRangePartial = createOneWayRangePartialWrtParameter(
252  oneWayRangeLinkEnds, bodyMap, parameterIterator->second, oneWayRangeScaling,
253  lightTimeCorrectionPartialObjects );
254  }
255  else
256  {
257  currentRangePartial = createObservationPartialWrtLinkProperty< 1 >(
258  oneWayRangeLinkEnds, observation_models::one_way_range, parameterIterator->second, useBiasPartials );
259  }
260 
261  // Check if partial is non-nullptr (i.e. whether dependency exists between current range and current parameter)
262  if( currentRangePartial != nullptr )
263  {
264  // Add partial to the list.
265  currentPair = std::pair< int, int >( parameterIterator->first,
266  parameterIterator->second->getParameterSize( ) );
267  rangePartials[ currentPair ] = currentRangePartial;
268  }
269  }
270 
271  // Return complete set of partials and scaling object.
272  return std::make_pair( rangePartials, oneWayRangeScaling );
273 }

291 template< typename ParameterType >
292 std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
293 std::shared_ptr< PositionPartialScaling > > > createOneWayRangePartials(
294  const std::vector< observation_models::LinkEnds >& linkEnds,
295  const simulation_setup::NamedBodyMap& bodyMap,
296  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
297  const std::map< observation_models::LinkEnds,
298  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >& lightTimeCorrections =
299  std::map< observation_models::LinkEnds,
300  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >( ),
301  const bool useBiasPartials = true )
302 {
303  // Declare return list.
304  std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
305  std::shared_ptr< PositionPartialScaling > > > rangePartials;
306 
307  // Iterate over all link ends.
308  for( unsigned int i = 0; i < linkEnds.size( ); i++ )
309  {
310  // Check if required link end types are present
311  if( ( linkEnds[ i ].count( observation_models::receiver ) == 0 ) ||
312  ( linkEnds[ i ].count( observation_models::transmitter ) == 0 ) )
313  {
314  throw std::runtime_error( "Error when making 1-way range partials, did not find both receiver and transmitter in link ends" );
315 
316  }
317 
318  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > singleLinkLightTimeCorrections;
319  if( lightTimeCorrections.count( linkEnds.at( i ) ) > 0 )
320  {
321  if( lightTimeCorrections.at( linkEnds.at( i ) ).size( ) > 1 )
322  {
323  std::cerr << "Error when making one way range partials, light time corrections for "
324  << lightTimeCorrections.at( linkEnds.at( i ) ).size( ) << " links found" << std::endl;
325  }
326  else if( lightTimeCorrections.at( linkEnds.at( i ) ).size( ) == 1 )
327  {
328  singleLinkLightTimeCorrections = lightTimeCorrections.at( linkEnds.at( i ) ).at( 0 );
329  }
330  }
331 
332  // Create range partials for current link ends
333  rangePartials[ linkEnds[ i ] ] = createOneWayRangePartials< ParameterType >(
334  linkEnds[ i ], bodyMap, parametersToEstimate, singleLinkLightTimeCorrections, useBiasPartials );
335  }
336 
337  // Return complete set of link ends.
338  return rangePartials;
339 }
