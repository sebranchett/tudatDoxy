51 void ShapeBasedMethod::getTrajectory(
52  std::vector< double >& epochsVector,
53  std::map< double, Eigen::Vector6d >& propagatedTrajectory )
54 {
55  propagatedTrajectory.clear( );
56 
57  for ( unsigned int i = 0 ; i < epochsVector.size() ; i++ )
58  {
59  if ( ( i > 0 ) && ( epochsVector[ i ] < epochsVector[ i - 1 ] ) )
60  {
61  throw std::runtime_error( "Error when retrieving the mass profile of a shape-based trajectory, "
62  "epochs are not provided in increasing order." );
63  }
64 
65  double currentIndependentVariable = convertTimeToIndependentVariable( epochsVector[ i ] );
66  propagatedTrajectory[ epochsVector[ i ] ] = computeCurrentStateVector( currentIndependentVariable );
67  }

87 void ShapeBasedMethod::getThrustForceProfile(
88  std::vector< double >& epochsVector,
89  std::map< double, Eigen::VectorXd >& thrustProfile,
90  std::function< double ( const double ) > specificImpulseFunction,
91  std::shared_ptr<numerical_integrators::IntegratorSettings< double > > integratorSettings )
92 
93 {
94  thrustProfile.clear( );
95  std::map< double, Eigen::VectorXd > massProfile;
96 
97  getMassProfile( epochsVector, massProfile, specificImpulseFunction, integratorSettings );
98 
99  for ( unsigned int i = 0 ; i < epochsVector.size() ; i++ )
100  {
101  if ( ( i > 0 ) && ( epochsVector[ i ] < epochsVector[ i - 1 ] ) )
102  {
103  throw std::runtime_error( "Error when retrieving the thrust profile of a shape-based trajectory, "
104  "epochs are not provided in increasing order." );
105  }
106 
107  double independentVariable = convertTimeToIndependentVariable( epochsVector[ i ] );
108 
109  double currentMass = massProfile[ epochsVector[ i ] ][ 0 ];
110  thrustProfile[ epochsVector[ i ] ] = currentMass * computeCurrentThrustAccelerationMagnitude(
111  independentVariable, specificImpulseFunction, integratorSettings )
112  * computeCurrentThrustAccelerationDirection( independentVariable, specificImpulseFunction, integratorSettings );
113  }
114 }
