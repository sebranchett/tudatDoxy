107  virtual StateType performIntegrationStep( const TimeStepType stepSize )
108  {
109  lastIndependentVariable_ = currentIndependentVariable_;
110  lastState_ = currentState_;
111 
112  // Calculate k1-k4.
113  StateDerivativeType k1, k2, k3, k4;
114  for ( unsigned int i = 1; i <= 4; i++ )
115  {
116  IndependentVariableType time;
117  StateType state;
118  switch ( i )
119  {
120  case 1:
121  time = currentIndependentVariable_;
122  state = currentState_;
123  k1 = stepSize * this->stateDerivativeFunction_( time, state );
124  break;
125  case 2:
126  time = currentIndependentVariable_ + stepSize / 2.0;
127  state = static_cast< StateType >( currentState_ + k1 / 2.0 );
128  k2 = stepSize * this->stateDerivativeFunction_( time, state );
129  break;
130  case 3:
131  time = currentIndependentVariable_ + stepSize / 2.0;
132  state = static_cast< StateType >( currentState_ + k2 / 2.0 );
133  k3 = stepSize * this->stateDerivativeFunction_( time, state );
134  break;
135  case 4:
136  time = currentIndependentVariable_ + stepSize;
137  state = static_cast< StateType >( currentState_ + k3 );
138  k4 = stepSize * this->stateDerivativeFunction_( time, state );
139  break;
140  }
141 
142  // Check if propagation should terminate because the propagation termination condition has been reached
143  // while computing k1, k2, k3 or k4. If so, return immediately the current state (not recomputed yet),
144  // which will be discarded.
145  if ( this->propagationTerminationFunction_( static_cast< double >( time ), TUDAT_NAN ) )
146  {
147  this->propagationTerminationConditionReachedDuringStep_ = true;
148  return currentState_;
149  }
150  }
151 
152  stepSize_ = stepSize;
153  currentIndependentVariable_ += stepSize_;
154  currentState_ += ( k1 + 2.0 * k2 + 2.0 * k3 + k4 ) / 6.0;
155 
156  // Return the integration result.
157  return currentState_;
158  }
