114 template< typename ParameterType >
115 std::pair< std::map< std::pair< int, int >, std::shared_ptr< ObservationPartial< 2 > > >,
116 std::shared_ptr< PositionPartialScaling > >
117 createAngularPositionPartials(
118  const observation_models::LinkEnds angularPositionLinkEnds,
119  const simulation_setup::NamedBodyMap& bodyMap,
120  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
121  const std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >& lightTimeCorrections =
122  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >( ) )
123 
124 {
125  std::vector< std::shared_ptr< observation_partials::LightTimeCorrectionPartial > > lightTimeCorrectionPartialObjects;
126  if( lightTimeCorrections.size( ) > 0 )
127  {
128  lightTimeCorrectionPartialObjects = observation_partials::createLightTimeCorrectionPartials( lightTimeCorrections );
129  }
130 
131  // Create scaling object, to be used for all angular position partials in current link end.
132  std::shared_ptr< AngularPositionScaling > angularPositionScaling =
133  std::make_shared< AngularPositionScaling >( );
134 
135  SingleLinkObservationTwoPartialList angularPositionPartials;
136 
137 
138  // Initialize vector index variables.
139  int currentIndex = 0;
140  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
141 
142  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
143  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
144  parametersToEstimate->getEstimatedInitialStateParameters( );
145 
146  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
147  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
148  {
149 
150  std::string acceleratedBody;
151  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state )
152  {
153  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
154  }
155  else if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
156  {
157  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
158  }
159  else
160  {
161  throw std::runtime_error( "Error when making angular position partials, could not identify parameter" );
162  }
163 
164  // Create position angular position partial for current body
165  std::shared_ptr< AngularPositionPartial > currentAngularPositionPartial = createAngularPositionPartialWrtBodyPosition(
166  angularPositionLinkEnds, bodyMap, acceleratedBody, angularPositionScaling,
167  lightTimeCorrectionPartialObjects );
168 
169  // Check if partial is non-nullptr (i.e. whether dependency exists between current angular position and current body)
170  if( currentAngularPositionPartial != nullptr )
171  {
172  // Add partial to the list.
173  currentPair = std::pair< int, int >( currentIndex, 6 );
174  angularPositionPartials[ currentPair ] = currentAngularPositionPartial;
175  }
176 
177  // Increment current index by size of body initial state (6).
178  currentIndex += 6;
179  }
180 
181  // Iterate over all double parameters that are to be estimated.
182  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParametersToEstimate =
183  parametersToEstimate->getDoubleParameters( );
184  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
185  parameterIterator = doubleParametersToEstimate.begin( );
186  parameterIterator != doubleParametersToEstimate.end( ); parameterIterator++ )
187  {
188  // Create position angular position partial for current parameter
189  std::shared_ptr< AngularPositionPartial > currentAngularPositionPartial = createAngularPositionPartialWrtParameter(
190  angularPositionLinkEnds, bodyMap, parameterIterator->second, angularPositionScaling,
191  lightTimeCorrectionPartialObjects );
192 
193  if( currentAngularPositionPartial != nullptr )
194  {
195  // Add partial to the list.
196  currentPair = std::pair< int, int >( parameterIterator->first, 1 );
197  angularPositionPartials[ currentPair ] = currentAngularPositionPartial;
198  }
199  }
200 
201  // Iterate over all vector parameters that are to be estimated.
202  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
203  vectorParametersToEstimate = parametersToEstimate->getVectorParameters( );
204  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
205  parameterIterator = vectorParametersToEstimate.begin( );
206  parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
207  {
208 
209  // Create position angular position partial for current parameter
210  std::shared_ptr< ObservationPartial< 2 > > currentAngularPositionPartial;
211 
212  if( !isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
213  {
214  currentAngularPositionPartial = createAngularPositionPartialWrtParameter(
215  angularPositionLinkEnds, bodyMap, parameterIterator->second, angularPositionScaling,
216  lightTimeCorrectionPartialObjects );
217  }
218  else
219  {
220  currentAngularPositionPartial = createObservationPartialWrtLinkProperty< 2 >(
221  angularPositionLinkEnds, observation_models::angular_position, parameterIterator->second );
222  }
223 
224  // Check if partial is non-nullptr (i.e. whether dependency exists between current observable and current parameter)
225  if( currentAngularPositionPartial != nullptr )
226  {
227  // Add partial to the list.
228  currentPair = std::pair< int, int >( parameterIterator->first,
229  parameterIterator->second->getParameterSize( ) );
230  angularPositionPartials[ currentPair ] = currentAngularPositionPartial;
231  }
232 
233  }
234  return std::make_pair( angularPositionPartials, angularPositionScaling );
235 }

253 template< typename ParameterType >
254 std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationTwoPartialList,
255 std::shared_ptr< PositionPartialScaling > > >
256 createAngularPositionPartials(
257  const std::vector< observation_models::LinkEnds > linkEnds,
258  const simulation_setup::NamedBodyMap& bodyMap,
259  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
260  const std::map< observation_models::LinkEnds,
261  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >& lightTimeCorrections =
262  std::map< observation_models::LinkEnds,
263  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > >( ) )
264 {
265  // Declare return list.
266  std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationTwoPartialList,
267  std::shared_ptr< PositionPartialScaling > > > angularPositionPartials;
268 
269  // Iterate over all link ends.
270  for( unsigned int i = 0; i < linkEnds.size( ); i++ )
271  {
272  // Check if required link end types are present
273  if( ( linkEnds[ i ].count( observation_models::receiver ) == 0 ) ||
274  ( linkEnds[ i ].count( observation_models::transmitter ) == 0 ) )
275  {
276  throw std::runtime_error( "Error when making angular position partials, did not find both receiver and transmitter in link ends" );
277 
278  }
279 
280  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > singleLinkLightTimeCorrections;
281  if( lightTimeCorrections.count( linkEnds.at( i ) ) > 0 )
282  {
283  if( lightTimeCorrections.at( linkEnds.at( i ) ).size( ) != 1 )
284  {
285  std::cerr << "Error when making angular position partials, light time corrections for " <<
286  lightTimeCorrections.at( linkEnds.at( i ) ).size( ) << " links found" << std::endl;
287  }
288  singleLinkLightTimeCorrections = lightTimeCorrections.at( linkEnds.at( i ) ).at( 0 );
289  }
290 
291  // Create angular position partials for current link ends
292  angularPositionPartials[ linkEnds[ i ] ] = createAngularPositionPartials(
293  linkEnds[ i ], bodyMap, parametersToEstimate, singleLinkLightTimeCorrections );
294  }
295  return angularPositionPartials;
296 }
