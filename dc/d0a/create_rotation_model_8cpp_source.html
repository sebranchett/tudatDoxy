47 std::function< Eigen::Vector6d( const double, bool ) > createRelativeStateFunction(
48  const NamedBodyMap& bodyMap,
49  const std::string orbitingBody,
50  const std::string centralBody )
51 {
52  // Retrieve state functions for relevant bodies (obtained from current state of body objects)
53  std::function< Eigen::Vector6d( const double ) > bodyInertialStateFunction =
54  std::bind( &Body::getState, bodyMap.at( orbitingBody ) );
55  std::function< Eigen::Vector6d( const double ) > centralBodyInertialStateFunction =
56  std::bind( &Body::getState, bodyMap.at( centralBody ) );
57 
58  // Define relative state function from body object
59  std::function< Eigen::Vector6d( const double ) > fromBodyStateFunction =
60  std::bind(
61  &ephemerides::getDifferenceBetweenStates, bodyInertialStateFunction,
62  centralBodyInertialStateFunction, std::placeholders::_1 );
63 
64  // Define state function from ephemeris
65  std::function< Eigen::Vector6d( const double ) > fromEphemerisStateFunction;
66 
67  if( bodyMap.at( orbitingBody )->getEphemeris( )->getReferenceFrameOrigin( ) == centralBody )
68  {
69  fromEphemerisStateFunction = std::bind( &ephemerides::Ephemeris::getCartesianState,
70  bodyMap.at( orbitingBody )->getEphemeris( ), std::placeholders::_1 );
71 
72  }
73  else
74  {
75  std::function< Eigen::Vector6d( const double ) > ephemerisInertialStateFunction =
76  std::bind( &Body::getStateInBaseFrameFromEphemeris< double, double >, bodyMap.at( orbitingBody ),
77  std::placeholders::_1 );
78  std::function< Eigen::Vector6d( const double ) > ephemerisCentralBodyInertialStateFunction =
79  std::bind( &Body::getStateInBaseFrameFromEphemeris< double, double >, bodyMap.at( centralBody ),
80  std::placeholders::_1 );
81  fromEphemerisStateFunction = std::bind(
82  &ephemerides::getDifferenceBetweenStates,
83  ephemerisInertialStateFunction,
84  ephemerisCentralBodyInertialStateFunction, std::placeholders::_1 );
85  }
86 
87  return std::bind( &getStateFromSelectedStateFunction, std::placeholders::_1, std::placeholders::_2,
88  fromBodyStateFunction, fromEphemerisStateFunction );
89 }

92 std::shared_ptr< ephemerides::RotationalEphemeris > createRotationModel(
93  const std::shared_ptr< RotationModelSettings > rotationModelSettings,
94  const std::string& body,
95  const NamedBodyMap& bodyMap )
96 {
97  using namespace tudat::ephemerides;
98 
99  // Declare return object.
100  std::shared_ptr< RotationalEphemeris > rotationalEphemeris;
101 
102  // Check which type of rotation model is to be created.
103  switch( rotationModelSettings->getRotationType( ) )
104  {
105  case simple_rotation_model:
106  {
107  // Check whether settings for simple rotation model are consistent with its type.
108  std::shared_ptr< SimpleRotationModelSettings > simpleRotationSettings =
109  std::dynamic_pointer_cast< SimpleRotationModelSettings >( rotationModelSettings );
110  if( simpleRotationSettings == nullptr )
111  {
112  throw std::runtime_error(
113  "Error, expected simple rotation model settings for " + body );
114  }
115  else
116  {
117  // Create and initialize simple rotation model.
118  rotationalEphemeris = std::make_shared< SimpleRotationalEphemeris >(
119  simpleRotationSettings->getInitialOrientation( ),
120  simpleRotationSettings->getRotationRate( ),
121  simpleRotationSettings->getInitialTime( ),
122  simpleRotationSettings->getOriginalFrame( ),
123  simpleRotationSettings->getTargetFrame( ) );
124  }
125  break;
126  }
127 #if USE_SOFA
128  case gcrs_to_itrs_rotation_model:
129  {
130 
131  // Check whether settings for simple rotation model are consistent with its type.
132  std::shared_ptr< GcrsToItrsRotationModelSettings > gcrsToItrsRotationSettings =
133  std::dynamic_pointer_cast< GcrsToItrsRotationModelSettings >( rotationModelSettings );
134  if( gcrsToItrsRotationSettings == nullptr )
135  {
136  throw std::runtime_error(
137  "Error, expected GCRS to ITRS rotation model settings for " + body );
138  }
139  else
140  {
141  std::shared_ptr< earth_orientation::EOPReader > eopReader = std::make_shared< earth_orientation::EOPReader >(
142  gcrsToItrsRotationSettings->getEopFile( ),
143  gcrsToItrsRotationSettings->getEopFileFormat( ),
144  gcrsToItrsRotationSettings->getNutationTheory( ) );
145 
146  // Load polar motion corrections
147  std::shared_ptr< interpolators::LinearInterpolator< double, Eigen::Vector2d > > cipInItrsInterpolator =
148  std::make_shared< interpolators::LinearInterpolator< double, Eigen::Vector2d > >(
149  eopReader->getCipInItrsMapInSecondsSinceJ2000( ) );
150 
151  // Load nutation corrections
152  std::shared_ptr< interpolators::LinearInterpolator< double, Eigen::Vector2d > > cipInGcrsCorrectionInterpolator =
153  std::make_shared< interpolators::LinearInterpolator< double, Eigen::Vector2d > >(
154  eopReader->getCipInGcrsCorrectionMapInSecondsSinceJ2000( ) );
155 
156  // Create polar motion correction (sub-diural frequencies) object
157  std::shared_ptr< earth_orientation::ShortPeriodEarthOrientationCorrectionCalculator< Eigen::Vector2d > >
158  shortPeriodPolarMotionCalculator =
159  std::make_shared< earth_orientation::ShortPeriodEarthOrientationCorrectionCalculator< Eigen::Vector2d > >(
160  gcrsToItrsRotationSettings->getPolarMotionCorrectionSettings( )->conversionFactor_,
161  gcrsToItrsRotationSettings->getPolarMotionCorrectionSettings( )->minimumAmplitude_,
162  gcrsToItrsRotationSettings->getPolarMotionCorrectionSettings( )->amplitudesFiles_,
163  gcrsToItrsRotationSettings->getPolarMotionCorrectionSettings( )->argumentMultipliersFile_ );
164 
165  // Create full polar motion calculator
166  std::shared_ptr< earth_orientation::PolarMotionCalculator > polarMotionCalculator =
167  std::make_shared< earth_orientation::PolarMotionCalculator >
168  ( cipInItrsInterpolator, shortPeriodPolarMotionCalculator );
169 
170  // Create IAU 2006 precession/nutation calculator
171  std::shared_ptr< earth_orientation::PrecessionNutationCalculator > precessionNutationCalculator =
172  std::make_shared< earth_orientation::PrecessionNutationCalculator >(
173  gcrsToItrsRotationSettings->getNutationTheory( ), cipInGcrsCorrectionInterpolator );
174 
175  // Create UT1 correction (sub-diural frequencies) object
176  std::shared_ptr< earth_orientation::ShortPeriodEarthOrientationCorrectionCalculator< double > >
177  ut1CorrectionSettings =
178  std::make_shared< earth_orientation::ShortPeriodEarthOrientationCorrectionCalculator< double > >(
179  gcrsToItrsRotationSettings->getUt1CorrectionSettings( )->conversionFactor_,
180  gcrsToItrsRotationSettings->getUt1CorrectionSettings( )->minimumAmplitude_,
181  gcrsToItrsRotationSettings->getUt1CorrectionSettings( )->amplitudesFiles_,
182  gcrsToItrsRotationSettings->getUt1CorrectionSettings( )->argumentMultipliersFile_ );
183 
184  std::shared_ptr< interpolators::OneDimensionalInterpolator < double, double > > dailyUtcUt1CorrectionInterpolator =
185  std::make_shared< interpolators::JumpDataLinearInterpolator< double, double > >(
186  eopReader->getUt1MinusUtcMapInSecondsSinceJ2000( ), 0.5, 1.0 );
187 
188  // Create default time scale converter
189  std::shared_ptr< earth_orientation::TerrestrialTimeScaleConverter > terrestrialTimeScaleConverter =
190  std::make_shared< earth_orientation::TerrestrialTimeScaleConverter >
191  ( dailyUtcUt1CorrectionInterpolator, ut1CorrectionSettings );
192 
193  // Create rotation model
194  std::shared_ptr< earth_orientation::EarthOrientationAnglesCalculator > earthOrientationCalculator =
195  std::make_shared< earth_orientation::EarthOrientationAnglesCalculator >(
196  polarMotionCalculator, precessionNutationCalculator, terrestrialTimeScaleConverter );
197  rotationalEphemeris = std::make_shared< ephemerides::GcrsToItrsRotationModel >(
198  earthOrientationCalculator, gcrsToItrsRotationSettings->getInputTimeScale( ),
199  gcrsToItrsRotationSettings->getOriginalFrame( ) );
200 
201  break;
202  }
203 
204  }
205 #endif
206 
207 #if USE_CSPICE
208  case spice_rotation_model:
209  {
210  // Create rotational ephemeris directly from Spice.
211  rotationalEphemeris = std::make_shared< SpiceRotationalEphemeris >(
212  rotationModelSettings->getOriginalFrame( ),
213  rotationModelSettings->getTargetFrame( ) );
214  break;
215  }
216 #endif
217  case planetary_rotation_model:
218  {
219  std::shared_ptr< PlanetaryRotationModelSettings > planetaryRotationModelSettings =
220  std::dynamic_pointer_cast< PlanetaryRotationModelSettings >( rotationModelSettings );
221  if( planetaryRotationModelSettings == nullptr )
222  {
223  std::cerr<<"Error, expected planetary rotation model settings for "<<body<<std::endl;
224  }
225  else
226  {
227 
228  Eigen::VectorXd bodyKeplerElements = orbital_element_conversions::convertCartesianToKeplerianElements(
229  spice_interface::getBodyCartesianStateAtEpoch(
230  body, planetaryRotationModelSettings->getCentralBody( ), "ECLIPJ2000", "NONE", 0.0 ),
231  spice_interface::getBodyGravitationalParameter( planetaryRotationModelSettings->getCentralBody( ) ) );
232 
233  double meanAnomalyAtJ2000 = orbital_element_conversions::convertEccentricAnomalyToMeanAnomaly(
234  orbital_element_conversions::convertTrueAnomalyToEccentricAnomaly(
235  bodyKeplerElements( 5 ), bodyKeplerElements( 1 ) ), bodyKeplerElements( 1 ) );
236 
237  double meanMotion = orbital_element_conversions::convertSemiMajorAxisToEllipticalMeanMotion(
238  bodyKeplerElements( 0 ), spice_interface::getBodyGravitationalParameter(
239  planetaryRotationModelSettings->getCentralBody( ) ) + spice_interface::getBodyGravitationalParameter( body ) );
240 
241  std::shared_ptr< PlanetaryOrientationAngleCalculator > planetaryOrientationAnglesCalculator
242  = std::make_shared< PlanetaryOrientationAngleCalculator >(
243  planetaryRotationModelSettings->getAnglePsiAtEpoch( ),
244  planetaryRotationModelSettings->getAnglePsiRateAtEpoch( ),
245  planetaryRotationModelSettings->getAngleIAtEpoch( ),
246  planetaryRotationModelSettings->getAngleIRateAtEpoch( ),
247  planetaryRotationModelSettings->getAnglePhiAtEpoch( ),
248  planetaryRotationModelSettings->getAnglePhiRateAtEpoch( ),
249  planetaryRotationModelSettings->getCoreFactor( ),
250  planetaryRotationModelSettings->getFreeCoreNutationRate( ),
251  meanMotion, meanAnomalyAtJ2000,
252  planetaryRotationModelSettings->getOriginalFrame( ),
253  planetaryRotationModelSettings->getMeanMotionDirectNutationCorrections( ),
254  planetaryRotationModelSettings->getMeanMotionTimeDependentPhaseNutationCorrections( ),
255  planetaryRotationModelSettings->getTimeDependentPhaseCorrectionFunctions( ),
256  planetaryRotationModelSettings->getRotationRateCorrections( ),
257  planetaryRotationModelSettings->getxPolarMotionCoefficients( ),
258  planetaryRotationModelSettings->getyPolarMotionCoefficients( ) );
259 
260 
261  rotationalEphemeris = std::make_shared< PlanetaryRotationModel >(
262  planetaryRotationModelSettings->getAngleN( ),
263  planetaryRotationModelSettings->getAngleJ( ),
264  planetaryOrientationAnglesCalculator,
265  planetaryRotationModelSettings->getOriginalFrame( ), planetaryRotationModelSettings->getTargetFrame( ) );
266  }
267 
268  break;
269  }
270  case synchronous_rotation_model:
271  {
272  std::shared_ptr< SynchronousRotationModelSettings > synchronousRotationSettings =
273  std::dynamic_pointer_cast< SynchronousRotationModelSettings >( rotationModelSettings );
274  if( synchronousRotationSettings == NULL )
275  {
276  throw std::runtime_error( "Error, expected synchronous rotation model settings for " + body );
277  }
278  else
279  {
280  if( bodyMap.at( body )->getEphemeris( )->getReferenceFrameOrigin( ) == synchronousRotationSettings->getCentralBodyName( ) )
281  {
282  if( bodyMap.at( body )->getEphemeris( )->getReferenceFrameOrientation( ) !=
283  synchronousRotationSettings->getOriginalFrame( ) )
284  {
285  throw std::runtime_error( "Error, ephemeris of body " + body + " is in " +
286  bodyMap.at( body )->getEphemeris( )->getReferenceFrameOrientation( ) +
287  " frame when making synchronous rotation model, expected " +
288  synchronousRotationSettings->getOriginalFrame( ) + " frame." );
289  }
290  }
291  std::shared_ptr< SynchronousRotationalEphemeris > synchronousRotationalEphemeris = std::make_shared< SynchronousRotationalEphemeris >(
292  createRelativeStateFunction( bodyMap, body, synchronousRotationSettings->getCentralBodyName( ) ),
293  synchronousRotationSettings->getCentralBodyName( ),
294  synchronousRotationSettings->getOriginalFrame( ),
295  synchronousRotationSettings->getTargetFrame( ) );
296 
297  rotationalEphemeris = synchronousRotationalEphemeris;
298  }
299  break;
300  }
301  default:
302  throw std::runtime_error(
303  "Error, did not recognize rotation model settings type " +
304  std::to_string( rotationModelSettings->getRotationType( ) ) );
305  }
306 
307  return rotationalEphemeris;
308 }
