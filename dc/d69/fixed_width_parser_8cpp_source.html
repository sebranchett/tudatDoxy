22 FixedWidthParser::FixedWidthParser( int numberOfFields, ... ) : TextParser( false ), doTrim( true )
23 {
24  numberOfFields_ = numberOfFields;
25 
26  // Create a fancy vector (list) of all the fields:

29 
30  // Initialize list. Point to last defined argument.
31  va_start( listOfArguments, numberOfFields );
32 
33  for ( unsigned int i=0; i < numberOfFields_; i++ )
34  {
35  // Populate typeList with arguments from constructor.
36  typeList.push_back( va_arg(listOfArguments, FieldType ) );

38

50 void FixedWidthParser::parseLine( std::string& line )
51 {
52  using namespace parsed_data_vector_utilities;
53 
54  // Define a new type: field type and pointer to value pair
55  typedef std::pair< FieldType, FieldValuePtr > FieldDataPair;
56 
57  // Define a new type: vector of strings.
58  typedef std::vector< std::string > vectorOfIndividualStrings;
59 
60  // Create a vector of individual strings.
61  vectorOfIndividualStrings vectorOfIndividualStrings_;
62 
63  // Size of vector of individual strings equals number of fields.
64  vectorOfIndividualStrings_.resize( numberOfFields_ );
65 
66  // Temporary string.
67  std::string temp;
68 
69  // Start index of current field in the line.
70  int currentFieldIndex = 0;
71 
72  for ( int unsigned currentFieldNumber = 0; currentFieldNumber < numberOfFields_;
73  currentFieldNumber++ )
74  {
75  // Generate string of the field.
76  temp = line.substr( currentFieldIndex, sizeList[currentFieldNumber] );
77 
78  // If we need to trim whitespace, do so
79  if ( doTrim )
80  {
81  boost::trim( temp );
82  }
83 
84  // Copy field value.
85  vectorOfIndividualStrings_[currentFieldNumber] = temp;
86 
87  // Increase current field index to index of next field.
88  currentFieldIndex += sizeList[currentFieldNumber];
89 
90  // Clear temporary string.
91  temp.clear( );
92  }
93 
94  // Verify that number of individual vectors corresponds to the specified number of fields.
95  if ( vectorOfIndividualStrings_.size( ) != numberOfFields_ )
96  {
97  std::cerr << "Number of elements in the line (" << vectorOfIndividualStrings_.size( )
98  << ") does not match the specified number of fields (" << numberOfFields_ << ")"
99  << std::endl;
100  }
101 
102  // Create a new data line.
103  ParsedDataLineMapPtr currentLineData( new std::map< FieldType, FieldValuePtr >( ) );
104 
105  // Register the data line with the global current parsed data vector.
106  parsedData->push_back( currentLineData );
107 
108  //Loop over all field type and field value pairs
109  for ( int unsigned currentFieldNumber = 0; currentFieldNumber < numberOfFields_;
110  currentFieldNumber++ )
111  {
112  // Get the corresponding field type
113  FieldType type ( typeList.at( currentFieldNumber ) );
114 
115  // Define unit transformer
116  std::shared_ptr< FieldTransform > transformer;
117 
118  // If type corresponds to one of the entries of the unit transformation map
119  if ( unitTransformationMap_.find( type ) != unitTransformationMap_.end( ) )
120  {
121  // Set corresponding transformer
122  transformer = unitTransformationMap_.find( type )->second;
123  }
124 
125  else
126  {
127  // Else, do nothing.
128  transformer = std::shared_ptr< FieldTransform >( );
129  }
130 
131  // Store the resulting string.
132  FieldValuePtr value( new FieldValue( type,
133  vectorOfIndividualStrings_.at( currentFieldNumber ),
134  transformer ) );
135 
136  // Store the type and value in the current line data.
137  currentLineData->insert( FieldDataPair( type, value ) );
138  }
139 }
