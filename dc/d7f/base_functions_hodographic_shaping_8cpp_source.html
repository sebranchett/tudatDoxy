243 double PowerSineFunctionHodographicShaping::evaluateIntegral( const double independentVariable )
244 {
245 
246  double integralValue = 0.0;
247  double integrationScaleFactor = 1.0;
248 
249  // Integration by parts.
250  for ( int counterSuccessiveIntegrations = 1; counterSuccessiveIntegrations <= exponentPowerFunction_ + 1 ; counterSuccessiveIntegrations++ )
251  {
252  double doubleCounterSuccessiveIntegrations = static_cast<double>( counterSuccessiveIntegrations );
253 
254  if ( ( counterSuccessiveIntegrations % 2 ) == 0.0 )
255  {
256  integralValue += std::pow( - 1.0, doubleCounterSuccessiveIntegrations / 2.0 + 1.0 ) * integrationScaleFactor
257  * std::pow( independentVariable, exponentPowerFunction_ - doubleCounterSuccessiveIntegrations + 1.0 )
258  / std::pow( frequencySineFunction_, doubleCounterSuccessiveIntegrations )
259  * std::sin( frequencySineFunction_ * independentVariable );
260  }
261  else
262  {
263  integralValue += std::pow( - 1.0, ( doubleCounterSuccessiveIntegrations + 1.0 ) / 2.0 ) * integrationScaleFactor
264  * std::pow( independentVariable, exponentPowerFunction_ - doubleCounterSuccessiveIntegrations + 1.0 )
265  / std::pow( frequencySineFunction_, doubleCounterSuccessiveIntegrations )
266  * std::cos( frequencySineFunction_ * independentVariable );
267  }
268 
269  integrationScaleFactor *= exponentPowerFunction_ - doubleCounterSuccessiveIntegrations + 1.0;
270  }
271 
272  return integralValue;
273 }
