140  virtual void updateCurrentCoefficients( const std::vector< double >& independentVariables,
141  const double currentTime = TUDAT_NAN )
142  {
143  // Check if the correct number of aerodynamic coefficients is provided.
144  if( independentVariables.size( ) != numberOfIndependentVariables_ )
145  {
146  throw std::runtime_error(
147  "Error in CustomAerodynamicCoefficientInterface, number of input variables is inconsistent " +
148  std::to_string( independentVariables.size( ) ) + ", " +
149  std::to_string( numberOfIndependentVariables_ ) );
150  }
151 
152  // Update current coefficients.
153  if( timeUpdateFunction_ != nullptr )
154  {
155  timeUpdateFunction_( currentTime );
156  }
157 
158  Eigen::Vector6d currentCoefficients = coefficientFunction_(
159  independentVariables );
160  currentForceCoefficients_ = currentCoefficients.segment( 0, 3 );
161  currentMomentCoefficients_ = currentCoefficients.segment( 3, 3 );
162  }
