59 template< template< typename ... > class MapType, typename KeyType, typename ValueType >
60 MapType< KeyType, ValueType > mapFromJson( const nlohmann::json& jsonObject )
61 {
62  MapType< KeyType, ValueType > map;
63  for ( nlohmann::json::const_iterator it = jsonObject.begin( ); it != jsonObject.end( ); ++it )
64  {
65  const std::string key = it.key( );
66  if ( ! contains( SpecialKeys::all, key ) )
67  {
68  map[ boost::lexical_cast< KeyType >( key ) ] = getValue< ValueType >( jsonObject, key );
69  }
70  }
71  return map;
72 }

198 template< typename ScalarType, int rows, int cols >
199 void from_json( const nlohmann::json& jsonObject, Matrix< ScalarType, rows, cols >& matrix )
200 {
201  nlohmann::json jsonArray;
202  if ( jsonObject.is_array( ) )
203  {
204  if ( jsonObject.empty( ) )
205  {
206  return;
207  }
208  jsonArray = jsonObject;
209  }
210  else if ( jsonObject.is_number( ) )
211  {
212  jsonArray.push_back( jsonObject );
213  }
214  else
215  {
216  throw nlohmann::detail::type_error::create( 0, "" );
217  }
218 
219  nlohmann::json jsonArrayOfArrays;
220  if ( jsonArray.front( ).is_array( ) ) // provided matrix
221  {
222  jsonArrayOfArrays = jsonArray;
223  }
224  else // provided vector
225  {
226  if ( rows == 1 ) // expected row vector
227  {
228  jsonArrayOfArrays.push_back( jsonArray );
229  }
230  else if ( cols == 1 ) // expected column vector
231  {
232  for ( unsigned int i = 0; i < jsonArray.size( ); ++i )
233  {
234  jsonArrayOfArrays.push_back( { jsonArray.at( i ) } );
235  }
236  }
237  else // expected matrix
238  {
239  std::cerr << "Expected a matrix, received a vector." << std::endl;
240  throw nlohmann::detail::type_error::create( 0, "" );
241  }
242  }
243 
244  const unsigned int providedRows = jsonArrayOfArrays.size( );
245  const unsigned int providedCols = jsonArrayOfArrays.front( ).size( );
246  if ( ( rows >= 0 && int( providedRows ) != rows ) || ( cols >= 0 && int( providedCols ) != cols ) )
247  {
248  std::cerr << "Expected matrix of size " << rows << "x" << cols
249  << ", received matrix of size " << providedRows << "x" << providedCols << "." << std::endl;
250  throw nlohmann::detail::type_error::create( 0, "" );
251  }
252 
253  matrix.resize( providedRows, providedCols );
254  for ( unsigned int r = 0; r < providedRows; ++r )
255  {
256  if ( jsonArrayOfArrays.at( r ).size( ) != providedCols )
257  {
258  std::cerr << "Unconsistent matrix size: some rows have different number of columns." << std::endl;
259  throw nlohmann::detail::type_error::create( 0, "" );
260  }
261  for ( unsigned int c = 0; c < providedCols; ++c )
262  {
263  matrix( r, c ) = jsonArrayOfArrays.at( r ).at( c );
264  }
265  }
266 }
