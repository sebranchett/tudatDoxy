29 template< typename ObservationScalarType = double, typename TimeType = double >
30 std::vector< TimeType > getConcatenatedTimeVector(
31  const typename OrbitDeterminationManager< ObservationScalarType, TimeType >::PodInputType& measurementData )
32 {
33  // Iterate over all observations and concatenate the time vectors.
34  std::vector< TimeType > concatenatedTimes;
35  for( typename OrbitDeterminationManager< ObservationScalarType, TimeType >::PodInputType::const_iterator
36  observablesIterator = measurementData.begin( ); observablesIterator != measurementData.end( ); observablesIterator++ )
37  {
38  // Get data type size.
39  int currentObservableSize = getObservableSize( observablesIterator->first );
40 
41  // Iterate over all link ends
42  for( typename OrbitDeterminationManager< ObservationScalarType, TimeType >::SingleObservablePodInputType::const_iterator
43  dataIterator = observablesIterator->second.begin( ); dataIterator != observablesIterator->second.end( );
44  dataIterator++ )
45  {
46  // Copy data in the case that observable size is 1
47  if( currentObservableSize == 1 )
48  {
49  concatenatedTimes.insert( concatenatedTimes.end( ), dataIterator->second.second.first.begin( ),
50  dataIterator->second.second.first.end( ) );
51  }
52  // For observable size N>1, coopy time tag N times for each observation
53  else
54  {
55  std::vector< TimeType > originalTimesList = dataIterator->second.second.first;
56  std::vector< TimeType > currentTimesList;
57  for( unsigned int i = 0; i < originalTimesList.size( ); i++ )
58  {
59  for( int j = 0; j < currentObservableSize; j++ )
60  {
61  currentTimesList.push_back( originalTimesList.at( i ) );
62  }
63  }
64 
65  concatenatedTimes.insert( concatenatedTimes.end( ), currentTimesList.begin( ),
66  currentTimesList.end( ) );
67 
68  }
69  }
70  }
71 
72  return concatenatedTimes;
73 }

193  measurementData.begin( ); observablesIterator != measurementData.end( ); observablesIterator++ )

197  observablesIterator->second.begin( ); dataIterator != observablesIterator->second.end( ); dataIterator++ )

199  concatenatedObservations.segment( currentIndex, dataIterator->second.first.rows( ) ) = dataIterator->second.first;

221 template< typename ObservationScalarType = double, typename TimeType = double >
222 std::pair< Eigen::MatrixXd, std::vector< TimeType > > getTimeOrderedInformationMatrix(
223  const typename OrbitDeterminationManager< ObservationScalarType, TimeType >::PodInputType& measurementData,
224  const Eigen::MatrixXd& typeAndLinkSortedInformationMatrix,
225  std::vector< int >& timeOrder )
226 {
227  // Retrieve unordered vector of times
228  std::vector< TimeType > concatenatedTimes = getConcatenatedTimeVector< ObservationScalarType, TimeType >( measurementData ) ;
229 
230  // Sort the concatesnated time vector, and get the order of the sorting.
231  std::pair< std::vector< int >, std::vector< TimeType > > sortOutput = utilities::getSortOrderOfVectorAndSortedVector(
232  concatenatedTimes );
233  timeOrder = sortOutput.first;
234 
235  // Retrieve sort order and check consistency of its size.
236  std::vector< int > timeVectorSortOrder = sortOutput.first;
237  if( static_cast< int >( timeVectorSortOrder.size( ) ) != typeAndLinkSortedInformationMatrix.rows( ) )
238  {
239  throw std::runtime_error( "Error when sorting information matrix by time, sizes incompatible" );
240  }
241 
242  // Create information matrix with entries in order of time.
243  int numberOfColumns = typeAndLinkSortedInformationMatrix.cols( );
244  Eigen::MatrixXd sortedMatrix = Eigen::MatrixXd::Zero( typeAndLinkSortedInformationMatrix.rows( ),
245  typeAndLinkSortedInformationMatrix.cols( ) );
246  for( unsigned int i = 0; i < timeVectorSortOrder.size( ); i++ )
247  {
248  sortedMatrix.block( i, 0, 1, numberOfColumns ) =
249  typeAndLinkSortedInformationMatrix.block( timeVectorSortOrder[ i ], 0, 1, numberOfColumns );
250  }
251 
252  // Return time-sorted information matrix and time order.
253  return std::make_pair( sortedMatrix, sortOutput.second );
254 }

270 template< typename ObservationScalarType = double, typename TimeType = double >
271 std::map< TimeType, Eigen::MatrixXd > calculateCovarianceUsingDataUpToEpoch(
272  const typename OrbitDeterminationManager< ObservationScalarType, TimeType >::PodInputType& measurementData,
273  const Eigen::MatrixXd& typeAndLinkSortedNormalizedInformationMatrix,
274  const Eigen::VectorXd& normalizationFactors,
275  const std::vector< double >& outputTimes,
276  const Eigen::VectorXd& diagonalOfWeightMatrix,
277  const Eigen::MatrixXd& unnormalizedInverseAPrioriCovariance )
278 {
279  int totalNumberOfParameters = unnormalizedInverseAPrioriCovariance.cols( );
280 
281  Eigen::MatrixXd normalizedInverseAPrioriCovariance = Eigen::MatrixXd( totalNumberOfParameters, totalNumberOfParameters );
282  for( int j = 0; j < totalNumberOfParameters; j++ )
283  {
284  for( int k = 0; k < totalNumberOfParameters; k++ )
285  {
286  normalizedInverseAPrioriCovariance( j, k ) = unnormalizedInverseAPrioriCovariance( j, k ) /
287  ( normalizationFactors( j ) * normalizationFactors( k ) );
288  }
289  }
290 
291  // Check consistency of input data
292  if( normalizedInverseAPrioriCovariance.cols( ) != normalizedInverseAPrioriCovariance.rows( ) )
293  {
294  throw std::runtime_error( "Error when calculating covariance as function of time, a priori covariance is not square" );
295  }
296 
297  if( typeAndLinkSortedNormalizedInformationMatrix.cols( ) != totalNumberOfParameters )
298  {
299  throw std::runtime_error(
300  "Error when calculating covariance as function of time, number of parameters is inconsistent with information matrix" );
301  }
302 
303  if( normalizationFactors.rows( ) != totalNumberOfParameters )
304  {
305  throw std::runtime_error(
306  "Error when calculating covariance as function of time, number of parameters is inconsistent with normalization factors" );
307  }
308  if( typeAndLinkSortedNormalizedInformationMatrix.rows( ) != diagonalOfWeightMatrix.rows( ) )
309  {
310  throw std::runtime_error(
311  "Error when calculating covariance as function of time, weights are inconsistent with partials" );
312  }
313 
314  // Order information matrix by time of observations
315  std::vector< int > timeOrder;
316  std::pair< Eigen::MatrixXd, std::vector< TimeType > > timeOrderedMatrixOutput =
317  getTimeOrderedInformationMatrix< ObservationScalarType, TimeType >(
318  measurementData, typeAndLinkSortedNormalizedInformationMatrix, timeOrder );
319  std::vector< TimeType > orderedTimeVector = timeOrderedMatrixOutput.second;
320 
321  Eigen::VectorXd timeOrderedDiagonalOfWeightMatrix = Eigen::VectorXd::Zero( diagonalOfWeightMatrix.rows( ) );
322  for( unsigned int i = 0; i < timeOrder.size( ); i++ )
323  {
324  timeOrderedDiagonalOfWeightMatrix( i ) = diagonalOfWeightMatrix( timeOrder.at( i ) );
325  }
326 
327  // Create lookupn scheme for time value
328  interpolators::BinarySearchLookupScheme< TimeType > timeLookup =
329  interpolators::BinarySearchLookupScheme< TimeType >( orderedTimeVector );
330 
331  // Declare return map.
332  std::map< TimeType, Eigen::MatrixXd > covarianceMatrixHistory;
333 
334  // Initialize loop variables.
335  TimeType currentTime = orderedTimeVector[ 0 ];
336  unsigned int currentIndex;
337  Eigen::MatrixXd currentInverseNormalizedCovarianceMatrix;
338 
339  // Loop over matrix at given time interval.
340  for( unsigned int i = 0; i < outputTimes.size( ); i++ )
341  {
342  // Increment time (no covariance computed for t=t0)
343  currentTime = outputTimes.at( i );
344 
345  // Find index in list of times.
346  currentIndex = timeLookup.findNearestLowerNeighbour( currentTime );
347 
348  if( currentIndex != orderedTimeVector.size( ) - 1 )
349  {
350  while( orderedTimeVector.at( currentIndex) == orderedTimeVector.at( currentIndex + 1 ) )
351  {
352  currentIndex++;
353  if( currentIndex == orderedTimeVector.size( ) - 1 )
354  {
355  break;
356  }
357  }
358  }
359 
360  if( orderedTimeVector.size( ) <= currentIndex )
361  {
362  throw std::runtime_error( "Error when getting covariance as a function of time, output time not found" );
363  }
364 
365  // Create information matrix up to current time.
366  Eigen::MatrixXd currentInformationMatrix = timeOrderedMatrixOutput.first.block(
367  0, 0, currentIndex + 1, timeOrderedMatrixOutput.first.cols( ) );
368 
369  // Create inverse of covariance matrix
370  currentInverseNormalizedCovarianceMatrix = linear_algebra::calculateInverseOfUpdatedCovarianceMatrix(
371  currentInformationMatrix, timeOrderedDiagonalOfWeightMatrix.segment( 0, currentIndex + 1 ), normalizedInverseAPrioriCovariance );
372  Eigen::MatrixXd covarianceMatrix = currentInverseNormalizedCovarianceMatrix.inverse( );
373 
374  for( int i = 0; i < covarianceMatrix.rows( ); i++ )
375  {
376  for( int j = 0; j < covarianceMatrix.rows( ); j++ )
377  {
378  covarianceMatrix( i, j ) /= normalizationFactors( i ) * normalizationFactors( j );
379  }
380  }
381 
382  covarianceMatrixHistory[ currentTime ] = covarianceMatrix;
383  }
384 
385 
386  return covarianceMatrixHistory;
387 }

389 template< typename ObservationScalarType = double, typename TimeType = double >
390 std::map< TimeType, Eigen::MatrixXd > calculateCovarianceUsingDataUpToEpoch(
391  const typename OrbitDeterminationManager< ObservationScalarType, TimeType >::PodInputType& measurementData,
392  const Eigen::MatrixXd& typeAndLinkSortedNormalizedInformationMatrix,
393  const Eigen::VectorXd& normalizationFactors,
394  const double outputTimeStep,
395  const Eigen::VectorXd& diagonalOfWeightMatrix,
396  const Eigen::MatrixXd& unnormalizedInverseAPrioriCovariance )

398  Eigen::VectorXd timeVector = utilities::convertStlVectorToEigenVector(
399  getConcatenatedTimeVector( measurementData ) );
400  double minimumTime = timeVector.minCoeff( );
401  double maximumTime = timeVector.maxCoeff( );
402  double currentTime = minimumTime;
403  std::vector< double > outputTimes;
404 
405  // Loop over matrix at given time interval.
406  while( currentTime < maximumTime )
407  {
408  // Increment time (no covariance computed for t=t0)
409  currentTime += outputTimeStep;
410  outputTimes.push_back( currentTime );
411  }
412 
413  return calculateCovarianceUsingDataUpToEpoch(
414  measurementData, typeAndLinkSortedNormalizedInformationMatrix, normalizationFactors,
415  outputTimes, diagonalOfWeightMatrix, unnormalizedInverseAPrioriCovariance );
