140  template< typename StateScalarType >
141  void getParameterPartialMatrix(
142  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > currentMatrixDerivative )
143  {
144  // Initialize matrix to zeros
145  variationalParameterMatrix_.setZero( );
146 
147  // Iterate over all bodies undergoing accelerations for which initial condition is to be estimated.
148  for( std::map< IntegratedStateType, std::vector< std::multimap< std::pair< int, int >,
149  std::function< void( Eigen::Block< Eigen::MatrixXd > ) > > > >::iterator typeIterator =
150  parameterPartialList_.begin( ); typeIterator != parameterPartialList_.end( ); typeIterator++ )
151  {
152  int startIndex = stateTypeStartIndices_.at( typeIterator->first );
153  int currentStateSize = getSingleIntegrationSize( typeIterator->first );
154  int entriesToSkipPerEntry = currentStateSize - getGeneralizedAccelerationSize( typeIterator->first );
155 
156  // Iterate over all bodies being estimated.
157  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
158  {
159  // Iterate over all parameter partial functions determined by setParameterPartialFunctionList( )
160  for( functionIterator = typeIterator->second[ i ].begin( );
161  functionIterator != typeIterator->second[ i ].end( );
162  functionIterator++ )
163  {
164  functionIterator->second(
165  variationalParameterMatrix_.block(
166  startIndex + entriesToSkipPerEntry + currentStateSize * i,
167  functionIterator->first.first - totalDynamicalStateSize_,
168  currentStateSize - entriesToSkipPerEntry,
169  functionIterator->first.second ) );
170  }
171  }
172  }
173 
174  for( unsigned int i = 0; i < inertiaTensorsForMultiplication_.size( ); i++ )
175  {
176  variationalParameterMatrix_.block( inertiaTensorsForMultiplication_.at( i ).first, 0, 3,
177  numberOfParameterValues_ - totalDynamicalStateSize_ ) =
178  ( inertiaTensorsForMultiplication_.at( i ).second( ).inverse( ) ) *
179  variationalParameterMatrix_.block(
180  inertiaTensorsForMultiplication_.at( i ).first, 0, 3,
181  numberOfParameterValues_ - totalDynamicalStateSize_ ).eval( );
182  }
183 
184  currentMatrixDerivative.block( 0, totalDynamicalStateSize_, totalDynamicalStateSize_,
185  numberOfParameterValues_ - totalDynamicalStateSize_ ) +=
186  variationalParameterMatrix_.template cast< StateScalarType >( );

322  void addParameterPartialToList(
323  const std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< CurrentParameterType > > >&
324  parameterList,
325  const std::shared_ptr< orbit_determination::StateDerivativePartial > partialObject,
326  std::multimap< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >&
327  functionListOfBody,
328  const int totalParameterVectorIndicesToSubtract = 0 )
329  {
330  using namespace acceleration_partials;
331 
332  // Iterate over all parameters.
333  for( typename std::map< int,
334  std::shared_ptr< estimatable_parameters::EstimatableParameter< CurrentParameterType > > >::const_iterator
335  parameterIterator = parameterList.begin( ); parameterIterator != parameterList.end( ); parameterIterator++ )
336  {
337  // Add current parameter to list of partials to be computed for current acceleration (if dependency exists)
338  int functionToEvaluate =
339  partialObject->setParameterPartialUpdateFunction( parameterIterator->second );
340 
341  // If function is non-nullptr, add to list
342  if( functionToEvaluate != 0 )
343  {
344  // Make pair of indices for generating parameter partial matrix:
345  //first is start column in matrix, second is number of entries (1 for double parameter)
346  std::pair< int, int > indexPair = std::make_pair(
347  parameterIterator->first - totalParameterVectorIndicesToSubtract, functionToEvaluate );
348 
349  // Add to list.
350  functionListOfBody.insert(
351  std::pair< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
352  ( indexPair, std::bind(
353  static_cast< void ( orbit_determination::StateDerivativePartial::* )
354  ( const std::shared_ptr<
355  estimatable_parameters::EstimatableParameter< CurrentParameterType > >,
356  Eigen::Block< Eigen::MatrixXd > )>

371  template< typename ParameterType >
372  void setParameterPartialFunctionList(
373  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > >
374  parametersToEstimate )
375  {
376  // Get double parameters.
377  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >
378  doubleParametersToEstimate =
379  parametersToEstimate->getDoubleParameters( );
380 
381  // Get vector parameters.
382  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
383  vectorParametersToEstimate =
384  parametersToEstimate->getVectorParameters( );
385 
386  int totalParameterVectorIndicesToSubtract = parametersToEstimate->getInitialDynamicalStateParameterSize( ) -
387  estimatable_parameters::getSingleArcInitialDynamicalStateParameterSetSize( parametersToEstimate );
388 
389  for( std::map< propagators::IntegratedStateType,
390  orbit_determination::StateDerivativePartialsMap >::iterator
391  stateDerivativeTypeIterator = stateDerivativePartialList_.begin( );
392  stateDerivativeTypeIterator != stateDerivativePartialList_.end( );
393  stateDerivativeTypeIterator++ )
394  {
395 
396  // Initialize vector of lists to correct size.
397  parameterPartialList_[ stateDerivativeTypeIterator->first ].resize(
398  stateDerivativeTypeIterator->second.size( ) );
399 
400  // Iterate over all bodies of which initial position is being estimated.
401  for( unsigned int i = 0; i < stateDerivativeTypeIterator->second.size( ); i++ )
402  {
403  // Initialize list of parameter partial functions for single body.
404  std::multimap< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
405  functionListOfBody;
406 
407  // Iterate over all accelerations due to this body on current body.
408  for( unsigned int j = 0; j < stateDerivativeTypeIterator->second.at( i ).size( ); j++ )
409  {
410  addParameterPartialToList< double >(
411  doubleParametersToEstimate, stateDerivativeTypeIterator->second.at( i ).at( j ),
412  functionListOfBody, totalParameterVectorIndicesToSubtract );
413  addParameterPartialToList< Eigen::VectorXd >(
414  vectorParametersToEstimate, stateDerivativeTypeIterator->second.at( i ).at( j ),
415  functionListOfBody, totalParameterVectorIndicesToSubtract );
416  }
417 
418  // Add generated parameter partial list of current body.
419  parameterPartialList_[ stateDerivativeTypeIterator->first ][ i ] = functionListOfBody;
420  }
421  }

432  template< typename ParameterType >
433  void setTranslationalStatePartialFrameScalingFunctions(
434  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > >
435  parametersToEstimate,
436  const int currentArcIndex = -1 )
437  {
438  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
439  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
440  parametersToEstimate->getEstimatedInitialStateParameters( );
441 
442  std::vector< std::string > propagatedBodies;
443  std::vector< std::string > centralBodies;
444 
445  // Retrieve propagated bodies and central bodies of estimation.
446  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
447  {
448  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state )
449  {
450  propagatedBodies.push_back(
451  initialDynamicalParameters.at( i )->getParameterName( ).second.first );
452  centralBodies.push_back( std::dynamic_pointer_cast
453  < estimatable_parameters::InitialTranslationalStateParameter< ParameterType > >(
454  initialDynamicalParameters.at( i ) )->getCentralBody( ) );
455  }
456  else if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
457  {
458  if( currentArcIndex < 0 )
459  {
460  throw std::runtime_error( "Error in setTranslationalStatePartialFrameScalingFunctions, cpuld not find xurrent arc index " );
461  }
462  propagatedBodies.push_back(
463  initialDynamicalParameters.at( i )->getParameterName( ).second.first );
464  centralBodies.push_back( std::dynamic_pointer_cast< estimatable_parameters::ArcWiseInitialTranslationalStateParameter< ParameterType > >(
465  initialDynamicalParameters.at( i ) )->getCentralBodies( ).at( currentArcIndex ) );
466  }
467  }
468 
469  if( propagatedBodies.size( ) > 0 )
470  {
471  // Get order in which ephemerides were to be updated.
472  std::vector< std::string > updateOrder = determineEphemerisUpdateorder(
473  propagatedBodies, centralBodies, centralBodies );
474 
475  // Iterate over central bodies and propagated bodies and check for dependencies
476  for( int i = updateOrder.size( ) - 1; i >= 0 ; i-- )
477  {
478  int currentBodyIndex = std::distance(
479  propagatedBodies.begin( ),
480  std::find( propagatedBodies.begin( ), propagatedBodies.end( ), updateOrder.at( i ) ) );
481  for( unsigned int j = 0; j < propagatedBodies.size( ); j++ )
482  {
483  if( centralBodies.at( currentBodyIndex ) == propagatedBodies.at( j ) )
484  {
485 
486  statePartialAdditionIndices_.push_back(
487  std::make_pair( stateTypeStartIndices_[ propagators::translational_state ] +
488  currentBodyIndex * propagators::getSingleIntegrationSize( propagators::translational_state ),
489  stateTypeStartIndices_[ propagators::translational_state ] +
490  j * propagators::getSingleIntegrationSize( propagators::translational_state ) ) );
491  }
492  }
493  }
494  }
495  }
