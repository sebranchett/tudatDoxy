37 template< typename DirectGravityPartial >
38 basic_astrodynamics::AvailableAcceleration getAccelerationTypeOfThirdBodyGravity(
39  const std::shared_ptr< DirectGravityPartial > directGravityPartial )
40 {
41  using namespace basic_astrodynamics;
42  AvailableAcceleration accelerationType;
43 
44  // Check type of direct partial derivative.
45  if( std::dynamic_pointer_cast< CentralGravitationPartial >( directGravityPartial ) != nullptr )
46  {
47  accelerationType = third_body_central_gravity;
48  }
49  else if( std::dynamic_pointer_cast< SphericalHarmonicsGravityPartial >( directGravityPartial ) != nullptr )
50  {
51  accelerationType = third_body_spherical_harmonic_gravity;
52  }
53  else if( std::dynamic_pointer_cast< MutualSphericalHarmonicsGravityPartial >( directGravityPartial ) != nullptr )
54  {
55  accelerationType = third_body_mutual_spherical_harmonic_gravity;
56  }
57 
58  else
59  {
60  throw std::runtime_error( "Error when getting third body partial type, type not identified" );
61  }
62  return accelerationType;
63 }

365  int setParameterPartialUpdateFunction(
366  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
367  {
368  // Check parameter dependency of direct acceleration.
369  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromDirectGravity =
370  partialOfDirectGravityOnBodyUndergoingAcceleration_->getParameterPartialFunction( parameter );
371  if( partialFunctionFromDirectGravity.second > 0 )
372  {
373  partialOfDirectGravityOnBodyUndergoingAcceleration_->setParameterPartialUpdateFunction( parameter );
374  }
375 
376  // Check parameter dependency of indirect acceleration.
377  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromCentralGravity =
378  partialOfDirectGravityOnCentralBody_->getParameterPartialFunction( parameter );
379  if( partialFunctionFromCentralGravity.second > 0 )
380  {
381  partialOfDirectGravityOnCentralBody_->setParameterPartialUpdateFunction( parameter );
382  }
383 
384  // If either dependency exists, create combined partial.
385  if( partialFunctionFromCentralGravity.second > 0 || partialFunctionFromDirectGravity.second > 0 )
386  {
387  parameterDoublePartialFunctions_[ parameter ] =
388  getCombinedCurrentDoubleParameterFunction(
389  partialOfDirectGravityOnBodyUndergoingAcceleration_,
390  partialOfDirectGravityOnCentralBody_,
391  parameter, partialFunctionFromDirectGravity.second, partialFunctionFromCentralGravity.second, 1 );
392  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
393  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
394  }
395  return std::max( partialFunctionFromDirectGravity.second, partialFunctionFromCentralGravity.second );
396  }

519 inline std::string getCentralBodyNameFromThirdBodyAccelerationPartial(
520  const std::shared_ptr< AccelerationPartial > accelerationPartial )
521 {
522  std::string centralBody;
523  if( !basic_astrodynamics::isAccelerationFromThirdBody( accelerationPartial->getAccelerationType( ) ) )
524  {
525  throw std::runtime_error( "Error, requested third body from acceleration partial, but input is incompatible." );
526  }
527  else
528  {
529  if( accelerationPartial->getAccelerationType( ) == basic_astrodynamics::third_body_point_mass_gravity )
530  {
531  centralBody = std::dynamic_pointer_cast< ThirdBodyGravityPartial< CentralGravitationPartial > >(
532  accelerationPartial )->getCentralBodyName( );
533  }
534  else if( accelerationPartial->getAccelerationType( ) == basic_astrodynamics::third_body_spherical_harmonic_gravity )
535  {
536  centralBody = std::dynamic_pointer_cast< ThirdBodyGravityPartial< SphericalHarmonicsGravityPartial > >(
537  accelerationPartial )->getCentralBodyName( );
538  }
539  else if( accelerationPartial->getAccelerationType( ) == basic_astrodynamics::third_body_mutual_spherical_harmonic_gravity )
540  {
541  centralBody = std::dynamic_pointer_cast< ThirdBodyGravityPartial< MutualSphericalHarmonicsGravityPartial > >(
542  accelerationPartial )->getCentralBodyName( );
543  }
544  else
545  {
546  throw std::runtime_error( "Error, requested third body from acceleration partial, input third-body type not recognized." );
547  }
548  }
549 
550  return centralBody;
551 }
