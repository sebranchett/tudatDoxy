23 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< AtmosphereSettings >& atmosphereSettings )
24 {
25  if ( ! atmosphereSettings )
26  {
27  return;
28  }
29  using namespace json_interface;
30  using K = Keys::Body::Atmosphere;
31 
32  const AtmosphereTypes atmosphereType = atmosphereSettings->getAtmosphereType( );
33  jsonObject[ K::type ] = atmosphereType;
34 
35  // ExponentialAtmosphereSettings
36  switch ( atmosphereType )
37  {
38  case exponential_atmosphere:
39  {
40  std::shared_ptr< ExponentialAtmosphereSettings > exponentialAtmosphereSettings =
41  std::dynamic_pointer_cast< ExponentialAtmosphereSettings >( atmosphereSettings );
42  assertNonnullptrPointer( exponentialAtmosphereSettings );
43  jsonObject[ K::densityScaleHeight ] = exponentialAtmosphereSettings->getDensityScaleHeight( );
44  jsonObject[ K::constantTemperature ] = exponentialAtmosphereSettings->getConstantTemperature( );
45  jsonObject[ K::densityAtZeroAltitude ] = exponentialAtmosphereSettings->getDensityAtZeroAltitude( );
46  jsonObject[ K::specificGasConstant ] = exponentialAtmosphereSettings->getSpecificGasConstant( );
47  jsonObject[ K::ratioOfSpecificHeats ] = exponentialAtmosphereSettings->getRatioOfSpecificHeats( );
48  return;
49  }
50  case tabulated_atmosphere:
51  {
52  std::shared_ptr< TabulatedAtmosphereSettings > tabulatedAtmosphereSettings =
53  std::dynamic_pointer_cast< TabulatedAtmosphereSettings >( atmosphereSettings );
54  assertNonnullptrPointer( tabulatedAtmosphereSettings );
55  jsonObject[ K::file ] = tabulatedAtmosphereSettings->getAtmosphereFile( );
56  jsonObject[ K::independentVariablesNames ] = tabulatedAtmosphereSettings->getIndependentVariables( );
57  jsonObject[ K::dependentVariablesNames ] = tabulatedAtmosphereSettings->getDependentVariables( );
58  jsonObject[ K::specificGasConstant ] = tabulatedAtmosphereSettings->getSpecificGasConstant( );
59  jsonObject[ K::ratioOfSpecificHeats ] = tabulatedAtmosphereSettings->getRatioOfSpecificHeats( );
60  jsonObject[ K::boundaryHandling ] = tabulatedAtmosphereSettings->getBoundaryHandling( );
61  return;
62  }
63  case nrlmsise00:
64  {
65  std::shared_ptr< NRLMSISE00AtmosphereSettings > nrlmsise00AtmosphereSettings =
66  std::dynamic_pointer_cast< NRLMSISE00AtmosphereSettings >( atmosphereSettings );
67  if ( nrlmsise00AtmosphereSettings )
68  {
69  jsonObject[ K::spaceWeatherFile ] = boost::filesystem::path( nrlmsise00AtmosphereSettings->getSpaceWeatherFile( ) );
70  return;
71  }
72  // If not a NRLMSISE00AtmosphereSettings, it is a AtmosphereSettings with default space weather file.
73  return;
74  }
75  default:
76  handleUnimplementedEnumValue( atmosphereType, atmosphereTypes, unsupportedAtmosphereTypes );
77  }
78 }

81 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< AtmosphereSettings >& atmosphereSettings )
82 {
83  using namespace json_interface;
84  using K = Keys::Body::Atmosphere;
85 
86  // Get atmosphere model type
87  const AtmosphereTypes atmosphereType = getValue< AtmosphereTypes >( jsonObject, K::type );
88 
89  switch ( atmosphereType ) {
90  case exponential_atmosphere:
91  {
92  ExponentialAtmosphereSettings defaults( 0.0, 0.0, 0.0 );
93  atmosphereSettings = std::make_shared< ExponentialAtmosphereSettings >(
94  getValue< double >( jsonObject, K::densityScaleHeight ),
95  getValue< double >( jsonObject, K::constantTemperature ),
96  getValue< double >( jsonObject, K::densityAtZeroAltitude ),
97  getValue< double >( jsonObject, K::specificGasConstant, defaults.getSpecificGasConstant( ) ),
98  getValue< double >( jsonObject, K::ratioOfSpecificHeats, defaults.getRatioOfSpecificHeats( ) ) );
99  return;
100  }
101  case tabulated_atmosphere:
102  {
103  TabulatedAtmosphereSettings defaults( getValue< std::map< int, std::string > >( jsonObject, K::file ) );
104  atmosphereSettings = std::make_shared< TabulatedAtmosphereSettings >(
105  getValue< std::map< int, std::string > >( jsonObject, K::file ),
106  getValue< std::vector< AtmosphereIndependentVariables > >( jsonObject, K::independentVariablesNames,
107  defaults.getIndependentVariables( ) ),
108  getValue< std::vector< AtmosphereDependentVariables > >( jsonObject, K::dependentVariablesNames,
109  defaults.getDependentVariables( ) ),
110  getValue< double >( jsonObject, K::specificGasConstant, defaults.getSpecificGasConstant( ) ),
111  getValue< double >( jsonObject, K::ratioOfSpecificHeats, defaults.getRatioOfSpecificHeats( ) ),
112  getValue< std::vector< interpolators::BoundaryInterpolationType > >( jsonObject, K::boundaryHandling,
113  defaults.getBoundaryHandling( ) ) );
114  return;
115  }
116  case nrlmsise00:
117  {
118  if ( isDefined( jsonObject, K::spaceWeatherFile ) )
119  {
120  atmosphereSettings = std::make_shared< NRLMSISE00AtmosphereSettings >(
121  getValue< boost::filesystem::path >( jsonObject, K::spaceWeatherFile ).string( ) );
122  }
123  else
124  {
125  atmosphereSettings = std::make_shared< AtmosphereSettings >( nrlmsise00 );
126  }
127  return;
128  }
129  default:
130  handleUnimplementedEnumValue( atmosphereType, atmosphereTypes, unsupportedAtmosphereTypes );
131  }
132 }
