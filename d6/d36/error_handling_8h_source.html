391 template< typename EnumType >
392 EnumType enumFromString( const std::string& stringValue, const std::map< EnumType, std::string >& stringValues )
393 {
394  for ( auto entry : stringValues )
395  {
396  if ( stringValue == entry.second )
397  {
398  return entry.first;
399  }
400  }
401  std::cerr << "Unknown string \"" << stringValue << "\" for enum " <<
402  boost::core::demangled_name( typeid( EnumType ) ) << std::endl;
403  std::cerr << "Recognized strings:" << std::endl;
404  for ( auto entry : stringValues )
405  {
406  std::cerr << " " << entry.second << std::endl;
407  }
408  throw UnknownEnumError( );
409 }

419 template< typename EnumType >
420 std::string stringFromEnum( const EnumType enumValue, const std::map< EnumType, std::string >& stringValues )
421 {
422  if ( stringValues.count( enumValue ) > 0 )

429  << boost::core::demangled_name( typeid( EnumType ) ) << "::" << enumValue << std::endl;
430  throw UnknownEnumError( );
