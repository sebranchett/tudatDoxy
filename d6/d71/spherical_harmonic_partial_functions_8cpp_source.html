29 void computePotentialSphericalHessian(
30  const double distance,
31  const double radiusPowerTerm,
32  const double cosineOfOrderLongitude,
33  const double sineOfOrderLongitude,
34  const double cosineOfLatitude,
35  const double sineOfLatitude,
36  const double preMultiplier,
37  const int degree,
38  const int order,
39  const double cosineHarmonicCoefficient,
40  const double sineHarmonicCoefficient,
41  const double legendrePolynomial,
42  const double legendrePolynomialDerivative,
43  const double legendrePolynomialSecondDerivative,
44  Eigen::Matrix3d& sphericalHessian )
45 {
46  sphericalHessian( 0, 0 ) = static_cast< double >( ( degree + 1 ) * ( degree + 2 ) )/
47  ( distance * distance ) * legendrePolynomial *
48  ( cosineHarmonicCoefficient * cosineOfOrderLongitude + sineHarmonicCoefficient * sineOfOrderLongitude );
49  sphericalHessian( 1, 0 ) = -static_cast< double >( degree + 1 ) / distance * cosineOfLatitude *
50  legendrePolynomialDerivative *
51  ( cosineHarmonicCoefficient * cosineOfOrderLongitude + sineHarmonicCoefficient * sineOfOrderLongitude );
52  sphericalHessian( 2, 0 ) = -static_cast< double >( order * ( degree + 1 ) ) / distance *
53  legendrePolynomial *
54  ( -cosineHarmonicCoefficient * sineOfOrderLongitude + sineHarmonicCoefficient * cosineOfOrderLongitude );
55 
56  sphericalHessian( 0, 1 ) = sphericalHessian( 1, 0 );
57  sphericalHessian( 1, 1 ) = ( cosineOfLatitude * cosineOfLatitude * legendrePolynomialSecondDerivative -
58  sineOfLatitude * legendrePolynomialDerivative ) *
59  ( cosineHarmonicCoefficient * cosineOfOrderLongitude + sineHarmonicCoefficient * sineOfOrderLongitude );
60  sphericalHessian( 2, 1 ) = static_cast< double >( order ) * cosineOfLatitude * legendrePolynomialDerivative *
61  ( -cosineHarmonicCoefficient * sineOfOrderLongitude + sineHarmonicCoefficient * cosineOfOrderLongitude );
62 
63  sphericalHessian( 0, 2 ) = sphericalHessian( 2, 0 );
64  sphericalHessian( 1, 2 ) = sphericalHessian( 2, 1 );
65  sphericalHessian( 2, 2 ) = static_cast< double >( order * order ) * legendrePolynomial *
66  ( -cosineHarmonicCoefficient * cosineOfOrderLongitude - sineHarmonicCoefficient * sineOfOrderLongitude );
67 
68 
69  sphericalHessian *= preMultiplier * radiusPowerTerm;
70 }

123 Eigen::Matrix3d computeCumulativeSphericalHessian(
124  const Eigen::Vector3d& sphericalPosition,
125  const double referenceRadius,
126  const double gravitionalParameter,
127  const Eigen::MatrixXd cosineHarmonicCoefficients,
128  const Eigen::MatrixXd sineHarmonicCoefficients,
129  const std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache )
130 {
131  double preMultiplier = gravitionalParameter / referenceRadius;
132 
133  Eigen::Matrix3d sphericalHessian, sphericalHessianTerm;
134 
135  sphericalHessian.setZero( );
136  for( int i = 0; i < cosineHarmonicCoefficients.rows( ); i++ )
137  {
138  for( int j = 0; ( j <= i && j < cosineHarmonicCoefficients.cols( ) ); j++ )
139  {
140  computePotentialSphericalHessian(
141  sphericalPosition, preMultiplier, i, j, cosineHarmonicCoefficients( i, j ),
142  sineHarmonicCoefficients( i, j ), sphericalHarmonicsCache, sphericalHessianTerm );
143  sphericalHessian += sphericalHessianTerm;
144  }
145  }
146  return sphericalHessian;
147 
148 }

152 Eigen::Matrix3d computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
153  const Eigen::Vector3d& cartesianPosition,
154  const Eigen::Vector3d& sphericalPosition,
155  const double referenceRadius,
156  const double gravitionalParameter,
157  const Eigen::MatrixXd cosineHarmonicCoefficients,
158  const Eigen::MatrixXd sineHarmonicCoefficients,
159  const std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache,
160  const Eigen::Vector3d& sphericalPotentialGradient,
161  const Eigen::Matrix3d& sphericalToCartesianGradientMatrix )
162 {
163  // Compute Hessian in spherical coordinates.
164  Eigen::Matrix3d sphericalHessian = computeCumulativeSphericalHessian(
165  sphericalPosition, referenceRadius, gravitionalParameter, cosineHarmonicCoefficients,
166  sineHarmonicCoefficients, sphericalHarmonicsCache );
167 
168  // Convert to Cartesian Hessian
169  Eigen::Matrix3d accelerationPartial =
170  sphericalToCartesianGradientMatrix * sphericalHessian * sphericalToCartesianGradientMatrix.transpose( );
171 
172  // Add effect of direct change in rotation matrix
173  accelerationPartial += coordinate_conversions::getDerivativeOfSphericalToCartesianGradient(
174  sphericalPotentialGradient, cartesianPosition );
175 
176  return accelerationPartial;
177 }

181 Eigen::Matrix3d computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
182  const Eigen::Vector3d& cartesianPosition,
183  const double referenceRadius,
184  const double gravitionalParameter,
185  const Eigen::MatrixXd cosineHarmonicCoefficients,
186  const Eigen::MatrixXd sineHarmonicCoefficients,
187  const std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache )
188 {
189  // Compute spherical position.
190  Eigen::Vector3d sphericalPosition =
191  coordinate_conversions::convertCartesianToSpherical( cartesianPosition );
192  sphericalPosition( 1 ) = mathematical_constants::PI / 2.0 - sphericalPosition( 1 );
193 
194  // Compute spherical to Cartesian gradient transformation.
195  Eigen::Matrix3d gradientTransformationMatrix =
196  coordinate_conversions::getSphericalToCartesianGradientMatrix( cartesianPosition );
197 
198  // Compute spherical gradient.
199  std::map< std::pair< int, int >, Eigen::Vector3d > dummyMap;
200  Eigen::Vector3d sphericalPotentialGradient = gradientTransformationMatrix.inverse( ) *
201  gravitation::computeGeodesyNormalizedGravitationalAccelerationSum(
202  cartesianPosition, gravitionalParameter, referenceRadius, cosineHarmonicCoefficients,
203  sineHarmonicCoefficients, sphericalHarmonicsCache, dummyMap );
204 
205  return computePartialDerivativeOfBodyFixedSphericalHarmonicAcceleration(
206  cartesianPosition, sphericalPosition, referenceRadius, gravitionalParameter, cosineHarmonicCoefficients,
207  sineHarmonicCoefficients, sphericalHarmonicsCache, sphericalPotentialGradient,
208  gradientTransformationMatrix );
209 }

212 void calculateSphericalHarmonicGravityWrtCCoefficients(
213  const Eigen::Vector3d& sphericalPosition,
214  const double referenceRadius,
215  const double gravitionalParameter,
216  const std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache,
217  const std::vector< std::pair< int, int > >& blockIndices,
218  const Eigen::Matrix3d& sphericalToCartesianGradientMatrix,
219  const Eigen::Matrix3d& bodyFixedToIntegrationFrame,
220  Eigen::MatrixXd& partialsMatrix,
221  const int maximumAccelerationDegree,
222  const int maximumAccelerationOrder )
223 {
224  double preMultiplier = gravitionalParameter / referenceRadius;
225  const std::shared_ptr< basic_mathematics::LegendreCache > legendreCache = sphericalHarmonicsCache->getLegendreCache( );
226 
227  int degree, order;
228  for( unsigned int i = 0; i < blockIndices.size( ); i++ )
229  {
230  degree = blockIndices.at( i ).first;
231  order = blockIndices.at( i ).second;
232 
233  // Calculate and set partial of current degree and order.
234  if( degree <= maximumAccelerationDegree && order <= maximumAccelerationOrder )
235  {
236  partialsMatrix.block( 0, i, 3, 1 ) =
237  basic_mathematics::computePotentialGradient(
238  sphericalPosition( radiusIndex ),
239  sphericalHarmonicsCache->getReferenceRadiusRatioPowers( degree + 1 ),
240  sphericalHarmonicsCache->getCosineOfMultipleLongitude( order ),
241  sphericalHarmonicsCache->getSineOfMultipleLongitude( order ),
242  sphericalHarmonicsCache->getLegendreCache( )->getCurrentPolynomialParameterComplement( ),
243  preMultiplier, degree, order,
244  1.0, 0.0, legendreCache->getLegendrePolynomial( degree, order ),
245  legendreCache->getLegendrePolynomialDerivative( degree, order ) );
246  }
247  else
248  {
249  partialsMatrix.block( 0, i, 3, 1 ).setZero( );
250  }
251 
252  }
253 
254  // Transform partials to Cartesian position and integration frame.
255  partialsMatrix = bodyFixedToIntegrationFrame * sphericalToCartesianGradientMatrix * partialsMatrix;
256 }
