65 template< typename TimeType, typename StateScalarType >
66 void determineInitialStates(
67  nlohmann::json& jsonObject,
68  const simulation_setup::NamedBodyMap& bodyMap,
69  const std::shared_ptr< numerical_integrators::IntegratorSettings< TimeType > >& integratorSettings )
70 {
71  using namespace propagators;
72  using namespace basic_astrodynamics;
73  using namespace simulation_setup;
74  using K = Keys::Propagator;
75 
76  // Get propagators
77  nlohmann::json jsonPropagators = jsonObject.at( Keys::propagators );
78 
79  // Update propagators at jsonObject with initial states stored at JSON bodies settings
80  for ( nlohmann::json& jsonPropagator : jsonPropagators )
81  {
82  if ( ! isDefined( jsonPropagator, K::initialStates ) )
83  {
84  // Integrated state type
85  const IntegratedStateType integratedStateType =
86  getValue( jsonPropagator, K::integratedStateType, translational_state );
87 
88  // State size and associated stateKey
89  const unsigned int stateSize = getSingleIntegrationSize( integratedStateType );
90  const std::string stateKey = getAssociatedKey( integratedStateType );
91 
92  // Bodies to propagate
93  const std::vector< std::string > bodiesToPropagate =
94  getValue< std::vector< std::string > >( jsonPropagator, K::bodiesToPropagate );
95 
96  // System initial state
97  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialStates =
98  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero(
99  stateSize * bodiesToPropagate.size( ), 1 );
100 
101  // Get state for each body
102  for ( unsigned int i = 0; i < bodiesToPropagate.size( ); ++i )
103  {
104  const KeyPath stateKeyPath = Keys::bodies / bodiesToPropagate.at( i ) / stateKey;
105  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > bodyState;
106  try
107  {
108  if ( integratedStateType == translational_state )
109  {
110  Eigen::Matrix< StateScalarType, 6, 1 > stateToAdd;
111  std::string initialStateOrigin;
112 
113  stateToAdd.setZero( );
114 
115  const std::string centralBodyName = getValue< std::vector< std::string > >(
116  jsonPropagator, K::centralBodies ).at( i );
117  try
118  {
119  const nlohmann::json jsonState = getValue< nlohmann::json >( jsonObject, stateKeyPath );
120  initialStateOrigin = getValue< std::string >(
121  jsonState, Keys::Body::initialStateOrigin );
122 
123  if( initialStateOrigin != centralBodyName )
124  {
125  stateToAdd = getInitialStateOfBody< TimeType, StateScalarType >(
126  initialStateOrigin, centralBodyName,
127  bodyMap, integratorSettings->initialTime_ );
128  }
129  }
130  catch( ... )
131  {
132  initialStateOrigin = centralBodyName;
133  }
134 
135  std::shared_ptr< simulation_setup::Body > stateOriginBody;
136  if( bodyMap.count( initialStateOrigin ) == 0 )
137  {
138  stateOriginBody = nullptr;
139  }
140  else
141  {
142  stateOriginBody = bodyMap.at( initialStateOrigin );
143  }
144  bodyState = getCartesianState< StateScalarType >(
145  jsonObject, stateKeyPath, stateOriginBody,
146  integratorSettings->initialTime_ ) + stateToAdd;
147  }
148  else
149  {
150  bodyState = getValue< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >(
151  jsonObject, stateKeyPath );
152  }
153  }
154  catch( ... )
155  {
156  if ( integratedStateType == translational_state )
157  {
158  bodyState =
159  getInitialStateOfBody< TimeType, StateScalarType >(
160  bodiesToPropagate.at( i ),
161  getValue< std::vector< std::string > >( jsonPropagator, K::centralBodies ).at( i ),
162  bodyMap,
163  integratorSettings->initialTime_ );
164  jsonPropagator[ K::initialStates ] = initialStates;
165  }
166  else
167  {
168  throw std::runtime_error( "Error when getting initial state from JSON file, state type not recognized" );
169  }
170  }
171  initialStates.segment( i * stateSize, stateSize ) = bodyState;
172  }
173 
174  // Update system initial states
175  jsonPropagator[ K::initialStates ] = initialStates;
176  }
177  }
178 
179  jsonObject[ Keys::propagators ] = jsonPropagators;
180 }

222 template< typename TimeType >
223 TimeType getTerminationEpoch(
224  const std::shared_ptr< propagators::PropagationTerminationSettings >& terminationSettings )
225 {
226  using namespace propagators;
227 
228  std::shared_ptr< PropagationTimeTerminationSettings > timeTerminationSettings =
229  std::dynamic_pointer_cast< PropagationTimeTerminationSettings >( terminationSettings );
230  if ( timeTerminationSettings )
231  {
232  return timeTerminationSettings->terminationTime_;
233  }
234 
235  std::shared_ptr< PropagationHybridTerminationSettings > hybridTerminationSettings =
236  std::dynamic_pointer_cast< PropagationHybridTerminationSettings >( terminationSettings );
237  if ( hybridTerminationSettings )
238  {
239  for ( unsigned int i = 0; i < hybridTerminationSettings->terminationSettings_.size( ); ++i )
240  {
241  const TimeType endEpoch =
242  getTerminationEpoch< TimeType >( hybridTerminationSettings->terminationSettings_.at( i ) );
243  if ( ! isNaN( endEpoch ) )
244  {
245  return endEpoch;
246  }
247  }
248  }
249 
250  return TUDAT_NAN;
251 }

373 template< typename StateScalarType >
374 void from_json( const nlohmann::json& jsonObject,
375  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > >& multiTypePropagatorSettings )
376 {
377  using namespace simulation_setup;
378  using namespace json_interface;
379 
380  // Termination settings. If not provided, stop when epoch > simulation.finalEpoch
381  std::vector< std::shared_ptr< PropagationTerminationSettings > > terminationConditions;
382 
383  // Find user-defined conditions (and determine if time condition is missing)
384  bool timeConditionMissing = true;
385  if ( isDefined( jsonObject, Keys::termination ) )
386  {
387  std::shared_ptr< PropagationHybridTerminationSettings > userConditions =
388  getValue< std::shared_ptr< PropagationHybridTerminationSettings > >( jsonObject, Keys::termination );
389  terminationConditions.push_back( userConditions );
390 
391  for ( std::shared_ptr< PropagationTerminationSettings > condition : userConditions->terminationSettings_ )
392  {
393  std::shared_ptr< PropagationTimeTerminationSettings > timeCondition =
394  std::dynamic_pointer_cast< PropagationTimeTerminationSettings >( condition );
395  if ( timeCondition )
396  {
397  timeConditionMissing = false;
398  break;
399  }
400  }
401  }
402 
403  // If user did not provide conditions, or if finalEpoch is defined but the time condition is missing, create it
404  if ( ! isDefined( jsonObject, Keys::termination ) ||
405  ( isDefined( jsonObject, Keys::finalEpoch ) && timeConditionMissing ) )
406  {
407  terminationConditions.push_back( std::make_shared< PropagationTimeTerminationSettings >(
408  getValue< double >( jsonObject, Keys::finalEpoch ) ) );
409  }
410 
411  // If there's only one condition in total (either user-provided time, user-provided dependent or created time)
412  // use it as termination settings.
413  std::shared_ptr< PropagationTerminationSettings > terminationSettings;
414  if ( terminationConditions.size( ) == 1 )
415  {
416  terminationSettings = terminationConditions.at( 0 );
417  }
418  // If there are two conditions (those provided by the user, and the created time condition), combine them into
419  // hybrid termination settings satisfying any of the two conditions.
420  else
421  {
422  terminationSettings = std::make_shared< PropagationHybridTerminationSettings >( terminationConditions, true );
423  }
424 
425  multiTypePropagatorSettings = std::make_shared< MultiTypePropagatorSettings< StateScalarType > >(
426  getValue< std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >(
427  jsonObject, Keys::propagators ),
428  terminationSettings,
429  std::shared_ptr< DependentVariableSaveSettings >( ),
430  getValue< double >( jsonObject, Keys::options / Keys::Options::printInterval, TUDAT_NAN ) );
431 }

447 template< typename StateScalarType >
448 void to_json( nlohmann::json& jsonObject,
449  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > >& singleArcPropagatorSettings )
450 {
451  if ( ! singleArcPropagatorSettings )
452  {
453  return;
454  }
455  using namespace simulation_setup;
456  using namespace json_interface;
457  using K = Keys::Propagator;
458 
459  if( singleArcPropagatorSettings->getStateType( ) == hybrid )
460  {
461  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
462  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( singleArcPropagatorSettings );
463  if ( ! multiTypePropagatorSettings )
464  {
465  return;
466  }
467  using namespace simulation_setup;
468  using namespace json_interface;
469 
470  jsonObject[ Keys::propagators ] = getFlattenedMapValues< std::map, IntegratedStateType,
471  std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > >(
472  multiTypePropagatorSettings->propagatorSettingsMap_ );
473  jsonObject[ Keys::termination ] = multiTypePropagatorSettings->getTerminationSettings( );
474  if ( ! isNaN( multiTypePropagatorSettings->getPrintInterval( ) ) )
475  {
476  jsonObject[ Keys::options ][ Keys::Options::printInterval ] = multiTypePropagatorSettings->getPrintInterval( );
477  }
478  }
479  else
480  {
481 
482  // Common keys
483  const IntegratedStateType integratedStateType = singleArcPropagatorSettings->getStateType( );
484  jsonObject[ K::integratedStateType ] = integratedStateType;
485  if ( singleArcPropagatorSettings->getInitialStates( ).rows( ) > 0 )
486  {
487  jsonObject[ K::initialStates ] = singleArcPropagatorSettings->getInitialStates( );
488  }
489 
490  switch ( integratedStateType )
491  {
492  case translational_state:
493  {
494  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
495  translationalStatePropagatorSettings = std::dynamic_pointer_cast<
496  TranslationalStatePropagatorSettings< StateScalarType > >( singleArcPropagatorSettings );
497  assertNonnullptrPointer( translationalStatePropagatorSettings );
498  jsonObject[ K::type ] = translationalStatePropagatorSettings->propagator_;
499  jsonObject[ K::centralBodies ] = translationalStatePropagatorSettings->centralBodies_;
500  jsonObject[ K::bodiesToPropagate ] = translationalStatePropagatorSettings->bodiesToIntegrate_;
501  jsonObject[ K::accelerations ] = translationalStatePropagatorSettings->getAccelerationSettingsMap( );
502  return;
503  }
504  case body_mass_state:
505  {
506  std::shared_ptr< MassPropagatorSettings< StateScalarType > > massPropagatorSettings =
507  std::dynamic_pointer_cast< MassPropagatorSettings< StateScalarType > >( singleArcPropagatorSettings );
508  assertNonnullptrPointer( massPropagatorSettings );
509  jsonObject[ K::bodiesToPropagate ] = massPropagatorSettings->bodiesWithMassToPropagate_;
510  jsonObject[ K::massRateModels ] = massPropagatorSettings->getMassRateSettingsMap( );
511  return;
512  }
513  case rotational_state:
514  {
515  std::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationalStatePropagatorSettings =
516  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >(
517  singleArcPropagatorSettings );
518 
519  assertNonnullptrPointer( rotationalStatePropagatorSettings );
520  jsonObject[ K::type ] = rotationalStatePropagatorSettings->propagator_;
521  jsonObject[ K::bodiesToPropagate ] = rotationalStatePropagatorSettings->bodiesToIntegrate_;
522  jsonObject[ K::torques ] = rotationalStatePropagatorSettings->getTorqueSettingsMap( );
523  return;
524  }
525  case hybrid:
526  {
527  std::cerr << "Multitype (hybrid) propagation is implicitly supported by providing a list of propagators, "
528  << "but multitype propagators cannot be nested inside multitype propagators." << std::endl;
529  throw;
530  }
531  default:
532  handleUnimplementedEnumValue( integratedStateType, integratedStateTypes, unsupportedIntegratedStateTypes );
533  }
534  }

538 template< typename StateScalarType >
539 void from_json( const nlohmann::json& jsonObject,
540  std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > >& singleArcPropagatorSettings )
541 {
542  using namespace simulation_setup;
543  using namespace json_interface;
544  using K = Keys::Propagator;
545 
546  // Integrated state type
547  const IntegratedStateType integratedStateType =
548  getValue( jsonObject, K::integratedStateType, translational_state );
549 
550  // Named of bodies to propagate
551  const std::vector< std::string > bodiesToPropagate =
552  getValue< std::vector< std::string > >( jsonObject, K::bodiesToPropagate );
553 
554  // Initial states
555  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialStates =
556  getValue< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >( jsonObject, K::initialStates );
557 
558  // No termination settings ( epoch > TUDAT_NAN will always be false )
559  std::shared_ptr< PropagationTerminationSettings > terminationSettings =
560  std::make_shared< PropagationTimeTerminationSettings >( TUDAT_NAN );
561 
562  switch ( integratedStateType )
563  {
564  case translational_state:
565  {
566  TranslationalStatePropagatorSettings< StateScalarType > defaults(
567  { }, SelectedAccelerationMap( ), { }, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( ),
568  std::shared_ptr< PropagationTerminationSettings >( ) );
569  singleArcPropagatorSettings = std::make_shared< TranslationalStatePropagatorSettings< StateScalarType > >(
570  getValue< std::vector< std::string > >( jsonObject, K::centralBodies ),
571  getValue< SelectedAccelerationMap >( jsonObject, K::accelerations ),
572  bodiesToPropagate,
573  initialStates,
574  terminationSettings,
575  getValue( jsonObject, K::type, defaults.propagator_ ) );
576  return;
577  }
578  case body_mass_state:
579  {
580  MassPropagatorSettings< StateScalarType > defaults(
581  { }, SelectedMassRateModelMap( ), Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( ), nullptr );
582  singleArcPropagatorSettings = std::make_shared< MassPropagatorSettings< StateScalarType > >(
583  bodiesToPropagate,
584  getValue< SelectedMassRateModelMap >( jsonObject, K::massRateModels ),
585  initialStates,
586  terminationSettings );
587  return;
588  }
589  case rotational_state:
590  {
591  RotationalStatePropagatorSettings< StateScalarType > defaults(
592  SelectedTorqueMap( ), { }, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >( ), nullptr );
593  singleArcPropagatorSettings = std::make_shared< RotationalStatePropagatorSettings< StateScalarType > >(
594  getValue< SelectedTorqueMap >( jsonObject, K::torques ),
595  bodiesToPropagate,
596  initialStates,
597  terminationSettings,
598  getValue( jsonObject, K::type, defaults.propagator_ ) );
599  return;
600  }
601  case hybrid:
602  {
603  std::cerr << "Multitype (hybrid) propagation is implicitly supported by providing a list of propagators, "
604  << "but multitype propagators cannot be nested inside multitype propagators." << std::endl;
605  throw;
606  }
607  default:
608  handleUnimplementedEnumValue( integratedStateType, integratedStateTypes, unsupportedIntegratedStateTypes );
609  }
