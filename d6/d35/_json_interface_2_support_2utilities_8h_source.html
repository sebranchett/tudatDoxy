198 inline std::string url_encode( const std::string &value )
199 {
200  std::ostringstream escaped;
201  escaped.fill( '0' );
202  escaped << std::hex;
203 
204  for ( std::string::const_iterator i = value.begin(), n = value.end(); i != n; ++i )
205  {
206  std::string::value_type c = (*i);
207 
208  // Keep alphanumeric and other accepted characters intact
209  if ( isalnum(c) || c == '-' || c == '_' || c == '.' || c == '~' )
210  {
211  escaped << c;
212  continue;
213  }
214 
215  // Any other characters are percent-encoded
216  escaped << std::uppercase;
217  unsigned char uc = c;
218  escaped << '%' << std::setw( 2 ) << int( uc );
219  escaped << std::nouppercase;
220  }
221 
222  return escaped.str( );
223 }
