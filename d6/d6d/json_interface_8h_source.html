663 template< typename TimeType, typename StateScalarType >
664 void to_json( nlohmann::json& jsonObject,
665  const std::shared_ptr< JsonSimulationManager< TimeType, StateScalarType > >& jsonSimulationManager )
666 {
667  if ( ! jsonSimulationManager )
668  {
669  return;
670  }
671 
672  // integrator
673  jsonObject[ Keys::integrator ] = jsonSimulationManager->getIntegratorSettings( );
674 
675  // spice
676  assignIfNotnullptr( jsonObject, Keys::spice, jsonSimulationManager->getSpiceSettings( ) );
677 
678  // bodies
679  jsonObject[ Keys::globalFrameOrigin ] = jsonSimulationManager->getGlobalFrameOrigin( );
680  jsonObject[ Keys::globalFrameOrientation ] = jsonSimulationManager->getGlobalFrameOrientation( );
681  jsonObject[ Keys::bodies ] = jsonSimulationManager->getBodySettingsMap( );
682 
683  // export
684  assignIfNotEmpty( jsonObject, Keys::xport, jsonSimulationManager->getExportSettingsVector( ) );
685 
686  // options
687  jsonObject[ Keys::options ] = jsonSimulationManager->getApplicationOptions( );
688 
689  // propagation + termination + options.printInterval
690  propagators::to_json( jsonObject, std::dynamic_pointer_cast<
691  propagators::SingleArcPropagatorSettings< StateScalarType > >( jsonSimulationManager->getPropagatorSettings( ) ) );
692 }
