43 Eigen::VectorXd ConicalFrustum::getSurfacePoint( double azimuthAngle, double lengthFraction )
44 {
45  // Determines the radius of the cone at the given length fraction.
46  double localRadius_ = startRadius_ + length_ * lengthFraction * std::tan( coneHalfAngle_ );
47 
48 
49  // Set x and y coordinate of untransformed cone.
50  cartesianPositionVector_ = coordinate_conversions::
51  convertCylindricalToCartesian( Eigen::Vector3d( localRadius_, azimuthAngle, 0.0 ) );
52 
53  // Set z coordinate of untransformed cone.
54  cartesianPositionVector_( 2 ) = -length_ * lengthFraction;
55 
56  // Transform conical frustum to desired position and orientation.
57  transformPoint( cartesianPositionVector_ );
58 
59  // Return Cartesian position vector.
60  return cartesianPositionVector_;
61 }

64 Eigen::VectorXd ConicalFrustum::getSurfaceDerivative(
65  const double lengthFraction, const double azimuthAngle,
66  const int powerOfLengthFractionDerivative, const int powerOfAzimuthAngleDerivative )
67 {
68  // Declare and set size of derivative vector.
69  Eigen::VectorXd derivative_ = Eigen::VectorXd( 3 );
70 
71  // No negative derivatives may be retrieved, a zero vector is returned in
72  // this case.
73  if ( powerOfLengthFractionDerivative < 0 || powerOfAzimuthAngleDerivative < 0 )
74  {
75  derivative_( 0 ) = 0.0;
76  derivative_( 1 ) = 0.0;
77  derivative_( 2 ) = 0.0;
78 
79  throw std::runtime_error( " No negative derivatives allowed when retrieving cone derivative " );
80  }
81 
82  // When requesting the zeroth derivative with respect to the two
83  // independent variables, the surface point is returned.
84  else if ( powerOfLengthFractionDerivative == 0 && powerOfAzimuthAngleDerivative == 0 )
85  {
86  derivative_ = getSurfacePoint( lengthFraction, azimuthAngle );
87  }
88 
89  // The contributions of the two parametrizing variables are determined and
90  // then multiplied component-wise.
91  else
92  {
93  // Declare and set sizes of contribution vectors.
94  Eigen::VectorXd derivative1Contribution_ = Eigen::VectorXd( 3 );
95  Eigen::VectorXd derivative2Contribution_ = Eigen::VectorXd( 3 );
96 
97  switch( powerOfLengthFractionDerivative )
98  {
99  case( 0 ):
100 
101  derivative1Contribution_( 0 ) = 0.0;
102  derivative1Contribution_( 1 ) = 0.0;
103  derivative1Contribution_( 2 ) = -length_ * lengthFraction;
104  break;
105 
106  case( 1 ):
107 
108  derivative1Contribution_( 0 ) = 0.0;
109  derivative1Contribution_( 1 ) = 0.0;
110  derivative1Contribution_( 2 ) = -length_;
111  break;
112 
113  // For all higher derivatives, all components are zero.
114  default:
115 
116  derivative1Contribution_( 0 ) = 0.0;
117  derivative1Contribution_( 1 ) = 0.0;
118  derivative1Contribution_( 2 ) = 0.0;
119  break;
120  }
121 
122  // Since this derivative is "cyclical", as it is only dependant on sines
123  // and cosines, only the "modulo 4"th derivative need be determined.
124  // Derivatives are determined from the form of the cylindrical coordinates,
125  // see basic_mathematics::coordinateConversions::convertSphericalToCartesian
126  switch( powerOfAzimuthAngleDerivative % 4 )
127  {
128  case( 0 ):
129 
130  derivative2Contribution_( 0 ) = -sin( azimuthAngle );
131  derivative2Contribution_( 1 ) = cos( azimuthAngle );
132  derivative2Contribution_( 2 ) = 0.0;
133  break;
134 
135  case( 1 ):
136 
137  derivative2Contribution_( 0 ) = -cos( azimuthAngle );
138  derivative2Contribution_( 1 ) = -sin( azimuthAngle );
139  derivative2Contribution_( 2 ) = 0.0;
140  break;
141 
142  case( 2 ):
143 
144  derivative2Contribution_( 0 ) = sin( azimuthAngle );
145  derivative2Contribution_( 1 ) = -cos( azimuthAngle );
146  derivative2Contribution_( 2 ) = 0.0;
147  break;
148 
149  case( 3 ):
150 
151  derivative2Contribution_( 0 ) = cos( azimuthAngle );
152  derivative2Contribution_( 1 ) = sin( azimuthAngle );
153  derivative2Contribution_( 2 ) = 0.0;
154  break;
155 
156  default:
157 
158  throw std::runtime_error( " Bad value for powerOfAzimuthAngleDerivative ( mod 4 ) of value is not 0, 1, 2 or 3 ");
159  }
160 
161  // Combine contributions to derivative.
162  derivative_( 0 ) = derivative1Contribution_( 0 ) * derivative2Contribution_( 0 );
163  derivative_( 1 ) = derivative1Contribution_( 1 ) * derivative2Contribution_( 1 );
164  derivative_( 2 ) = derivative1Contribution_( 2 ) * derivative2Contribution_( 2 );
165 
166  // Rotate derivatives to take into account rotation of cone.
167  derivative_ = scalingMatrix_ * rotationMatrix_ * derivative_;
168  }
169 
170  // Return derivative vector.
171  return derivative_;
172 }
