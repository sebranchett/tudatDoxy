105 template< typename ObservationScalarType = double >
106 ObservationScalarType computeOneWayFirstOrderDopplerTaylorSeriesExpansion(
107  Eigen::Matrix< ObservationScalarType, 6, 1 >& transmitterState,
108  Eigen::Matrix< ObservationScalarType, 6, 1 >& receiverState,
109  Eigen::Matrix< ObservationScalarType, 1, 3 >& lightTimeWrtTransmitterPositionPartial,
110  Eigen::Matrix< ObservationScalarType, 1, 3 >& lightTimeWrtReceiverPositionPartial,
111  const int taylorSeriesOrder )
112 {
113  // Compute projected velocity components
114  ObservationScalarType transmitterTerm =
115  ( -lightTimeWrtTransmitterPositionPartial * ( transmitterState.segment( 3, 3 ) ) /
116  physical_constants::getSpeedOfLight< ObservationScalarType >( ) )( 0 );
117  ObservationScalarType receiverTerm =
118  ( lightTimeWrtReceiverPositionPartial * ( receiverState.segment( 3, 3 ) ) /
119  physical_constants::getSpeedOfLight< ObservationScalarType >( ) )( 0 );
120 
121  // Compute Taylor series of 1/(1-r21*v2) up to required order
122  ObservationScalarType currentTaylorSeriesTerm =
123  mathematical_constants::getFloatingInteger< ObservationScalarType >( 1 );
124  ObservationScalarType currentTaylorSeries = mathematical_constants::getFloatingInteger< ObservationScalarType >( 0 );
125  for( int i = 0; i < taylorSeriesOrder; i++ )
126  {
127  currentTaylorSeriesTerm *= transmitterTerm;
128  currentTaylorSeries += currentTaylorSeriesTerm;
129  }
130 
131  // Compute Doppler term
132  return -receiverTerm + currentTaylorSeries *
133  ( mathematical_constants::getFloatingInteger< ObservationScalarType >( 1 ) - receiverTerm );
134 }

145 template< typename ObservationScalarType = double >
146 ObservationScalarType computeDopplerProperTimeInfluenceTaylorSeriesExpansion(
147  const ObservationScalarType transmitterProperTimeRateDifference,
148  const ObservationScalarType receiverProperTimeRateDifference,
149  const int taylorSeriesOrder )
150 {
151  ObservationScalarType currentTaylorSeriesTerm = mathematical_constants::getFloatingInteger< ObservationScalarType >( 1 );
152  ObservationScalarType currentTaylorSeries = mathematical_constants::getFloatingInteger< ObservationScalarType >( 0 );
153  for( int i = 0; i < taylorSeriesOrder; i++ )
154  {
155  currentTaylorSeriesTerm *= -receiverProperTimeRateDifference;
156  currentTaylorSeries += currentTaylorSeriesTerm;
157  }
158 
159  return ( currentTaylorSeries + transmitterProperTimeRateDifference ) +
160  transmitterProperTimeRateDifference * currentTaylorSeries;
161 }

548  Eigen::Matrix< ObservationScalarType, 1, 1 > computeIdealObservationsWithLinkEndData(
549  const TimeType time,
550  const LinkEndType linkEndAssociatedWithTime,
551  std::vector< double >& linkEndTimes,
552  std::vector< Eigen::Matrix< double, 6, 1 > >& linkEndStates )
553  {
554  ObservationScalarType lightTime = TUDAT_NAN;
555  TimeType transmissionTime = TUDAT_NAN, receptionTime = TUDAT_NAN;
556 
557  // Compute light time
558  switch( linkEndAssociatedWithTime )
559  {
560  case receiver:
561  lightTime = lightTimeCalculator_->calculateLightTimeWithLinkEndsStates(
562  receiverState_, transmitterState_, time, true );
563  transmissionTime = time - lightTime;
564  receptionTime = time;
565  break;
566 
567  case transmitter:
568  lightTime = lightTimeCalculator_->calculateLightTimeWithLinkEndsStates(
569  receiverState_, transmitterState_, time, false );
570  transmissionTime = time;
571  receptionTime = time + lightTime;
572  break;
573  default:
574  throw std::runtime_error(
575  "Error when calculating one way Doppler observation, link end is not transmitter or receiver" );
576  }
577 
578  linkEndTimes.clear( );
579  linkEndStates.clear( );
580 
581  // Save link end times and states
582  linkEndTimes.push_back( transmissionTime );
583  linkEndTimes.push_back( receptionTime );
584 
585  linkEndStates.push_back( transmitterState_.template cast< double >( ) );
586  linkEndStates.push_back( receiverState_.template cast< double >( ) );
587 
588  // Compute transmitter and receiver proper time rate
589  ObservationScalarType transmitterProperTimeDifference =
590  mathematical_constants::getFloatingInteger< ObservationScalarType >( 0 );
591  if( transmitterProperTimeRateCalculator_ != nullptr )
592  {
593  transmitterProperTimeDifference = static_cast< ObservationScalarType >(
594  transmitterProperTimeRateCalculator_->getOberverProperTimeDeviation(
595  linkEndTimes, linkEndStates ) );
596  }
597  ObservationScalarType receiverProperTimeDifference =
598  mathematical_constants::getFloatingInteger< ObservationScalarType >( 0 );
599  if( receiverProperTimeRateCalculator_ != nullptr )
600  {
601  receiverProperTimeDifference = static_cast< ObservationScalarType >(
602  receiverProperTimeRateCalculator_->getOberverProperTimeDeviation(
603  linkEndTimes, linkEndStates ) );
604  }
605 
606  // Compute proper time correction term
607  ObservationScalarType properTimeCorrectionTerm =
608  computeDopplerProperTimeInfluenceTaylorSeriesExpansion(
609  transmitterProperTimeDifference, receiverProperTimeDifference, taylorSeriesExpansionOrder_ );
610 
611  // Compute first-order (geometrical) one-way Doppler contribution
612  lightTimePartialWrtReceiverPosition_ =
613  lightTimeCalculator_->getPartialOfLightTimeWrtLinkEndPosition(
614  transmitterState_, receiverState_, transmissionTime, receptionTime, true );
615  lightTimePartialWrtTransmitterPosition_ =
616  lightTimeCalculator_->getPartialOfLightTimeWrtLinkEndPosition(
617  transmitterState_, receiverState_, transmissionTime, receptionTime, false );
618  ObservationScalarType firstOrderDopplerObservable =
619  computeOneWayFirstOrderDopplerTaylorSeriesExpansion<
620  ObservationScalarType >(
621  transmitterState_, receiverState_,
622  lightTimePartialWrtTransmitterPosition_, lightTimePartialWrtReceiverPosition_,
623  taylorSeriesExpansionOrder_ );
624 
625  // Compute full Doppler observable and return
626  ObservationScalarType totalDopplerObservable = firstOrderDopplerObservable *
627  ( mathematical_constants::getFloatingInteger< ObservationScalarType >( 1 ) + properTimeCorrectionTerm ) +
628  properTimeCorrectionTerm;
629  return ( Eigen::Matrix< ObservationScalarType, 1, 1 >( ) << totalDopplerObservable ).finished( );
630  }
