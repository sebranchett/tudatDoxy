40 double computeKeplerRadialDistance( const double semiMajorAxis,
41  const double eccentricity,
42  const double trueAnomaly )
43 {
44  return ( semiMajorAxis * ( 1.0 - eccentricity * eccentricity ) ) / ( 1.0 + eccentricity * std::cos( trueAnomaly ) );
