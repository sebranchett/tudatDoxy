97 void TwoLineElementsTextFileReader::readAndStoreData( )
98 {
99  // Reset the datafile.
100  containerOfDataFromFile_.clear( );
101 
102  // Whilst the end of the data file has not been reached, continue reading
103  // from lines from data file.
104  while( !dataFile_.eof( ) )
105  {
106  // Get next line of data from data file and store in a string.
107  getline( dataFile_, stringOfData_ );
108 
109  // Check if line of data is header line.
110  if ( lineCounter_ <= numberOfHeaderLines_ )
111  {
112  // Store header line data.
113  containerOfHeaderDataFromFile_[ lineCounter_ ] = stringOfData_;
114  }
115 
116  // Else process non-header data line.
117  else
118  {
119  // Check if string doesn't start with set starting character, if string
120  // is not empty, and if the skip keyword is not in the string.
121  if ( ( ( !startingCharacter_.empty( ) && stringOfData_.substr( 0, 1 )
122  .compare( startingCharacter_ ) != 0 )
123  || ( !skipKeyword_.empty( ) && stringOfData_.find( skipKeyword_ )
124  == std::string::npos )
125  || ( startingCharacter_.empty( ) && skipKeyword_.empty( ) ) )
126  && !stringOfData_.empty( ) )
127  {
128  // Store string in container.
129  containerOfDataFromFile_[ lineCounter_ ] = stringOfData_;
130  }
131  }
132 
133  // Increment line counter.
134  lineCounter_++;
135  }
136 }

160 void TwoLineElementsTextFileReader::stripEndOfLineCharacters(
161  LineBasedStringDataMap& containerOfLinesOfData )
162 {
163  // Declare local variables.
164  // Declare string iterator.
165  std::string::iterator iteratorString_;
166 
167  // Loop through all the strings stored in the container.
168  for ( LineBasedStringDataMap::iterator iteratorContainerOfDataFromFile_
169  = containerOfLinesOfData.begin( );
170  iteratorContainerOfDataFromFile_ != containerOfLinesOfData.end( );
171  iteratorContainerOfDataFromFile_++ )
172  {
173  // Loop through all the characters in the string.
174  for ( iteratorString_ = iteratorContainerOfDataFromFile_->second.begin( );
175  iteratorString_ != iteratorContainerOfDataFromFile_->second.end( );
176  iteratorString_++ )
177  {
178  // Check if end-of-line characters are present in string.
179  // The end-of-line characters are checked for their integer
180  // equivalents. See: http://www.asciitable.com/.
181  if ( static_cast< int >( *iteratorString_ ) == 10
182  || static_cast< int >( *iteratorString_ ) == 13 )
183  {
184  // Strip end-of-line character from string.
185  iteratorContainerOfDataFromFile_->second.erase( iteratorString_ );
186 
187  // Decrement string iterator since character was erased from string.
188  iteratorString_--;
189  }
190  }
191  }

195 void TwoLineElementsTextFileReader::storeTwoLineElementData( )
196 {
197  using boost::algorithm::trim_copy;
198 
199  // Strip End-Of-Line characters from data container.
200  stripEndOfLineCharacters( containerOfDataFromFile_ );
201 
202  // Calculate number of objects in catalog file.
203  numberOfObjects_ = static_cast< int >( lineCounter_ / numberOfLinesPerTwoLineElementDatum_ );
204 
205  // Set TLE data vector size.
206  twoLineElementData_.resize( numberOfObjects_ );
207 
208  // Create vector of the three lines of a single object's TLE data as strings.
209  std::vector< std::string > twoLineElementString_( 3 );
210 
211  // Create the object counter.
212  unsigned int objectNumberCounter_ = 0;
213 
214  // Declare Keplerian elements variables.
215  double inclination_;
216  double rightAscensionOfAscendingNode_;
217  double eccentricity_;
218  double argumentOfPerigee_;
219  double meanMotion_;
220 
221  // Declare approximate number of revolutions, remainder, and lost part.
222  int approximateNumberOfRevolutions_;
223  int approximateNumberOfRevolutionsRemainder_;
224  int lostNumberOfRevolutions_;
225 
226  // Reference: Table 2 in (Vallado, D.A., et al., 2006).
227  const double earthWithWorldGeodeticSystem72GravitationalParameter = 398600.8e9;
228 
229  // For every 3 lines, read the data from 3 consecutive strings of TLE data
230  // and convert them to the TLE data variables
231  for ( unsigned int i = 1; i < lineCounter_; i += numberOfLinesPerTwoLineElementDatum_ )
232  {
233  // General setup for variable storing.
234  //---------------------------------------------------------------------
235 
236  // Fill the vector with the line strings from the data container.
237  if ( numberOfLinesPerTwoLineElementDatum_ == 3 )
238  {
239  twoLineElementString_.at( 0 ) = containerOfDataFromFile_.at( i );
240  twoLineElementString_.at( 1 ) = containerOfDataFromFile_.at( i + 1 );
241  twoLineElementString_.at( 2 ) = containerOfDataFromFile_.at( i + 2 );
242  }
243 
244  else if ( numberOfLinesPerTwoLineElementDatum_ == 2 )
245  {
246  twoLineElementString_.at( 1 ) = containerOfDataFromFile_.at( i );
247  twoLineElementString_.at( 2 ) = containerOfDataFromFile_.at( i + 1 );
248  }
249 
250  // Push the TLE strings to the TLE data container.
251  twoLineElementData_[ objectNumberCounter_ ].twoLineElementStrings = twoLineElementString_;
252 
253  // Push the line numbers to the TLE data container.
254  twoLineElementData_[ objectNumberCounter_ ].lineNumbers.push_back( i );
255  twoLineElementData_[ objectNumberCounter_ ].lineNumbers.push_back( i + 1 );
256  if ( numberOfLinesPerTwoLineElementDatum_ == 3 )
257  {
258  twoLineElementData_[ objectNumberCounter_ ].lineNumbers.push_back( i + 2 );
259  }
260 
261  // Initiate a stringstream for each line.
262  std::stringstream line0StringStream_( std::stringstream::in | std::stringstream::out );
263  std::stringstream line1StringStream_( std::stringstream::in | std::stringstream::out );
264  std::stringstream line2StringStream_( std::stringstream::in | std::stringstream::out );
265 
266  // Insert the strings into the stringstreams
267  if ( numberOfLinesPerTwoLineElementDatum_ == 3 )
268  {
269  line0StringStream_ << twoLineElementString_.at( 0 );
270  }
271  line1StringStream_ << twoLineElementString_.at( 1 );
272  line2StringStream_ << twoLineElementString_.at( 2 );
273 
274  // Line-0 variable storing.
275  //---------------------------------------------------------------------
276  if ( numberOfLinesPerTwoLineElementDatum_ == 3 )
277  {
278  // Declare string containing part of name of object.
279  std::string namePart_;
280 
281  // Loop through the stringstream and read words that constitute name of
282  // object. Store name parts in objectName storage container.
283  while ( line0StringStream_ >> namePart_ )
284  {
285  twoLineElementData_[ objectNumberCounter_ ].objectName.push_back( namePart_ );
286  }
287 
288  // Insert the entire line-0 string in the object name string.
289  twoLineElementData_[ objectNumberCounter_ ].objectNameString
290  = twoLineElementString_.at( 0 );
291  }
292 
293  // Line-1 variable storing.
294  //---------------------------------------------------------------------
295  // Fill all line-1 variables of the object structure with substrings
296  // of the line-1 string using the template function to transform the
297  // string to another type.
298  // See reference for which columns are assigned to which variable.
299 
300  // Get line number integer of line-1 from string.
301  twoLineElementData_[ objectNumberCounter_ ].lineNumberLine1 =
302  std::stoul(
303  trim_copy( twoLineElementString_.at( 1 ).substr( 0, 1 ) ) );
304 
305  // Get object indentification number integer of line-1 from string
306  twoLineElementData_[ objectNumberCounter_ ].objectIdentificationNumber =
307  std::stoul(
308  trim_copy( twoLineElementString_.at( 1 ).substr( 2, 5 ) ) );
309 
310 
311  // Get classification character from string.
312  twoLineElementData_[ objectNumberCounter_ ].tleClassification =
313  twoLineElementString_.at( 1 ) [ 7 ];
314 
315  // Get launch year integer from string.
316  twoLineElementData_[ objectNumberCounter_ ].launchYear =
317  std::stoul(
318  trim_copy( twoLineElementString_.at( 1 ).substr( 9, 2 ) ) );
319 
320  // Calculate four-digit launch year from the above.
321  if ( twoLineElementData_[ objectNumberCounter_ ].launchYear > 56 )
322  {
323  twoLineElementData_[ objectNumberCounter_ ].fourDigitlaunchYear =
324  twoLineElementData_[ objectNumberCounter_ ].launchYear + 1900;
325  }
326 
327  else
328  {
329  twoLineElementData_[ objectNumberCounter_ ].fourDigitlaunchYear =
330  twoLineElementData_[ objectNumberCounter_ ].launchYear + 2000;
331  }
332 
333  // Get launch number integer from string.
334  twoLineElementData_[ objectNumberCounter_ ].launchNumber =
335  std::stoul(
336  trim_copy( twoLineElementString_.at( 1 ).substr( 11, 3 ) ) );
337 
338  // Get launch part string from string.
339  twoLineElementData_[ objectNumberCounter_ ].launchPart =
340  twoLineElementString_.at( 1 ).substr( 14, 3 );
341 
342  // Get epoch year integer from string.
343  twoLineElementData_[ objectNumberCounter_ ].epochYear =
344  std::stoul(
345  trim_copy( twoLineElementString_.at( 1 ).substr( 18, 2 ) ) );
346 
347  // Calculate four-digit epoch year from the above.
348  if ( twoLineElementData_[ objectNumberCounter_ ].epochYear > 56 )
349  {
350  twoLineElementData_[ objectNumberCounter_ ].fourDigitEpochYear =
351  twoLineElementData_[ objectNumberCounter_ ].epochYear + 1900;
352  }
353 
354  else
355  {
356  twoLineElementData_[ objectNumberCounter_ ].fourDigitEpochYear =
357  twoLineElementData_[ objectNumberCounter_ ].epochYear + 2000;
358  }
359 
360  // Get epoch day double from string.
361  twoLineElementData_[ objectNumberCounter_ ].epochDay =
362  std::stod(
363  trim_copy( twoLineElementString_.at( 1 ).substr( 20, 12 ) ) );
364 
365  // Get "first-derivative of mean motion divided by two" double from string.
366  twoLineElementData_[ objectNumberCounter_].firstDerivativeOfMeanMotionDividedByTwo =
367  std::stod(
368  trim_copy( twoLineElementString_.at( 1 ).substr( 33, 10 ) ) );
369 
370  // Get coefficient of scientific notation of "second-derivative of mean motion divided
371  // by six" double from string,
372  // Apply implied leading decimal point.
373  twoLineElementData_[ objectNumberCounter_ ]
374  .coefficientOfSecondDerivativeOfMeanMotionDividedBySix =
375  std::stod(
376  trim_copy( twoLineElementString_.at( 1 ).substr( 44, 6 ) ) ) / 100000.0;
377 
378  // Get exponent of scientific notation of "second-derivative of mean motion divided
379  // by six" integer from string.
380  twoLineElementData_[ objectNumberCounter_]
381  .exponentOfSecondDerivativeOfMeanMotionDividedBySix =
382  std::stod(
383  trim_copy( twoLineElementString_.at( 1 ).substr( 50, 2 ) ) );
384 
385  // Calculate "second-derivative of mean motion divided by six" double from the above two.
386  twoLineElementData_[ objectNumberCounter_ ].secondDerivativeOfMeanMotionDividedBySix =
387  twoLineElementData_[ objectNumberCounter_]
388  .coefficientOfSecondDerivativeOfMeanMotionDividedBySix
389  * pow( 10, twoLineElementData_[ objectNumberCounter_ ]
390  .exponentOfSecondDerivativeOfMeanMotionDividedBySix );
391 
392  // Get coefficient of scientific notation of "B* divided by six" double
393  // from string; apply implied leading decimal point.
394  twoLineElementData_[ objectNumberCounter_ ].coefficientOfBStar =
395  std::stod(
396  trim_copy( twoLineElementString_.at( 1 ).substr( 53, 6 ) ) ) /
397  100000.0;
398 
399  // Get exponent of scientific notation of B* integer from string
400  twoLineElementData_[ objectNumberCounter_ ].exponentOfBStar =
401  std::stoi(
402  trim_copy( twoLineElementString_.at( 1 ).substr( 59, 2 ) ) );
403 
404  // Calculate B* double from the above two.
405  twoLineElementData_[ objectNumberCounter_ ].bStar =
406  twoLineElementData_[ objectNumberCounter_ ].coefficientOfBStar
407  * pow( 10.0, twoLineElementData_[ objectNumberCounter_ ].exponentOfBStar );
408 
409  // Get orbital model integer from string.
410  twoLineElementData_[ objectNumberCounter_ ].orbitalModel =
411  std::stoul(
412  trim_copy( twoLineElementString_.at( 1 ).substr( 62, 1 ) ) );
413 
414  // Get TLE number integer from string.
415  twoLineElementData_[ objectNumberCounter_ ].tleNumber =
416  std::stoul(
417  trim_copy( twoLineElementString_.at( 1 ).substr( 64, 4 ) ) );
418 
419  // Get modulo-10 checksum integer from string.
420  twoLineElementData_[ objectNumberCounter_ ].modulo10CheckSumLine1 =
421  std::stoul(
422  trim_copy( twoLineElementString_.at( 1 ).substr( 68, 1 ) ) );
423 
424  // Line-2 variable storing.
425  //---------------------------------------------------------------------
426  // Fill all line-2 variables of the object structure partly with the
427  // stringstream and partly with with substrings of the line-2 string
428  // using the template function to transform the string to another type.
429  // See reference for which columns are assigned to which variable.
430 
431  // Get line number integer of line-2 from stringstream.
432  line2StringStream_ >> twoLineElementData_[ objectNumberCounter_ ].lineNumberLine2;
433 
434  // Get object identification number integer of line-2 from stringstream.
435  line2StringStream_ >> twoLineElementData_[ objectNumberCounter_ ]
436  .objectIdentificationNumberLine2;
437 
438  // Get inclination double from stringstream.
439  line2StringStream_ >> inclination_;
440  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
441  orbital_element_conversions::inclinationIndex ) = inclination_;
442 
443  // Get right ascension of ascending node double from stringstream.
444  line2StringStream_ >> rightAscensionOfAscendingNode_;
445  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
446  orbital_element_conversions::longitudeOfAscendingNodeIndex )
447  = rightAscensionOfAscendingNode_;
448 
449  // Get eccetricity double from stringstream.
450  line2StringStream_ >> eccentricity_;
451  eccentricity_ /= 10000000;
452  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
453  orbital_element_conversions::eccentricityIndex ) = eccentricity_;
454 
455  // Get argument of perigee double from stringstream
456  line2StringStream_ >> argumentOfPerigee_;
457  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
458  orbital_element_conversions::argumentOfPeriapsisIndex ) = argumentOfPerigee_;
459 
460  // Get mean anomaly double from stringstream.
461  line2StringStream_ >> twoLineElementData_[ objectNumberCounter_ ].meanAnomaly;
462 
463  // Get mean motion double from line-2 string.
464  twoLineElementData_[ objectNumberCounter_ ].meanMotionInRevolutionsPerDay =
465  std::stod(
466  trim_copy( twoLineElementString_[ 2 ].substr( 52, 11 ) ) );
467 
468  // Get revolution number integer from line-2 string.
469  twoLineElementData_[ objectNumberCounter_ ].revolutionNumber =
470  std::stoi(
471  trim_copy( twoLineElementString_[ 2 ].substr( 63, 5 ) ) );
472 
473  // Get modulo-10 checksum integer of line-2 from line-2 string.
474  twoLineElementData_[ objectNumberCounter_ ].modulo10CheckSumLine2 =
475  std::stoul(
476  trim_copy( twoLineElementString_[ 2 ].substr( 68, 1 ) ) );
477 
478  // Calculate the approximate total number of revolutions, as the counter resets to 0 after
479  // passing by 99999, insert the current year in the following equation.
480  approximateNumberOfRevolutions_ = twoLineElementData_[ objectNumberCounter_ ]
481  .meanMotionInRevolutionsPerDay
482  * ( currentYear_
483  - twoLineElementData_[ objectNumberCounter_ ].fourDigitlaunchYear )
484  * physical_constants::JULIAN_YEAR_IN_DAYS;
485  approximateNumberOfRevolutionsRemainder_ = approximateNumberOfRevolutions_ % 100000;
486  lostNumberOfRevolutions_ = approximateNumberOfRevolutions_
487  - approximateNumberOfRevolutionsRemainder_;
488 
489  // Check if the counter has been reset after passing 99999.
490  if ( ( twoLineElementData_[ objectNumberCounter_ ].revolutionNumber -
491  approximateNumberOfRevolutionsRemainder_ ) <=
492  ( approximateNumberOfRevolutionsRemainder_ + 100000 -
493  twoLineElementData_[ objectNumberCounter_ ].revolutionNumber ) )
494  {
495  twoLineElementData_[ objectNumberCounter_ ].totalRevolutionNumber =
496  lostNumberOfRevolutions_ + twoLineElementData_[ objectNumberCounter_ ]
497  .revolutionNumber;
498  }
499 
500  else
501  {
502  twoLineElementData_[ objectNumberCounter_ ].totalRevolutionNumber =
503  lostNumberOfRevolutions_ - 100000
504  + twoLineElementData_[ objectNumberCounter_ ].revolutionNumber;
505  }
506 
507  // Check if total number of revolutions is negative.
508  if ( twoLineElementData_[ objectNumberCounter_ ].totalRevolutionNumber < 0 )
509  {
510  twoLineElementData_[ objectNumberCounter_ ].totalRevolutionNumber =
511  twoLineElementData_[ objectNumberCounter_ ].revolutionNumber;
512  }
513 
514  // Semi-major axis of the object is calculated from the other TLE variables.
515  meanMotion_ = twoLineElementData_[ objectNumberCounter_ ].meanMotionInRevolutionsPerDay
516  * 2.0 * PI / physical_constants::JULIAN_DAY;
517  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
518  orbital_element_conversions::semiMajorAxisIndex )
519  = orbital_element_conversions::
520  convertEllipticalMeanMotionToSemiMajorAxis(
521  meanMotion_, earthWithWorldGeodeticSystem72GravitationalParameter );
522 
523  // Perigee of the object is calculated from the other TLE variables.
524  twoLineElementData_[ objectNumberCounter_ ].perigee =
525  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
526  orbital_element_conversions::semiMajorAxisIndex )
527  * ( 1.0 - twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
528  orbital_element_conversions::eccentricityIndex ) );
529 
530  // Apogee of the object is calculated from the other TLE variables.
531  twoLineElementData_[ objectNumberCounter_ ].apogee =
532  twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
533  orbital_element_conversions::semiMajorAxisIndex )
534  * ( 1.0 + twoLineElementData_[ objectNumberCounter_ ].TLEKeplerianElements(
535  orbital_element_conversions::eccentricityIndex ) );
536 
537  // Increment object number counter by one.
538  objectNumberCounter_++;
539  }

543 std::multimap< int, std::string > TwoLineElementsTextFileReader::checkTwoLineElementsFileIntegrity( )
544 {
545  // Create vector of the three lines of a TLE as strings
546  std::vector< std::string > twoLineElementString_( 3 );
547 
548  // Boolean which turns to true if one of the tests is not passed.
549  bool isObjectErroneous = false;
550 
551  // Vector of object numbers of objects with corrupted TLE data.
552  std::vector< int > corruptedTwoLineElementDataPositions_;
553 
554  // Multimap of corrupted TLE errors.
555  std::multimap< int, std::string > corruptedTwoLineElementDataErrors_;
556 
557  // Loop is started over the entire structure of objects,
558  // multiple checks are performed per TLE, if a check is not passed boolean
559  // is set to true for that object, and amount of corrupted TLEs integer is
560  // incremented.
561  for ( unsigned int i = 0; i < numberOfObjects_; i++ )
562  {
563  // Fill the vector with the line strings from the data container.
564  if ( numberOfLinesPerTwoLineElementDatum_ == 3 )
565  {
566  twoLineElementString_.at( 1 ) = containerOfDataFromFile_[ i + 1 + ( i * 2 ) + 1 ];
567  twoLineElementString_.at( 2 ) = containerOfDataFromFile_[ i + 1 + ( i * 2 ) + 2 ];
568  }
569 
570  else if ( numberOfLinesPerTwoLineElementDatum_ == 2 )
571  {
572  twoLineElementString_.at( 1 ) = containerOfDataFromFile_[ 2 * i + 1 ];
573  twoLineElementString_.at( 2 ) = containerOfDataFromFile_[ 2 * i + 1 + 1 ];
574  }
575 
576  // Set boolean to false.
577  isObjectErroneous = false;
578 
579  // If a line-1 line number is not 1, error message is given and boolean is set to true.
580  if ( twoLineElementData_.at( i ).lineNumberLine1 != 1 )
581  {
582  corruptedTwoLineElementDataErrors_.insert(
583  std::pair< int, std::string >( i, "Incorrect line-1 leading integer." ) );
584 
585  isObjectErroneous = true;
586  }
587 
588  // If a line-2 line number is not 2, error message is given and boolean is set to true.
589  if ( twoLineElementData_.at( i ).lineNumberLine2 != 2 )
590  {
591  corruptedTwoLineElementDataErrors_.insert(
592  std::pair< int, std::string >( i, "Incorrect line-2 leading integer." ) );
593 
594  isObjectErroneous = true;
595  }
596 
597  if ( twoLineElementData_.at( i ).tleClassification != 'U'
598  && twoLineElementData_.at( i ).tleClassification != 'C' )
599  {
600  corruptedTwoLineElementDataErrors_.insert(
601  std::pair< int, std::string >( i, "Invalid TLE classification." ) );
602 
603  isObjectErroneous = true;
604  }
605 
606  // Check if orbital model is 0, else error message is given and boolean is set to true.
607  if ( twoLineElementData_.at( i ).orbitalModel != 0 )
608  {
609  corruptedTwoLineElementDataErrors_.insert(
610  std::pair< int, std::string >( i, "Incorrect orbital model." ) );
611 
612  isObjectErroneous = true;
613  }
614 
615  // Line-1 modulo-10 checker, add all integers of line 1 together,
616  // and minus signs count as 1, rest is not counted.
617 
618  // Create calculated modulo-10 checksum of line 1 and temporary integer for addition.
619  unsigned int line1Modulo10Sum_ = 0;
620 
621  // Run loop over the 69-character long, line-2 string, but not the last character,
622  // since this is the modulo-10 checksum itself and is not added.
623  for ( unsigned int j = 0; j < 68; j++ )
624  {
625  // Check if the char is not a space or a dot, or a plus or a minus sign or any of the
626  // alphabetical positions, because these can not be added.
627  if ( twoLineElementString_.at( 1 )[ j ] != ' ' &&
628  twoLineElementString_.at( 1 )[ j ] != '.' &&
629  twoLineElementString_.at( 1 )[ j ] != '+' &&
630  twoLineElementString_.at( 1 )[ j ] != '-' &&
631  j != 7 && j != 14 && j != 15 && j != 16 )
632  {
633  // Add int to checksum.
634  line1Modulo10Sum_ += std::stoi(
635  twoLineElementString_.at( 1 ).substr( j , 1 ) );
636  }
637 
638  // Check is the char is a minus sign
639  else if ( twoLineElementString_.at( 1 )[ j ] == '-' )
640  {
641  // Add 1 to checksum.
642  line1Modulo10Sum_++;
643  }
644  }
645 
646  // Take modulo 10 of checksum.
647  line1Modulo10Sum_ %= 10;
648 
649  // Comparison of calculated modulo-10 checksum with modulo-10 checksum from TLE for line 1,
650  // if they do not compare, error message is given and boolean is set to true.
651  if ( line1Modulo10Sum_ != twoLineElementData_.at( i ).modulo10CheckSumLine1 )
652  {
653  corruptedTwoLineElementDataErrors_.insert(
654  std::pair< int, std::string >( i, "Incorrect line-1 modulo-10 checksum." ) );
655 
656  isObjectErroneous = true;
657  }
658 
659  // Line-2 modulo-10 checker, add all integers of line 2 together.
660  unsigned int line2Modulo10Sum_ = 0;
661 
662  // Run loop over the 69-character long, line-2 string, but not the last character,
663  // since this is the modulo-10 checksum itself and is not added.
664  for ( unsigned int k = 0; k < 68; k++ )
665  {
666  // Check if the char is not a space or a dot, because these can not be added.
667  if ( twoLineElementString_.at( 2 )[ k ] != ' ' &&
668  twoLineElementString_.at( 2 )[ k ] != '.' )
669  {
670  // Add int to checksum.
671  line2Modulo10Sum_ += std::stoi(
672  twoLineElementString_[ 2 ].substr( k , 1 ) );
673  }
674  }
675 
676  // Take modulo 10 of checksum.
677  line2Modulo10Sum_ %= 10;
678 
679  // Comparison of calculated modulo-10 checksum with modulo-10 checksum from TLE for line 2,
680  // if they do not compare error message is given and boolean is set to true.
681  if ( line2Modulo10Sum_ != twoLineElementData_.at( i ).modulo10CheckSumLine2 )
682  {
683  corruptedTwoLineElementDataErrors_.insert(
684  std::pair< int, std::string >( i, "Incorrect line-2 modulo-10 checksum." ) );
685 
686  isObjectErroneous = true;
687  }
688 
689  // When the object's line-1 and line-2 identification numbers do not match an
690  // error message is given and boolean is set to true.
691  if ( twoLineElementData_.at( i ).objectIdentificationNumber !=
692  twoLineElementData_.at( i ).objectIdentificationNumberLine2 )
693  {
694  corruptedTwoLineElementDataErrors_.insert(
695  std::pair< int, std::string >(
696  i, "Line-1 and line-2 object idenfitication number mismatch." ) );
697 
698  isObjectErroneous = true;
699  }
700 
701  // If object is erroneous, add to list.
702  if ( isObjectErroneous )
703  {
704  corruptedTwoLineElementDataPositions_.push_back( i );
705  }
706  }
707 
708  // Erase all the corrupted TLEs from the TLE data vector.
709 
710  for ( unsigned int q = 0; q < corruptedTwoLineElementDataPositions_.size( ); q++ )
711  {
712  twoLineElementData_.erase( twoLineElementData_.begin( ) + (
713  corruptedTwoLineElementDataPositions_.at( q ) - q ) );
714  numberOfObjects_--;
715  }
716 
717  // Return the amount of corrupted TLEs
718  return corruptedTwoLineElementDataErrors_;
719 }
