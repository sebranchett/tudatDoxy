98 void AtmosphericFlightConditions::setAerodynamicCoefficientsIndependentVariableFunction(
99  const AerodynamicCoefficientsIndependentVariables independentVariable,
100  const std::function< double( ) > coefficientDependency )
101 {
102  if( ( independentVariable == mach_number_dependent ) ||
103  ( independentVariable == angle_of_attack_dependent ) ||
104  ( independentVariable == angle_of_sideslip_dependent )||
105  ( independentVariable == altitude_dependent ) )
106  {
107  throw std::runtime_error(
108  std::string( "Error when setting aerodynamic coefficient function dependency, value of parameter " ) +
109  std::to_string( independentVariable ) +
110  std::string(", will not be used." ) );
111  }
112  else
113  {
114  customCoefficientDependencies_[ independentVariable ] = coefficientDependency;
115  }

119 void AtmosphericFlightConditions::updateConditions( const double currentTime )
120 {
121  if( !( currentTime == currentTime_ ) )
122  {
123  currentTime_ = currentTime;
124 
125  // Update aerodynamic angles (but not angles w.r.t. body-fixed frame).
126  if( aerodynamicAngleCalculator_!= nullptr )
127  {
128  aerodynamicAngleCalculator_->update( currentTime, false );
129  }
130 
131  // Calculate state of vehicle in global frame and corotating frame.
132  currentBodyCenteredAirspeedBasedBodyFixedState_ = bodyCenteredPseudoBodyFixedStateFunction_( );
133 
134  updateAerodynamicCoefficientInput( );
135 
136  // Update angles from aerodynamic to body-fixed frame (if relevant).
137  if( aerodynamicAngleCalculator_!= nullptr )
138  {
139  aerodynamicAngleCalculator_->update( currentTime, true );
140  updateAerodynamicCoefficientInput( );
141  }
142 
143  // Update aerodynamic coefficients.
144  aerodynamicCoefficientInterface_->updateFullCurrentCoefficients(
145  aerodynamicCoefficientIndependentVariables_, controlSurfaceAerodynamicCoefficientIndependentVariables_,
146  currentTime_ );
147  }

151 double AtmosphericFlightConditions::getAerodynamicCoefficientIndependentVariable(
152  const AerodynamicCoefficientsIndependentVariables independentVariableType,
153  const std::string& secondaryIdentifier )
154 {
155  double currentIndependentVariable;
156  switch( independentVariableType )
157  {
158  //Calculate Mach number if needed.
159  case mach_number_dependent:
160  currentIndependentVariable = getCurrentMachNumber( );
161  break;
162  //Get angle of attack if needed.
163  case angle_of_attack_dependent:
164  if( aerodynamicAngleCalculator_== nullptr )
165  {
166  throw std::runtime_error( "Error, aerodynamic angle calculator is nullptr, but require angle of attack" );
167  }
168  currentIndependentVariable = aerodynamicAngleCalculator_->getAerodynamicAngle(
169  reference_frames::angle_of_attack );
170  break;
171  //Get angle of sideslip if needed.
172  case angle_of_sideslip_dependent:
173  if( aerodynamicAngleCalculator_== nullptr )
174  {
175  throw std::runtime_error( "Error, aerodynamic angle calculator is nullptr, but require angle of sideslip" );
176  }
177  currentIndependentVariable = aerodynamicAngleCalculator_->getAerodynamicAngle(
178  reference_frames::angle_of_sideslip );
179  break;
180  case altitude_dependent:
181  currentIndependentVariable = getCurrentAltitude( );
182  break;
183  case control_surface_deflection_dependent:
184  {
185  try
186  {
187  currentIndependentVariable = controlSurfaceDeflectionFunction_( secondaryIdentifier );
188  }
189  catch( std::runtime_error )
190  {
191  throw std::runtime_error( "Error, control surface " + secondaryIdentifier + "not recognized when updating coefficients" );
192  }
193  break;
194  }
195  default:
196  if( customCoefficientDependencies_.count( independentVariableType ) == 0 )
197  {
198  throw std::runtime_error( "Error, did not recognize aerodynamic coefficient dependency "
199  + std::to_string( independentVariableType ) );
200  }
201  else
202  {
203  currentIndependentVariable = customCoefficientDependencies_.at( independentVariableType )( );
204  }
205  }
206 
207  return currentIndependentVariable;
208 }

211 void AtmosphericFlightConditions::updateAerodynamicCoefficientInput( )
212 {
213  aerodynamicCoefficientIndependentVariables_.clear( );
214  // Calculate independent variables for aerodynamic coefficients.
215  for( unsigned int i = 0; i < aerodynamicCoefficientInterface_->getNumberOfIndependentVariables( ); i++ )
216  {
217  aerodynamicCoefficientIndependentVariables_.push_back(
218  getAerodynamicCoefficientIndependentVariable(
219  aerodynamicCoefficientInterface_->getIndependentVariableName( i ) ) );
220  }
221 
222  controlSurfaceAerodynamicCoefficientIndependentVariables_.clear( );
223  for( unsigned int i = 0; i < aerodynamicCoefficientInterface_->getNumberOfControlSurfaces( ); i++ )
224  {
225  std::string currentControlSurface = aerodynamicCoefficientInterface_->getControlSurfaceName( i );
226  for( unsigned int j = 0; j < aerodynamicCoefficientInterface_->getNumberOfControlSurfaceIndependentVariables( currentControlSurface ); j++ )
227  {
228  controlSurfaceAerodynamicCoefficientIndependentVariables_[ currentControlSurface ].push_back(
229  getAerodynamicCoefficientIndependentVariable(
230  aerodynamicCoefficientInterface_->getControlSurfaceIndependentVariableName(
231  currentControlSurface, j ), currentControlSurface ) );
232  }
233  }
234 }

237 std::shared_ptr< TrimOrientationCalculator > setTrimmedConditions(
238  const std::shared_ptr< AtmosphericFlightConditions > flightConditions )
239 {
240  // Create trim object.
241  std::shared_ptr< TrimOrientationCalculator > trimOrientation =
242  std::make_shared< TrimOrientationCalculator >(
243  flightConditions->getAerodynamicCoefficientInterface( ) );
244 
245  // Create angle-of-attack function from trim object.
246  std::function< std::vector< double >( ) > untrimmedIndependentVariablesFunction =
247  std::bind( &AtmosphericFlightConditions::getAerodynamicCoefficientIndependentVariables,
248  flightConditions );
249  std::function< std::map< std::string, std::vector< double > >( ) > untrimmedControlSurfaceIndependentVariablesFunction =
250  std::bind( &AtmosphericFlightConditions::getControlSurfaceAerodynamicCoefficientIndependentVariables,
251  flightConditions );
252  flightConditions->getAerodynamicAngleCalculator( )->setOrientationAngleFunctions(
253  std::bind( &TrimOrientationCalculator::findTrimAngleOfAttackFromFunction, trimOrientation,
254  untrimmedIndependentVariablesFunction, untrimmedControlSurfaceIndependentVariablesFunction ) );
255 
256  return trimOrientation;
257 }
