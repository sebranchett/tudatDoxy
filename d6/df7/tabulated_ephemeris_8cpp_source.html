233 std::pair< double, double > getTabulatedEphemerisSafeInterval( const std::shared_ptr< Ephemeris > ephemeris )
234 {
235  // Initialize return pair
236  std::pair< double, double > safeInterval = std::make_pair( TUDAT_NAN, TUDAT_NAN );
237 
238  // Check input consistency
239  if( !isTabulatedEphemeris( ephemeris ) )
240  {
241  throw std::runtime_error( "Error wgen getting tabulated ephemeris safe interval, input is not a tabulated ephemeris" );
242  }
243  // Identify type of tabulated ephemeris, and call associated safe interval function
244  else if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, double > >( ephemeris ) != nullptr )
245  {
246  safeInterval = std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, double > >(
247  ephemeris )->getSafeInterpolationInterval( );
248  }
249  else if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, double > >( ephemeris ) != nullptr )
250  {
251  safeInterval = std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, double > >(
252  ephemeris )->getSafeInterpolationInterval( );
253  }
254  else if ( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, Time > >( ephemeris ) != nullptr )
255  {
256  safeInterval = std::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, Time > >(
257  ephemeris )->getSafeInterpolationInterval( );
258  }
259  else if( std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, Time > >( ephemeris ) != nullptr )
260  {
261  safeInterval = std::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, Time > >(
262  ephemeris )->getSafeInterpolationInterval( );
263  }
264  return safeInterval;
265 }
