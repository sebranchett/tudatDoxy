23 double computeEquilibiumWallTemperature( const std::function< double( const double ) > heatTransferFunction,
24  const double wallEmmisivity,
25  const double adiabaticWallTemperature )
26 {
27  // Create the object that contains the function who's root needs to be found.
28  std::shared_ptr< EquilibriumTemperatureFunction > equilibriumTemperatureFunction
29  = std::make_shared< EquilibriumTemperatureFunction >(
30  heatTransferFunction, wallEmmisivity, adiabaticWallTemperature );
31 
32  // Compute wall temperature, first try secant method, use bisection as backup if secany unsuccesfull.
33  double wallTemperature = TUDAT_NAN;
34  try
35  {
36  root_finders::SecantRootFinder::TerminationFunction terminationConditionFunction =
37  std::bind( &root_finders::termination_conditions::RootRelativeToleranceTerminationCondition< double >::
38  checkTerminationCondition,
39  std::make_shared< root_finders::termination_conditions::
40  RootRelativeToleranceTerminationCondition< double > >(
41  ), std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4, std::placeholders::_5 );
42  root_finders::SecantRootFinder secant( terminationConditionFunction );
43  wallTemperature = secant.execute(
44  equilibriumTemperatureFunction, equilibriumTemperatureFunction->getInitialGuess( ) );
45  }
46  catch ( std::runtime_error )
47  {
48  try
49  {
50  root_finders::Bisection::TerminationFunction terminationConditionFunction =
51  std::bind( &root_finders::termination_conditions::RootRelativeToleranceTerminationCondition< double >::
52  checkTerminationCondition,
53  std::make_shared< root_finders::termination_conditions::
54  RootRelativeToleranceTerminationCondition< double > >(
55  ), std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4, std::placeholders::_5 );
56  root_finders::Bisection bisection( terminationConditionFunction );
57  wallTemperature = bisection.execute(
58  equilibriumTemperatureFunction, equilibriumTemperatureFunction->getInitialGuess( ) );
59  }
60  catch ( std::runtime_error )
61  {
62  throw std::runtime_error( "Error, could not find equilibrium wall temperature" );
63  }
64 
65  }
66 
67  return wallTemperature;
68 }
