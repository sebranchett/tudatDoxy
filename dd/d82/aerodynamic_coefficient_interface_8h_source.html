148  void updateCurrentControlSurfaceCoefficientsCoefficients(
149  const std::string& currentControlSurface,
150  std::vector< double > controlSurfaceIndependentVariables )
151  {
152  if( controlSurfaceIncrementInterfaces_.count( currentControlSurface ) == 0 )
153  {
154  throw std::runtime_error( "Error when updating coefficients, could not fid control surface " + currentControlSurface );
155  }
156  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->updateCurrentCoefficients(
157  controlSurfaceIndependentVariables );
158  currentForceCoefficients_ +=
159  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentForceCoefficients( );
160  currentMomentCoefficients_ +=
161  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentMomentCoefficients( );
162  }

173  void updateFullCurrentCoefficients(
174  const std::vector< double >& independentVariables,
175  const std::map< std::string, std::vector< double > >& controlSurfaceIndependentVariables =
176  std::map< std::string, std::vector< double > > ( ),
177  const double currentTime = TUDAT_NAN )
178  {
179  updateCurrentCoefficients( independentVariables, currentTime );
180 
181  for( std::map< std::string, std::vector< double > >::const_iterator controlSurfaceIterator =
182  controlSurfaceIndependentVariables.begin( ); controlSurfaceIterator != controlSurfaceIndependentVariables.end( );
183  controlSurfaceIterator++ )
184  {
185  updateCurrentControlSurfaceCoefficientsCoefficients(
