327  void updateMembers( const double currentTime = TUDAT_NAN )
328  {
329  if( !( this->currentTime_ == currentTime ) )
330  {
331 
332  cosineHarmonicCoefficients = getCosineHarmonicsCoefficients( );
333  sineHarmonicCoefficients = getSineHarmonicsCoefficients( );
334 
335  rotationToIntegrationFrame_ = rotationFromBodyFixedToIntegrationFrameFunction_( );
336  this->updateBaseMembers( );
337 
338  currentInertialRelativePosition_ =
339  this->positionOfBodySubjectToAcceleration - this->positionOfBodyExertingAcceleration ;
340 
341  currentRelativePosition_ = rotationToIntegrationFrame_.inverse( ) * (
342  currentInertialRelativePosition_ );
343 
344  currentAcceleration_ =
345  computeGeodesyNormalizedGravitationalAccelerationSum(
346  currentRelativePosition_,
347  gravitationalParameter,
348  equatorialRadius,
349  cosineHarmonicCoefficients,
350  sineHarmonicCoefficients, sphericalHarmonicsCache_,
351  accelerationPerTerm_,
352  saveSphericalHarmonicTermsSeparately_,
353  rotationToIntegrationFrame_.toRotationMatrix( ) );
354  currentAccelerationInBodyFixedFrame_ = rotationToIntegrationFrame_.inverse( ) * currentAcceleration_;
