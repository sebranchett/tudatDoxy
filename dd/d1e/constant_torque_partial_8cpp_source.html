22 ConstantTorquePartial::getParameterPartialFunction( std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
23 {
24  using namespace estimatable_parameters;
25 
26  std::pair< std::function< void( Eigen::MatrixXd& ) >, int > partialFunction = std::make_pair(
27  std::function< void( Eigen::MatrixXd& ) >( ), 0 );
28 
29  if( parameter->getParameterName( ).second.first == bodyUndergoingTorque_ )

132  Eigen::MatrixXd& momentOfInertiaPartial )
