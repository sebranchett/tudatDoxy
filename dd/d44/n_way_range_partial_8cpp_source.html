21 void NWayRangeScaling::update( const std::vector< Eigen::Vector6d >& linkEndStates,
22  const std::vector< double >& times,
23  const observation_models::LinkEndType fixedLinkEnd,
24  const Eigen::VectorXd currentObservation )
25 {
26  Eigen::Vector3d currentRangeVector;
27  int currentIndex;
28 
29  // Define lists of link end states and tines to be used for eah one-way link.
30  std::vector< Eigen::Vector6d > singleLinkEndStates;
31  singleLinkEndStates.resize( 2 );
32  std::vector< double > singleLinkTimes;
33  singleLinkTimes.resize( 2 );
34 
35  // Find index in link ends for fixed (reference) link ends
36  int fixedLinkEndIndex = observation_models::getNWayLinkIndexFromLinkEndType(
37  fixedLinkEnd, numberOfLinkEnds_ );
38 
39  // Iterate over all constituent one-way range scalings
40  observation_models::LinkEndType referenceLinkEnd;
41  for( constituentRangeScalingIterator_ = constituentRangeScalings_.begin( );
42  constituentRangeScalingIterator_ != constituentRangeScalings_.end( ); constituentRangeScalingIterator_++ )
43  {
44  // Set times, states and reference link end for current one-way range
45  currentIndex = constituentRangeScalingIterator_->first;
46  singleLinkEndStates[ 0 ] = linkEndStates.at( 2 * currentIndex );
47  singleLinkEndStates[ 1 ] = linkEndStates.at( 2 * currentIndex + 1 );
48  singleLinkTimes[ 0 ] = times.at( 2 * currentIndex );
49  singleLinkTimes[ 1 ] = times.at( 2 * currentIndex + 1 );
50 
51  if( fixedLinkEndIndex <= constituentRangeScalingIterator_->first )
52  {
53  referenceLinkEnd = observation_models::transmitter;
54  }
55  else
56  {
57  referenceLinkEnd = observation_models::receiver;
58  }
59 
60  // Update current one-way range scaling
61  constituentRangeScalingIterator_->second->update(
62  singleLinkEndStates, singleLinkTimes, referenceLinkEnd );
63  }
64 
65  // Compute scaling factors by which one-way range partias are to be multiplied before being included in the n-way range
66  // partial
67  for( unsigned int i = 0; i < linkEndStates.size( ) - 1; i += 2 )
68  {
69  currentRangeVector = ( linkEndStates[ i + 1 ] - linkEndStates[ i ] ).segment( 0, 3 ).normalized( );
70  projectedRelativeVelocityRatios_[ i / 2 ] =
71  ( 1.0 - currentRangeVector.dot( ( linkEndStates[ i ] ).segment( 3, 3 ) ) /
72  physical_constants::SPEED_OF_LIGHT ) /
73  ( 1.0 - currentRangeVector.dot( ( linkEndStates[ i + 1 ] ).segment( 3, 3 ) ) /
74  physical_constants::SPEED_OF_LIGHT );
75  }
76 }

101 
102  // Set link end times and states for current one-way range
103  subLinkStates[ 0 ] = states[ 2 * rangePartialIterator_->first ];
104  subLinkStates[ 1 ] = states[ 2 * rangePartialIterator_->first + 1 ];
105  subLinkTimes[ 0 ] = times[ 2 * rangePartialIterator_->first ];
106  subLinkTimes[ 1 ] = times[ 2 * rangePartialIterator_->first + 1 ];
107 
108  // Compute value by which one-way range should be scaled for inclusion into n-way range
109  currentPartialMultiplier = 1.0;
110  if( rangePartialIterator_->first >= referenceStartLinkEndIndex )
111  {
112  subLinkReference = observation_models::transmitter;

117  }
118  }
119  else

121  subLinkReference = observation_models::receiver;
122  for( int i = rangePartialIterator_->first; i > 0; i-- )
123  {
124  currentPartialMultiplier += 1.0 / nWayRangeScaler_->getProjectedRelativeVelocityRatio( i - 1 ) - 1.0;
125  }
126  }
127 
128  // Compute one-way range partials
129  currentPartialSet = rangePartialIterator_->second->calculatePartial( subLinkStates, subLinkTimes, subLinkReference );
130 
131  // Scale partials by required amount and add to return map.
132  for( unsigned int i = 0; i < currentPartialSet.size( ); i++ )
133  {
134  currentPartialSet[ i ].first *= currentPartialMultiplier;
135  }
136  completePartialSet.insert( completePartialSet.end( ), currentPartialSet.begin( ), currentPartialSet.end( ) );
137  }
