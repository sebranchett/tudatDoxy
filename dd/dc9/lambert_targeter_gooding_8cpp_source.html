122 double LambertTargeterGooding::getSemiMajorAxis( )
123 {
124  // Compute specific orbital energy: eps = v^2/ - mu/r.
125  const double specificOrbitalEnergy = cartesianVelocityAtDeparture.squaredNorm( ) / 2.0
126  - gravitationalParameter / cartesianPositionAtDeparture.norm( );
127 
128  // Compute semi-major axis: a = -mu / 2*eps.
129  return -gravitationalParameter / ( 2.0 * specificOrbitalEnergy );
130 }
