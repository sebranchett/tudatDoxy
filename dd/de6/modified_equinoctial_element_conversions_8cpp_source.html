53 Eigen::Vector6d convertKeplerianToModifiedEquinoctialElements(
54  const Eigen::Vector6d& keplerianElements,
55  const bool avoidSingularityAtPiInclination )
56 // Based on Hintz, 2008.
57 {
58  using mathematical_constants::PI;
59 
60  // Declaring eventual output vector.
61  Eigen::Vector6d modifiedEquinoctialState( 6 );
62 
63  // Compute semi-latus rectum.
64  double singularityTolerance = 1.0e-15; // Based on tolerance chosen in
65  // orbitalElementConversions.cpp in Tudat Core.
66 
67  // Extracting eccentricity for ease of referencing.
68  double eccentricity = keplerianElements( eccentricityIndex );
69 
70  // If e is (very near) one, then semi-major axis is undefined and thus the first kepler
71  // element is the semi-latus rectum.
72  if ( std::fabs( eccentricity - 1.0 ) < singularityTolerance )
73  {
74  modifiedEquinoctialState( semiLatusRectumIndex )
75  = keplerianElements( semiLatusRectumIndex );
76  }
77  else // eccentricity is significantly away from singularity and can be computed with p=a(1-e).
78  {
79  modifiedEquinoctialState( semiLatusRectumIndex ) = keplerianElements( semiMajorAxisIndex )
80  * ( 1.0 - eccentricity * eccentricity );
81  }
82 
83  // Determine prograde-ness, as other five parameters depend on that fact.
84  double inclination = keplerianElements( inclinationIndex );
85 
86  // If inclination is outside range [0,PI].
87  if ( ( inclination < 0.0 ) || ( inclination > PI ) )
88  {
89  // Define the error message.
90  std::stringstream errorMessage;
91  errorMessage << "Inclination is expected in range [0," << PI << "]\n"
92  << "Specified inclination: " << inclination << " rad." << std::endl;
93 
94  // Throw exception.
95  throw std::runtime_error( errorMessage.str( ) );
96  }
97  //Else, nothing wrong and continue.
98 
99  // Compute set dependant helper parameters.
100  double argumentOfPeriapsisAndAscendingNode = 0.0;
101  double tangentOfHalfInclination = 0.0;
102 
103  // If normal set of equations is to be used (i.e. not inverse set for singularity).
104  if ( !avoidSingularityAtPiInclination )
105  {
106  // Add (+) argument of periapsis and longitude of ascending node.
107  argumentOfPeriapsisAndAscendingNode = keplerianElements( argumentOfPeriapsisIndex )
108  + keplerianElements( longitudeOfAscendingNodeIndex );
109 
110  // Take the tangent of half the inclination.
111  tangentOfHalfInclination = std::tan( inclination / 2.0 );
112  }
113  else // the orbit is retrograde.
114  {
115  // Subtract (-) longitude of ascending node from argument of periapsis.
116  argumentOfPeriapsisAndAscendingNode = keplerianElements( argumentOfPeriapsisIndex )
117  - keplerianElements( longitudeOfAscendingNodeIndex );
118 
119  // Take the inverse of the tangent of half the inclination to avoid singularity at tan
120  // PI/2.
121  tangentOfHalfInclination = 1.0 / std::tan( inclination / 2.0 );
122  }
123 
124  // Compute f-element.
125  modifiedEquinoctialState( fElementIndex ) = eccentricity
126  * std::cos( argumentOfPeriapsisAndAscendingNode );
127 
128  // Compute g-element.
129  modifiedEquinoctialState( gElementIndex ) = eccentricity
130  * std::sin( argumentOfPeriapsisAndAscendingNode );
131 
132  // Compute h-element.
133  modifiedEquinoctialState( hElementIndex ) = tangentOfHalfInclination
134  * std::cos( keplerianElements( longitudeOfAscendingNodeIndex ) );
135 
136  // Compute k-element.
137  modifiedEquinoctialState( kElementIndex ) = tangentOfHalfInclination
138  * std::sin( keplerianElements( longitudeOfAscendingNodeIndex ) );
139 
140  // Compute true longitude (modulo 2 PI to keep within interval -2PI to 2PI).
141  modifiedEquinoctialState( trueLongitudeIndex )
142  = basic_mathematics::computeModulo(
143  argumentOfPeriapsisAndAscendingNode
144  + keplerianElements( trueAnomalyIndex ), 2.0 * PI );
145 
146  // Give back result.
147  return modifiedEquinoctialState;
148 }

151 Eigen::Vector6d convertModifiedEquinoctialToKeplerianElements(
152  const Eigen::Vector6d& modifiedEquinoctialElements,
153  const bool avoidSingularityAtPiInclination )
154 // Using unknown source pdf, code archive E. Heeren and personal derivation based on Hintz 2008.
155 {
156  using mathematical_constants::PI;
157 
158  // Declaration of output vector.
159  Eigen::Vector6d convertedKeplerianElements = Eigen::Vector6d::Zero( 6 );
160 
161  // for ease of referencing, almost all modified equinoctial elements.
162  double fElement = modifiedEquinoctialElements( fElementIndex );
163  double gElement = modifiedEquinoctialElements( gElementIndex );
164  double hElement = modifiedEquinoctialElements( hElementIndex );
165  double kElement = modifiedEquinoctialElements( kElementIndex );
166 
167  // Tolerance for singularities.
168  double singularityTolerance = 1.0e-15;
169 
170  // Compute eccentricity.
171  double eccentricity = std::sqrt( fElement * fElement + gElement * gElement );
172  convertedKeplerianElements( eccentricityIndex ) = eccentricity;
173 
174  // Compute semi-major axis.
175  // If eccentricity is not near-parabolic.
176  if ( std::fabs( eccentricity - 1.0 ) > singularityTolerance )
177  {
178  // Use semi-latus rectum and eccentricity to calculate semi-major axis with a=p/(1-e^2).
179  convertedKeplerianElements( semiMajorAxisIndex ) =
180  modifiedEquinoctialElements( semiLatusRectumIndex ) /
181  ( 1.0 - eccentricity * eccentricity );
182  }
183  // This is (almost) a parabola and the semimajor axis will tend to infinity and be useless.
184  else
185  {
186  // Give semi-latus rectum instead.
187  convertedKeplerianElements( semiLatusRectumIndex ) =
188  modifiedEquinoctialElements( semiLatusRectumIndex );
189  }
190 
191  // Compute longitude of ascending node.
192 
193  // Compute solution in [-PI,PI] interval with atan2.
194  double longitudeOfAscendingNode = std::atan2( kElement, hElement );
195 
196  // Store longitude of ascending node.
197  convertedKeplerianElements( longitudeOfAscendingNodeIndex ) = basic_mathematics::
198  computeModulo( longitudeOfAscendingNode, 2.0 * PI );
199 
200  // Compute inclination.
201 
202  // If prograde factor must be positive, otherwise negative.
203  double retrogradeFactor = avoidSingularityAtPiInclination ? -1.0 : 1.0;
204 
205  // Calculate magnitude of inclination with retrogradeFactor (derived based on Hintz, 2008).
206  double hSquaredPlusKSquared = std::pow( hElement * hElement + kElement * kElement ,
207  retrogradeFactor );
208 
209  convertedKeplerianElements( inclinationIndex ) =
210  2.0 * std::atan( std::sqrt( hSquaredPlusKSquared ) );
211  // Was: std::atan2(2.0 * std::sqrt(hSquaredPlusKSquared) , (1.0 - hSquaredPlusKSquared));
212 
213  // Compute argument of periapsis.
214 
215  // Helper quantity (omega + I * OMEGA ), for argument of periapsis and true anomaly.
216  double argumentOfPeriapsisAndLongitude;
217 
218  // If eccentricity is (near) circular
219  if ( eccentricity < singularityTolerance )
220  // Then the composite argument of periapsis and longitude together cannot be determined
221  // because both arguments of the atan2 functions will be zero.
222  {
223  // Set argument of periapsis to zero.
224  convertedKeplerianElements( argumentOfPeriapsisIndex ) = 0.0;
225 
226  // Set composite argument to longitude only, since argument of periapsis is now zero.
227  argumentOfPeriapsisAndLongitude = retrogradeFactor * longitudeOfAscendingNode;
228  }
229  else
230  // Argument of periapsis can be found from composite argument.
231  {
232  // Compute composite argument.
233  argumentOfPeriapsisAndLongitude = std::atan2( gElement, fElement );
234 
235  // Compute argument of periapsis.
236  convertedKeplerianElements( argumentOfPeriapsisIndex ) = basic_mathematics::
237  computeModulo( argumentOfPeriapsisAndLongitude
238  - retrogradeFactor * longitudeOfAscendingNode, 2.0 * PI );
239  }
240 
241  // Compute true anomaly.
242  convertedKeplerianElements( trueAnomalyIndex ) = basic_mathematics::
243  computeModulo( modifiedEquinoctialElements( trueLongitudeIndex )
244  - argumentOfPeriapsisAndLongitude, 2.0 * PI );
245 
246  // Return converted elements.
247  return convertedKeplerianElements;
248 }

252  const Eigen::Vector6d& cartesianElements,

254 {

284 Eigen::Vector6d convertModifiedEquinoctialToCartesianElements(
285  const Eigen::Vector6d& modifiedEquinoctialElements,
286  const double centralBodyGravitationalParameter,
287  const bool avoidSingularityAtPiInclination )
288 // Using unnamed pdf and code archive Bart Rmgens.
289 {
290  // Creating output vector.
291  Eigen::Vector6d convertedCartesianElements = Eigen::Vector6d::Zero( 6 );
292 
293  // If the prograde equations are to be used.
294  if ( !avoidSingularityAtPiInclination )
295  {
296  // Local storage of elements for ease of access.
297  double semiLatusRectum = modifiedEquinoctialElements( semiLatusRectumIndex );
298  double fElement = modifiedEquinoctialElements( fElementIndex );
299  double gElement = modifiedEquinoctialElements( gElementIndex );
300  double hElement = modifiedEquinoctialElements( hElementIndex );
301  double kElement = modifiedEquinoctialElements( kElementIndex );
302  double trueLongitude = modifiedEquinoctialElements( trueLongitudeIndex );
303 
304  // Computing helper parameters.
305 
306  // Square-root
307  double squareRootOfGravitationalParameterOverSemiLatusRectum =
308  std::sqrt( centralBodyGravitationalParameter / semiLatusRectum );
309 
310  // Cosines and sine.
311  double cosineTrueLongitude = std::cos( trueLongitude );
312  double sineTrueLongitude = std::sin( trueLongitude );
313 
314  // s, a, w, r from references.
315  double sSquaredParameter = 1.0 + hElement * hElement + kElement * kElement;
316  double aSquaredParameter = hElement * hElement - kElement*kElement;
317  double wParameter = 1.0 + fElement * cosineTrueLongitude
318  + gElement * sineTrueLongitude;
319  double radius = semiLatusRectum / wParameter;
320 
321  // Computing position and storing (using code archive Bart Rmgens and unnamed pdf).
322  convertedCartesianElements( xCartesianPositionIndex )
323  = radius / sSquaredParameter * ( cosineTrueLongitude + aSquaredParameter
324  * cosineTrueLongitude + 2.0 * hElement * kElement
325  * sineTrueLongitude );
326  convertedCartesianElements( yCartesianPositionIndex )
327  = radius / sSquaredParameter * ( sineTrueLongitude - aSquaredParameter
328  * sineTrueLongitude + 2.0 * hElement * kElement
329  * cosineTrueLongitude );
330  convertedCartesianElements( zCartesianPositionIndex ) = 2.0 * radius / sSquaredParameter
331  * ( hElement * sineTrueLongitude - kElement * cosineTrueLongitude );
332 
333  // Computing velocities (these can probably use more optimal computation).
334  convertedCartesianElements( xCartesianVelocityIndex ) = -1.0 / sSquaredParameter
335  * squareRootOfGravitationalParameterOverSemiLatusRectum
336  * ( sineTrueLongitude + aSquaredParameter * sineTrueLongitude
337  - 2.0 * hElement * kElement * cosineTrueLongitude + gElement
338  - 2.0 * fElement * hElement * kElement + aSquaredParameter * gElement );
339  convertedCartesianElements( yCartesianVelocityIndex ) = -1.0 / sSquaredParameter
340  * squareRootOfGravitationalParameterOverSemiLatusRectum
341  * ( -cosineTrueLongitude + aSquaredParameter * cosineTrueLongitude
342  + 2.0 * hElement * kElement * sineTrueLongitude - fElement
343  + 2.0 * gElement * hElement * kElement + aSquaredParameter * fElement );
344  convertedCartesianElements( zCartesianVelocityIndex ) = 2.0 / sSquaredParameter
345  * squareRootOfGravitationalParameterOverSemiLatusRectum
346  * ( hElement * cosineTrueLongitude + kElement * sineTrueLongitude
347  + fElement * hElement + gElement * kElement );
348  }
349  else // use the indirect transformation via an intermediate Kepler state.
350  {
351  // Use the Keplerian state as an intermediary step, as there is no direct transformation
352  // available in literature (personal derivation forgone for now due to time constraints).
353  convertedCartesianElements =
354  convertKeplerianToCartesianElements(
355  convertModifiedEquinoctialToKeplerianElements(
356  modifiedEquinoctialElements, avoidSingularityAtPiInclination ),
357  centralBodyGravitationalParameter );
358  }
359 
360  // Return converted set of elements.
361  return convertedCartesianElements;
362 }
