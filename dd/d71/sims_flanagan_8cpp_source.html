25 std::vector< double > convertToSimsFlanaganThrustModel( std::function< Eigen::Vector3d( const double ) > thrustModelWrtTime,
26  const double maximumThrust,
27  const double timeOfFlight, const int numberSegmentsForwardPropagation,
28  const int numberSegmentsBackwardPropagation )
29 {
30  double segmentDurationForwardPropagation = timeOfFlight / ( 2.0 * numberSegmentsForwardPropagation );
31  double segmentDurationBackwardPropagation = timeOfFlight / ( 2.0 * numberSegmentsBackwardPropagation );
32 
33  std::vector< double > simsFlanaganThrustModel;
34  for ( int i = 0 ; i < numberSegmentsForwardPropagation ; i++ )
35  {
36  Eigen::Vector3d currentThrustVector = thrustModelWrtTime( segmentDurationForwardPropagation * ( i + 1.0 / 2.0 ) ) / maximumThrust;
37  for ( int j = 0 ; j < 3 ; j++ )
38  {
39  simsFlanaganThrustModel.push_back( currentThrustVector[ j ] );
40  }
41  }
42  for ( int i = 0 ; i < numberSegmentsBackwardPropagation ; i++ )
43  {
44  Eigen::Vector3d currentThrustVector = thrustModelWrtTime( timeOfFlight / 2.0
45  + segmentDurationBackwardPropagation * ( i + 1.0 / 2.0 ) ) / maximumThrust;
46  for ( int j = 0 ; j < 3 ; j++ )
47  {
48  simsFlanaganThrustModel.push_back( currentThrustVector[ j ] );
49  }
50  }
51 
52  return simsFlanaganThrustModel;
53 }

56 std::function< Eigen::Vector3d( const double ) > getInitialGuessFunctionFromShaping(
57  std::shared_ptr< shape_based_methods::ShapeBasedMethod > shapeBasedLeg,
58  const int numberSegmentsSimsFlanagan,
59  const double timeOfFlight,
60  std::function< double( const double ) > specificImpulseFunction,
61  std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings )
62 {
63  // Calculate number of segments for both the forward propagation (from departure to match point)
64  // and the backward propagation (from arrival to match point).
65  int numberSegmentsForwardPropagation = ( numberSegmentsSimsFlanagan + 1 ) / 2;
66  int numberSegmentsBackwardPropagation = numberSegmentsSimsFlanagan / 2;
67  int segmentDurationForwardPropagation = timeOfFlight / ( 2.0 * numberSegmentsForwardPropagation );
68  int segmentDurationBackwardPropagation = timeOfFlight / ( 2.0 * numberSegmentsBackwardPropagation );
69 
70  // Compute time at each node of the Sims-Flanagan trajectory.
71  std::vector< double > timesAtNodes;
72  for ( int i = 0 ; i < numberSegmentsForwardPropagation ; i++)
73  {
74  timesAtNodes.push_back( i * segmentDurationForwardPropagation );
75  }
76  for ( int i = 0 ; i <= numberSegmentsBackwardPropagation ; i++ )
77  {
78  timesAtNodes.push_back( timeOfFlight / 2.0 + i * segmentDurationBackwardPropagation );
79  }
80 
81  // Retrieve thrust profile from shaping method.
82  std::map< double, Eigen::VectorXd > thrustProfileShapedTrajectory;
83  shapeBasedLeg->getThrustForceProfile( timesAtNodes, thrustProfileShapedTrajectory, specificImpulseFunction, integratorSettings );
84 
85  // Compute averaged thrust vector per segment.
86  std::vector< Eigen::Vector3d > thrustVectorPerSegment;
87  for ( int i = 0 ; i < numberSegmentsSimsFlanagan ; i++ )
88  {
89  thrustVectorPerSegment.push_back( ( thrustProfileShapedTrajectory[ timesAtNodes[ i ] ]
90  + thrustProfileShapedTrajectory[ timesAtNodes[ i + 1 ] ] ) / 2.0 );
91  }
92 
93  // Define function returning Sims-Flanagan thrust vector as a function of time.
94  std::function< Eigen::Vector3d( const double ) > initialGuessThrustFromShaping = [ = ]( const double currentTime )
95  {
96 
97  // Compute index of the current segment, dependent on currentTime.
98  int indexSegment;
99 
100  if ( currentTime <= timeOfFlight / 2.0 )
101  {
102  indexSegment = currentTime / segmentDurationForwardPropagation;
103  }
104  else if ( currentTime == timeOfFlight )
105  {
106  indexSegment = numberSegmentsSimsFlanagan - 1;
107  }
108  else
109  {
110  indexSegment = numberSegmentsForwardPropagation + ( currentTime - ( timeOfFlight / 2.0 ) ) / segmentDurationBackwardPropagation;
111  }
112 
113  // Retrieve the constant thrust vector corresponding to this segment.
114  Eigen::Vector3d thrustVectorFromInitialGuess = thrustVectorPerSegment[ indexSegment ];
115  return thrustVectorFromInitialGuess;
116 
117  };
118 
119  return initialGuessThrustFromShaping;
120 }

124 std::pair< std::vector< double >, std::vector< double > > SimsFlanagan::performOptimisation( )
125 {
126  //Set seed for reproducible results
127  pagmo::random_device::set_seed( 456 );
128 
129  // Create object to compute the problem fitness
130  problem prob{ SimsFlanaganProblem(
131  stateAtDeparture_, stateAtArrival_, centralBodyGravitationalParameter_, initialMass_, maximumThrust_,
132  specificImpulseFunction_, numberSegments_, timeOfFlight_, initialGuessThrustModel_,
133  optimisationSettings_->relativeToleranceConstraints_ ) };
134 
135 
136  std::vector< double > constraintsTolerance;
137  for ( unsigned int i = 0 ; i < ( prob.get_nec() + prob.get_nic() ) ; i++ )
138  {
139  constraintsTolerance.push_back( 1.0e-3 );
140  }
141  prob.set_c_tol( constraintsTolerance );
142 
143 
144  algorithm algo{ optimisationSettings_->optimisationAlgorithm_ };
145 
146  unsigned long long populationSize = optimisationSettings_->numberOfIndividualsPerPopulation_;
147  island islUnconstrained{ algo, prob, populationSize };
148 
149  // Evolve for a given number of generations.
150  for( int i = 0 ; i < optimisationSettings_->numberOfGenerations_ ; i++ )
151  {
152  islUnconstrained.evolve( );
153  while( islUnconstrained.status( ) != pagmo::evolve_status::idle &&
154  islUnconstrained.status( ) != pagmo::evolve_status::idle_error )
155  {
156  islUnconstrained.wait( );
157  }
158  islUnconstrained.wait_check( ); // Raises errors
159  }
160 
161  std::vector< double > championFitness = islUnconstrained.get_population().champion_f();
162  std::vector< double > championDesignVariables = islUnconstrained.get_population().champion_x();
163 
164  std::pair< std::vector< double >, std::vector< double > > output;
165  output.first = championFitness;
166  output.second = championDesignVariables;
167 
168  championFitness_ = championFitness;
169  championDesignVariables_ = championDesignVariables;
170 
171  return output;
172 
173 }

187 double SimsFlanagan::computeCurrentThrustAccelerationMagnitude(
188  double currentTime, std::function< double ( const double ) > specificImpulseFunction,
189  std::shared_ptr<numerical_integrators::IntegratorSettings< double > > integratorSettings )
190 {
191  std::map< double, Eigen::VectorXd > massProfile;
192  getMassProfile( std::vector< double >( { currentTime } ), massProfile, specificImpulseFunction, integratorSettings );

194

213 void SimsFlanagan::getThrustAccelerationProfile(
214  std::vector< double >& epochsVector,
215  std::map< double, Eigen::VectorXd >& thrustAccelerationProfile,
216  std::function< double ( const double ) > specificImpulseFunction,
217  std::shared_ptr<numerical_integrators::IntegratorSettings< double > > integratorSettings )
218 {
219  thrustAccelerationProfile.clear();
220 
221  std::map< double, Eigen::VectorXd > thrustProfile;
222  getThrustForceProfile( epochsVector, thrustProfile, specificImpulseFunction, integratorSettings );
223 
224  std::map< double, Eigen::VectorXd > massProfile;
225  getMassProfile( epochsVector, massProfile, specificImpulseFunction, integratorSettings );
226 
227  for ( unsigned int i = 0 ; i < epochsVector.size() ; i++ )
228  {
229  if ( ( i > 0 ) && ( epochsVector[ i ] < epochsVector[ i - 1 ] ) )
230  {
231  throw std::runtime_error( "Error when retrieving the thrust profile of a low-thrust trajectory, "
232  "epochs are not provided in increasing order." );
233  }
234 
235  thrustAccelerationProfile[ epochsVector[ i ] ] =
236  thrustProfile[ epochsVector[ i ] ] / massProfile[ epochsVector[ i ] ][ 0 ];
237 
238  }

271 std::pair< std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > >,
272 std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > >
273 SimsFlanagan::createLowThrustTranslationalStatePropagatorSettings(
274  const std::string& bodyToPropagate,
275  const std::string& centralBody,
276  const basic_astrodynamics::AccelerationMap& accelerationModelMap,
277  const std::shared_ptr< propagators::DependentVariableSaveSettings > dependentVariablesToSave )
278 {
279 
280  // Create termination conditions settings.
281  std::pair< std::shared_ptr< propagators::PropagationTerminationSettings >,
282  std::shared_ptr< propagators::PropagationTerminationSettings > > terminationConditions;
283 
284  terminationConditions.first = std::make_shared< propagators::PropagationTimeTerminationSettings >( 0.0, true );
285  terminationConditions.second = std::make_shared< propagators::PropagationTimeTerminationSettings >( timeOfFlight_, true );
286 
287 
288  // Compute state at half of the time of flight after forward propagation.
289  Eigen::Vector6d stateHalfOfTimeOfFlightForwardPropagation = simsFlanaganModel_->propagateTrajectoryForward(
290  0.0, timeOfFlight_ / 2.0, stateAtDeparture_, timeOfFlight_ / ( 2.0 * numberSegmentsForwardPropagation_ ) );
291 
292  // Compute state at half of the time of flight after backward propagation.
293  Eigen::Vector6d stateHalfOfTimeOfFlightBackwardPropagation = simsFlanaganModel_->propagateTrajectoryBackward(
294  timeOfFlight_, timeOfFlight_ / 2.0, stateAtArrival_, timeOfFlight_ / ( 2.0 * numberSegmentsBackwardPropagation_ ) );
295 
296 
297  // Compute state at half of the time of flight (averaged between forward and backward propagation results).
298  Eigen::Vector6d stateHalfOfTimeOfFlight = 1.0 / 2.0 * ( stateHalfOfTimeOfFlightForwardPropagation + stateHalfOfTimeOfFlightBackwardPropagation );
299 
300  // Define translational state propagator settings.
301  std::pair< std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > >,
302  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > > translationalStatePropagatorSettings;
303 
304  // Define backward translational state propagation settings.
305  translationalStatePropagatorSettings.first = std::make_shared< propagators::TranslationalStatePropagatorSettings< double > >
306  ( std::vector< std::string >{ centralBody }, accelerationModelMap,
307  std::vector< std::string >{ bodyToPropagate }, stateHalfOfTimeOfFlightForwardPropagation,
308  terminationConditions.first,
309  propagators::cowell, dependentVariablesToSave );
310 
311  // Define forward translational state propagation settings.
312  translationalStatePropagatorSettings.second = std::make_shared< propagators::TranslationalStatePropagatorSettings< double > >
313  ( std::vector< std::string >{ centralBody }, accelerationModelMap,
314  std::vector< std::string >{ bodyToPropagate }, stateHalfOfTimeOfFlightBackwardPropagation,
315  terminationConditions.second,
316  propagators::cowell, dependentVariablesToSave );
317 
318  return translationalStatePropagatorSettings;
319 }
