40 inline bool checkMaximumIterationsExceeded( const unsigned int numberOfIterations,
41  const unsigned int maximumNumberOfIterations,
42  const bool throwRunTimeException = true )
43 {
44  // Check if maximum number of iterations have been exceeded.
45  const bool isMaximumIterationsExceeded = numberOfIterations > maximumNumberOfIterations;
46 
47  // If exceeded, and flag is set to throw run-time exception, proceed.
48  if ( isMaximumIterationsExceeded && throwRunTimeException )
49  {
50  std::string errorMessage
51  = "Root-finder did not converge within maximum number of iterations!";
52 
53  throw std::runtime_error( errorMessage );
54  }
55 
56  // Else, simply return whether maximum number of iterations have been exceeded.
57  return isMaximumIterationsExceeded;
58 }

71  const ScalarType previousRootGuess,

87 inline bool checkRootRelativeTolerance( const ScalarType currentRootGuess,
88  const ScalarType previousRootGuess,
89  const ScalarType relativeTolerance )
90 {
