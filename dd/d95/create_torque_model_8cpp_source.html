36 std::shared_ptr< aerodynamics::AerodynamicTorque > createAerodynamicTorqueModel(
37  const std::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
38  const std::shared_ptr< simulation_setup::Body > bodyExertingTorque,
39  const std::string& nameOfBodyUndergoingTorque,
40  const std::string& nameOfBodyExertingTorque )
41 {
42  // Check existence of required environment models
43  if( bodyUndergoingTorque->getAerodynamicCoefficientInterface( ) == nullptr )
44  {
45  throw std::runtime_error( "Error when making aerodynamic torque, body " +
46  nameOfBodyUndergoingTorque +
47  "has no aerodynamic coefficients." );
48  }
49 
50  if( bodyExertingTorque->getAtmosphereModel( ) == nullptr )
51  {
52  throw std::runtime_error( "Error when making aerodynamic torque, central body " +
53  nameOfBodyExertingTorque + " has no atmosphere model.");
54  }
55 
56  if( bodyExertingTorque->getShapeModel( ) == nullptr )
57  {
58  throw std::runtime_error( "Error when making aerodynamic torque, central body " +
59  nameOfBodyExertingTorque + " has no shape model." );
60  }
61 
62  // Retrieve flight conditions; create object if not yet extant.
63  std::shared_ptr< aerodynamics::AtmosphericFlightConditions > bodyFlightConditions =
64  std::dynamic_pointer_cast< aerodynamics::AtmosphericFlightConditions >(
65  bodyUndergoingTorque->getFlightConditions( ) );
66 
67  if( bodyFlightConditions == nullptr && bodyUndergoingTorque->getFlightConditions( ) == nullptr )
68  {
69  bodyFlightConditions = createAtmosphericFlightConditions( bodyUndergoingTorque,
70  bodyExertingTorque,
71  nameOfBodyUndergoingTorque,
72  nameOfBodyExertingTorque ) ;
73  bodyUndergoingTorque->setFlightConditions(
74  bodyFlightConditions );
75  }
76  else if( bodyFlightConditions == nullptr && bodyUndergoingTorque->getFlightConditions( ) != nullptr )
77  {
78  throw std::runtime_error( "Error when making aerodynamic torque, found flight conditions that are not atmospheric." );
79  }
80 
81  // Retrieve frame in which aerodynamic coefficients are defined.
82  std::shared_ptr< aerodynamics::AerodynamicCoefficientInterface > aerodynamicCoefficients =
83  bodyUndergoingTorque->getAerodynamicCoefficientInterface( );
84  reference_frames::AerodynamicsReferenceFrames torqueFrame;
85  if( aerodynamicCoefficients->getAreCoefficientsInAerodynamicFrame( ) )
86  {
87  torqueFrame = reference_frames::aerodynamic_frame;
88  }
89  else
90  {
91  torqueFrame = reference_frames::body_frame;
92  }
93 
94  // Create function to transform from frame of aerodynamic coefficienrs to that of propagation.
95  std::function< Eigen::Vector3d( const Eigen::Vector3d& ) > toPropagationFrameTransformation;
96  toPropagationFrameTransformation =
97  reference_frames::getAerodynamicForceTransformationFunction(
98  bodyFlightConditions->getAerodynamicAngleCalculator( ),
99  torqueFrame,
100  std::bind( &Body::getCurrentRotationToGlobalFrame, bodyExertingTorque ),
101  reference_frames::body_frame );
102 
103 
104  std::function< Eigen::Vector3d( ) > coefficientFunction =
105  std::bind( &aerodynamics::AerodynamicCoefficientInterface::getCurrentMomentCoefficients,
106  aerodynamicCoefficients );
107  std::function< Eigen::Vector3d( ) > coefficientInPropagationFrameFunction =
108  std::bind( &reference_frames::transformVectorFunctionFromVectorFunctions,
109  coefficientFunction, toPropagationFrameTransformation );
110 
111  // Create torque model.
112  return std::make_shared< aerodynamics::AerodynamicTorque >(
113  coefficientInPropagationFrameFunction,
114  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentDensity, bodyFlightConditions ),
115  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentAirspeed, bodyFlightConditions ),
116  std::bind( &aerodynamics::AerodynamicCoefficientInterface::getReferenceArea, aerodynamicCoefficients ),
117  std::bind( &aerodynamics::AerodynamicCoefficientInterface::getReferenceLengths, aerodynamicCoefficients ),
118  aerodynamicCoefficients->getAreCoefficientsInNegativeAxisDirection( ) );
119 }

122 std::shared_ptr< gravitation::SecondDegreeGravitationalTorqueModel > createSecondDegreeGravitationalTorqueModel(
123  const std::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
124  const std::shared_ptr< simulation_setup::Body > bodyExertingTorque,
125  const std::string& nameOfBodyUndergoingTorque,
126  const std::string& nameOfBodyExertingTorque )
127 {
128  // Retrieve state functions
129  std::function< Eigen::Vector3d( ) > positionOfBodySubjectToTorqueFunction =
130  std::bind( &simulation_setup::Body::getPosition, bodyUndergoingTorque );
131  std::function< Eigen::Vector3d( ) > positionOfBodyExertingTorqueFunction =
132  std::bind( &simulation_setup::Body::getPosition, bodyExertingTorque );
133 
134  // Check model availability
135  std::shared_ptr< gravitation::GravityFieldModel > gravityFieldModel = bodyExertingTorque->getGravityFieldModel( );
136  std::function< double( ) > gravitationalParameterOfAttractingBodyFunction;
137  if( gravityFieldModel == nullptr )
138  {
139  throw std::runtime_error( "Error when making second degree gravitational torque, " + nameOfBodyExertingTorque +
140  " does not possess a gravity field" );
141  }
142  else
143  {
144  gravitationalParameterOfAttractingBodyFunction = std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
145  gravityFieldModel );
146  }
147 
148  // Retrieve environment parameters
149  std::function< Eigen::Matrix3d( ) > inertiaTensorOfRotatingBodyFunction =
150  std::bind( &simulation_setup::Body::getBodyInertiaTensor, bodyUndergoingTorque );
151  std::function< Eigen::Quaterniond( ) > rotationToBodyFixedFrameFunction =
152  std::bind( &simulation_setup::Body::getCurrentRotationToLocalFrame, bodyUndergoingTorque );
153 
154  return std::make_shared<gravitation::SecondDegreeGravitationalTorqueModel >(
155  positionOfBodySubjectToTorqueFunction, gravitationalParameterOfAttractingBodyFunction,
156  inertiaTensorOfRotatingBodyFunction, positionOfBodyExertingTorqueFunction, rotationToBodyFixedFrameFunction );

194 std::shared_ptr< basic_astrodynamics::TorqueModel > createTorqueModel(
195  const std::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
196  const std::shared_ptr< simulation_setup::Body > bodyExertingTorque,
197  const std::shared_ptr< TorqueSettings > torqueSettings,
198  const std::string& nameOfBodyUndergoingTorque,
199  const std::string& nameOfBodyExertingTorque )
200 {
201  std::shared_ptr< basic_astrodynamics::TorqueModel > torqueModel;
202 
203  switch( torqueSettings->torqueType_ )
204  {
205  case basic_astrodynamics::inertial_torque:
206  {
207  torqueModel = createInertialTorqueModel(
208  bodyUndergoingTorque, nameOfBodyUndergoingTorque );
209  break;
210  }
211  case basic_astrodynamics::second_order_gravitational_torque:
212  {
213  torqueModel = createSecondDegreeGravitationalTorqueModel(
214  bodyUndergoingTorque, bodyExertingTorque, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
215  break;
216  }
217  case basic_astrodynamics::aerodynamic_torque:
218  {
219  torqueModel = createAerodynamicTorqueModel(
220  bodyUndergoingTorque, bodyExertingTorque, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
221  break;
222  }
223  case basic_astrodynamics::spherical_harmonic_gravitational_torque:
224  {
225  torqueModel = createSphericalHarmonicGravitationalTorqueModel(
226  bodyUndergoingTorque, bodyExertingTorque, torqueSettings, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
227  break;
228  }
229  default:
230  throw std::runtime_error(
231  "Error, did not recognize type " + std::to_string( torqueSettings->torqueType_ ) +
232  " when making torque model" );
233  }
234 
235  return torqueModel;
236 }

240 basic_astrodynamics::TorqueModelMap createTorqueModelsMap(
241  const NamedBodyMap& bodyMap,
242  SelectedTorqueMap selectedTorquePerBody,
243  const std::vector< std::string >& propagatedBodies )
244 {
245  for( unsigned int i = 0; i < propagatedBodies.size( ); i++ )
246  {
247  if( selectedTorquePerBody.count( propagatedBodies.at( i ) ) == 0 )
248  {
249  selectedTorquePerBody[ propagatedBodies.at( i ) ] =
250  std::map< std::string, std::vector< std::shared_ptr< TorqueSettings > > >( );
251  }
252  }
253 
254  basic_astrodynamics::TorqueModelMap torqueModelMap;
255 
256  for( SelectedTorqueMap::const_iterator acceleratedBodyIterator = selectedTorquePerBody.begin( );
257  acceleratedBodyIterator != selectedTorquePerBody.end( ); acceleratedBodyIterator++ )
258  {
259  if( bodyMap.count( acceleratedBodyIterator->first ) == 0 )
260  {
261  throw std::runtime_error(
262  "Error, could not find body " + acceleratedBodyIterator->first + " when making torque model map." );
263  }
264  else
265  {
266  torqueModelMap[ acceleratedBodyIterator->first ][ acceleratedBodyIterator->first ].push_back(
267  createTorqueModel(
268  bodyMap.at( acceleratedBodyIterator->first ), bodyMap.at( acceleratedBodyIterator->first ),
269  std::make_shared< TorqueSettings >( basic_astrodynamics::inertial_torque ),
270  acceleratedBodyIterator->first, acceleratedBodyIterator->first ) );
271 
272  for( std::map< std::string, std::vector< std::shared_ptr< TorqueSettings > > >::const_iterator
273  acceleratingBodyIterator = acceleratedBodyIterator->second.begin( );
274  acceleratingBodyIterator != acceleratedBodyIterator->second.end( ); acceleratingBodyIterator++ )
275  {
276  if( bodyMap.count( acceleratingBodyIterator->first ) == 0 )
277  {
278  throw std::runtime_error(
279  "Error, could not find body " + acceleratingBodyIterator->first + " when making torque model map." );
280  }
281 
282  for( unsigned int i = 0; i < acceleratingBodyIterator->second.size( ); i++ )
283  {
284  torqueModelMap[ acceleratedBodyIterator->first ][ acceleratingBodyIterator->first ].push_back(
285  createTorqueModel(
286  bodyMap.at( acceleratedBodyIterator->first ), bodyMap.at( acceleratingBodyIterator->first ),
287  acceleratingBodyIterator->second.at( i ),
288  acceleratedBodyIterator->first, acceleratingBodyIterator->first ) );
289  }
290  }
291  }
292  }
293 
294  return torqueModelMap;
295 }
