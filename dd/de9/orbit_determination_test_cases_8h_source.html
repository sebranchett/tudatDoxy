261  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialParameterEstimate =

271  }
272 
273  // Define estimation input
274  std::shared_ptr< PodInput< StateScalarType, TimeType > > podInput =
275  std::make_shared< PodInput< StateScalarType, TimeType > >(
276  observationsAndTimes, initialParameterEstimate.rows( ), inverseAPrioriCovariance,
277  initialParameterEstimate - truthParameters );
278  if( observableType == 4 )
279  {
280  std::map< observation_models::ObservableType, double > weightPerObservable;
281  weightPerObservable[ one_way_range ] = 1.0 / ( 1.0 * 1.0 );
282  weightPerObservable[ angular_position ] = 1.0 / ( 1.0E-9 * 1.0E-9 );
283  weightPerObservable[ one_way_doppler ] = 1.0 / ( 1.0E-12 * 1.0E-12 );
284 
285  podInput->setConstantPerObservableWeightsMatrix( weightPerObservable );
286  }
287  else
288  {
289  podInput->setConstantWeightsMatrix( weight );
290  }
291  podInput->defineEstimationSettings( true, true, false, false, false );
292 
293  // Perform estimation
294  std::shared_ptr< PodOutput< StateScalarType, TimeType > > podOutput = orbitDeterminationManager.estimateParameters(
295  podInput, std::make_shared< EstimationConvergenceChecker >( ) );
296 
297  return std::make_pair( podOutput,
298  ( podOutput->parameterEstimate_.template cast< double >( ) -
299  truthParameters .template cast< double >( ) ) );
300 }
