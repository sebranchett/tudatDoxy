31 std::shared_ptr< aerodynamics::WindModel > createWindModel(
32  const std::shared_ptr< WindModelSettings > windSettings,
33  const std::string& body )
34 {
35  std::shared_ptr< aerodynamics::WindModel > windModel;
36 
37  // Check wind model type and create requested model
38  switch( windSettings->getWindModelType( ) )
39  {
40  case custom_wind_model:
41  {
42  // Check input consistency
43  std::shared_ptr< CustomWindModelSettings > customWindModelSettings =
44  std::dynamic_pointer_cast< CustomWindModelSettings >( windSettings );
45 
46  if( customWindModelSettings == nullptr )
47  {
48  throw std::runtime_error( "Error when making custom wind model for body " + body + ", input is incompatible" );
49  }
50  windModel = std::make_shared< aerodynamics::CustomWindModel >(
51  customWindModelSettings->getWindFunction( ) );
52  break;
53  }
54  default:
55  throw std::runtime_error( "Error when making wind model for body " + body + ", input type not recognized" );
56  }
57 
58  return windModel;
59

63 std::shared_ptr< aerodynamics::AtmosphereModel > createAtmosphereModel(
64  const std::shared_ptr< AtmosphereSettings > atmosphereSettings,
65  const std::string& body )
66 {
67  using namespace tudat::aerodynamics;
68 
69  // Declare return object.
70  std::shared_ptr< AtmosphereModel > atmosphereModel;
71 
72  // Check which type of atmosphere model is to be created.
73  switch( atmosphereSettings->getAtmosphereType( ) )
74  {
75  case exponential_atmosphere:
76  {
77  // Check whether settings for atmosphere are consistent with its type.
78  std::shared_ptr< ExponentialAtmosphereSettings > exponentialAtmosphereSettings =
79  std::dynamic_pointer_cast< ExponentialAtmosphereSettings >( atmosphereSettings );
80 
81  if( exponentialAtmosphereSettings == nullptr )
82  {
83  throw std::runtime_error( "Error, expected exponential atmosphere settings for body " + body );
84  }
85  else
86  {
87  // Create and initialize exponential atmosphere model.
88  std::shared_ptr< ExponentialAtmosphere > exponentialAtmosphereModel;
89  if ( exponentialAtmosphereSettings->getBodyName( ) == undefined_body )
90  {
91  exponentialAtmosphereModel = std::make_shared< ExponentialAtmosphere >(
92  exponentialAtmosphereSettings->getDensityScaleHeight( ) ,
93  exponentialAtmosphereSettings->getConstantTemperature( ),
94  exponentialAtmosphereSettings->getDensityAtZeroAltitude( ),
95  exponentialAtmosphereSettings->getSpecificGasConstant( ),
96  exponentialAtmosphereSettings->getRatioOfSpecificHeats( ) );
97  }
98  else
99  {
100  exponentialAtmosphereModel = std::make_shared< ExponentialAtmosphere >(
101  exponentialAtmosphereSettings->getBodyName( ) );
102  }
103  atmosphereModel = exponentialAtmosphereModel;
104  }
105  break;
106  }
107  case custom_constant_temperature_atmosphere:
108  {
109  // Check whether settings for atmosphere are consistent with its type.
110  std::shared_ptr< CustomConstantTemperatureAtmosphereSettings > customConstantTemperatureAtmosphereSettings =
111  std::dynamic_pointer_cast< CustomConstantTemperatureAtmosphereSettings >( atmosphereSettings );
112  if( customConstantTemperatureAtmosphereSettings == nullptr )
113  {
114  throw std::runtime_error( "Error, expected exponential atmosphere settings for body " + body );
115  }
116  else
117  {

119  std::shared_ptr< CustomConstantTemperatureAtmosphere > customConstantTemperatureAtmosphereModel;
120  if ( customConstantTemperatureAtmosphereSettings->getModelSpecificParameters( ).empty( ) )
121  {
122  customConstantTemperatureAtmosphereModel = std::make_shared< CustomConstantTemperatureAtmosphere >(
123  customConstantTemperatureAtmosphereSettings->getDensityFunction( ) ,
124  customConstantTemperatureAtmosphereSettings->getConstantTemperature( ),
125  customConstantTemperatureAtmosphereSettings->getSpecificGasConstant( ),
126  customConstantTemperatureAtmosphereSettings->getRatioOfSpecificHeats( ) );
127  }
128  else
129  {
130  customConstantTemperatureAtmosphereModel = std::make_shared< CustomConstantTemperatureAtmosphere >(
131  customConstantTemperatureAtmosphereSettings->getDensityFunctionType( ),
132  customConstantTemperatureAtmosphereSettings->getConstantTemperature( ),
133  customConstantTemperatureAtmosphereSettings->getSpecificGasConstant( ),
134  customConstantTemperatureAtmosphereSettings->getRatioOfSpecificHeats( ),
135  customConstantTemperatureAtmosphereSettings->getModelSpecificParameters( ) );
136  }
137  atmosphereModel = customConstantTemperatureAtmosphereModel;
138  }
139  break;
140  }
141  case tabulated_atmosphere:
142  {
143  // Check whether settings for atmosphere are consistent with its type
144  std::shared_ptr< TabulatedAtmosphereSettings > tabulatedAtmosphereSettings =
145  std::dynamic_pointer_cast< TabulatedAtmosphereSettings >( atmosphereSettings );
146  if( tabulatedAtmosphereSettings == nullptr )
147  {
148  throw std::runtime_error( "Error, expected tabulated atmosphere settings for body " + body );
149  }
150  else
151  {
152  // Create and initialize tabulated atmosphere model.
153  atmosphereModel = std::make_shared< TabulatedAtmosphere >(
154  tabulatedAtmosphereSettings->getAtmosphereFile( ),
155  tabulatedAtmosphereSettings->getIndependentVariables( ),
156  tabulatedAtmosphereSettings->getDependentVariables( ),
157  tabulatedAtmosphereSettings->getSpecificGasConstant( ),
158  tabulatedAtmosphereSettings->getRatioOfSpecificHeats( ),
159  tabulatedAtmosphereSettings->getBoundaryHandling( ),
160  tabulatedAtmosphereSettings->getDefaultExtrapolationValue( ) );
161  }
162  break;
163  }
164 #if USE_NRLMSISE00
165  case nrlmsise00:
166  {
167  std::string spaceWeatherFilePath;
168  std::shared_ptr< NRLMSISE00AtmosphereSettings > nrlmsise00AtmosphereSettings =
169  std::dynamic_pointer_cast< NRLMSISE00AtmosphereSettings >( atmosphereSettings );
170

172  {
173  // Use default space weather file stored in tudatBundle.
174  spaceWeatherFilePath = input_output::getSpaceWeatherDataPath( ) + "sw19571001.txt";
175  }
176  else

179  spaceWeatherFilePath = nrlmsise00AtmosphereSettings->getSpaceWeatherFile( );

181 
182  tudat::input_output::solar_activity::SolarActivityDataMap solarActivityData =

184 
185  // Create atmosphere model using NRLMISE00 input function
186  std::function< tudat::aerodynamics::NRLMSISE00Input( double, double, double, double ) > inputFunction =
187  std::bind( &tudat::aerodynamics::nrlmsiseInputFunction,
188  std::placeholders::_1, std::placeholders::_2, std::placeholders::_3, std::placeholders::_4,
189  solarActivityData, false, TUDAT_NAN );
190  atmosphereModel = std::make_shared< aerodynamics::NRLMSISE00Atmosphere >( inputFunction );
191  break;
192  }
193 #endif
194  default:
195  throw std::runtime_error( "Error, did not recognize atmosphere model settings type " +
196  std::to_string( atmosphereSettings->getAtmosphereType( ) ) );
197  }
198 
199  if( atmosphereSettings->getWindSettings( ) != nullptr )
200  {
201  atmosphereModel->setWindModel( createWindModel( atmosphereSettings->getWindSettings( ), body ) );
202  }
203 
204  return atmosphereModel;
205 }
