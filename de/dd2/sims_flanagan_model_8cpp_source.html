22 int SimsFlanaganModel::convertTimeToLegSegment( double currentTime )
23 {
24  int indexSegment;
25 
26  double segmentDurationForwardPropagation = timeAtMatchPoint_ / numberSegmentsForwardPropagation_;
27  double segmentDurationBackwardPropagation = timeAtMatchPoint_ / numberSegmentsBackwardPropagation_;
28 
29  if ( currentTime <= timeAtMatchPoint_ )
30  {
31  indexSegment = currentTime / segmentDurationForwardPropagation;
32  }
33  else if ( currentTime == timeOfFlight_ )
34  {
35  indexSegment = numberSegments_ - 1;
36  }
37  else
38  {
39  indexSegment = numberSegmentsForwardPropagation_ + ( currentTime - timeAtMatchPoint_ ) / segmentDurationBackwardPropagation;
40  }
41 
42  return indexSegment;
43 }

45 std::shared_ptr< simulation_setup::ThrustAccelerationSettings > SimsFlanaganModel::getConstantThrustAccelerationSettingsPerSegment(
46  unsigned int indexSegment )
47 {
48  // Define (constant) thrust magnitude function.
49  std::function< double( const double ) > thrustMagnitudeFunction = [ = ]( const double currentTime )
50  {
51  return maximumThrust_ * throttles_[ indexSegment ].norm( );
52  };
53 
54  // Define thrust magnitude settings from thrust magnitude function.
55  std::shared_ptr< simulation_setup::FromFunctionThrustMagnitudeSettings > thrustMagnitudeSettings
56  = std::make_shared< simulation_setup::FromFunctionThrustMagnitudeSettings >(
57  thrustMagnitudeFunction, specificImpulseFunction_ );
58 
59  // Define thrust direction function (constant over one leg segment).
60  std::function< Eigen::Vector3d( const double ) > thrustDirectionFunction = [ = ]( const double currentTime )
61  {
62  return throttles_[ indexSegment ].normalized( );
63  };
64 
65 
66  // Define thrust direction settings from the direction of thrust acceleration retrieved from the shaping method.
67  std::shared_ptr< simulation_setup::CustomThrustDirectionSettings > thrustDirectionSettings =
68  std::make_shared< simulation_setup::CustomThrustDirectionSettings >( thrustDirectionFunction );
69 
70  // Define thrust acceleration settings.
71  std::shared_ptr< simulation_setup::ThrustAccelerationSettings > thrustAccelerationSettings =
72  std::make_shared< simulation_setup::ThrustAccelerationSettings >(
73  thrustDirectionSettings, thrustMagnitudeSettings );
74 
75  return thrustAccelerationSettings;
76 }

169 double SimsFlanaganModel::getTotalDeltaV( )
170 {
171  totalDeltaV_ = 0.0;
172 
173  for( int currentSegment = 0 ; currentSegment < numberSegments_ ; currentSegment++ )
174  {
175  double segmentDuration;
176  if ( currentSegment < numberSegmentsForwardPropagation_ )
177  {
178  segmentDuration = segmentDurationForwardPropagation_;
179  }
180  else
181  {
182  segmentDuration = segmentDurationBackwardPropagation_;
183  }
184 
185  Eigen::Vector3d deltaVvector =
186  maximumThrust_ / segmentMasses_.at( currentSegment ) * segmentDuration * throttles_[ currentSegment ];
187  totalDeltaV_ += deltaVvector.norm( );
188  }
189 
190  return totalDeltaV_;
191 }

194 void SimsFlanaganModel::propagateMassToSegments( )
195 {
196  double currentMass = initialSpacecraftMass_;
197  segmentMasses_.resize( numberSegments_ + 1 );
198  segmentMasses_[ 0 ] = currentMass;
199  for( int currentSegment = 0 ; currentSegment < numberSegments_ ; currentSegment++ )
200  {
201  // Compute time at half of the current segment.
202  double currentTime = timesAtNodes_[ currentSegment ] +
203  ( timesAtNodes_[ currentSegment + 1 ] - timesAtNodes_[ currentSegment ] ) / 2.0;
204 
205  // Compute segment duration.
206  double segmentDuration;
207  if ( currentSegment < numberSegmentsForwardPropagation_ )
208  {
209  segmentDuration = timeAtMatchPoint_ / numberSegmentsForwardPropagation_;
210  }
211  else
212  {
213  segmentDuration = timeAtMatchPoint_ / numberSegmentsBackwardPropagation_;
214  }
215 
216  // Compute current deltaV vector.
217  Eigen::Vector3d currentDeltaV;
218  for( int i = 0 ; i < 3 ; i++ )
219  {
220  currentDeltaV[ i ] = maximumThrust_ / currentMass * segmentDuration * throttles_[ currentSegment ][ i ];
221  }
222 
223  // Update mass of the spacecraft.
224  currentMass *= std::exp(
225  - currentDeltaV.norm( ) /
226  ( specificImpulseFunction_( currentTime ) * physical_constants::SEA_LEVEL_GRAVITATIONAL_ACCELERATION ) );
227  segmentMasses_[ currentSegment + 1 ] = currentMass;
228 
229  }

261 Eigen::Vector6d SimsFlanaganModel::propagateInsideForwardSegment(
262  double initialTime, double finalTime, double segmentDuration, Eigen::Vector6d initialState )
263 {
264 
265  // Compute time elapsed since start of the current leg segment.
266  double timeElapsedCurrentSegment = finalTime - initialTime;
267 
268  // Compute current segment.
269  int currentSegment = convertTimeToLegSegment( initialTime );
270 
271  // Initialise propagated state.
272  Eigen::Vector6d propagatedState = initialState;
273 
274  // Propagate from start of the current leg segment to the required propagation final time.
275  if ( ( finalTime <= ( timesAtNodes_[ currentSegment ] + segmentDuration / 2.0 ) )
276  || ( initialTime > ( timesAtNodes_[ currentSegment ] + segmentDuration / 2.0 ) ) )
277  {
278  // Directly propagate the current state over the time elapsed since the start of the current leg segment.
279  propagatedState = orbital_element_conversions::convertKeplerianToCartesianElements(
280  orbital_element_conversions::propagateKeplerOrbit(
281  orbital_element_conversions::convertCartesianToKeplerianElements(
282  propagatedState, centralBodyGravitationalParameter_ ),
283  timeElapsedCurrentSegment, centralBodyGravitationalParameter_ ), centralBodyGravitationalParameter_ );
284  }
285  else
286  {
287  // First propagate the current state to half of the current segment.
288  if ( ( timesAtNodes_[ currentSegment ] + segmentDuration / 2.0 ) - initialTime > 0.0 )
289  {
290  propagatedState = orbital_element_conversions::convertKeplerianToCartesianElements(
291  orbital_element_conversions::propagateKeplerOrbit(
292  orbital_element_conversions::convertCartesianToKeplerianElements(
293  propagatedState, centralBodyGravitationalParameter_ ),
294  ( timesAtNodes_[ currentSegment ] + segmentDuration / 2.0 ) - initialTime, centralBodyGravitationalParameter_ ),
295  centralBodyGravitationalParameter_ );
296  }
297 
298  // Compute the deltaV that needs to be applied at half of the current segment.
299  Eigen::Vector3d deltaVvector;
300  for( unsigned int i = 0 ; i < 3 ; i++ )
301  {
302  deltaVvector[ i ] = maximumThrust_ / segmentMasses_[ currentSegment ]
303  * segmentDuration * throttles_[ currentSegment ][ i ];
304  }
305 
306  // Add the deltaV vector to the state propagated until half of the current segment.
307  propagatedState.segment( 3, 3 ) += deltaVvector;
308 
309  // Propagate the updated state from half of the current segment to required propagation final time.
310  if ( finalTime - ( timesAtNodes_[ currentSegment ] + segmentDuration / 2.0 ) > 0.0 )
311  {
312  propagatedState = orbital_element_conversions::convertKeplerianToCartesianElements(
313  orbital_element_conversions::propagateKeplerOrbit(
314  orbital_element_conversions::convertCartesianToKeplerianElements(
315  propagatedState, centralBodyGravitationalParameter_ ),
316  finalTime - ( timesAtNodes_[ currentSegment ] + segmentDuration / 2.0 ), centralBodyGravitationalParameter_ ),
317  centralBodyGravitationalParameter_ );
318  }
319  }
320 
321  return propagatedState;
322 }

391 Eigen::Vector6d SimsFlanaganModel::propagateTrajectoryForward(
392  double initialTime, double finalTime, Eigen::Vector6d initialState, double segmentDuration )
393 {
394  // Compute index of leg segment which corresponds to the propagation initial time.
395  int initialSegment = convertTimeToLegSegment( initialTime );
396 
397  // Compute index of leg segment which corresponds to the propagation final time.
398  int finalSegment = convertTimeToLegSegment( finalTime );
399 
400  // Initialise propagated state.
401  Eigen::Vector6d propagatedState = initialState;
402 
403  // Initialise current segment.
404  int currentSegment = initialSegment;
405 
406  if ( initialSegment == finalSegment )
407  {
408  // Directly propagate to final time inside the current segment.
409  if ( finalTime - initialTime > 0 )
410  {
411  propagatedState = propagateInsideForwardSegment( initialTime, finalTime, segmentDuration, propagatedState );
412  }
413  }
414  else
415  {
416  // First propagate to beginning of next segment.
417  propagatedState = propagateInsideForwardSegment(
418  initialTime, timesAtNodes_[ initialSegment + 1 ], segmentDuration, propagatedState );
419 
420  // Update current segment.
421  currentSegment++;
422 
423  // Compute number of segments the trajectory should be propagated through before reaching the segment corresponding
424  // to the final propagation time.
425  int propagatedSegments = ( finalTime - timesAtNodes_[ currentSegment ] ) / segmentDuration;
426 
427  // Propagate through these segments.
428  for( int i = 0 ; i < propagatedSegments ; i++ )
429  {
430  propagatedState = propagateInsideForwardSegment(
431  timesAtNodes_[ currentSegment ], timesAtNodes_[ currentSegment + 1 ],
432  segmentDuration, propagatedState );
433  currentSegment++;
434  }
435 
436  // Propagate inside last segment.
437  if ( ( finalTime - timesAtNodes_[ currentSegment ] ) > 0.0 )
438  {
439  propagatedState = propagateInsideForwardSegment(
440  timesAtNodes_[ currentSegment ], finalTime, segmentDuration, propagatedState );
441  currentSegment++;
442  }
443 
444  }
445 
446  return propagatedState;
447 
448 }

510 std::map< double, Eigen::Vector6d > SimsFlanaganModel::propagateTrajectoryForward(
511  std::vector< double > epochs, std::map< double, Eigen::Vector6d >& propagatedTrajectory, double segmentDuration )
512 {
513  // Initialise propagated state.
514  Eigen::Vector6d propagatedState = stateAtDeparture_;
515 
516  for( unsigned int epochIndex = 0 ; epochIndex < epochs.size( ) ; epochIndex++ )
517  {
518  double currentTime = epochs[ epochIndex ];
519  if ( epochIndex > 0 )
520  {
521  if ( currentTime < epochs[ epochIndex - 1 ] )
522  {
523  throw std::runtime_error(
524  "Error when propagating trajectory in Sims-Flanagan, epochs at which the trajectory should be computed are not in increasing order." );
525  }
526  }
527  if ( ( currentTime < 0.0 ) || ( currentTime > timeOfFlight_ ) )
528  {
529  throw std::runtime_error(
530  "Error when propagating trajectory in Sims-Flanagan, epochs at which the trajectory should be computed are not constrained between 0.0 and timeOfFlight." );
531  }
532 
533 
534  if ( epochIndex == 0 )
535  {
536  if ( currentTime > 0.0 )
537  {
538  propagatedState = propagateTrajectoryForward(
539  0.0, currentTime, propagatedState, segmentDuration );
540  }
541  propagatedTrajectory[ currentTime ] = propagatedState;
542  }
543  else
544  {
545  propagatedState = propagateTrajectoryForward(
546  epochs[ epochIndex - 1 ], currentTime, propagatedState, segmentDuration );
547  propagatedTrajectory[ currentTime ] = propagatedState;
548  }
549 
550  }
551 
552  return propagatedTrajectory;
553 }

603 void SimsFlanaganModel::propagateTrajectory(
604  std::vector< double > epochs, std::map< double, Eigen::Vector6d >& propagatedTrajectory )
605 {
606 
607  // Index of the last epoch in the forward propagation leg.
608  int index = epochs.size( ) - 1;
609 
610  for( unsigned int epochIndex = 0 ; epochIndex < epochs.size( ) ; epochIndex++ )
611  {
612  double currentTime = epochs[ epochIndex ];
613  if ( ( epochIndex != 0 ) && ( epochs[ epochIndex - 1 ] > epochs[ epochIndex ] ) )
614  {
615  throw std::runtime_error( "Error when propagating trajectory in Sims-Flanagan, epochs at which the trajectory should be "
616  "computed are not in increasing order." );
617  }
618  if ( ( currentTime < 0.0 ) || ( currentTime > timeOfFlight_ ) )
619  {
620  throw std::runtime_error( "Error when propagating trajectory backward in Sims-Flanagan, epochs at which the trajectory should be "
621  "computed are not constrained between 0.0 and timeOfFlight." );
622  }
623 
624 
625  if ( ( epochIndex == 0 ) && ( currentTime >= timeOfFlight_ / 2.0 ) )
626  {
627  index = epochIndex;
628  }
629  if ( ( epochIndex != epochs.size( ) - 1 )
630  && ( ( currentTime <= timeOfFlight_ / 2.0 )
631  && ( epochs[ epochIndex + 1 ] >= timeOfFlight_ / 2.0 ) ) )
632  {
633  index = epochIndex;
634  }
635 
636  }
637 
638  // Retrieve epochs corresponding to the forward and backward propagation legs in two separate vectors.
639  std::vector< double > epochsVectorForwardPropagation;
640  std::vector< double > epochsVectorBackwardPropagation;
641  for( int epochIndex = 0 ; epochIndex <= index ; epochIndex++ )
642  {
643  epochsVectorForwardPropagation.push_back( epochs[ epochIndex ] );
644  }
645  for( int epochIndex = epochs.size( ) - 1; epochIndex > index ; epochIndex-- )
646  {
647  epochsVectorBackwardPropagation.push_back( epochs[ epochIndex ] );
648  }
649 
650  // Propagate the forward part of the trajectory.
651  propagatedTrajectory = propagateTrajectoryForward(
652  epochsVectorForwardPropagation, propagatedTrajectory, segmentDurationForwardPropagation_ );
653 
654  // Propagate the backward part of the trajectory.
655  propagatedTrajectory = propagateTrajectoryBackward(
656  epochsVectorBackwardPropagation, propagatedTrajectory, segmentDurationBackwardPropagation_ );
657 
658 }
