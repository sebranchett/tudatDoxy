217 Eigen::Vector6d computeCartesianStateFromKeplerianOrbit(
218  const Eigen::Vector6d& initialState,
219  const double finalPropagationTime,
220  const double gravitationalParameter)
221 {
222  Eigen::Vector6d keplerianInitialState = orbital_element_conversions::convertCartesianToKeplerianElements(
223  initialState, gravitationalParameter);
224  // Retrieve the semi-major axis and eccentricty of the keplerian orbit.
225  double semiMajorAxis = keplerianInitialState[orbital_element_conversions::semiMajorAxisIndex];
226  double eccentricity = keplerianInitialState[orbital_element_conversions::eccentricityIndex];
227 
228  // Calculate the initial mean anomaly.
229  double initialTrueAnomaly = keplerianInitialState[orbital_element_conversions::trueAnomalyIndex];
230  double initialMeanAnomaly;
231 
232  if( eccentricity < 1.0 )
233  {
234  initialMeanAnomaly = orbital_element_conversions::convertEccentricAnomalyToMeanAnomaly(
235  orbital_element_conversions::convertTrueAnomalyToEccentricAnomaly(initialTrueAnomaly, eccentricity), eccentricity);
236 
237  }
238  else
239  {
240  initialMeanAnomaly = orbital_element_conversions::convertHyperbolicEccentricAnomalyToMeanAnomaly(
241  orbital_element_conversions::convertTrueAnomalyToHyperbolicEccentricAnomaly(
242  initialTrueAnomaly, eccentricity), eccentricity);
243  }
244 
245  // Calculate the mean anomaly at the final time.
246  double meanAnomalyEndPropagation = initialMeanAnomaly + orbital_element_conversions::convertElapsedTimeToMeanAnomalyChange(
247  finalPropagationTime, gravitationalParameter, semiMajorAxis );
248 
249  double trueAnomalyEndPropagation;
250 
251  if( eccentricity < 1.0 )
252  {
253  trueAnomalyEndPropagation = orbital_element_conversions::convertEccentricAnomalyToTrueAnomaly(
254  orbital_element_conversions::convertMeanAnomalyToEccentricAnomaly(
255  eccentricity, meanAnomalyEndPropagation), eccentricity );
256  }
257  else
258  {
259  trueAnomalyEndPropagation = orbital_element_conversions::convertHyperbolicEccentricAnomalyToTrueAnomaly(
260  orbital_element_conversions::convertMeanAnomalyToHyperbolicEccentricAnomaly(
261  eccentricity, meanAnomalyEndPropagation), eccentricity );
262  }
263 
264  // Determine the keplerian state at final time.
265  Eigen::Vector6d finalKeplerianState = keplerianInitialState;
266  finalKeplerianState[orbital_element_conversions::trueAnomalyIndex] = trueAnomalyEndPropagation;
267 
268  // Convert keplerian to cartesian state at final time.
269  Eigen::Vector6d cartesianStateLambertSolution = orbital_element_conversions::convertKeplerianToCartesianElements(
270  finalKeplerianState, gravitationalParameter);
271 
272  return cartesianStateLambertSolution;
273 }

278 void propagateLambertTargeterAndFullProblem(
279  const double timeOfFlight,
280  const double initialTime,
281  const simulation_setup::NamedBodyMap& bodyMap,
282  const std::string& centralBody,
283  std::pair< std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > >,
284  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > > propagatorSettings,
285  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
286  std::map< double, Eigen::Vector6d >& lambertTargeterResult,
287  std::map< double, Eigen::Vector6d >& fullProblemResult,
288  std::map< double, Eigen::VectorXd >& dependentVariableResult,
289  const std::vector<std::string>& departureAndArrivalBodies,
290  const double centralBodyGravitationalParameter,
291  const Eigen::Vector3d& cartesianPositionAtDeparture,
292  const Eigen::Vector3d& cartesianPositionAtArrival )
293 {
294  // Clear output maps
295  lambertTargeterResult.clear( );
296  fullProblemResult.clear( );
297  dependentVariableResult.clear( );
298 
299  // Retrieve the gravitational parameter of the relevant bodies.
300  double gravitationalParameterCentralBody = ( centralBodyGravitationalParameter == centralBodyGravitationalParameter ) ?
301  centralBodyGravitationalParameter :
302  bodyMap.at( centralBody )->getGravityFieldModel( )->getGravitationalParameter( );
303 
304  // Get halved value of the time of flight, later used as initial time for the propagation.
305  double halvedTimeOfFlight = timeOfFlight / 2.0;
306 
307  // Time at the end of the transfer
308  double finalTime = initialTime + timeOfFlight;
309 
310  // Retrieve positions of departure and arrival bodies from ephemerides
311  Eigen::Vector3d cartesianPositionAtDepartureForLambertTargeter, cartesianPositionAtArrivalForLambertTargeter;
312  if ( cartesianPositionAtDeparture != cartesianPositionAtDeparture )
313  {
314  // Cartesian state at departure
315  if ( bodyMap.at( departureAndArrivalBodies.at( 0 ) )->getEphemeris( ) == nullptr)
316  {
317  throw std::runtime_error( "Ephemeris not defined for departure body." );
318  }
319  else
320  {
321  Eigen::Vector6d cartesianStateDepartureBody =
322  bodyMap.at( departureAndArrivalBodies.at( 0 ) )->getEphemeris( )->getCartesianState( initialTime);
323  cartesianPositionAtDepartureForLambertTargeter = cartesianStateDepartureBody.segment(0,3);
324  }
325  }
326  else
327  {
328  cartesianPositionAtDepartureForLambertTargeter = cartesianPositionAtDeparture;
329  }
330 
331  if( cartesianPositionAtArrival != cartesianPositionAtArrival )
332  {
333 
334  // Cartesian state at arrival
335  if ( bodyMap.at( departureAndArrivalBodies.at( 1 ) )->getEphemeris( ) == nullptr){
336  throw std::runtime_error( "Ephemeris not defined for arrival body." );
337  }
338  else{
339  Eigen::Vector6d cartesianStateArrivalBody =
340  bodyMap.at( departureAndArrivalBodies.at( 1 ) )->getEphemeris( )->getCartesianState(finalTime);
341  cartesianPositionAtArrivalForLambertTargeter = cartesianStateArrivalBody.segment(0,3);
342  }
343  }
344  else
345  {
346  cartesianPositionAtArrivalForLambertTargeter = cartesianPositionAtArrival;
347  }
348 
349 
350  // Run the Lambert targeter.
351  mission_segments::LambertTargeterIzzo lambertTargeter(
352  cartesianPositionAtDepartureForLambertTargeter, cartesianPositionAtArrivalForLambertTargeter,
353  timeOfFlight, gravitationalParameterCentralBody );
354 
355  // Retrieve cartesian state at departure.
356  Eigen::Vector3d cartesianVelocityAtDeparture = lambertTargeter.getInertialVelocityAtDeparture( );
357  Eigen::Vector6d cartesianStateAtDeparture;
358  cartesianStateAtDeparture.segment(0,3) = cartesianPositionAtDepartureForLambertTargeter;
359  cartesianStateAtDeparture.segment(3,3) = cartesianVelocityAtDeparture;
360 
361 
362  // Convert into keplerian elements
363  Eigen::Vector6d keplerianStateAtDeparture = orbital_element_conversions::convertCartesianToKeplerianElements(
364  cartesianStateAtDeparture, gravitationalParameterCentralBody );
365 
366  // Propagate the keplerian elements until half of the time of flight.
367  Eigen::Vector6d keplerianStateAtHalvedTimeOfFlight = orbital_element_conversions::propagateKeplerOrbit( keplerianStateAtDeparture,
368  halvedTimeOfFlight, gravitationalParameterCentralBody );
369 
370  // Convert the keplerian elements back into Cartesian elements.
371  Eigen::Vector6d initialStatePropagationCartesianElements = orbital_element_conversions::convertKeplerianToCartesianElements(
372  keplerianStateAtHalvedTimeOfFlight, gravitationalParameterCentralBody );
373 
374 
375  Eigen::Vector6d cartesianStateLambertSolution;
376 
377 
378  // Define forward propagator settings variables.
379  integratorSettings->initialTime_ = initialTime + halvedTimeOfFlight;
380 
381  // Define forward propagation settings
382  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > propagatorSettingsForwardPropagation;
383  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > propagatorSettingsBackwardPropagation;
384 
385 
386  propagatorSettingsForwardPropagation = propagatorSettings.second;
387  propagatorSettingsForwardPropagation->resetInitialStates( initialStatePropagationCartesianElements );
388 
389  propagatorSettingsBackwardPropagation = propagatorSettings.first;
390  propagatorSettingsBackwardPropagation->resetInitialStates( initialStatePropagationCartesianElements );
391 
392  // Perform forward propagation.
393  propagators::SingleArcDynamicsSimulator< > dynamicsSimulatorIntegrationForwards(
394  bodyMap, integratorSettings, propagatorSettingsForwardPropagation );
395  std::map< double, Eigen::VectorXd > stateHistoryFullProblemForwardPropagation = dynamicsSimulatorIntegrationForwards.
396  getEquationsOfMotionNumericalSolution( );
397  std::map< double, Eigen::VectorXd > dependentVariableHistoryForwardPropagation =
398  dynamicsSimulatorIntegrationForwards.getDependentVariableHistory( );
399 
400  // Calculate the difference between the full problem and the Lambert targeter solution along the forward propagation direction.
401  for( std::map< double, Eigen::VectorXd >::iterator itr = stateHistoryFullProblemForwardPropagation.begin( );
402  itr != stateHistoryFullProblemForwardPropagation.end( ); itr++ )
403  {
404  cartesianStateLambertSolution = computeCartesianStateFromKeplerianOrbit(
405  initialStatePropagationCartesianElements, itr->first - ( initialTime + halvedTimeOfFlight ),
406  gravitationalParameterCentralBody );
407  lambertTargeterResult[ itr->first ] = cartesianStateLambertSolution;
408  fullProblemResult[ itr->first ] = itr->second;
409  dependentVariableResult[ itr->first ] = dependentVariableHistoryForwardPropagation[ itr->first ];
410  }
411 
412 
413  // Define backward propagator settings variables.
414  integratorSettings->initialTimeStep_ = -integratorSettings->initialTimeStep_;
415  integratorSettings->initialTime_ = initialTime + halvedTimeOfFlight;
416 
417  // Perform the backward propagation.
418  propagators::SingleArcDynamicsSimulator< > dynamicsSimulatorIntegrationBackwards(bodyMap, integratorSettings, propagatorSettingsBackwardPropagation );
419  std::map< double, Eigen::VectorXd > stateHistoryFullProblemBackwardPropagation =
420  dynamicsSimulatorIntegrationBackwards.getEquationsOfMotionNumericalSolution( );
421  std::map< double, Eigen::VectorXd > dependentVariableHistoryBackwardPropagation =
422  dynamicsSimulatorIntegrationBackwards.getDependentVariableHistory( );
423 
424  // Calculate the difference between the full problem and the Lambert targeter solution along the backward propagation direction.
425  for( std::map< double, Eigen::VectorXd >::iterator itr = stateHistoryFullProblemBackwardPropagation.begin( );
426  itr != stateHistoryFullProblemBackwardPropagation.end( ); itr++ )
427  {
428  cartesianStateLambertSolution = computeCartesianStateFromKeplerianOrbit(
429  initialStatePropagationCartesianElements, - (initialTime + halvedTimeOfFlight) + itr->first,
430  gravitationalParameterCentralBody);
431 
432  lambertTargeterResult[ itr->first ] = cartesianStateLambertSolution;
433  fullProblemResult[ itr->first ] = itr->second;
434  dependentVariableResult[ itr->first ] = dependentVariableHistoryBackwardPropagation[ itr->first ];
435 
436  }
437 
438 // std::cout<<"In loop: "<<std::endl<<std::setprecision( 16 )<<
439 // lambertTargeterResult.begin( )->second.segment( 0, 3 ).transpose( )<<std::endl<<
440 // cartesianPositionAtDeparture.transpose( )<<std::endl<<
441 // lambertTargeterResult.rbegin( )->second.segment( 0, 3 ).transpose( )<<std::endl<<
442 // cartesianPositionAtArrival.transpose( )<<std::endl<<std::endl;
443 
444 
445 
446  integratorSettings->initialTimeStep_ = -integratorSettings->initialTimeStep_;
447 
448 }

453  const double timeOfFlight,
454  const double initialTime,
455  const simulation_setup::NamedBodyMap& bodyMap,
456  const basic_astrodynamics::AccelerationMap& accelerationModelMap,
457  const std::string& bodyToPropagate,
458  const std::string& centralBody,
459  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
460  std::map< double, Eigen::Vector6d >& lambertTargeterResult,
461  std::map< double, Eigen::Vector6d >& fullProblemResult,
462  std::map< double, Eigen::VectorXd >& dependentVariableResult,
463  const std::vector<std::string>& departureAndArrivalBodies,
464  const bool terminationSphereOfInfluence,
465  const Eigen::Vector3d& cartesianPositionAtDeparture,
466  const Eigen::Vector3d& cartesianPositionAtArrival,
467  const double departureBodyGravitationalParameter,
468  const double arrivalBodyGravitationalParameter,
469  const double centralBodyGravitationalParameter,
470  const std::shared_ptr< DependentVariableSaveSettings > dependentVariablesToSave = std::shared_ptr< DependentVariableSaveSettings >( ),
471  const TranslationalPropagatorType propagator = cowell)
472 {
473 
474  // Retrieve the gravitational parameter of the relevant bodies.
475  double gravitationalParameterCentralBody = ( centralBodyGravitationalParameter == centralBodyGravitationalParameter ) ?
476  centralBodyGravitationalParameter :
477  bodyMap.at( centralBody )->getGravityFieldModel( )->getGravitationalParameter( );
478 
479 
480  // Retrieve positions of departure and arrival bodies from ephemerides if not defined as inputs.
481  Eigen::Vector3d cartesianPositionAtDepartureForLambertTargeter, cartesianPositionAtArrivalForLambertTargeter;
482  if ( cartesianPositionAtDeparture != cartesianPositionAtDeparture )
483  {
484  // Cartesian state at departure
485  if ( bodyMap.at( departureAndArrivalBodies.at( 0 ) )->getEphemeris( ) == nullptr)
486  {
487  throw std::runtime_error( "Ephemeris not defined for departure body." );
488  }
489  else
490  {
491  Eigen::Vector6d cartesianStateDepartureBody =
492  bodyMap.at( departureAndArrivalBodies.at( 0 ) )->getEphemeris( )->getCartesianState( initialTime);
493  cartesianPositionAtDepartureForLambertTargeter = cartesianStateDepartureBody.segment(0,3);
494  }
495  }
496  else
497  {
498  cartesianPositionAtDepartureForLambertTargeter = cartesianPositionAtDeparture;
499  }
500 
501  if( cartesianPositionAtArrival != cartesianPositionAtArrival )
502  {
503 
504  // Cartesian state at arrival
505  if ( bodyMap.at( departureAndArrivalBodies.at( 1 ) )->getEphemeris( ) == nullptr){
506  throw std::runtime_error( "Ephemeris not defined for arrival body." );
507  }
508  else{
509  Eigen::Vector6d cartesianStateArrivalBody =
510  bodyMap.at( departureAndArrivalBodies.at( 1 ) )->getEphemeris( )->getCartesianState( initialTime + timeOfFlight );
511  cartesianPositionAtArrivalForLambertTargeter = cartesianStateArrivalBody.segment(0,3);
512  }
513  }
514  else
515  {
516  cartesianPositionAtArrivalForLambertTargeter = cartesianPositionAtArrival;
517  }
518 
519 
520 
521  // Calculate radii sphere of influence about departure and arrival bodies
522  double radiusSphereOfInfluenceDeparture;
523  double radiusSphereOfInfluenceArrival;
524 
525  std::pair< std::shared_ptr< propagators::PropagationTerminationSettings >,
526  std::shared_ptr< propagators::PropagationTerminationSettings > > terminationSettings;
527 
528  if (terminationSphereOfInfluence == true)
529  {
530 
531  double gravitationalParameterDepartureBody = ( departureBodyGravitationalParameter == departureBodyGravitationalParameter ) ?
532  departureBodyGravitationalParameter :
533  bodyMap.at( departureAndArrivalBodies[0] )->getGravityFieldModel( )->getGravitationalParameter( );
534  double gravitationalParameterArrivalBody = ( arrivalBodyGravitationalParameter == arrivalBodyGravitationalParameter ) ?
535  arrivalBodyGravitationalParameter :
536  bodyMap.at( departureAndArrivalBodies[1] )->getGravityFieldModel( )->getGravitationalParameter( );
537 
538  double distanceDepartureToCentralBodies =
539  bodyMap.at( centralBody )->getEphemeris( )->getCartesianState(
540  initialTime ).segment( 0, 3 ).norm( ) - cartesianPositionAtDepartureForLambertTargeter.segment( 0, 3 ).norm( );
541  double distanceArrivalToCentralBodies =
542  bodyMap.at( centralBody )->getEphemeris( )->getCartesianState(
543  initialTime + timeOfFlight ).segment( 0, 3 ).norm( ) - cartesianPositionAtArrivalForLambertTargeter.segment( 0, 3 ).norm( );
544 
545 
546  // Calculate radius sphere of influence for departure body.
547  radiusSphereOfInfluenceDeparture = tudat::mission_geometry::computeSphereOfInfluence(
548  distanceDepartureToCentralBodies, gravitationalParameterDepartureBody, gravitationalParameterCentralBody);
549 
550  // Calculate radius sphere of influence for arrival body.
551  radiusSphereOfInfluenceArrival = tudat::mission_geometry::computeSphereOfInfluence(
552  distanceArrivalToCentralBodies, gravitationalParameterArrivalBody, gravitationalParameterCentralBody);
553 
554  // Calculate the synodic period.
555  double orbitalPeriodDepartureBody = basic_astrodynamics::computeKeplerOrbitalPeriod(
