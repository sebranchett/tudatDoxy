130  void updateMembers( const double currentTime = TUDAT_NAN )
131  {
132  if( !( this->currentTime_ == currentTime ) )
133  {
134  // Calculate current normal vector and distance from accelerated body to source
135  currentNormalizedVectorToSource_ = ( sourcePositionFunction_( ) - acceleratedBodyPositionFunction_( ) );
136  currentDistanceToSource_ = currentNormalizedVectorToSource_.norm( );
137  currentNormalizedVectorToSource_.normalize( );
138 
139  // Retrieve current radiation pressure and mass
140  currentRadiationPressure_ = radiationPressureFunction_( );
141  currentMass_ = massFunction_( );
142 
143  // Iterate over all panels and calculate acceleration due to radiation pressure on panel
144  currentAcceleration_.setZero( );
145  if( currentRadiationPressure_ > 0.0 )
146  {
147  for( int i = 0; i < numberOfPanels_; i++ )
148  {
149  currentPanelAccelerations_[ i ] = currentRadiationPressure_ / massFunction_( ) * computeSinglePanelNormalizedRadiationPressureForce(
150  currentNormalizedVectorToSource_ , panelSurfaceNormalFunctions_[ i ]( ), panelAreaFunctions_[ i ]( ),
151  panelEmissivittyFunctions_[ i ]( ), panelDiffuseReflectionCoefficientFunctions_[ i ]( ) );
152  currentAcceleration_ += currentPanelAccelerations_[ i ];
153  }
154  }
155  this->currentTime_ = currentTime;
156  }
157  }
