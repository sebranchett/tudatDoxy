20 void DepartureLegMga1DsmVelocity::calculateLeg( Eigen::Vector3d& velocityBeforeArrivalBody,
21  double& deltaV )
22 {
23  // Calculate the DSM time of application from the time of flight fraction
24  dsmTime_ = dsmTimeOfFlightFraction_ * timeOfFlight_;
25 
26  // Calculate unit vectors as described in [Vinko and Izzo, 2008].
27  const Eigen::Vector3d unitVector1 = departureBodyVelocity_ / departureBodyVelocity_.norm( );
28  const Eigen::Vector3d unitVector3 = departureBodyPosition_.cross( departureBodyVelocity_ ) /
29  departureBodyPosition_.cross( departureBodyVelocity_ ).norm( );
30  const Eigen::Vector3d unitVector2 = unitVector3.cross( unitVector1 );
31 
32  // Calculate the velocity after departure as described in [Vinko and Izzo, 2008].
33  // First add the departure body velocity
34  velocityAfterDeparture_ = departureBodyVelocity_ +
35  // Then add the velocity in the direction of the departure velocity
36  excessVelocityMagnitude_ * cos( excessVelocityInPlaneAngle_ ) *
37  cos( excessVelocityOutOfPlaneAngle_ ) * unitVector1 +
38  // Then add the velocity in the other direction of the 2D plane of the departure velocity
39  excessVelocityMagnitude_ * sin( excessVelocityInPlaneAngle_ ) *
40  cos( excessVelocityOutOfPlaneAngle_ ) * unitVector2 +
41  // Finally add the 3D component
42  excessVelocityMagnitude_ * sin( excessVelocityOutOfPlaneAngle_ ) * unitVector3;
43 
44  // Transfer the initial position and velocity into a vectorXd object with cartesian coordinates.
45  Eigen::Vector6d cartesianElements (6), keplerianElements (6);
46  cartesianElements.segment( 0, 3 ) = departureBodyPosition_;
47  cartesianElements.segment( 3, 3 ) = velocityAfterDeparture_;
48 
49  // Convert the cartesian elements into keplerian elements.
50  keplerianElements = orbital_element_conversions::convertCartesianToKeplerianElements(
51  cartesianElements, centralBodyGravitationalParameter_ );
52 
53  // Propagate the keplerian elements until the moment of application of the DSM.
54  keplerianElements = orbital_element_conversions::propagateKeplerOrbit( keplerianElements,
55  dsmTime_, centralBodyGravitationalParameter_ );
56  // Convert the keplerian elements back into Cartesian elements.
57  cartesianElements = orbital_element_conversions::convertKeplerianToCartesianElements(
58  keplerianElements, centralBodyGravitationalParameter_ );
59 
60  // Set the corresponding position and velocity vectors.
61  dsmLocation_ = cartesianElements.segment( 0, 3 );
62  velocityBeforeDsm_ = cartesianElements.segment( 3, 3 );
63 
64  // Calculate the velocities after the DSM and before the arrival body.
65  mission_segments::solveLambertProblemIzzo( dsmLocation_, arrivalBodyPosition_, timeOfFlight_ -
66  dsmTime_, centralBodyGravitationalParameter_,
67  velocityAfterDsm_, velocityBeforeArrivalBody );
68 
69 
70  // Calculate the deltaV originating from the departure maneuver and the DSM.
71  escapeDeltaV_ = mission_segments::computeEscapeOrCaptureDeltaV(
72  departureBodyGravitationalParameter_, semiMajorAxis_, eccentricity_,
73  excessVelocityMagnitude_ );
74 
75  deltaVDsm_ = ( velocityAfterDsm_ - velocityBeforeDsm_ ).norm( );
76 
77  //Calculate the total deltaV.
78  if( includeDepartureDeltaV_ )
79  {
80  deltaV_ = escapeDeltaV_ + deltaVDsm_;
81  }
82  else
83  {
84  deltaV_ = deltaVDsm_;
85  }
86 
87  // Return the deltaV
88  deltaV = deltaV_;
89 }
