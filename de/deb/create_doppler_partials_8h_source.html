145 template< typename ParameterType >
146 std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > createOneWayDopplerPartials(
147  const observation_models::LinkEnds oneWayDopplerLinkEnds,
148  const simulation_setup::NamedBodyMap& bodyMap,
149  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
150  const std::shared_ptr< observation_models::DopplerProperTimeRateInterface > transmitterDopplerProperTimeInterface,
151  const std::shared_ptr< observation_models::DopplerProperTimeRateInterface > receiverDopplerProperTimeInterface,
152  const std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >& lightTimeCorrections =
153  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > >( ),
154  const bool useBiasPartials = true )
155 {
156  std::vector< std::shared_ptr< observation_partials::LightTimeCorrectionPartial > > lightTimeCorrectionPartialObjects;
157  if( lightTimeCorrections.size( ) > 0 )
158  {
159  lightTimeCorrectionPartialObjects = observation_partials::createLightTimeCorrectionPartials( lightTimeCorrections );
160  }
161 
162  std::function< Eigen::Vector6d( const double )> transmitterNumericalStateDerivativeFunction =
163  std::bind( &numerical_derivatives::computeCentralDifferenceFromFunction< Eigen::Vector6d, double >,
164  observation_models::getLinkEndCompleteEphemerisFunction< double, double >(
165  oneWayDopplerLinkEnds.at( observation_models::transmitter ), bodyMap ), std::placeholders::_1, 100.0,
166  numerical_derivatives::order8 );
167  std::function< Eigen::Vector6d( const double )> receiverNumericalStateDerivativeFunction =
168  std::bind( numerical_derivatives::computeCentralDifferenceFromFunction< Eigen::Vector6d, double >,
169  observation_models::getLinkEndCompleteEphemerisFunction< double, double >(
170  oneWayDopplerLinkEnds.at( observation_models::receiver ), bodyMap ), std::placeholders::_1, 100.0,
171  numerical_derivatives::order8 );
172 
173  // Create scaling object, to be used for all one-way doppler partials in current link end.
174  std::shared_ptr< OneWayDopplerProperTimeComponentScaling > transmitterProperTimePartials =
175  createDopplerProperTimePartials( transmitterDopplerProperTimeInterface, oneWayDopplerLinkEnds,
176  observation_models::transmitter );
177  std::shared_ptr< OneWayDopplerProperTimeComponentScaling > receiverProperTimePartials =
178  createDopplerProperTimePartials( receiverDopplerProperTimeInterface, oneWayDopplerLinkEnds,
179  observation_models::receiver );
180 
181  std::shared_ptr< OneWayDopplerScaling > oneWayDopplerScaling = std::make_shared< OneWayDopplerScaling >(
182  std::bind( &linear_algebra::evaluateSecondBlockInStateVector, transmitterNumericalStateDerivativeFunction, std::placeholders::_1 ),
183  std::bind( &linear_algebra::evaluateSecondBlockInStateVector, receiverNumericalStateDerivativeFunction, std::placeholders::_1 ),
184  transmitterProperTimePartials,
185  receiverProperTimePartials );
186 
187 
188  // Initialize vector index variables.
189  int currentIndex = 0;
190  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
191 
192  SingleLinkObservationPartialList dopplerPartials;
193 
194  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
195  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
196  parametersToEstimate->getEstimatedInitialStateParameters( );
197 
198  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
199  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
200  {
201  std::string acceleratedBody;
202  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state ||
203  ( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state ) )
204  {
205  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
206  }
207  else
208  {
209  throw std::runtime_error( "Error when making one way doppler partials, could not identify parameter " +
210  std::to_string(
211  initialDynamicalParameters.at( i )->getParameterName( ).first ) );
212  }
213 
214 
215  // Create position one-way doppler partial for current body
216  std::shared_ptr< ObservationPartial< 1 > > currentDopplerPartialWrtPosition = createOneWayDopplerPartialWrtBodyState(
217  oneWayDopplerLinkEnds, bodyMap, acceleratedBody, oneWayDopplerScaling, lightTimeCorrectionPartialObjects );
218 
219  // Check if partial is non-nullptr (i.e. whether dependency exists between current doppler and current body)
220  if( currentDopplerPartialWrtPosition != nullptr )
221  {
222  // Add partial to the list.
223  currentPair = std::pair< int, int >( currentIndex, 6 );
224  dopplerPartials[ currentPair ] = currentDopplerPartialWrtPosition;
225  }
226 
227  // Increment current index by size of body initial state (6).
228  currentIndex += 6;
229  }
230 
231  // Iterate over all double parameters that are to be estimated.
232  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParametersToEstimate =
233  parametersToEstimate->getDoubleParameters( );
234  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
235  parameterIterator =
236  doubleParametersToEstimate.begin( ); parameterIterator != doubleParametersToEstimate.end( ); parameterIterator++ )
237  {
238  // Create position one-way doppler partial for current parameter
239  std::shared_ptr< ObservationPartial< 1 > > currentDopplerPartial = createOneWayDopplerPartialWrtParameter(
240  oneWayDopplerLinkEnds, bodyMap, parameterIterator->second,
241  oneWayDopplerScaling, lightTimeCorrectionPartialObjects );
242 
243  // Check if partial is non-nullptr (i.e. whether dependency exists between current doppler and current parameter)
244  if( currentDopplerPartial != nullptr )
245  {
246  // Add partial to the list.
247  currentPair = std::pair< int, int >( parameterIterator->first, 1 );
248  dopplerPartials[ currentPair ] = currentDopplerPartial;
249  }
250  }
251 
252  // Iterate over all vector parameters that are to be estimated.
253  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
254  vectorParametersToEstimate =
255  parametersToEstimate->getVectorParameters( );
256  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
257  parameterIterator =
258  vectorParametersToEstimate.begin( ); parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
259  {
260 
261  std::shared_ptr< ObservationPartial< 1 > > currentDopplerPartial;
262  if( !isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
263  {
264  currentDopplerPartial = createOneWayDopplerPartialWrtParameter(
265  oneWayDopplerLinkEnds, bodyMap, parameterIterator->second, oneWayDopplerScaling,
266  lightTimeCorrectionPartialObjects );
267  }
268  else
269  {
270  currentDopplerPartial = createObservationPartialWrtLinkProperty< 1 >(
271  oneWayDopplerLinkEnds, observation_models::one_way_doppler, parameterIterator->second, useBiasPartials );
272  }
273 
274 
275  // Check if partial is non-nullptr (i.e. whether dependency exists between current doppler and current parameter)
276  if( currentDopplerPartial != nullptr )
277  {
278  // Add partial to the list.
279  currentPair = std::pair< int, int >( parameterIterator->first,
280  parameterIterator->second->getParameterSize( ) );
281  dopplerPartials[ currentPair ] = currentDopplerPartial;
282  }
283  }
284 
285  // Return complete set of partials and scaling object.
286  return std::make_pair( dopplerPartials, oneWayDopplerScaling );
287 }

304 template< typename ObservationScalarType, typename TimeType >
305 std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
306 std::shared_ptr< PositionPartialScaling > > > createOneWayDopplerPartials(
307  const std::map< observation_models::LinkEnds,
308  std::shared_ptr< observation_models::ObservationModel< 1, ObservationScalarType, TimeType > > > observationModelList,
309  const simulation_setup::NamedBodyMap& bodyMap,
310  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > > parametersToEstimate,
311  const bool useBiasPartials = true )
312 {
313  // Declare return list.
314  std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
315  std::shared_ptr< PositionPartialScaling > > > dopplerPartials;
316 
317  // Iterate over all link ends.
318  for( auto linkIterator: observationModelList )
319  {
320  // Check if required link end types are present
321  if( ( linkIterator.first.count( observation_models::receiver ) == 0 ) ||
322  ( linkIterator.first.count( observation_models::transmitter ) == 0 ) )
323  {
324  throw std::runtime_error( "Error when making 1-way doppler partials, did not find both receiver and transmitter in link ends" );
325 
326  }
327 
328 
329  std::shared_ptr< observation_models::OneWayDopplerObservationModel< ObservationScalarType, TimeType > >
330  dopplerObservationModel =
331  std::dynamic_pointer_cast< observation_models::OneWayDopplerObservationModel
332  < ObservationScalarType, TimeType > >( linkIterator.second );
333 
334  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > singleLinkLightTimeCorrections;
335  if( dopplerObservationModel == nullptr )
336  {
337  throw std::runtime_error( "Error when making one-way Doppler partials. Type is inconsistent" );
338  }
339  else
340  {
341  singleLinkLightTimeCorrections = dopplerObservationModel->getLightTimeCalculator( )->getLightTimeCorrection( );
342  }
343 
344  // Create doppler partials for current link ends
345  dopplerPartials[ linkIterator.first ] = createOneWayDopplerPartials< ObservationScalarType >(
346  linkIterator.first, bodyMap, parametersToEstimate,
347  dopplerObservationModel->getTransmitterProperTimeRateCalculator( ),
348  dopplerObservationModel->getReceiverProperTimeRateCalculator( ),
349  singleLinkLightTimeCorrections, useBiasPartials );
350  }
351 
352  // Return complete set of link ends.
353  return dopplerPartials;
354 }

378 template< typename ParameterType, typename TimeType >
379 std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > createTwoWayDopplerPartials(
380  const observation_models::LinkEnds& twoWayDopplerLinkEnds,
381  const std::shared_ptr< observation_models::TwoWayDopplerObservationModel< ParameterType, TimeType > >
382  twoWayObservationModel,
383  const simulation_setup::NamedBodyMap& bodyMap,
384  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
385  const std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > > lightTimeCorrections =
386  std::vector< std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > >( ) )
387 
388 {
389 
390  // Define list of constituent one-way partials.
391  typedef std::vector< std::pair< SingleLinkObservationPartialList, std::shared_ptr< PositionPartialScaling > > >
392  OneWayPartialList;
393  OneWayPartialList constituentOneWayDopplerPartials;
394  OneWayPartialList constituentOneWayRangePartials;
395 
396  int numberOfLinkEnds = 3;
397  observation_models::LinkEnds currentLinkEnds;
398 
399  // Iterate over all links in the two-way Doppler observable
400  std::vector< std::function< double( const double, const observation_models::LinkEndType ) > > oneWayDopplerModels;
401  std::vector< std::shared_ptr< observation_models::LightTimeCorrection > > currentLightTimeCorrections;
402  for( int i = 0; i < numberOfLinkEnds - 1; i++ )
403  {
404  currentLightTimeCorrections.clear( );
405  if( lightTimeCorrections.size( ) > 0 )
406  {
407  currentLightTimeCorrections = lightTimeCorrections.at( i );
408  }
409 
410  // Define links for current one-way Doppler link
411  currentLinkEnds.clear( );
412  currentLinkEnds[ observation_models::transmitter ] = twoWayDopplerLinkEnds.at(
413  observation_models::getNWayLinkEnumFromIndex( i, 3 ) );
414  currentLinkEnds[ observation_models::receiver ] = twoWayDopplerLinkEnds.at(
415  observation_models::getNWayLinkEnumFromIndex( i + 1, 3 ) );
416 
417  std::shared_ptr< observation_models::DopplerProperTimeRateInterface > transmitterDopplerProperTimeInterface;
418  std::shared_ptr< observation_models::DopplerProperTimeRateInterface > receiverDopplerProperTimeInterface;
419  if( i == 0 )
420  {
421  transmitterDopplerProperTimeInterface =
422  twoWayObservationModel->getUplinkDopplerCalculator( )->getTransmitterProperTimeRateCalculator( );
423  receiverDopplerProperTimeInterface =
424  twoWayObservationModel->getUplinkDopplerCalculator( )->getReceiverProperTimeRateCalculator( );
425  oneWayDopplerModels.push_back(
426  observation_models::getSizeOneObservationFunctionAtDoublePrecisionFromObservationModel<
427  ParameterType, TimeType >( twoWayObservationModel->getUplinkDopplerCalculator( ) ) );
428 
429  }
430  else if( i == 1 )
431  {
432  transmitterDopplerProperTimeInterface =
433  twoWayObservationModel->getDownlinkDopplerCalculator( )->getTransmitterProperTimeRateCalculator( );
434  receiverDopplerProperTimeInterface =
435  twoWayObservationModel->getDownlinkDopplerCalculator( )->getReceiverProperTimeRateCalculator( );
436  oneWayDopplerModels.push_back(
437  observation_models::getSizeOneObservationFunctionAtDoublePrecisionFromObservationModel<
438  ParameterType, TimeType >( twoWayObservationModel->getDownlinkDopplerCalculator( ) ) );
439  }
440 
441 
442  // Create one-way Doppler partials for current link
443  constituentOneWayDopplerPartials.push_back(
444  createOneWayDopplerPartials(
445  currentLinkEnds, bodyMap, parametersToEstimate, transmitterDopplerProperTimeInterface,
446  receiverDopplerProperTimeInterface, currentLightTimeCorrections, false ) );
447  constituentOneWayRangePartials.push_back(
448  createOneWayRangePartials( currentLinkEnds, bodyMap, parametersToEstimate,
449  currentLightTimeCorrections ) );
450  }
451 
452  // Retrieve sorted (by parameter index and link index) one-way range partials and (by link index) opne-way range partials
453  std::vector< std::shared_ptr< OneWayDopplerScaling > > oneWayDopplerScalers;
454  std::vector< std::shared_ptr< OneWayRangeScaling > > oneWayRangeScalings;
455 
456  std::map< std::pair< int, int >, std::map< int, std::shared_ptr< ObservationPartial< 1 > > > > sortedOneWayDopplerPartials;
457  std::map< std::pair< int, int >, std::map< int, std::shared_ptr< ObservationPartial< 1 > > > > sortedOneWayRangePartials;
458 
459  std::map< std::pair< int, int >, estimatable_parameters::EstimatebleParameterIdentifier > parameterIdList;
460  for( unsigned int i = 0; i < constituentOneWayDopplerPartials.size( ); i++ )
461  {
462  // Retrieve one-way Doppler paritals
463  oneWayDopplerScalers.push_back( std::dynamic_pointer_cast< OneWayDopplerScaling >
464  ( constituentOneWayDopplerPartials.at( i ).second ) );
465  oneWayRangeScalings.push_back( std::dynamic_pointer_cast< OneWayRangeScaling >
466  ( constituentOneWayRangePartials.at( i ).second ) );
467 
468  // Iterate over all one-way Doppler partials of current link
469  for( SingleLinkObservationPartialList::iterator parameterIterator = constituentOneWayDopplerPartials.at( i ).first.begin( );
470  parameterIterator != constituentOneWayDopplerPartials.at( i ).first.end( ); parameterIterator++ )
471  {
472  sortedOneWayDopplerPartials[ parameterIterator->first ][ i ] = parameterIterator->second;
473  if( parameterIdList.count( parameterIterator->first ) == 0 )
474  {
475  parameterIdList[ parameterIterator->first ] = parameterIterator->second->getParameterIdentifier( );
476  }
477  else if( parameterIdList.at( parameterIterator->first ) != parameterIterator->second->getParameterIdentifier( ) )
478  {
479  throw std::runtime_error( "Error when making two way Doppler partial, parameter indices in Doppler are inconsistent" );
480  }
481  }
482 
483  // Iterate over all one-way range partials of current link
484  for( SingleLinkObservationPartialList::iterator parameterIterator = constituentOneWayRangePartials.at( i ).first.begin( );
485  parameterIterator != constituentOneWayRangePartials.at( i ).first.end( ); parameterIterator++ )
486  {
487  sortedOneWayRangePartials[ parameterIterator->first ][ i ] = parameterIterator->second;
488  if( parameterIdList.count( parameterIterator->first ) == 0 )
489  {
490  parameterIdList[ parameterIterator->first ] = parameterIterator->second->getParameterIdentifier( );
491  }
492  else if( parameterIdList.at( parameterIterator->first ) != parameterIterator->second->getParameterIdentifier( ) )
493  {
494  throw std::runtime_error( "Error when making two way Doppler partial, parameter indices in range are inconsistent" );
495  }
496  }
497  }
498 
499  // Create two-way Doppler scaling object
500  std::shared_ptr< TwoWayDopplerScaling > twoWayDopplerScaler = std::make_shared< TwoWayDopplerScaling >(
501  oneWayDopplerScalers, oneWayRangeScalings, oneWayDopplerModels );
502 
503  // Define return partial list
504  SingleLinkObservationPartialList twoWayDopplerPartialList;
505 
506  // Create two-way Doppler partial objects
507  for( std::map< std::pair< int, int >, std::map< int, std::shared_ptr< ObservationPartial< 1 > > > >::iterator sortedPartialIterator =
508  sortedOneWayDopplerPartials.begin( ); sortedPartialIterator != sortedOneWayDopplerPartials.end( ); sortedPartialIterator++ )
509  {
510  std::map< int, std::shared_ptr< ObservationPartial< 1 > > > currentDopplerPartialList = sortedPartialIterator->second;
511  std::map< int, std::shared_ptr< ObservationPartial< 1 > > > currentRangePartialList;
512  if( sortedOneWayRangePartials.count( sortedPartialIterator->first ) != 0 )
513  {
514  currentRangePartialList = sortedOneWayRangePartials.at( sortedPartialIterator->first );
515  }
516 
517  twoWayDopplerPartialList[ sortedPartialIterator->first ] = std::make_shared< TwoWayDopplerPartial >(
518  twoWayDopplerScaler, currentDopplerPartialList, currentRangePartialList,
519  parameterIdList.at( sortedPartialIterator->first ), numberOfLinkEnds );
520  }
521 
522  // Create two-way Doppler partial objects that only have range dependencies.
523  for( std::map< std::pair< int, int >, std::map< int, std::shared_ptr< ObservationPartial< 1 > > > >::iterator sortedPartialIterator =
524  sortedOneWayRangePartials.begin( ); sortedPartialIterator != sortedOneWayRangePartials.end( ); sortedPartialIterator++ )
525  {
526  if( sortedOneWayDopplerPartials.count( sortedPartialIterator->first ) == 0 )
527  {
528  std::map< int, std::shared_ptr< ObservationPartial< 1 > > > currentDopplerPartialList;
529  std::map< int, std::shared_ptr< ObservationPartial< 1 > > > currentRangePartialList = sortedPartialIterator->second;
530 
531  twoWayDopplerPartialList[ sortedPartialIterator->first ] = std::make_shared< TwoWayDopplerPartial >(
532  twoWayDopplerScaler, currentDopplerPartialList, currentRangePartialList,
533  parameterIdList.at( sortedPartialIterator->first ), numberOfLinkEnds );
534  }
535  }
536 
537  // Create two-way Doppler partials
538  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
539  vectorParametersToEstimate = parametersToEstimate->getVectorParameters( );
540  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
541  parameterIterator =
542  vectorParametersToEstimate.begin( ); parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
543  {
544 
545  std::shared_ptr< ObservationPartial< 1 > > currentTwoWayDopplerPartial;
546  if( isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
547  {
548  currentTwoWayDopplerPartial = createObservationPartialWrtLinkProperty< 1 >(
549  twoWayDopplerLinkEnds, observation_models::two_way_doppler, parameterIterator->second );
550  }
551 
552  // Check if partial is non-nullptr (i.e. whether dependency exists between current doppler and current parameter)
553  if( currentTwoWayDopplerPartial != nullptr )
554  {
555  // Add partial to the list.
556  std::pair< double, double > currentPair = std::pair< int, int >( parameterIterator->first,
557  parameterIterator->second->getParameterSize( ) );
558  twoWayDopplerPartialList[ currentPair ] = currentTwoWayDopplerPartial;
559  }
560  }
561 
562  return std::make_pair( twoWayDopplerPartialList, twoWayDopplerScaler );
563 }
