486 template< typename ObservationScalarType = double, typename TimeType = double >
487 std::shared_ptr< DopplerProperTimeRateInterface > createOneWayDopplerProperTimeCalculator(
488  std::shared_ptr< DopplerProperTimeRateSettings > properTimeRateSettings,
489  const LinkEnds& linkEnds,
490  const simulation_setup::NamedBodyMap &bodyMap,
491  const LinkEndType linkEndForCalculator )
492 {
493  std::shared_ptr< DopplerProperTimeRateInterface > properTimeRateInterface;
494 
495  // Check tyope of proper time rate model
496  switch( properTimeRateSettings->dopplerProperTimeRateType_ )
497  {
498  case direct_first_order_doppler_proper_time_rate:
499  {
500  // Check input consistency
501  std::shared_ptr< DirectFirstOrderDopplerProperTimeRateSettings > directFirstOrderDopplerProperTimeRateSettings =
502  std::dynamic_pointer_cast< DirectFirstOrderDopplerProperTimeRateSettings >( properTimeRateSettings );
503  if( directFirstOrderDopplerProperTimeRateSettings == nullptr )
504  {
505  throw std::runtime_error( "Error when making DopplerProperTimeRateInterface, input type (direct_first_order_doppler_proper_time_rate) is inconsistent" );
506  }
507  else if( linkEnds.count( linkEndForCalculator ) == 0 )
508  {
509  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not find link end " +
510  std::to_string( linkEndForCalculator );
511  throw std::runtime_error( errorMessage );
512  }
513  else
514  {
515  if( bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->getGravityFieldModel( ) == nullptr )
516  {
517  throw std::runtime_error( "Error when making DirectFirstOrderDopplerProperTimeRateInterface, no gravity field found for " +
518  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ );
519  }
520  else
521  {
522  // Retrieve gravitational parameter
523  std::function< double( ) > gravitationalParameterFunction =
524  std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
525  bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->
526  getGravityFieldModel( ) );
527 
528  // Create calculation object.
529  LinkEndId referencePointId =
530  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, "" );
531  if( ( linkEnds.at( receiver ) != referencePointId ) && ( linkEnds.at( transmitter ) != referencePointId ) )
532  {
533  properTimeRateInterface = std::make_shared<
534  DirectFirstOrderDopplerProperTimeRateInterface >(
535  linkEndForCalculator, gravitationalParameterFunction,
536  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, unidentified_link_end,
537  getLinkEndCompleteEphemerisFunction< double, double >(
538  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, ""), bodyMap ) );
539  }
540  else
541  {
542  throw std::runtime_error(
543  "Error, proper time reference point as link end not yet implemented for DopplerProperTimeRateInterface creation" );
544  }
545  }
546  }
547  break;
548  }
549  default:
550  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not recognize type " +
551  std::to_string( properTimeRateSettings->dopplerProperTimeRateType_ );
552  throw std::runtime_error( errorMessage );
553  }
554  return properTimeRateInterface;
555 }

597 template< int ObservationSize = 1 >
598 std::shared_ptr< ObservationBias< ObservationSize > > createObservationBiasCalculator(
599  const LinkEnds linkEnds,
600  const ObservableType observableType,
601  const std::shared_ptr< ObservationBiasSettings > biasSettings,
602  const simulation_setup::NamedBodyMap &bodyMap )
603 {
604  std::shared_ptr< ObservationBias< ObservationSize > > observationBias;
605  switch( biasSettings->observationBiasType_ )
606  {
607  case constant_absolute_bias:
608  {
609  // Check input consistency
610  std::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = std::dynamic_pointer_cast<
611  ConstantObservationBiasSettings >( biasSettings );
612  if( constantBiasSettings == nullptr )
613  {
614  throw std::runtime_error( "Error when making constant observation bias, settings are inconsistent" );
615  }
616 
617  if( !constantBiasSettings->useAbsoluteBias_ )
618  {
619  throw std::runtime_error( "Error when making constant observation bias, class settings are inconsistent" );
620  }
621 
622  // Check if size of bias is consistent with requested observable size
623  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
624  {
625  throw std::runtime_error( "Error when making constant observation bias, bias size is inconsistent" );
626  }
627  observationBias = std::make_shared< ConstantObservationBias< ObservationSize > >(
628  constantBiasSettings->observationBias_ );
629  break;
630  }
631  case arc_wise_constant_absolute_bias:
632  {
633  // Check input consistency
634  std::shared_ptr< ArcWiseConstantObservationBiasSettings > arcwiseBiasSettings = std::dynamic_pointer_cast<
635  ArcWiseConstantObservationBiasSettings >( biasSettings );
636  if( arcwiseBiasSettings == nullptr )
637  {
638  throw std::runtime_error( "Error when making arc-wise observation bias, settings are inconsistent" );
639  }
640  else if( !arcwiseBiasSettings->useAbsoluteBias_ )
641  {
642  throw std::runtime_error( "Error when making arc-wise observation bias, class contents are inconsistent" );
643  }
644 
645  std::vector< Eigen::Matrix< double, ObservationSize, 1 > > observationBiases;
646  for( unsigned int i = 0; i < arcwiseBiasSettings->observationBiases_.size( ); i++ )
647  {
648  // Check if size of bias is consistent with requested observable size
649  if( arcwiseBiasSettings->observationBiases_.at( i ).rows( ) != ObservationSize )
650  {
651  throw std::runtime_error( "Error when making arc-wise observation bias, bias size is inconsistent" );
652  }
653  else
654  {
655  observationBiases.push_back( arcwiseBiasSettings->observationBiases_.at( i ) );
656  }
657  }
658  observationBias = std::make_shared< ConstantArcWiseObservationBias< ObservationSize > >(
659  arcwiseBiasSettings->arcStartTimes_, observationBiases,
660  observation_models::getLinkEndIndicesForLinkEndTypeAtObservable(
661  observableType, arcwiseBiasSettings->linkEndForTime_, linkEnds.size( ) ).at( 0 ) );
662  break;
663  }
664  case constant_relative_bias:
665  {
666  // Check input consistency
667  std::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = std::dynamic_pointer_cast<
668  ConstantObservationBiasSettings >( biasSettings );
669  if( constantBiasSettings == nullptr )
670  {
671  throw std::runtime_error( "Error when making constant relative observation bias, settings are inconsistent" );
672  }
673 
674  if( constantBiasSettings->useAbsoluteBias_ )
675  {
676  throw std::runtime_error( "Error when making constant relative observation bias, class settings are inconsistent" );
677  }
678 
679  // Check if size of bias is consistent with requested observable size
680  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
681  {
682  throw std::runtime_error( "Error when making constant relative observation bias, bias size is inconsistent" );
683  }
684  observationBias = std::make_shared< ConstantRelativeObservationBias< ObservationSize > >(
685  constantBiasSettings->observationBias_ );
686  break;
687  }
688  case arc_wise_constant_relative_bias:
689  {
690  // Check input consistency
691  std::shared_ptr< ArcWiseConstantObservationBiasSettings > arcwiseBiasSettings = std::dynamic_pointer_cast<
692  ArcWiseConstantObservationBiasSettings >( biasSettings );
693  if( arcwiseBiasSettings == nullptr )
694  {
695  throw std::runtime_error( "Error when making arc-wise relative observation bias, settings are inconsistent" );
696  }
697  else if( arcwiseBiasSettings->useAbsoluteBias_ )
698  {
699  throw std::runtime_error( "Error when making arc-wise relative observation bias, class contents are inconsistent" );
700  }
701 
702  std::vector< Eigen::Matrix< double, ObservationSize, 1 > > observationBiases;
703  for( unsigned int i = 0; i < arcwiseBiasSettings->observationBiases_.size( ); i++ )
704  {
705  // Check if size of bias is consistent with requested observable size
706  if( arcwiseBiasSettings->observationBiases_.at( i ).rows( ) != ObservationSize )
707  {
708  throw std::runtime_error( "Error when making arc-wise observation bias, bias size is inconsistent" );
709  }
710  else
711  {
712  observationBiases.push_back( arcwiseBiasSettings->observationBiases_.at( i ) );
713  }
714  }
715  observationBias = std::make_shared< ConstantRelativeArcWiseObservationBias< ObservationSize > >(
716  arcwiseBiasSettings->arcStartTimes_, observationBiases,
717  observation_models::getLinkEndIndicesForLinkEndTypeAtObservable(
718  observableType, arcwiseBiasSettings->linkEndForTime_, linkEnds.size( ) ).at( 0 ) );
719  break;
720  }
721  case multiple_observation_biases:
722  {
723  // Check input consistency
724  std::shared_ptr< MultipleObservationBiasSettings > multiBiasSettings = std::dynamic_pointer_cast<
725  MultipleObservationBiasSettings >( biasSettings );
726  if( multiBiasSettings == nullptr )
727  {
728  throw std::runtime_error( "Error when making multiple observation biases, settings are inconsistent" );
729  }
730 
731  // Create list of biases
732  std::vector< std::shared_ptr< ObservationBias< ObservationSize > > > observationBiasList;
733  for( unsigned int i = 0; i < multiBiasSettings->biasSettingsList_.size( ); i++ )
734  {
735  observationBiasList.push_back( createObservationBiasCalculator< ObservationSize >(
736  linkEnds, observableType, multiBiasSettings->biasSettingsList_.at( i ) , bodyMap ) );
737  }
738 
739  // Create combined bias object
740  observationBias = std::make_shared< MultiTypeObservationBias< ObservationSize > >(
741  observationBiasList );
742  break;
743  }
744  default:
745  {
746  std::string errorMessage = "Error when making observation bias, bias type " +
747  std::to_string( biasSettings->observationBiasType_ ) + " not recognized";
748  throw std::runtime_error( errorMessage );
749  }
750  }
751  return observationBias;
752 }
