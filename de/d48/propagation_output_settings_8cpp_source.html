217 std::string getDependentVariableId(
218  const std::shared_ptr< SingleDependentVariableSaveSettings > dependentVariableSettings )
219 {
220  std::string variableId = getDependentVariableName( dependentVariableSettings->dependentVariableType_ );
221 
222  if( ( dependentVariableSettings->dependentVariableType_ == single_acceleration_dependent_variable ) ||
223  ( dependentVariableSettings->dependentVariableType_ == single_acceleration_norm_dependent_variable ) )
224  {
225  std::shared_ptr< SingleAccelerationDependentVariableSaveSettings > accelerationDependentVariableSettings =
226  std::dynamic_pointer_cast< SingleAccelerationDependentVariableSaveSettings >( dependentVariableSettings );
227  if( accelerationDependentVariableSettings == nullptr )
228  {
229  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (acceleration type )" );
230  }
231  else
232  {
233  variableId += basic_astrodynamics::getAccelerationModelName(
234  accelerationDependentVariableSettings->accelerationModelType_ );
235  }
236  }
237  else if( ( dependentVariableSettings->dependentVariableType_ == single_torque_dependent_variable ) ||
238  ( dependentVariableSettings->dependentVariableType_ == single_torque_norm_dependent_variable ) )
239  {
240  std::shared_ptr< SingleTorqueDependentVariableSaveSettings > torqueDependentVariableSettings =
241  std::dynamic_pointer_cast< SingleTorqueDependentVariableSaveSettings >( dependentVariableSettings );
242  if( torqueDependentVariableSettings == nullptr )
243  {
244  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (torque type )" );
245  }
246  else
247  {
248  variableId += basic_astrodynamics::getTorqueModelName(
249  torqueDependentVariableSettings->torqueModelType_ );
250  }
251  }
252  else if( dependentVariableSettings->dependentVariableType_ == intermediate_aerodynamic_rotation_matrix_variable )
253  {
254  std::shared_ptr< IntermediateAerodynamicRotationVariableSaveSettings > rotationDependentVariableSettings =
255  std::dynamic_pointer_cast< IntermediateAerodynamicRotationVariableSaveSettings >( dependentVariableSettings );
256  if( rotationDependentVariableSettings == nullptr )
257  {
258  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (rotation matrix)" );
259  }
260  else
261  {
262  variableId +=
263  reference_frames::getAerodynamicFrameName( rotationDependentVariableSettings->baseFrame_ ) + "to " +
264  reference_frames::getAerodynamicFrameName( rotationDependentVariableSettings->targetFrame_ );
265  }
266  }
267 
268  else if( dependentVariableSettings->dependentVariableType_ == relative_body_aerodynamic_orientation_angle_variable )
269  {
270  std::shared_ptr< BodyAerodynamicAngleVariableSaveSettings > angleDependentVariableSettings =
271  std::dynamic_pointer_cast< BodyAerodynamicAngleVariableSaveSettings >( dependentVariableSettings );
272  if( angleDependentVariableSettings == nullptr )
273  {
274  throw std::runtime_error( "Error when getting dependent variable ID, input is inconsistent (angle)" );
275  }
276  else
277  {
278  variableId +=
279  reference_frames::getAerodynamicAngleName( angleDependentVariableSettings->angle_ );
280  }
281  }
282 
283  if( ( dependentVariableSettings->dependentVariableType_ == single_acceleration_dependent_variable ) ||
284  ( dependentVariableSettings->dependentVariableType_ == single_acceleration_norm_dependent_variable ) ||
285  ( dependentVariableSettings->dependentVariableType_ == spherical_harmonic_acceleration_terms_dependent_variable ) ||
286  ( dependentVariableSettings->dependentVariableType_ == spherical_harmonic_acceleration_norm_terms_dependent_variable ) )
287  {
288  variableId += ", acting on " + dependentVariableSettings->associatedBody_;
289  if( dependentVariableSettings->secondaryBody_ != dependentVariableSettings->associatedBody_ )
290  {
291  variableId += ", exerted by " + dependentVariableSettings->secondaryBody_;
292  }
293  }
294  else
295  {
296  variableId += "of " + dependentVariableSettings->associatedBody_;
297  if( dependentVariableSettings->secondaryBody_ != "" )
298  {
299  variableId += " w.r.t. " + dependentVariableSettings->secondaryBody_;
300  }
301 
302  }
303  return variableId;
304 }
