650 template< typename StateScalarType = double, typename TimeType = double >
651 std::shared_ptr< ephemerides::Ephemeris > createTabulatedEphemerisFromSpice(
652  const std::string& body,
653  const TimeType initialTime,
654  const TimeType endTime,
655  const TimeType timeStep,
656  const std::string& observerName,
657  const std::string& referenceFrameName,
658  std::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
659  std::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
660 {
661  using namespace interpolators;
662 
663  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > timeHistoryOfState;
664 
665  // Calculate state from spice at given time intervals and store in timeHistoryOfState.
666  TimeType currentTime = initialTime;
667  while( currentTime < endTime )
668  {
669  timeHistoryOfState[ currentTime ] = spice_interface::getBodyCartesianStateAtEpoch(
670  body, observerName, referenceFrameName, "none", static_cast< double >( currentTime ) ).
671  template cast< StateScalarType >( );
672  currentTime += timeStep;
673  }
674 
675  // Create interpolator.
676  std::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > > interpolator =
677  interpolators::createOneDimensionalInterpolator(
678  timeHistoryOfState, interpolatorSettings );
679 
680  // Create ephemeris and return.
681  return std::make_shared< ephemerides::TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
682  interpolator, observerName, referenceFrameName );
683 }
684 #endif
