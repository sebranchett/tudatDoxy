22 double ShortPeriodEarthOrientationCorrectionCalculator< double >::sumCorrectionTerms( const Eigen::Vector6d& arguments )
23 {
24  // Initialize corrections to zero.
25  double currentCorrection = 0;
26  double tideAngle = 0.0;
27 
28  // Iterate over all correction types
29  for( unsigned int i = 0; i < argumentAmplitudes_.size( ); i++ )
30  {
31  Eigen::MatrixXd currentArgumentMultipliers = argumentMultipliers_.at( i );
32  Eigen::MatrixXd currentAmplitudes = argumentAmplitudes_.at( i );
33 
34  // Iterte over all libration corrections.
35  for( int i = 0; i < currentAmplitudes.rows( ); i++ )
36  {
37  // Calculate current phase angle of tide.
38  tideAngle = ( arguments.transpose( ) *
39  currentArgumentMultipliers.block( i, 0, 1, 6 ).transpose( ) )( 0, 0 );
40 
41  currentCorrection += currentAmplitudes( i, 0 ) * std::sin( tideAngle ) +
42  currentAmplitudes( i, 1 ) * std::cos( tideAngle );
43  }
44  }
45 
46 
47  return currentCorrection;
48 }
