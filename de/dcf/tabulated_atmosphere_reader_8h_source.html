51 {

59

61  for( std::map< int, std::string >::const_iterator fileIterator = fileNames.begin( ); fileIterator != fileNames.end( );
62  fileIterator++ )
63  {
64  // Read current coefficients/independent variables
65  std::pair< boost::multi_array< double, static_cast< size_t >( NumberOfDimensions ) >,
66  std::vector< std::vector< double > > > currentCoefficients =
67  MultiArrayFileReader< NumberOfDimensions >::readMultiArrayAndIndependentVariables( fileIterator->second );
68 
69  // Save/check consistency of independent variables
70  if( rawAtmosphereArrays.size( ) == 0 )
71  {
72  independentVariables = currentCoefficients.second;
73  }
74  else
75  {
76  bool areIndependentVariablesEqual = compareIndependentVariables(
77  independentVariables, currentCoefficients.second );
78 
79  if( !areIndependentVariablesEqual )
80  {
81  throw std::runtime_error( "Error when reading 1-Dimensional atmosphere parameters, inconsistent independent variables." );
82  }
83  }

85  // Save file contents into rawAtmosphereArrays
86  utilities::copyMultiArray< double, NumberOfDimensions >(
87  currentCoefficients.first, rawAtmosphereArrays[ fileIterator->first ] );
88  }
89 
90  // Check if anything has been read from files.

92  {

94  }
95  else

100

102  for( unsigned int i = 0; i < numberOfFiles; i++ )
103  {

105  if( rawAtmosphereArrays.count( i ) != 0 )
106  {

109  }

111  else

116 
117  atmosphereArrays[ i ].resize( sizeVector );

121  }
122  }
123  }
124
