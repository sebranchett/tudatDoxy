268  void updateMembers( const double currentTime = TUDAT_NAN )
269  {
270  if( !( this->currentTime_ == currentTime ) )
271  {
272  currentFrontEmissivityCoefficient_ = frontEmissivityCoefficientFunction_( );
273  currentBackEmissivityCoefficient_ = backEmissivityCoefficientFunction_( );
274  currentFrontLambertianCoefficient_ = frontLambertianCoefficientFunction_( );
275  currentBackLambertianCoefficient_ = backLambertianCoefficientFunction_( );
276  currentReflectivityCoefficient_ = reflectivityCoefficientFunction_( );
277  currentSpecularReflectionCoefficient_ = specularReflectionCoefficientFunction_( );
278  currentNormalizedVectorToSource_ = ( sourcePositionFunction_( ) - acceleratedBodyPositionFunction_( ) ).normalized( );
279  currentNormalizedVelocityVector_= ( acceleratedBodyVelocityFunction_( ) - centralBodyVelocityFunction_( ) ).normalized( );
280  currentRadiationPressure_ = radiationPressureFunction_( );
281  currentArea_ = areaFunction_( );
282  currentConeAngle_ = coneAngleFunction_( );
283  currentClockAngle_ = clockAngleFunction_( );
284  currentMass_ = massFunction_( );
285  currentDistanceToSource_ = ( sourcePositionFunction_( ) - acceleratedBodyPositionFunction_( ) ).norm();
286  currentVelocityWrtSource_ = ( acceleratedBodyVelocityFunction_( ) - centralBodyVelocityFunction_( ) ).norm( );
