26 Eigen::Matrix3d calculateAccelerationWrtStatePartials(
27  std::function< void( Eigen::Vector6d ) > setBodyState,
28  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > accelerationModel,
29  Eigen::Vector6d originalState,
30  Eigen::Vector3d statePerturbation,
31  int startIndex,
32  std::function< void( ) > updateFunction,
33  const double evaluationTime )
34 {
35  Eigen::Matrix3d upAccelerations = Eigen::Matrix3d::Zero( );
36  Eigen::Matrix3d downAccelerations = Eigen::Matrix3d::Zero( );
37 
38  Eigen::Vector6d perturbedState = originalState;
39 
40  accelerationModel->resetTime( TUDAT_NAN );
41 
42  // Calculate perturbed accelerations for up-perturbed state entries.
43  for( int i = 0; i < 3; i++ )
44  {
45  perturbedState( i + startIndex ) += statePerturbation( i );
46  setBodyState( perturbedState );
47  updateFunction( );
48  upAccelerations.block( 0, i, 3, 1 ) = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
49  accelerationModel, evaluationTime );
50  accelerationModel->resetTime( TUDAT_NAN );
51  perturbedState = originalState;
52  }
53 
54  // Calculate perturbed accelerations for down-perturbed state entries.
55  for( int i = 0; i < 3; i++ )
56  {
57  perturbedState( i + startIndex ) -= statePerturbation( i );
58  setBodyState( perturbedState );
59  updateFunction( );
60  downAccelerations.block( 0, i, 3, 1 ) = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
61  accelerationModel, evaluationTime );
62  accelerationModel->resetTime( TUDAT_NAN );
63  perturbedState = originalState;
64  }
65 
66 
67  // Reset state/environment to original state.
68  setBodyState( perturbedState );
69  updateFunction( );
70 
71  basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
72  accelerationModel, evaluationTime );
73 
74  // Numerically compute partial derivatives.
75  Eigen::Matrix3d accelerationPartials = upAccelerations - downAccelerations;
76  for( int i = 0; i < 3; i++ )
77  {
78  accelerationPartials.block( 0, i, 3, 1 ) /= ( 2.0 * statePerturbation( i ) );
79  }
80 
81  return accelerationPartials;

203 Eigen::MatrixXd calculateAccelerationDeviationDueToOrientationChange(
204  const std::function< void( Eigen::Vector7d ) > setBodyRotationalState,
205  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > accelerationModel,
206  const Eigen::Vector7d& originalRotationalState,
207  const Eigen::Vector4d& commandedQuaternionPerturbation,
208  std::vector< Eigen::Vector4d >& appliedQuaternionPerturbation,
209  std::function< void( ) > updateFunction,
210  const double evaluationTime )
211 {
212  appliedQuaternionPerturbation.resize( 4 );
213 
214  setBodyRotationalState( originalRotationalState );
215  Eigen::Vector3d nominalAcceleration = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
216  accelerationModel, evaluationTime );
217  Eigen::MatrixXd deviations = Eigen::MatrixXd::Zero( 3, 3 );
218 
219  Eigen::Vector7d perturbedState = originalRotationalState;
220 
221  accelerationModel->resetTime( TUDAT_NAN );
222 
223  // Calculate perturbed accelerations for up-perturbed state entries.
224  for( int i = 1; i < 4; i++ )
225  {
226  perturbedState( i ) += commandedQuaternionPerturbation( i );
227  perturbedState( 0 ) = ( originalRotationalState( 0 ) > 0 ? 1.0 : -1.0 ) * std::sqrt( 1.0 - std::pow( perturbedState.segment( 1, 3 ).norm( ), 2 ) );
228 
229  appliedQuaternionPerturbation[ i ] = perturbedState.segment( 0, 4 ).normalized( ) -
230  originalRotationalState.segment( 0, 4 );
231  setBodyRotationalState( perturbedState );
232  updateFunction( );
233  deviations.block( 0, i - 1, 3, 1 ) = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
234  accelerationModel, evaluationTime ) -
235  nominalAcceleration;
236  accelerationModel->resetTime( TUDAT_NAN );
237  perturbedState = originalRotationalState;
238 
239  }
240 
241  setBodyRotationalState( perturbedState );
242  updateFunction( );
243 
244  basic_astrodynamics::updateAndGetAcceleration( accelerationModel, evaluationTime );
245 
246  return deviations;
247 }

294 Eigen::Vector3d calculateAccelerationWrtParameterPartials(
295  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter,
296  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > accelerationModel,
297  double parameterPerturbation,
298  std::function< void( ) > updateDependentVariables,
299  const double currentTime,
300  std::function< void( const double ) > timeDependentUpdateDependentVariables )
301 {
302  // Store uperturbed value.
303  double unperturbedParameterValue = parameter->getParameterValue( );
304 
305  // Calculate up-perturbation
306  parameter->setParameterValue(
307  unperturbedParameterValue + parameterPerturbation );
308  updateDependentVariables( );
309  timeDependentUpdateDependentVariables( currentTime );
310  accelerationModel->resetTime( TUDAT_NAN );
311 
312  Eigen::Vector3d upPerturbedAcceleration = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
313  accelerationModel, currentTime );
314  accelerationModel->resetTime( TUDAT_NAN );
315  // Calculate down-perturbation.
316  parameter->setParameterValue(
317  unperturbedParameterValue - parameterPerturbation );
318  updateDependentVariables( );
319  timeDependentUpdateDependentVariables( currentTime );
320 
321  Eigen::Vector3d downPerturbedAcceleration = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
322  accelerationModel, currentTime );
323 
324  accelerationModel->resetTime( TUDAT_NAN );
325 
326  // Reset to original value.
327  parameter->setParameterValue(
328  unperturbedParameterValue ) ;
329  updateDependentVariables( );
330  timeDependentUpdateDependentVariables( currentTime );
331 
332  accelerationModel->resetTime( TUDAT_NAN );
333  basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
334  accelerationModel, currentTime );
335 
336  // Calculate partial using central difference.
337  return ( upPerturbedAcceleration - downPerturbedAcceleration ) / ( 2.0 * parameterPerturbation );
338 
339 }
