84 AvailableAcceleration getAccelerationModelType(
85  const std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > >
86  accelerationModel )
87 {
88  using namespace tudat::aerodynamics;
89  using namespace tudat::electro_magnetism;
90  using namespace tudat::gravitation;
91 
92  // Nominal type is undefined
93  AvailableAcceleration accelerationType = undefined_acceleration;
94 
95  // Check for each accelerarion mdoel type implemented as AvailableAcceleration.
96  if( std::dynamic_pointer_cast< CentralGravitationalAccelerationModel3d >(
97  accelerationModel ) != nullptr )
98  {
99  accelerationType = central_gravity;
100  }
101  else if( std::dynamic_pointer_cast< CannonBallRadiationPressureAcceleration >(
102  accelerationModel ) != nullptr )
103  {
104  accelerationType = cannon_ball_radiation_pressure;
105  }
106  else if( std::dynamic_pointer_cast< ThirdBodyCentralGravityAcceleration >(
107  accelerationModel ) != nullptr )
108  {
109  accelerationType = third_body_central_gravity;
110  }
111  else if( std::dynamic_pointer_cast< ThirdBodySphericalHarmonicsGravitationalAccelerationModel >(
112  accelerationModel ) != nullptr )
113  {
114  accelerationType = third_body_spherical_harmonic_gravity;
115  }
116  else if( std::dynamic_pointer_cast< ThirdBodyMutualSphericalHarmonicsGravitationalAccelerationModel >(
117  accelerationModel ) != nullptr )
118  {
119  accelerationType = third_body_mutual_spherical_harmonic_gravity;
120  }
121  else if( std::dynamic_pointer_cast< SphericalHarmonicsGravitationalAccelerationModel >(
122  accelerationModel ) != nullptr )
123  {
124  accelerationType = spherical_harmonic_gravity;
125  }
126  else if( std::dynamic_pointer_cast< MutualSphericalHarmonicsGravitationalAccelerationModel >( accelerationModel ) != nullptr )
127  {
128  accelerationType = mutual_spherical_harmonic_gravity;
129  }
130  else if( std::dynamic_pointer_cast< AerodynamicAcceleration >(
131  accelerationModel ) != nullptr )
132  {
133  accelerationType = aerodynamic;
134  }
135  else if( std::dynamic_pointer_cast< propulsion::ThrustAcceleration >(
136  accelerationModel ) != nullptr )
137  {
138  accelerationType = thrust_acceleration;
139  }
140  else if( std::dynamic_pointer_cast< propulsion::MomentumWheelDesaturationThrustAcceleration >(
141  accelerationModel ) != nullptr )
142  {
143  accelerationType = momentum_wheel_desaturation_acceleration;
144  }
145  else if( std::dynamic_pointer_cast< relativity::RelativisticAccelerationCorrection >(
146  accelerationModel ) != nullptr )
147  {
148  accelerationType = relativistic_correction_acceleration;
149  }
150  else if( std::dynamic_pointer_cast< basic_astrodynamics::EmpiricalAcceleration >( accelerationModel ) != nullptr )
151  {
152  accelerationType = empirical_acceleration;
153  }
154  else if( std::dynamic_pointer_cast< gravitation::DirectTidalDissipationAcceleration >( accelerationModel ) != nullptr )
155  {
156  std::shared_ptr< gravitation::DirectTidalDissipationAcceleration > dissipationAcceleration =
157  std::dynamic_pointer_cast< gravitation::DirectTidalDissipationAcceleration >( accelerationModel );
158  if( dissipationAcceleration->getModelTideOnPlanet( ) )
159  {
160  accelerationType = direct_tidal_dissipation_in_central_body_acceleration;
161  }
162  else
163  {
164  accelerationType = direct_tidal_dissipation_in_orbiting_body_acceleration;
165  }
166  }
167  else if( std::dynamic_pointer_cast< PanelledRadiationPressureAcceleration >( accelerationModel ) != NULL )
168  {
169  accelerationType = panelled_radiation_pressure_acceleration;
170  }
171  else if ( std::dynamic_pointer_cast< SolarSailAcceleration >( accelerationModel) != NULL )
172  {
173  accelerationType = solar_sail_acceleration;
174  }
175  else
176  {
177  throw std::runtime_error(
178  "Error, acceleration model not identified when getting acceleration type." );
179  }
180 
181  // Return identified type.
182  return accelerationType;
183 
184 }

212 std::vector< std::shared_ptr< AccelerationModel3d > > getAccelerationModelsOfType(
213  const std::vector< std::shared_ptr< AccelerationModel3d > >& fullList,
214  const AvailableAcceleration modelType )
215 {
216  std::vector< std::shared_ptr< AccelerationModel3d > > accelerationList;
217  for( unsigned int i = 0; i < fullList.size( ); i++ )
218  {
219  if( getAccelerationModelType( fullList.at( i ) ) == modelType )
220  {
221  accelerationList.push_back( fullList.at( i ) );
222  }
223  }
224  return accelerationList;
225 }

257 AvailableAcceleration getAssociatedThirdBodyAcceleration( const AvailableAcceleration accelerationType )
258 {
259  AvailableAcceleration thirdBodyAccelerationType;
260  if( !isAccelerationDirectGravitational( accelerationType ) )
261  {
262  std::string errorMessage = "Error when getting third-body gravity type, requested type: " +
263  std::to_string( accelerationType ) + " is not a direct gravity acceleration";
264  throw std::runtime_error( errorMessage );
265  }
266  else if( accelerationType == central_gravity )
267  {
268  thirdBodyAccelerationType = third_body_central_gravity;
269  }
270  else if( accelerationType == spherical_harmonic_gravity )
271  {
272  thirdBodyAccelerationType = third_body_spherical_harmonic_gravity;
273  }
274  else if( accelerationType == mutual_spherical_harmonic_gravity )
275  {
276  thirdBodyAccelerationType = third_body_mutual_spherical_harmonic_gravity;
277  }
278  else
279  { std::string errorMessage = "Error when getting thirdbody gravity type, requested type: " +
280  std::to_string( accelerationType ) + " is not recognized.";
281  throw std::runtime_error( errorMessage );
282 
283  }
284  return thirdBodyAccelerationType;
