23 Eigen::Vector3d computeDirectTidalAccelerationDueToTideOnPlanet(
24  const Eigen::Vector6d relativeStateOfBodyExertingTide, const Eigen::Vector3d planetAngularVelocityVector,
25  const double currentTidalAccelerationMultiplier, const double timeLag,
26  const bool includeDirectRadialComponent )
27 {
28  Eigen::Vector3d relativePosition = relativeStateOfBodyExertingTide.segment( 0, 3 );
29  Eigen::Vector3d relativeVelocity = relativeStateOfBodyExertingTide.segment( 3, 3 );
30 
31  double distance = relativePosition.norm( );
32  double distanceSquared = distance * distance;
33  double radialComponentMultiplier = ( includeDirectRadialComponent == true ) ? 1.0 : 0.0;
34 
35  return currentTidalAccelerationMultiplier * (
36  radialComponentMultiplier * relativePosition + timeLag * (
37  2.0 * ( relativePosition.dot( relativeVelocity ) * relativePosition / distanceSquared ) +
38  ( relativePosition.cross( planetAngularVelocityVector ) + relativeVelocity ) ) );
39 
40 }

49  Eigen::Vector3d relativeVelocity = relativeStateOfBodyExertingTide.segment( 3, 3 );
50 
51  double distance = relativePosition.norm( );
52  double distanceSquared = distance * distance;
53  double radialComponentMultiplier = ( includeDirectRadialComponent == true ) ? 1.0 : 0.0;
54 
55  return currentTidalAccelerationMultiplier * (
56  2.0 * radialComponentMultiplier * relativePosition + timeLag * (
