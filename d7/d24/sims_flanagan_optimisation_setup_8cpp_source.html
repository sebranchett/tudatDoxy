62 std::pair< std::vector< double >, std::vector< double > > SimsFlanaganProblem::get_bounds() const {
63 
64  // Define lower bounds.
65  std::vector< double > lowerBounds;
66  std::vector< double > upperBounds;
67 
68  if ( initialGuessThrottles_.size( ) != 0 )
69  {
70  if ( initialGuessThrottles_.size( ) != static_cast< unsigned int >( 3 * numberSegments_ ) )
71  {
72  throw std::runtime_error( "Error when providing an initial guess for Sims-Flanagan, vector size unconsistent with"
73  " number of segments into which the trajectory is subdivided." );
74  }
75  else
76  {
77  for ( int i = 0 ; i < numberSegments_ ; i++ )
78  {
79  for ( int k = 0 ; k < 3 ; k++ )
80  {
81  double lowerBoundFromInitialGuess;
82  double upperBoundFromInitialGuess;
83  if ( initialGuessThrottles_[ i * 3 + k ] >= 0 )
84  {
85  lowerBoundFromInitialGuess = ( 1.0 - relativeMarginWrtInitialGuess_ ) * initialGuessThrottles_[ i * 3 + k ];
86  upperBoundFromInitialGuess = ( 1.0 + relativeMarginWrtInitialGuess_ ) * initialGuessThrottles_[ i * 3 + k ];
87  }
88  else
89  {
90  lowerBoundFromInitialGuess = ( 1.0 + relativeMarginWrtInitialGuess_ ) * initialGuessThrottles_[ i * 3 + k ];
91  upperBoundFromInitialGuess = ( 1.0 - relativeMarginWrtInitialGuess_ ) * initialGuessThrottles_[ i * 3 + k ];
92  }
93 
94  if ( std::fabs( lowerBoundFromInitialGuess <= 1.0 ) )
95  {
96  lowerBounds.push_back( lowerBoundFromInitialGuess );
97  }
98  else
99  {
100  lowerBounds.push_back( - 1.0 );
101  }
102 
103  if ( std::fabs( upperBoundFromInitialGuess <= 1.0 ) )
104  {
105  upperBounds.push_back( upperBoundFromInitialGuess );
106  }
107  else
108  {
109  upperBounds.push_back( 1.0 );
110  }
111  }
112  }
113  }
114  }
115  else
116  {
117  for ( int i = 0 ; i < 3 * numberSegments_ ; i++ )
118  {
119  lowerBounds.push_back( - 1.0 );
120  upperBounds.push_back( 1.0 );
121  }
122  }
123 
124  return { lowerBounds, upperBounds };
125 }

128 std::vector< double > SimsFlanaganProblem::fitness( const std::vector< double > &designVariables ) const
129 {
130  // Transform vector of design variables into 3D vector of throttles.
131  std::vector< Eigen::Vector3d > throttles;
132 
133  // Check consistency of the size of the design variables vector.
134  if ( designVariables.size( ) != static_cast< unsigned int >( 3 * numberSegments_ ) )
135  {
136  throw std::runtime_error( "Error, size of the design variables vector unconsistent with number of segments." );
137  }
138 
139  for ( int i = 0 ; i < numberSegments_ ; i++ )
140  {
141  throttles.push_back( ( Eigen::Vector3d( ) << designVariables[ i * 3 ],
142  designVariables[ i * 3 + 1 ], designVariables[ i * 3 + 2 ] ).finished( ) );
143  }
144 
145  std::vector< double > fitness;
146 
147  // Create Sims Flanagan trajectory leg.
148  low_thrust_trajectories::SimsFlanaganModel currentLeg = low_thrust_trajectories::SimsFlanaganModel(
149  stateAtDeparture_, stateAtArrival_, centralBodyGravitationalParameter_, initialSpacecraftMass_,
150  maximumThrust_, specificImpulseFunction_, timeOfFlight_, throttles );
151 
152  // Forward propagation from departure to match point.
153  currentLeg.propagateForwardFromDepartureToMatchPoint();
154 
155  // Backward propagation from arrival to match point.
156  currentLeg.propagateBackwardFromArrivalToMatchPoint();
157 
158  // Fitness
159  double deltaV = currentLeg.getTotalDeltaV( );
160 
161  // Equality constraints (must be ... = 0 )
162  std::vector< double > equalityConstraints;
163 
164  // Spacecraft position and velocity must be continuous at the match point.
165  for ( int i = 0 ; i < 3 ; i++ )
166  {
167  equalityConstraints.push_back( std::fabs( currentLeg.getStateAtMatchPointForwardPropagation( )[ i ]
168  - currentLeg.getStateAtMatchPointBackwardPropagation( )[ i ] ) /
169  physical_constants::ASTRONOMICAL_UNIT );
170  equalityConstraints.push_back( std::fabs( currentLeg.getStateAtMatchPointForwardPropagation( )[ i + 3 ]
171  - currentLeg.getStateAtMatchPointBackwardPropagation( )[ i + 3 ] ) /
172  stateAtDeparture_.segment(3 ,3 ).norm() );
173  }
174 
175 
176  // Inequality constraints.
177  std::vector< double > inequalityConstraints;
178 
179  // Magnitude of the normalised thrust vector must be inferior or equal to 1 )
180  for ( unsigned int currentThrottle = 0 ; currentThrottle < throttles.size() ; currentThrottle++ )
181  {
182  if ( throttles[ currentThrottle ].norm( ) > 1.0 )
183  {
184  inequalityConstraints.push_back( throttles[ currentThrottle ][ 0 ] * throttles[ currentThrottle ][ 0 ]
185  + throttles[ currentThrottle ][ 1 ] * throttles[ currentThrottle ][ 1 ]
186  + throttles[ currentThrottle ][ 2 ] * throttles[ currentThrottle ][ 2 ] - 1.0 );
187  }
188  else
189  {
190  inequalityConstraints.push_back( 0.0 );
191  }
192  }
193 
194  // Compute auxiliary variables.
195  Eigen::VectorXd c; c.resize( equalityConstraints.size( ) + inequalityConstraints.size( ) );
196  Eigen::VectorXd r; r.resize( equalityConstraints.size( ) + inequalityConstraints.size( ) );
197  Eigen::VectorXd epsilon; epsilon.resize( equalityConstraints.size( ) + inequalityConstraints.size( ) );
198  for ( unsigned int i = 0 ; i < equalityConstraints.size( ) ; i++ )
199  {
200  c[ i ] = 1.0 / relativeToleranceConstraints_;
201  r[ i ] = 1.0 - relativeToleranceConstraints_ * c[ i ];
202  epsilon[ i ] = equalityConstraints[ i ] * c[ i ] + r[ i ];
203  }
204  for ( unsigned int i = 0 ; i < inequalityConstraints.size( ) ; i++ )
205  {
206  c[ equalityConstraints.size( ) + i ] = 1.0 / relativeToleranceConstraints_;
207  r[ equalityConstraints.size( ) + i ] = 1.0 - relativeToleranceConstraints_ * c[ equalityConstraints.size( ) + i ];
208  epsilon[ equalityConstraints.size( ) + i ] = inequalityConstraints[ i ] * c[ equalityConstraints.size( ) + i ] + r[ equalityConstraints.size( ) + i ];
209  }
210 
211  double weightDeltaV = 1.0;
212  double weightConstraints = 10.0;
213  double optimisationObjective = weightDeltaV * deltaV + weightConstraints * ( epsilon.norm( ) * epsilon.norm( ) );
214 
215 
216  // Optimisation objectives
217  fitness.push_back( optimisationObjective );
218 
219 
220  return fitness;
221 }
