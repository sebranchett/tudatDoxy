28 std::string printInFormattedScientificNotation( const double floatingPointNumber,
29  const int basePrecision,
30  const int exponentWidth )
31 {
32  // Declare string representation of the floating-point number.
33  std::string floatingPointNumberString;
34 
35  // Extract the decimal part (base) of the floating-point number with specified precision.
36  // Default precision is numeric_limits< double >::digits10.
37  std::stringstream buffer;
38  buffer << std::scientific
39  << std::setprecision( basePrecision )
40  << std::uppercase
41  << floatingPointNumber
42  << std::endl;
43 
44  // Write base to string.
45  buffer >> floatingPointNumberString;
46 
47  // Determine the location where the exponent part of the number starts.
48  std::string::size_type exponentLocation = floatingPointNumberString.find( 'E' );
49 
50  // Set the exponent to zero.
51  int exponent = 0;
52 
53  // Set flag indicating if exponent is positive or negative to positive.
54  bool isExponentPositive = true;
55 
56  // Check if the exponent is present.
57  if ( exponentLocation != std::string::npos )
58  {
59  // Extract the exponent to the buffer.
60  buffer << floatingPointNumberString.substr( exponentLocation + 1,
61  floatingPointNumberString.length( )
62  - exponentLocation - 1 )
63  << std::endl;
64 
65  // Write the exponent part from the buffer.
66  buffer >> exponent;
67 
68  // Check if the exponent is negative.
69  if ( exponent < 0 )
70  {
71  // If it is negative, set the flag to false, and switch signs of the exponent.
72  isExponentPositive = false;
73  exponent = -exponent;
74  }
75  }
76 
77  // Extract the exponent with a fixed field width.
78  // Default width is 2.
79  std::stringstream outputStream;
80  outputStream << floatingPointNumberString.substr( 0, exponentLocation + 1 )
81  << ( isExponentPositive ? '+' : '-' )
82  << std::setw( exponentWidth )
83  << std::setfill( '0' )
84  << exponent;
85 
86  // Declare output string and write buffer output to the string.
87  std::string outputString;
88  outputStream >> outputString;
89 
90  // Return the formatted string representation of the floating-point number.
91  return outputString;
92 }
