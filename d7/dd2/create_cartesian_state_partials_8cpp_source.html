51 std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > > createCartesianStatePartialsWrtBodyRotationalState(
52  const observation_models::LinkEnds& linkEnds,
53  const simulation_setup::NamedBodyMap& bodyMap,
54  const std::string& bodyToEstimate )
55 {
56  // Declare data map to return.
57  std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > > partialMap;
58 
59  // Declare local variable to use in loop
60  std::string currentBodyName;
61 
62  // Iterate over all like ends.
63  for( observation_models::LinkEnds::const_iterator linkEndIterator = linkEnds.begin( );
64  linkEndIterator != linkEnds.end( ); linkEndIterator++ )
65  {
66  // Check if current link end is on body that is requested.
67  currentBodyName = linkEndIterator->second.first;
68  if( bodyToEstimate == currentBodyName )
69  {
70  std::shared_ptr< simulation_setup::Body > currentBody = bodyMap.at( currentBodyName );
71 
72  if( currentBody->getGroundStationMap( ).count( linkEndIterator->second.second ) == 0 )

77  }
78 
79  // Set ground station position function
80  std::function< Eigen::Vector3d( const double ) > groundStationPositionFunction =
81  std::bind( &ground_stations::GroundStationState::getCartesianPositionInTime,

83  std::placeholders::_1, basic_astrodynamics::JULIAN_DAY_ON_J2000 );
84 
85  // Create partial
86  partialMap[ linkEndIterator->first ] = std::make_shared< CartesianStatePartialWrtRotationMatrixParameter >(
87  std::make_shared< RotationMatrixPartialWrtRotationalState >(
88  std::bind( &ephemerides::RotationalEphemeris::getRotationToBaseFrame,
89  currentBody->getRotationalEphemeris( ), std::placeholders::_1 ) ), groundStationPositionFunction );
90  }
91  }
92 
93  return partialMap;
94 }

97 std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > > createCartesianStatePartialsWrtParameter(
98  const observation_models::LinkEnds linkEnds,
99  const simulation_setup::NamedBodyMap& bodyMap,
100  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameterToEstimate )
101 {
102  using namespace ephemerides;
103 
104  // Declare data map to return.
105  std::map< observation_models::LinkEndType, std::shared_ptr< CartesianStatePartial > > partialMap;
106 
107  // Declare local variable to use in loop
108  std::shared_ptr< simulation_setup::Body > currentBody;
109  std::string currentBodyName;
110 
111  // Iterate over all like ends.
112  for( observation_models::LinkEnds::const_iterator linkEndIterator = linkEnds.begin( );
113  linkEndIterator != linkEnds.end( ); linkEndIterator++ )
114  {
115  // Check if current link end body corresponds to body with property to estimate.
116  if( ( linkEndIterator->second.first == parameterToEstimate->getParameterName( ).second.first ) &&
117  ( linkEndIterator->second.second != "" ) )
118  {
119  // Set current body name and object.
120  currentBodyName = linkEndIterator->second.first;
121  currentBody = bodyMap.at( currentBodyName );
122 
123  // Check if parameter is a rotation model property, in which case a CartesianStatePartialWrtRotationMatrixParameter,
124  // with the rotation matrix partial created from createRotationMatrixPartialsWrtParameter function.
125  if( estimatable_parameters::isParameterRotationMatrixProperty( parameterToEstimate->getParameterName( ).first ) )
126  {
127  // Set ground station position function
128  std::function< Eigen::Vector3d( const double ) > groundStationPositionFunction =
129  std::bind( &ground_stations::GroundStationState::getCartesianPositionInTime,

131  ->getNominalStationState( ), std::placeholders::_1, basic_astrodynamics::JULIAN_DAY_ON_J2000 );
132 
133  // Create parameter partial object.
134  partialMap[ linkEndIterator->first ] = std::make_shared< CartesianStatePartialWrtRotationMatrixParameter >(
135  createRotationMatrixPartialsWrtParameter( bodyMap, parameterToEstimate ),
136  groundStationPositionFunction );
137  }
138  else
139  {
140  // Check which parameter is requested and create position partial if direct dependency between position and
141  // parameter exists.
142  switch( parameterToEstimate->getParameterName( ).first )
143  {
144  case estimatable_parameters::gravitational_parameter:
145  break;
146  case estimatable_parameters::constant_drag_coefficient:
147  break;
148  case estimatable_parameters::radiation_pressure_coefficient:
149  break;
150  case estimatable_parameters::ppn_parameter_gamma:
151  break;
152  case estimatable_parameters::ppn_parameter_beta:
153  break;
154  case estimatable_parameters::equivalence_principle_lpi_violation_parameter:
155  break;
156  case estimatable_parameters::mean_moment_of_inertia:
157  break;
158  case estimatable_parameters::direct_dissipation_tidal_time_lag:
159  break;
160  default:
161 
162  std::string errorMessage =
163  "Parameter " + std::to_string(
164  parameterToEstimate->getParameterName( ).first ) +
165  " not implemented when making position partial";
166  throw std::runtime_error( errorMessage );
167  break;
168  }
169  }
170  }
171  }
172 
173  return partialMap;

308 std::shared_ptr< RotationMatrixPartial > createRotationMatrixPartialsWrtParameter(
309  const simulation_setup::NamedBodyMap& bodyMap,
310  const std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameterToEstimate )
311 {
312  using namespace simulation_setup;
313  using namespace ephemerides;
314 
315  // Declare return object.
316  std::shared_ptr< RotationMatrixPartial > rotationMatrixPartial;
317 
318  // Get body for rotation of which partial is to be created.
319  std::shared_ptr< Body > currentBody = bodyMap.at( parameterToEstimate->getParameterName( ).second.first );
320 
321  // Check for which rotation model parameter the partial object is to be created.
322  switch( parameterToEstimate->getParameterName( ).first )
323  {
324  case estimatable_parameters::constant_rotation_rate:
325 
326  if( std::dynamic_pointer_cast< ephemerides::SimpleRotationalEphemeris >(
327  currentBody->getRotationalEphemeris( ) ) == nullptr )
328  {
329  throw std::runtime_error( "Warning, body's rotation model is not simple when making position w.r.t. constant rtoation rate partial" ) ;
330  }
331 
332  // Create rotation matrix partial object
333  rotationMatrixPartial = std::make_shared< RotationMatrixPartialWrtConstantRotationRate >(
334  std::dynamic_pointer_cast< SimpleRotationalEphemeris>( currentBody->getRotationalEphemeris( ) ) );
335  break;
336 
337  case estimatable_parameters::core_factor:
338 
339  if( std::dynamic_pointer_cast< ephemerides::PlanetaryRotationModel >(
340  currentBody->getRotationalEphemeris() ) == nullptr )
341  {
342  std::string errorMessage = "Warning, body's rotation model is not a full planetary rotational model when making"
343  "position w.r.t. core factor partial";
344  throw std::runtime_error( errorMessage );
345  }
346 
347  // Create rotation matrix partial object
348  rotationMatrixPartial = std::make_shared< RotationMatrixPartialWrtCoreFactor >(
349  std::dynamic_pointer_cast< PlanetaryRotationModel >( currentBody->getRotationalEphemeris( ) ) );
350 
351  break;
352 
353  case estimatable_parameters::free_core_nutation_rate:
354 
355  if( std::dynamic_pointer_cast< ephemerides::PlanetaryRotationModel >(
356  currentBody->getRotationalEphemeris() ) == nullptr ){
357  std::string errorMessage = "Warning, body's rotation model is not a full planetary rotational model when making"
358  "position w.r.t. polar motion amplitude partial";
359  throw std::runtime_error( errorMessage );
360  }
361 
362  // Create rotation matrix partial object
363  rotationMatrixPartial = std::make_shared< RotationMatrixPartialWrtFreeCoreNutationRate >(
364  std::dynamic_pointer_cast< PlanetaryRotationModel >( currentBody->getRotationalEphemeris() ));
365 
366  break;
367  default:
368  std::string errorMessage = "Warning, rotation matrix partial not implemented for parameter " +
369  std::to_string( parameterToEstimate->getParameterName( ).first );
370  throw std::runtime_error( errorMessage );
371 
372  break;
373  }
374 
375  return rotationMatrixPartial;
376 
377 
378 }
