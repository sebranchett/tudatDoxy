26 double computeJacobiEnergy( const double massParameter,
27  const Eigen::VectorXd& stateInNormalizedUnits )
28 {
29  using namespace orbital_element_conversions;
30 
31  // Compute squared distances for efficient computation.
32  const double xCoordinateToPrimaryBodySquared =
33  ( massParameter + stateInNormalizedUnits( xCartesianPositionIndex ) )
34  * ( massParameter + stateInNormalizedUnits( xCartesianPositionIndex ) );
35 
36  const double yCoordinateSquared = stateInNormalizedUnits( yCartesianPositionIndex )
37  * stateInNormalizedUnits( yCartesianPositionIndex );
38 
39  const double zCoordinateSquared = stateInNormalizedUnits( zCartesianPositionIndex )
40  * stateInNormalizedUnits( zCartesianPositionIndex );
41 
42  const double xCoordinateToSecondaryBodySquared =
43  ( 1.0 - massParameter - stateInNormalizedUnits( xCartesianPositionIndex ) )
44  * ( 1.0 - massParameter - stateInNormalizedUnits( xCartesianPositionIndex ) );
45 
46  const double distanceToPrimaryBody = std::sqrt(
47  xCoordinateToPrimaryBodySquared + yCoordinateSquared + zCoordinateSquared );
48 
49  const double distanceToSecondaryBody = std::sqrt(
50  xCoordinateToSecondaryBodySquared + yCoordinateSquared + zCoordinateSquared );
51 
52  return stateInNormalizedUnits( xCartesianPositionIndex ) * stateInNormalizedUnits( xCartesianPositionIndex )
53  + yCoordinateSquared
54  + 2.0 * ( 1.0 - massParameter ) / distanceToPrimaryBody
55  + 2.0 * massParameter / distanceToSecondaryBody
56  - stateInNormalizedUnits.segment( xCartesianVelocityIndex, 3 ).squaredNorm( );
57 }
