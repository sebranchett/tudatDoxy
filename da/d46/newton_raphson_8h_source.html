100  {
101  // Set the root function.
102  this->rootFunction = rootFunction;
103 
104  // Start at initial guess, and compute the function value and its first derivative.
105  DataType currentRootValue = TUDAT_NAN;
106  DataType nextRootValue = initialGuess;
107  DataType currentFunctionValue = TUDAT_NAN;
108  DataType nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
109  DataType currentDerivativeValue = TUDAT_NAN;
110  DataType nextDerivativeValue = this->rootFunction->
111  computeDerivative( 1, nextRootValue );
112 
113  // Loop counter.
114  unsigned int counter = 1;
115 
116  // Loop until we have a solution with sufficient accuracy.
117  do
118  {
119  // Save the old values.
120  currentRootValue = nextRootValue;
121  currentFunctionValue = nextFunctionValue;
122  currentDerivativeValue = nextDerivativeValue;
123 
124  // Compute next value of root using the following algorithm (see class documentation):
125  nextRootValue = currentRootValue -
126  currentFunctionValue / currentDerivativeValue;
127  nextFunctionValue = this->rootFunction->evaluate( nextRootValue );
128  nextDerivativeValue = this->rootFunction->computeDerivative( 1, nextRootValue );
129 
130  // Update the counter.
131  counter++;
132  }
133  while( !this->terminationFunction( nextRootValue, currentRootValue, nextFunctionValue,
134  currentFunctionValue, counter ) );
135 
136  return nextRootValue;
137  }
