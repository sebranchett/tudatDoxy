107 std::pair< double, double > readGravityFieldFile(
108  const std::string& fileName, const int maximumDegree, const int maximumOrder,
109  std::pair< Eigen::MatrixXd, Eigen::MatrixXd >& coefficients,
110  const int gravitationalParameterIndex, const int referenceRadiusIndex )
111 {
112  // Attempt to open gravity file.
113  std::fstream stream( fileName.c_str( ), std::ios::in );
114  if( stream.fail( ) )
115  {
116  throw std::runtime_error( "Pds gravity field data file could not be opened: " + fileName );
117  }
118 
119  // Declare variables for reading file.
120  std::vector< std::string > vectorOfIndividualStrings;
121  vectorOfIndividualStrings.resize( 4 );
122  std::string line;
123 
124 
125  double gravitationalParameter = TUDAT_NAN;
126  double referenceRadius = TUDAT_NAN;
127 
128  if( ( gravitationalParameterIndex >= 0 ) &&
129  ( referenceRadiusIndex >= 0 ) )
130  {
131  // Get first line of file.
132  std::getline( stream, line );
133 
134  // Get reference radius and gravitational parameter from first line of file.
135  boost::algorithm::trim( line );
136  boost::algorithm::split( vectorOfIndividualStrings,
137  line,
138  boost::algorithm::is_any_of( "\t, " ),
139  boost::algorithm::token_compress_on );
140  if( gravitationalParameterIndex >= static_cast< int >( vectorOfIndividualStrings.size( ) ) ||
141  referenceRadiusIndex >= static_cast< int >( vectorOfIndividualStrings.size( ) ) )
142  {
143  throw std::runtime_error( "Error when reading gravity field file, requested header index exceeds file contents" );
144  }
145 
146  gravitationalParameter = std::stod( vectorOfIndividualStrings[ gravitationalParameterIndex ] );
147  referenceRadius = std::stod( vectorOfIndividualStrings[ referenceRadiusIndex ] );
148  }
149  else if( ( !( gravitationalParameterIndex >= 0 ) &&
150  ( referenceRadiusIndex >= 0 ) ) ||
151  ( ( gravitationalParameterIndex >= 0 ) &&
152  !( referenceRadiusIndex >= 0 ) ) )
153  {
154  throw std::runtime_error( "Error when reading gravity field file, must retrieve either both or neither of Re and mu" );
155  }
156 
157 
158  // Declare variables for reading in cosine and sine coefficients.
159  int currentDegree = 0, currentOrder = 0;
160  Eigen::MatrixXd cosineCoefficients = Eigen::MatrixXd( maximumDegree + 1, maximumOrder + 1 );
161  cosineCoefficients.setZero( );
162  Eigen::MatrixXd sineCoefficients = Eigen::MatrixXd( maximumDegree + 1, maximumOrder + 1 );
163  sineCoefficients.setZero( );
164 
165  // Read coefficients up to required maximum degree and order.
166  while ( !stream.fail( ) && !stream.eof( ) &&
167  ( currentDegree <= maximumDegree && currentOrder < maximumOrder ) )
168  {
169  // Read current line
170  std::getline( stream, line );
171 
172  // Trim input string (removes all leading and trailing whitespaces).
173  boost::algorithm::trim( line );
174 
175  // Split string into multiple strings, each containing one element from a line from the
176  // data file.
177  boost::algorithm::split( vectorOfIndividualStrings,
178  line,
179  boost::algorithm::is_any_of( ", " ),
180  boost::algorithm::token_compress_on );
181 
182  // Check current line for consistency
183  if( vectorOfIndividualStrings.size( ) != 0 )
184  {
185  if( vectorOfIndividualStrings.size( ) < 4 )
186  {
187  std::string errorMessage = "Error when reading pds gravity field file, number of fields is " +
188  std::to_string( vectorOfIndividualStrings.size( ) );
189  throw std::runtime_error( errorMessage );
190  }
191  else
192  {
193  // Read current degree and orde from line.
194  currentDegree = std::stoi( vectorOfIndividualStrings[ 0 ] );
195  currentOrder = std::stoi( vectorOfIndividualStrings[ 1 ] );
196 
197  // Set cosine and sine coefficients for current degree and order.
198  if( currentDegree <= maximumDegree && currentOrder <= maximumOrder )
199  {
200  cosineCoefficients( currentDegree, currentOrder ) =
201  std::stod( vectorOfIndividualStrings[ 2 ] );
202  sineCoefficients( currentDegree, currentOrder ) =
203  std::stod( vectorOfIndividualStrings[ 3 ] );
204  }
205  }
206  }
207  }
208 
209  // Set cosine coefficient at (0,0) to 1.
210  cosineCoefficients( 0, 0 ) = 1.0;
211  coefficients = std::make_pair( cosineCoefficients, sineCoefficients );
212 
213  return std::make_pair( gravitationalParameter, referenceRadius );
214 }

217 std::shared_ptr< gravitation::GravityFieldModel > createGravityFieldModel(
218  const std::shared_ptr< GravityFieldSettings > gravityFieldSettings,
219  const std::string& body,
220  const NamedBodyMap& bodyMap,
221  const std::vector< std::shared_ptr< GravityFieldVariationSettings > >& gravityFieldVariationSettings )
222 {
223  using namespace tudat::gravitation;
224 
225  // Declare return object.
226  std::shared_ptr< GravityFieldModel > gravityFieldModel;
227 
228  // Check which type of gravity field model is to be created.
229  switch( gravityFieldSettings->getGravityFieldType( ) )
230  {
231  case central:
232  {
233  // Check whether settings for point mass gravity field model are consistent with its type.
234  std::shared_ptr< CentralGravityFieldSettings > centralFieldSettings =
235  std::dynamic_pointer_cast< CentralGravityFieldSettings >( gravityFieldSettings );
236  if( centralFieldSettings == nullptr )
237  {
238  throw std::runtime_error(
239  "Error, expected central field settings when making gravity field model for body " +
240  body);
241  }
242  else if( gravityFieldVariationSettings.size( ) != 0 )
243  {
244  throw std::runtime_error( "Error, requested central gravity field, but field variations settings are not empty." );
245  }
246  else
247  {
248  // Create and initialize point mass gravity field model.
249  gravityFieldModel = std::make_shared< GravityFieldModel >(
250  centralFieldSettings->getGravitationalParameter( ) );
251  }
252  break;
253  }
254 #if USE_CSPICE
255  case central_spice:
256  {
257  if( gravityFieldVariationSettings.size( ) != 0 )
258  {
259  throw std::runtime_error( "Error, requested central gravity field, but field variations settings are not empty." );
260  }
261  else
262  {
263  // Create and initialize point mass gravity field model from Spice.
264  gravityFieldModel = std::make_shared< GravityFieldModel >(
265  spice_interface::getBodyGravitationalParameter( body ) );
266  }
267 
268  break;
269  }
270 #endif
271  case spherical_harmonic:
272  {
273  // Check whether settings for spherical harmonic gravity field model are consistent with
274  // its type.
275  std::shared_ptr< SphericalHarmonicsGravityFieldSettings > sphericalHarmonicFieldSettings =
276  std::dynamic_pointer_cast< SphericalHarmonicsGravityFieldSettings >(
277  gravityFieldSettings );
278 
279  if( sphericalHarmonicFieldSettings == nullptr )
280  {
281  throw std::runtime_error(
282  "Error, expected spherical harmonic field settings when making gravity field model of "
283  + body );
284  }
285  else
286  {
287  std::function< void( ) > inertiaTensorUpdateFunction;
288  if( bodyMap.count( body ) == 0 )
289  {
290  inertiaTensorUpdateFunction = std::function< void( ) >( );
291  }
292  else
293  {
294  inertiaTensorUpdateFunction =
295  std::bind( &Body::setBodyInertiaTensorFromGravityFieldAndExistingMeanMoment, bodyMap.at( body ), true );
296  }
297 
298  // Check consistency of cosine and sine coefficients.
299  if( ( sphericalHarmonicFieldSettings->getCosineCoefficients( ).rows( ) !=
300  sphericalHarmonicFieldSettings->getSineCoefficients( ).rows( ) ) ||
301  ( sphericalHarmonicFieldSettings->getCosineCoefficients( ).cols( ) !=
302  sphericalHarmonicFieldSettings->getSineCoefficients( ).cols( ) ) )
303  {
304  throw std::runtime_error(
305  std::string( "Error when making spherical harmonic field, sine and " ) +
306  std::string( "cosine matrix sizes are not equal for body " ) + body );
307  }
308  else
309  {
310 
311  if( gravityFieldVariationSettings.size( ) == 0 &&
312  sphericalHarmonicFieldSettings->getCreateTimeDependentField( ) == 0 )
313  {
314  // Create and initialize spherical harmonic gravity field model.
315  gravityFieldModel = std::make_shared< SphericalHarmonicsGravityField >(
316  sphericalHarmonicFieldSettings->getGravitationalParameter( ),
317  sphericalHarmonicFieldSettings->getReferenceRadius( ),
318  sphericalHarmonicFieldSettings->getCosineCoefficients( ),
319  sphericalHarmonicFieldSettings->getSineCoefficients( ),
320  sphericalHarmonicFieldSettings->getAssociatedReferenceFrame( ),
321  inertiaTensorUpdateFunction );
322  }
323  else
324  {
325  if( bodyMap.at( body )->getGravityFieldModel( ) != nullptr )
326  {
327  std::string errorMessage = "Warning when making time-dependent gravity field model for body " + body +
328  " existing gravity field is not empty but overwritten in Body! ";
329  throw std::runtime_error( errorMessage );
330  }
331 
332  // Create preliminary TimeDependentSphericalHarmonicsGravityField, without actual variation settings.
333  gravityFieldModel = std::make_shared< TimeDependentSphericalHarmonicsGravityField >(
334  sphericalHarmonicFieldSettings->getGravitationalParameter( ),
335  sphericalHarmonicFieldSettings->getReferenceRadius( ),
336  sphericalHarmonicFieldSettings->getCosineCoefficients( ),
337  sphericalHarmonicFieldSettings->getSineCoefficients( ),
338  sphericalHarmonicFieldSettings->getAssociatedReferenceFrame( ),
339  inertiaTensorUpdateFunction );
340  }
341 
342 
343  }
344  }
345  break;
346  }
347  default:
348  throw std::runtime_error(
349  "Error, did not recognize gravity field model settings type " +
350  std::to_string(
351  gravityFieldSettings->getGravityFieldType( ) ) );
352  }
353 
354  return gravityFieldModel;
355 }

358 std::shared_ptr< SphericalHarmonicsGravityFieldSettings > createHomogeneousTriAxialEllipsoidGravitySettings(
359  const double axisA, const double axisB, const double axisC, const double ellipsoidDensity,
360  const int maximumDegree, const int maximumOrder,
361  const std::string& associatedReferenceFrame )
362 {
363  // Compute reference quantities
364  double ellipsoidGravitationalParameter = gravitation::calculateTriAxialEllipsoidVolume(
365  axisA, axisB, axisC ) * ellipsoidDensity * physical_constants::GRAVITATIONAL_CONSTANT;
366  double ellipsoidReferenceRadius = gravitation::calculateTriAxialEllipsoidVolume(
367  axisA, axisB, axisC );
368 
369  // Compute coefficients
370  std::pair< Eigen::MatrixXd, Eigen::MatrixXd > coefficients =
371  gravitation::createTriAxialEllipsoidNormalizedSphericalHarmonicCoefficients(
372  axisA, axisB, axisC, maximumDegree, maximumOrder );
373 
374  return std::make_shared< SphericalHarmonicsGravityFieldSettings >(
375  ellipsoidGravitationalParameter, ellipsoidReferenceRadius, coefficients.first,
376  coefficients.second, associatedReferenceFrame );
377 }
