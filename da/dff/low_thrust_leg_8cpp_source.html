20 std::shared_ptr< simulation_setup::ThrustAccelerationSettings > LowThrustLeg::getLowThrustAccelerationSettings(
21  const simulation_setup::NamedBodyMap& bodyMapTest,
22  const std::string& bodyToPropagate,
23  const std::function< double( const double ) > specificImpulseFunction,
24  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
25  const double timeOffset )
26 {
27 
28  std::shared_ptr< simulation_setup::Body > vehicle = bodyMapTest.at( bodyToPropagate );
29 
30  // Define thrust magnitude function from the shaped trajectory.
31  std::function< double( const double ) > thrustMagnitudeFunction = [ = ]( const double currentTime )
32  {
33  // Compute current independent variable.
34  double currentIndependentVariable = convertTimeToIndependentVariable( currentTime - timeOffset );
35 
36  if( legModelIsForceBased_ )
37  {
38  // Compute current acceleration.
39  return computeCurrentThrustForce(
40  currentIndependentVariable, specificImpulseFunction, integratorSettings ).norm( );
41  }
42  else
43  {
44  // Compute current acceleration.
45  double currentAcceleration = computeCurrentThrustAccelerationMagnitude(
46  currentIndependentVariable, specificImpulseFunction, integratorSettings );
47 
48  // Compute current mass of the vehicle.
49  double currentMass = vehicle->getBodyMass();
50 
51  // Compute and return magnitude of the low-thrust force.
52  return currentAcceleration * currentMass;
53  }
54  };
55 
56  // Define thrust magnitude settings from thrust magnitude function.
57  std::shared_ptr< simulation_setup::FromFunctionThrustMagnitudeSettings > thrustMagnitudeSettings
58  = std::make_shared< simulation_setup::FromFunctionThrustMagnitudeSettings >(
59  thrustMagnitudeFunction, specificImpulseFunction );
60 
61 
62  // Define thrust direction function from the shaped trajectory.
63  std::function< Eigen::Vector3d( const double ) > thrustDirectionFunction = [ = ]( const double currentTime )
64  {
65  // Compute current independent variable.
66  double currentIndependentVariable = convertTimeToIndependentVariable( currentTime - timeOffset );
67 
68  // Compute current direction of the acceleration vector.
69  Eigen::Vector3d currentAccelerationDirection = computeCurrentThrustAccelerationDirection(
70  currentIndependentVariable, specificImpulseFunction, integratorSettings );
71 
72  // Return direction of the low-thrust acceleration.
73  return currentAccelerationDirection;
74  };
75 
76  // Define thrust direction settings from the direction of thrust acceleration retrieved from the shaping method.
77  std::shared_ptr< simulation_setup::CustomThrustDirectionSettings > thrustDirectionSettings =
78  std::make_shared< simulation_setup::CustomThrustDirectionSettings >( thrustDirectionFunction );
79 
80  // Define thrust acceleration settings.
81  std::shared_ptr< simulation_setup::ThrustAccelerationSettings > thrustAccelerationSettings =
82  std::make_shared< simulation_setup::ThrustAccelerationSettings >(
83  thrustDirectionSettings, thrustMagnitudeSettings );
84 
85  return thrustAccelerationSettings;
86 }

119 
120

134  std::make_shared< propagators::MassPropagatorSettings< double > >(

161 void LowThrustLeg::getMassProfile(
162  const std::vector< double >& epochsVector,
163  std::map< double, Eigen::VectorXd >& massProfile,
164  const std::function< double ( const double ) > specificImpulseFunction,
165  const std::shared_ptr<numerical_integrators::IntegratorSettings< double > > integratorSettings )
166 {
167  massProfile.clear( );
168 
169  if( std::isnan( initialMass_ ) )
170  {
171  throw std::runtime_error( "Error when getting mass profile, initial mass is NaN" );
172  }
173  double currentMass = initialMass_;
174 
175  for ( unsigned int i = 0 ; i < epochsVector.size() ; i++ )
176  {
177  if ( ( i > 0 ) && ( epochsVector.at( i ) < epochsVector.at( i - 1 ) ) )
178  {
179  throw std::runtime_error( "Error when retrieving the mass profile of a low-thrust trajectory, "
180  "epochs are not provided in increasing order." );
181  }
182 
183  if ( i == 0 )
184  {
185  currentMass = computeCurrentMass( 0.0, epochsVector.at( i ), currentMass, specificImpulseFunction, integratorSettings );
186  massProfile[ epochsVector.at( i ) ] = ( Eigen::Vector1d( ) << currentMass ).finished( );
187  }
188  else
189  {
190  currentMass = computeCurrentMass( epochsVector.at( i - 1 ), epochsVector.at( i ), currentMass, specificImpulseFunction, integratorSettings );
191  massProfile[ epochsVector.at( i ) ] = ( Eigen::Vector1d( ) << currentMass ).finished();
192  }
193  }
194 
195 }

257 void LowThrustLeg::computeSemiAnalyticalAndFullPropagation(
258  const simulation_setup::NamedBodyMap& bodyMapTest,
259  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
260  const std::pair< std::shared_ptr< propagators::PropagatorSettings< double > >,
261  std::shared_ptr< propagators::PropagatorSettings< double > > >& propagatorSettings,
262  std::map< double, Eigen::VectorXd >& fullPropagationResults,
263  std::map< double, Eigen::Vector6d >& semiAnalyticalResults,
264  std::map< double, Eigen::VectorXd >& dependentVariablesHistory )
265 {
266 
267  fullPropagationResults.clear();
268  semiAnalyticalResults.clear();
269  dependentVariablesHistory.clear();
270 
271  // Define backward propagator settings variables.
272  integratorSettings->initialTimeStep_ = - std::fabs( integratorSettings->initialTimeStep_ );
273  integratorSettings->initialTime_ = timeOfFlight_ / 2.0;
274 
275  // Perform the backward propagation.
276  propagators::SingleArcDynamicsSimulator< > dynamicsSimulatorIntegrationBackwards( bodyMapTest, integratorSettings, propagatorSettings.first );
277  std::map< double, Eigen::VectorXd > stateHistoryFullProblemBackwardPropagation = dynamicsSimulatorIntegrationBackwards.getEquationsOfMotionNumericalSolution( );
278  std::map< double, Eigen::VectorXd > dependentVariableHistoryBackwardPropagation = dynamicsSimulatorIntegrationBackwards.getDependentVariableHistory( );
279 
280  // Declare vector of epochs at which the trajectory has to be calculated.
281  std::vector< double > epochsVector;
282 
283  // Compute and save full propagation and shaping method results along the backward propagation direction
284  for( std::map< double, Eigen::VectorXd >::iterator itr = stateHistoryFullProblemBackwardPropagation.begin( );
285  itr != stateHistoryFullProblemBackwardPropagation.end( ); itr++ )
286  {
287  epochsVector.push_back( itr->first );
288  fullPropagationResults[ itr->first ] = itr->second;
289  dependentVariablesHistory[ itr->first ] = dependentVariableHistoryBackwardPropagation[ itr->first ];
290  }
291 
292  // Define forward propagator settings variables.
293  integratorSettings->initialTimeStep_ = std::fabs( integratorSettings->initialTimeStep_ );
294  integratorSettings->initialTime_ = timeOfFlight_ / 2.0;
295 
296  // Perform forward propagation.
297  propagators::SingleArcDynamicsSimulator< > dynamicsSimulatorIntegrationForwards(
298  bodyMapTest, integratorSettings, propagatorSettings.second );
299  std::map< double, Eigen::VectorXd > stateHistoryFullProblemForwardPropagation =
300  dynamicsSimulatorIntegrationForwards.getEquationsOfMotionNumericalSolution( );
301  std::map< double, Eigen::VectorXd > dependentVariableHistoryForwardPropagation =
302  dynamicsSimulatorIntegrationForwards.getDependentVariableHistory( );
303 
304  // Compute and save full propagation and shaping method results along the forward propagation direction.
305  for( std::map< double, Eigen::VectorXd >::iterator itr = stateHistoryFullProblemForwardPropagation.begin( );
306  itr != stateHistoryFullProblemForwardPropagation.end( ); itr++ )
307  {
308  epochsVector.push_back( itr->first );
309  fullPropagationResults[ itr->first ] = itr->second;
310  dependentVariablesHistory[ itr->first ] = dependentVariableHistoryForwardPropagation[ itr->first ];
311  }
312 
313  getTrajectory( epochsVector, semiAnalyticalResults );
314 
315  // Reset initial integrator settings.
316  integratorSettings->initialTimeStep_ = std::fabs( integratorSettings->initialTimeStep_ );
317 
318 }

322 std::pair< std::shared_ptr< propagators::PropagatorSettings< double > >,
323 std::shared_ptr< propagators::PropagatorSettings< double > > > LowThrustLeg::createLowThrustPropagatorSettings(
324  const simulation_setup::NamedBodyMap& bodyMapTest,
325  const std::string& bodyToPropagate,
326  const std::string& centralBody,
327  const std::function< double( const double ) > specificImpulseFunction,
328  const basic_astrodynamics::AccelerationMap perturbingAccelerationsMap,
329  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > >& integratorSettings,
330  const std::shared_ptr< propagators::DependentVariableSaveSettings >& dependentVariablesToSave )
331 {
332 
333  // Define propagator settings.
334  std::pair< std::shared_ptr< propagators::PropagatorSettings< double > >, std::shared_ptr< propagators::PropagatorSettings< double > > >
335  propagatorSettings;
336 
337  // Create termination conditions settings.
338  std::pair< std::shared_ptr< propagators::PropagationTerminationSettings >,
339  std::shared_ptr< propagators::PropagationTerminationSettings > > terminationConditions;
340 
341  terminationConditions.first = std::make_shared< propagators::PropagationTimeTerminationSettings >( 0.0, true );
342  terminationConditions.second = std::make_shared< propagators::PropagationTimeTerminationSettings >( timeOfFlight_, true );
343 
344  // Re-initialise spacecraft mass in body map.
345  bodyMapTest.at( bodyToPropagate )->setConstantBodyMass( initialMass_ );
346 
347  // Retrieve low-thrust trajectory nominal accelerations (thrust + central gravity accelerations).
348  basic_astrodynamics::AccelerationMap lowThrustTrajectoryAccelerations = retrieveLowThrustAccelerationMap(
349  bodyMapTest, bodyToPropagate, centralBody, specificImpulseFunction, integratorSettings );
350 
351  // Add perturbing accelerations given as input.
352  basic_astrodynamics::AccelerationMap accelerationModelMap = perturbingAccelerationsMap;
353  accelerationModelMap[ bodyToPropagate ][ bodyToPropagate ].push_back(
354  lowThrustTrajectoryAccelerations.at( bodyToPropagate ).at( bodyToPropagate ).at( 0 ) );
355  accelerationModelMap[ bodyToPropagate ][ centralBody ].push_back(
356  lowThrustTrajectoryAccelerations.at( bodyToPropagate ).at( centralBody ).at( 0 ) );
357 
358 
359  // Define translational state propagation settings
360  std::pair< std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > >,
361  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > > translationalStatePropagatorSettings =
362  createLowThrustTranslationalStatePropagatorSettings(
363  bodyToPropagate, centralBody, accelerationModelMap, dependentVariablesToSave );
364 
365  double massHalfOfTimeOfFlight = computeCurrentMass( timeOfFlight_ / 2.0, specificImpulseFunction, integratorSettings );
366 
367  // Create settings for propagating the mass of the vehicle.
368  std::pair< std::shared_ptr< propagators::MassPropagatorSettings< double > >,
369  std::shared_ptr< propagators::MassPropagatorSettings< double > > > massPropagatorSettings;
370 
371  // Create mass rate models
372  std::map< std::string, std::shared_ptr< basic_astrodynamics::MassRateModel > > massRateModels;
373  massRateModels[ bodyToPropagate ] = simulation_setup::createMassRateModel(
374  bodyToPropagate, std::make_shared< simulation_setup::FromThrustMassModelSettings >( 1 ), bodyMapTest, accelerationModelMap );
375 
376  // Define backward mass propagation settings.
377  massPropagatorSettings.first = std::make_shared< propagators::MassPropagatorSettings< double > >(
378  std::vector< std::string >{ bodyToPropagate }, massRateModels,
379  ( Eigen::Matrix< double, 1, 1 >( ) << massHalfOfTimeOfFlight ).finished( ), terminationConditions.first );
380 
381  // Define forward mass propagation settings.
382  massPropagatorSettings.second = std::make_shared< propagators::MassPropagatorSettings< double > >(
383  std::vector< std::string >{ bodyToPropagate }, massRateModels,
384  ( Eigen::Matrix< double, 1, 1 >( ) << massHalfOfTimeOfFlight ).finished( ), terminationConditions.second );
385 
386  // Create list of propagation settings.
387  std::pair< std::vector< std::shared_ptr< propagators::SingleArcPropagatorSettings< double > > >,
388  std::vector< std::shared_ptr< propagators::SingleArcPropagatorSettings< double > > > > propagatorSettingsVector;
389 
390  // Backward propagator settings vector.
391  propagatorSettingsVector.first.push_back( translationalStatePropagatorSettings.first );
392  propagatorSettingsVector.first.push_back( massPropagatorSettings.first );
393 
394  // Forward propagator settings vector.
395  propagatorSettingsVector.second.push_back( translationalStatePropagatorSettings.second );
396  propagatorSettingsVector.second.push_back( massPropagatorSettings.second );
397 
398  // Backward hybrid propagation settings.
399  propagatorSettings.first = std::make_shared< propagators::MultiTypePropagatorSettings< double > >(
400  propagatorSettingsVector.first, terminationConditions.first, dependentVariablesToSave );
401 
402  // Forward hybrid propagation settings.
403  propagatorSettings.second = std::make_shared< propagators::MultiTypePropagatorSettings< double > >(
404  propagatorSettingsVector.second, terminationConditions.second, dependentVariablesToSave );
405 
406 
407  return propagatorSettings;
408 }
