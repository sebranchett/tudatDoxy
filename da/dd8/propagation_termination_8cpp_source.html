61 bool HybridPropagationTerminationCondition::checkStopCondition( const double time, const double cpuTime )
62 {
63  // Check if single condition is fulfilled.
64  bool stopPropagation = -1;
65  unsigned int stopIndex = 0;
66  if( fulfillSingleCondition_ )
67  {
68  stopPropagation = false;
69  for( unsigned int i = 0; i < propagationTerminationCondition_.size( ); i++ )
70  {
71  if( propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime ) )
72  {
73  stopIndex = i;
74  stopPropagation = true;
75  isConditionMetWhenStopping_[ i ] = true;
76  break;
77  }
78  else
79  {
80  isConditionMetWhenStopping_[ i ] = false;
81  }
82  }
83  }
84  // Check all conditions are fulfilled.
85  else
86  {
87  stopPropagation = true;
88  for( unsigned int i = 0; i < propagationTerminationCondition_.size( ); i++ )
89  {
90  if( !propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime ) )
91  {
92  stopIndex = i;
93  stopPropagation = false;
94  isConditionMetWhenStopping_[ i ] = false;
95  break;
96  }
97  else
98  {
99  isConditionMetWhenStopping_[ i ] = true;
100  }
101  }
102  }
103 
104  // Save if conditions were met
105  if( stopPropagation )
106  {
107  for( unsigned int i = ( stopIndex + 1 ); i < propagationTerminationCondition_.size( ); i++ )
108  {
109  isConditionMetWhenStopping_[ i ] = propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime );
110  }
111  }
112 
113  return stopPropagation;
114 }

118 std::shared_ptr< PropagationTerminationCondition > createPropagationTerminationConditions(
119  const std::shared_ptr< PropagationTerminationSettings > terminationSettings,
120  const simulation_setup::NamedBodyMap& bodyMap,
121  const double initialTimeStep )
122 {
123  std::shared_ptr< PropagationTerminationCondition > propagationTerminationCondition;
124 
125  // Check termination type.
126  switch( terminationSettings->terminationType_ )
127  {
128  case time_stopping_condition:
129  {
130  // Create stopping time termination condition.
131  std::shared_ptr< PropagationTimeTerminationSettings > timeTerminationSettings =
132  std::dynamic_pointer_cast< PropagationTimeTerminationSettings >( terminationSettings );
133  propagationTerminationCondition = std::make_shared< FixedTimePropagationTerminationCondition >(
134  timeTerminationSettings->terminationTime_, ( initialTimeStep > 0 ),
135  timeTerminationSettings->terminateExactlyOnFinalCondition_ );
136  break;
137  }
138  case cpu_time_stopping_condition:
139  {
140  // Create stopping time termination condition.
141  std::shared_ptr< PropagationCPUTimeTerminationSettings > cpuTimeTerminationSettings =
142  std::dynamic_pointer_cast< PropagationCPUTimeTerminationSettings >( terminationSettings );
143  propagationTerminationCondition = std::make_shared< FixedCPUTimePropagationTerminationCondition >(
144  cpuTimeTerminationSettings->cpuTerminationTime_ );
145  break;
146  }
147  case dependent_variable_stopping_condition:
148  {
149  std::shared_ptr< PropagationDependentVariableTerminationSettings > dependentVariableTerminationSettings =
150  std::dynamic_pointer_cast< PropagationDependentVariableTerminationSettings >( terminationSettings );
151 
152  // Get dependent variable function
153  std::function< double( ) > dependentVariableFunction;
154  if( getDependentVariableSaveSize( dependentVariableTerminationSettings->dependentVariableSettings_ ) == 1 )
155  {
156  dependentVariableFunction =
157  getDoubleDependentVariableFunction(
158  dependentVariableTerminationSettings->dependentVariableSettings_, bodyMap );
159  }
160  else
161  {
162  throw std::runtime_error( "Error, cannot make stopping condition from vector dependent variable" );
163  }
164 
165  // Create dependent variable termination condition.
166  propagationTerminationCondition = std::make_shared< SingleVariableLimitPropagationTerminationCondition >(
167  dependentVariableTerminationSettings->dependentVariableSettings_,
168  dependentVariableFunction, dependentVariableTerminationSettings->limitValue_,
169  dependentVariableTerminationSettings->useAsLowerLimit_,
170  dependentVariableTerminationSettings->terminateExactlyOnFinalCondition_,
171  dependentVariableTerminationSettings->terminationRootFinderSettings_ );
172  break;
173  }
174  case custom_stopping_condition:
175  {
176  std::shared_ptr< PropagationCustomTerminationSettings > customTerminationSettings =
177  std::dynamic_pointer_cast< PropagationCustomTerminationSettings >( terminationSettings );
178 
179  // Create dependent variable termination condition.
180  propagationTerminationCondition = std::make_shared< CustomTerminationCondition >(
181  customTerminationSettings->checkStopCondition_,
182  customTerminationSettings->terminateExactlyOnFinalCondition_ );
183  break;
184  }
185  case hybrid_stopping_condition:
186  {
187  std::shared_ptr< PropagationHybridTerminationSettings > hybridTerminationSettings =
188  std::dynamic_pointer_cast< PropagationHybridTerminationSettings >( terminationSettings );
189 
190  // Recursively create termination condition list.
191  std::vector< std::shared_ptr< PropagationTerminationCondition > > propagationTerminationConditionList;
192  for( unsigned int i = 0; i < hybridTerminationSettings->terminationSettings_.size( ); i++ )
193  {
194  propagationTerminationConditionList.push_back(
195  createPropagationTerminationConditions(
196  hybridTerminationSettings->terminationSettings_.at( i ),
197  bodyMap, initialTimeStep ) );
198  }
199  propagationTerminationCondition = std::make_shared< HybridPropagationTerminationCondition >(
200  propagationTerminationConditionList, hybridTerminationSettings->fulfillSingleCondition_,
201  hybridTerminationSettings->terminateExactlyOnFinalCondition_ );
202  break;
203  }
204  default:
205  std::string errorMessage = "Error, stopping condition type " + std::to_string(
206  terminationSettings->terminationType_ ) + " not recognized when making stopping conditions object";
207  throw std::runtime_error( errorMessage );
208  }
209  return propagationTerminationCondition;
