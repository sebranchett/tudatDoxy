23 void PlanetaryOrientationAngleCalculator::updateCorrections( const double ephemerisTime )
24 {
25  currentEphemerisTime_ = ephemerisTime;
26  double currentMeanMotion = bodyMeanMotion_;
27  double currentMeanAnomaly = bodyMeanAnomalyAtEpoch_ + bodyMeanMotion_ * currentEphemerisTime_;
28 
29  currentAngleICorrection_ = 0.0;
30  currentAnglePsiCorrection_ = 0.0;
31 
32  for( std::map< double, std::pair< double, double > >::iterator correctionIterator = meanMotionDirectNutationCorrections_.begin( );
33  correctionIterator != meanMotionDirectNutationCorrections_.end( ); correctionIterator++ )
34  {
35  double a_m = correctionIterator->first * currentMeanMotion;
36 
37  double I_m = correctionIterator->second.first + coreFactor_ * a_m /
38  ( a_m * a_m - freeCoreNutationRate_ * freeCoreNutationRate_ ) *
39  ( a_m * correctionIterator->second.first + freeCoreNutationRate_ *
40  correctionIterator->second.second * std::sin( angleIAtEpoch_ ) );
41 
42  double Psi_m = correctionIterator->second.second + coreFactor_ * a_m /
43  ( a_m * a_m - freeCoreNutationRate_ * freeCoreNutationRate_ ) *
44  ( a_m * correctionIterator->second.second + freeCoreNutationRate_ *
45  correctionIterator->second.first / std::sin( angleIAtEpoch_ ) );
46 
47  currentAngleICorrection_ += I_m * std::cos(
48  correctionIterator->first * ( currentMeanMotion * currentEphemerisTime_ + bodyMeanAnomalyAtEpoch_ ) );
49  currentAnglePsiCorrection_ += Psi_m * std::sin(
50  correctionIterator->first * ( currentMeanMotion * currentEphemerisTime_ + bodyMeanAnomalyAtEpoch_ ) );
51  }
52 
53  double currentPhaseAngleCorrection;
54 
55  for( unsigned int i = 0; i < meanMotionTimeDependentPhaseNutationCorrections_.size( ); i++ )
56  {
57  currentPhaseAngleCorrection = phaseAngleCorrectionFunctions_[ i ]( currentEphemerisTime_ );
58  for( std::map< double, std::pair< double, double > >::iterator correctionIterator =
59  meanMotionTimeDependentPhaseNutationCorrections_[ i ].begin( );
60  correctionIterator != meanMotionTimeDependentPhaseNutationCorrections_[ i ].end( ); correctionIterator++ )
61  {
62  double a_m = correctionIterator->first * currentMeanMotion;
63 
64  double I_m = correctionIterator->second.first + coreFactor_ * a_m /
65  ( a_m * a_m - freeCoreNutationRate_ * freeCoreNutationRate_ ) *
66  ( a_m * correctionIterator->second.first + freeCoreNutationRate_ *
67  correctionIterator->second.second * std::sin( angleIAtEpoch_ ) );
68 
69  double Psi_m = correctionIterator->second.second + coreFactor_ * a_m /
70  ( a_m * a_m - freeCoreNutationRate_ * freeCoreNutationRate_ ) *
71  ( a_m * correctionIterator->second.second + freeCoreNutationRate_ *
72  correctionIterator->second.first / std::sin( angleIAtEpoch_ ) );
73 
74 
75  currentAngleICorrection_ += I_m * std::cos(
76  correctionIterator->first * ( currentMeanMotion * currentEphemerisTime_ + bodyMeanAnomalyAtEpoch_ ) +
77  currentPhaseAngleCorrection );
78  currentAnglePsiCorrection_ += Psi_m * std::sin(
79  correctionIterator->first * ( currentMeanMotion * currentEphemerisTime_ + bodyMeanAnomalyAtEpoch_ ) +
80  currentPhaseAngleCorrection );
81  }
82  }
83 
84  currentAnglePsi_ = anglePsiAtEpoch_ + anglePsiRateAtEpoch_ * currentEphemerisTime_ + currentAnglePsiCorrection_;
85 
86  currentAngleI_ = angleIAtEpoch_ + angleIRateAtEpoch_ * currentEphemerisTime_ + currentAngleICorrection_;
87 
88  currentAnglePhiCorrection_ = -currentAnglePsiCorrection_ * std::cos( currentAngleI_ );
89 
90  for( std::map< double, std::pair< double, double > >::iterator correctionIterator = rotationRateCorrections_.begin( );
91  correctionIterator != rotationRateCorrections_.end( ); correctionIterator++ )
92  {
93  currentAnglePhiCorrection_ += correctionIterator->second.first * std::cos(
94  correctionIterator->first * currentMeanAnomaly );
95  currentAnglePhiCorrection_ += correctionIterator->second.second * std::sin(
96  correctionIterator->first * currentMeanAnomaly );
97  }
98 
99  currentAnglePhi_ = anglePhiAtEpoch_ + anglePhiRateAtEpoch_ * currentEphemerisTime_ + currentAnglePhiCorrection_;
100 }

130 void PlanetaryOrientationAngleCalculator::calculateCurrentMeanPhiAngleDerivative( const double ephemerisTime )
131 {
132  currentEphemerisTime_ = ephemerisTime;
133  double currentMeanMotion = bodyMeanMotion_;
134  double currentMeanAnomaly = bodyMeanAnomalyAtEpoch_ + bodyMeanMotion_ * currentEphemerisTime_;
135 
136  currentMeanPhiAngleDerivative_ = anglePhiRateAtEpoch_ ;
137 
138  for( std::map< double, std::pair< double, double > >::iterator correctionIterator = rotationRateCorrections_.begin( );
139  correctionIterator != rotationRateCorrections_.end( ); correctionIterator++ )
140  {
141  currentMeanPhiAngleDerivative_ += - correctionIterator->first * currentMeanMotion * correctionIterator->second.first * std::sin(
142  correctionIterator->first * currentMeanAnomaly );
143  currentMeanPhiAngleDerivative_ += correctionIterator->first * currentMeanMotion * correctionIterator->second.second * std::cos(
144  correctionIterator->first * currentMeanAnomaly );
145  }

149 createInterpolatorForPlanetaryRotationAngles( double intervalStart,
150  double intervalEnd,
151  double timeStep,
152  boost::shared_ptr< PlanetaryOrientationAngleCalculator > planetaryOrientationCalculator )
153 {
154  using namespace interpolators;
155 
156  std::map< double, Eigen::Vector3d > orientationMap;
157 
158  double currentTime = intervalStart;
159  while( currentTime < intervalEnd )
160  {
161  orientationMap[ currentTime ] = planetaryOrientationCalculator->updateAndGetRotationAngles( currentTime );
162 
163  currentTime += timeStep;
164  }
165 
166  boost::shared_ptr< CubicSplineInterpolator< double, Eigen::Vector3d > > interpolator =
167  boost::make_shared< CubicSplineInterpolator< double, Eigen::Vector3d > >( orientationMap );
168  return interpolator;

195 Eigen::Matrix3d PlanetaryRotationModel::getDerivativeOfRotationToBaseFrame( const double ephemerisTime )
196 {
197  Eigen::Vector3d currentAngleCorrections = planetaryOrientationAnglesCalculator_->updateAndGetRotationAngles( ephemerisTime );
198 
199  double currentPhiAngle = currentAngleCorrections.z( );
200 
201  double meanPhiAngleDerivative = planetaryOrientationAnglesCalculator_->getcurrentMeanPhiAngleDerivative( ephemerisTime );
202 
203  return meanPhiAngleDerivative * ( rotationFromMeanOrbitToIcrf_ ).toRotationMatrix( ) *
204  ( Eigen::AngleAxisd( currentAngleCorrections.x( ), Eigen::Vector3d::UnitZ( ) ) *
205  Eigen::AngleAxisd( currentAngleCorrections.y( ), Eigen::Vector3d::UnitX( ) ) ).toRotationMatrix( ) *
206  ( Eigen::Matrix3d( ) << -std::sin( currentPhiAngle ), -std::cos( currentPhiAngle ), 0.0,
207  std::cos( currentPhiAngle ), -std::sin( currentPhiAngle ), 0.0,
208  0.0, 0.0, 0.0 ).finished( ) *
209  getPolarMotionRotation( ephemerisTime ).toRotationMatrix();
