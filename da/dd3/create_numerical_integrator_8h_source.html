563 template< typename IndependentVariableType, typename DependentVariableType,
564  typename IndependentVariableStepType = IndependentVariableType >
565 std::shared_ptr< numerical_integrators::NumericalIntegrator< IndependentVariableType, DependentVariableType,
566 DependentVariableType, IndependentVariableStepType > > createIntegrator(
567  std::function< DependentVariableType(
568  const IndependentVariableType, const DependentVariableType& ) > stateDerivativeFunction,
569  const DependentVariableType initialState,
570  std::shared_ptr< IntegratorSettings< IndependentVariableType > > integratorSettings )
571 {
572  // Declare eventual output
573  std::shared_ptr< NumericalIntegrator
574  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > > integrator;
575 
576  // Retrieve requested type of integrator
577  switch( integratorSettings->integratorType_ )
578  {
579  case euler:
580  {
581  // Create Euler integrator
582  integrator = std::make_shared< EulerIntegrator
583  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
584  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
585  break;
586  }
587  case rungeKutta4:
588  {
589  // Create Runge-Kutta 4 integrator
590  integrator = std::make_shared< RungeKutta4Integrator
591  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
592  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
593  break;
594  }
595  case rungeKuttaVariableStepSize:
596  {
597  // Cast integrator
598  std::shared_ptr< RungeKuttaVariableStepSizeBaseSettings< IndependentVariableType > >
599  variableStepIntegratorSettings = std::dynamic_pointer_cast< RungeKuttaVariableStepSizeBaseSettings<
600  IndependentVariableType > >( integratorSettings );
601 
602  // Check input consistency
603  if ( variableStepIntegratorSettings == nullptr )
604  {
605  throw std::runtime_error( "Error, type of integrator settings (rungeKuttaVariableStepSize) not compatible with "
606  "selected integrator (derived class of IntegratorSettings must be "
607  "RungeKuttaVariableStepSizeBaseSettings for this type)." );
608  }
609 
610  // Get requested RK coefficients
611  RungeKuttaCoefficients coefficients = RungeKuttaCoefficients::get( variableStepIntegratorSettings->coefficientSet_ );
612 
613  // Check which constructor is being used
614  if ( variableStepIntegratorSettings->areTolerancesDefinedAsScalar_ )
615  {
616  // Settings with scalar tolerances
617  std::shared_ptr< RungeKuttaVariableStepSizeSettingsScalarTolerances< IndependentVariableType > >
618  scalarTolerancesIntegratorSettings = std::dynamic_pointer_cast<
619  RungeKuttaVariableStepSizeSettingsScalarTolerances< IndependentVariableType > >( variableStepIntegratorSettings );
620 
621  // Check input consistency
622  if ( scalarTolerancesIntegratorSettings == nullptr )
623  {
624  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. Input class must be of "
625  "RungeKuttaVariableStepSizeSettingsScalarTolerances type." );
626  }
627 
628  // Create Runge-Kutta integrator with scalar tolerances
629  integrator = std::make_shared< RungeKuttaVariableStepSizeIntegrator
630  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
631  ( coefficients, stateDerivativeFunction, integratorSettings->initialTime_, initialState,
632  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->minimumStepSize_ ),
633  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->maximumStepSize_ ),
634  static_cast< typename DependentVariableType::Scalar >( scalarTolerancesIntegratorSettings->relativeErrorTolerance_ ),
635  static_cast< typename DependentVariableType::Scalar >( scalarTolerancesIntegratorSettings->absoluteErrorTolerance_ ),
636  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->safetyFactorForNextStepSize_ ),
637  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
638  static_cast< IndependentVariableStepType >( scalarTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
639  }
640  else
641  {
642  // Settings with vector tolerances
643  std::shared_ptr< RungeKuttaVariableStepSizeSettingsVectorTolerances< IndependentVariableType, DependentVariableType > >
644  vectorTolerancesIntegratorSettings = std::dynamic_pointer_cast<
645  RungeKuttaVariableStepSizeSettingsVectorTolerances< IndependentVariableType, DependentVariableType > >(
646  variableStepIntegratorSettings );
647 
648  // Check input consistency
649  if ( vectorTolerancesIntegratorSettings == nullptr )
650  {
651  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. Input class must be of "
652  "RungeKuttaVariableStepSizeSettingsVectorTolerances type." );
653  }
654 
655  // Check that sizes of tolerances and initial state match
656  DependentVariableType relativeErrorTolerance = vectorTolerancesIntegratorSettings->relativeErrorTolerance_.template cast<
657  typename DependentVariableType::Scalar >( );
658  DependentVariableType absoluteErrorTolerance = vectorTolerancesIntegratorSettings->absoluteErrorTolerance_.template cast<
659  typename DependentVariableType::Scalar >( );
660  if ( ( relativeErrorTolerance.rows( ) != initialState.rows( ) ) ||
661  ( relativeErrorTolerance.cols( ) != initialState.cols( ) ) ||
662  ( absoluteErrorTolerance.rows( ) != initialState.rows( ) ) ||
663  ( absoluteErrorTolerance.cols( ) != initialState.cols( ) ) )
664  {
665  throw std::runtime_error( "Error while creating Runge-Kutta variable step size integrator. The sizes of the "
666  "relative and absolute tolerance vectors do not match the size of the initial state. "
667  "This could be the case if you are propagating more than just one state, e.g., translational "
668  "and/or rotational dynamics and mass, or more than one body." );
669  }
670 
671  // Create Runge-Kutta integrator with vector tolerances
672  integrator = std::make_shared< RungeKuttaVariableStepSizeIntegrator
673  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
674  ( coefficients, stateDerivativeFunction, integratorSettings->initialTime_, initialState,
675  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->minimumStepSize_ ),
676  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->maximumStepSize_ ),
677  relativeErrorTolerance, absoluteErrorTolerance,
678  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->safetyFactorForNextStepSize_ ),
679  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
680  static_cast< IndependentVariableStepType >( vectorTolerancesIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
681  }
682  break;
683  }
684  case bulirschStoer:
685  {
686  // Check input consistency
687  std::shared_ptr< BulirschStoerIntegratorSettings< IndependentVariableType > > bulirschStoerIntegratorSettings =
688  std::dynamic_pointer_cast< BulirschStoerIntegratorSettings< IndependentVariableType > >(
689  integratorSettings );
690 
691  // Check that integrator type has been cast properly
692  if ( bulirschStoerIntegratorSettings == nullptr )
693  {
694  throw std::runtime_error( "Error, type of integrator settings (bulirschStoer) not compatible with "
695  "selected integrator (derived class of IntegratorSettings must be BulirschStoerIntegratorSettings "
696  "for this type)." );
697  }
698  else
699  {
700  integrator = std::make_shared< BulirschStoerVariableStepSizeIntegrator
701  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
702  ( getBulirschStoerStepSequence( bulirschStoerIntegratorSettings->extrapolationSequence_,
703  bulirschStoerIntegratorSettings->maximumNumberOfSteps_ ),
704  stateDerivativeFunction, integratorSettings->initialTime_, initialState,
705  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->minimumStepSize_ ),
706  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->maximumStepSize_ ),
707  bulirschStoerIntegratorSettings->relativeErrorTolerance_,
708  bulirschStoerIntegratorSettings->absoluteErrorTolerance_,
709  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->safetyFactorForNextStepSize_ ),
710  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
711  static_cast< IndependentVariableStepType >( bulirschStoerIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
712  }
713  break;
714  }
715  case adamsBashforthMoulton:
716  {
717  // Check input consistency
718  std::shared_ptr< AdamsBashforthMoultonSettings< IndependentVariableType > > adamsBashforthMoultonIntegratorSettings =
719  std::dynamic_pointer_cast< AdamsBashforthMoultonSettings< IndependentVariableType > >(
720  integratorSettings );
721 
722  // Check that integrator type has been cast properly
723  if ( adamsBashforthMoultonIntegratorSettings == nullptr )
724  {
725  throw std::runtime_error( "Error, type of integrator settings (AdamsBashforthMoultonSettings) not compatible with "
726  "selected integrator (derived class of IntegratorSettings must be AdamsBashforthMoultonSettings "
727  "for this type)." );
728  }
729  else
730  {
731  // Create integrator
732  integrator = std::make_shared< AdamsBashforthMoultonIntegrator
733  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >
734  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState,
735  static_cast< IndependentVariableStepType >( adamsBashforthMoultonIntegratorSettings->minimumStepSize_ ),
736  static_cast< IndependentVariableStepType >( adamsBashforthMoultonIntegratorSettings->maximumStepSize_ ),
737  adamsBashforthMoultonIntegratorSettings->relativeErrorTolerance_,
738  adamsBashforthMoultonIntegratorSettings->absoluteErrorTolerance_ ,
739  static_cast< IndependentVariableStepType >( adamsBashforthMoultonIntegratorSettings->bandwidth_ ) );
740 
741  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
742  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
743  integrator )->setMinimumOrder( adamsBashforthMoultonIntegratorSettings->minimumOrder_ );
744  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
745  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
746  integrator )->setMaximumOrder( adamsBashforthMoultonIntegratorSettings->maximumOrder_ );
747 
748  if ( adamsBashforthMoultonIntegratorSettings->minimumOrder_ == adamsBashforthMoultonIntegratorSettings->maximumOrder_ )
749  {
750  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
751  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
752  integrator )->setFixedOrder( true );
753  }
754 
755  if ( adamsBashforthMoultonIntegratorSettings->minimumStepSize_ ==
756  adamsBashforthMoultonIntegratorSettings->maximumStepSize_ )
757  {
758  std::dynamic_pointer_cast< AdamsBashforthMoultonIntegrator
759  < IndependentVariableType, DependentVariableType, DependentVariableType, IndependentVariableStepType > >(
760  integrator )->setFixedStepSize( true );
761  }
762  }
763  break;
764  }
765  default:
766  throw std::runtime_error( "Error, integrator " + std::to_string( integratorSettings->integratorType_ ) + " not found." );
767  }
768 
769  // Check that assignment of integrator went well
770  if ( integrator == nullptr )
771  {
772  throw std::runtime_error( "Error while creating integrator. The resulting integrator pointer is null." );
773  }
774 
775  // Give back integrator
776  return integrator;
777 }
