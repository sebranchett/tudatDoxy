48 TimeStepType getTerminationDependentVariableErrorForGivenTimeStep(
49  TimeStepType timeStep,
50  const std::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > >
51  integrator,
52  const std::shared_ptr< SingleVariableLimitPropagationTerminationCondition > dependentVariableTerminationCondition )

84 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
85 void getFinalStateForExactDependentVariableTerminationCondition(
86  const std::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > >
87  integrator,
88  const std::shared_ptr< SingleVariableLimitPropagationTerminationCondition > dependentVariableTerminationCondition,
89  const TimeType secondToLastTime,
90  const TimeType lastTime,
91  const StateType& secondToLastState,
92  const StateType& lastState,
93  TimeType& endTime,
94  StateType& endState )
95 {
96  TUDAT_UNUSED_PARAMETER( secondToLastState );
97 
98  // Function for which the root (zero value) occurs at the required end time/state
99  std::function< TimeStepType( TimeStepType ) > dependentVariableErrorFunction =
100  std::bind( &getTerminationDependentVariableErrorForGivenTimeStep< StateType, TimeType, TimeStepType >, std::placeholders::_1,
101  integrator, dependentVariableTerminationCondition );
102 
103  // Create root finder.
104  bool increasingTime = static_cast< double >( lastTime - secondToLastTime ) > 0.0;
105  std::shared_ptr< root_finders::RootFinderCore< TimeStepType > > finalConditionRootFinder;
106  if( increasingTime )
107  {
108  finalConditionRootFinder = root_finders::createRootFinder< TimeStepType >(
109  dependentVariableTerminationCondition->getTerminationRootFinderSettings( ),
110  static_cast< TimeStepType >( std::numeric_limits< double >::min( ) ),
111  static_cast< TimeStepType >( lastTime - secondToLastTime ),
112  static_cast< TimeStepType >( std::numeric_limits< double >::min( ) ) );
113  }
114  else
115  {
116  finalConditionRootFinder = root_finders::createRootFinder< TimeStepType >(
117  dependentVariableTerminationCondition->getTerminationRootFinderSettings( ),
118  static_cast< TimeStepType >( lastTime - secondToLastTime ),
119  static_cast< TimeStepType >( -std::numeric_limits< double >::min( ) ),
120  static_cast< TimeStepType >( lastTime - secondToLastTime ) );
121 
122  }
123 
124  // Solve root-finding problem.
125  TimeStepType finalTimeStep;
126  try
127  {
128  finalTimeStep = finalConditionRootFinder->execute(
129  std::make_shared< basic_mathematics::FunctionProxy< TimeStepType, TimeStepType > >(
130  dependentVariableErrorFunction ), ( lastTime - secondToLastTime ) / 2.0 );
131 
132  endState = integrator->performIntegrationStep( finalTimeStep );
133  endTime = integrator->getCurrentIndependentVariable( );
134  }
135  // If dependent variable has no root in given interval, set end time and state at NaN
136  catch( std::runtime_error& caughtException )
137  {
138  std::cerr << "Warning in propagation to exact dependent variable value. Root finder could not find a "
139  "root to the function. Returning time and state as NaNs. Caught exception: "
140  << caughtException.what( ) << std::endl;
141  endTime = TUDAT_NAN;
142  endState = StateType::Constant( lastState.rows( ), lastState.cols( ), TUDAT_NAN );
143  }
144 }

161 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
162 void getFinalStateForExactHybridVariableTerminationCondition(
163  const std::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > >
164  integrator,
165  const std::shared_ptr< HybridPropagationTerminationCondition > hyrbidTerminationCondition,
166  const TimeType secondToLastTime,
167  const TimeType lastTime,
168  const StateType& secondToLastState,
169  const StateType& lastState,
170  TimeType& endTime,
171  StateType& endState )
172 {
173 
174  std::vector< std::shared_ptr< PropagationTerminationCondition > > terminationConditionList =
175  hyrbidTerminationCondition->getPropagationTerminationConditions( );
176 
177  // Create list of converged times/states for each constituent condition
178  std::vector< TimeType > endTimes;
179  endTimes.resize( terminationConditionList.size( ) );
180  std::vector< StateType > endStates;
181  endStates.resize( terminationConditionList.size( ) );
182 
183  // Iterate over all constituent termination conditions, and determine separate end times/states
184  unsigned int minimumTimeIndex = 0, maximumTimeIndex = 0;
185  TimeStepType minimumTimeStep, maximumTimeStep;
186  bool timesAreSet = false;
187  for( unsigned int i = 0; i < terminationConditionList.size( ); i++ )
188  {
189  // Determine single termination condition
190  getFinalStateForExactTerminationCondition(
191  integrator, terminationConditionList.at( i ),secondToLastTime, lastTime, secondToLastState, lastState,
192  endTimes[ i ], endStates[ i ] );
193 
194  // If converged time is found, check if it is smallest/highest converged time
195  if( endTimes[ i ] == endTimes[ i ] )
196  {
197  TimeStepType currentFinalTimeStep = endTimes[ i ] - secondToLastTime;
198 
199  if( !timesAreSet )
200  {
201  maximumTimeStep = currentFinalTimeStep;
202  maximumTimeIndex = i;
203  minimumTimeStep = currentFinalTimeStep;
204  minimumTimeIndex = i;
205  timesAreSet = true;
206  }
207  else
208  {
209  if( currentFinalTimeStep < minimumTimeStep )
210  {
211  minimumTimeStep = currentFinalTimeStep;
212  minimumTimeIndex = i;
213  }
214 
215  if( currentFinalTimeStep > maximumTimeStep )
216  {
217  maximumTimeStep = currentFinalTimeStep;
218  maximumTimeIndex = i;
219  }
220  }
221  }
222  }
223 
224  // Check if any of the conditions were valid
225  if( !timesAreSet )
226  {
227  throw std::runtime_error( "Error when propagating exactly to hybrid condition, no conditions met" );
228  }
229 
230  // Set converged end time/state
231  bool propagationIsForwards = ( ( lastTime - secondToLastTime ) > 0.0 ) ? true : false;
232  if( ( propagationIsForwards && !hyrbidTerminationCondition->getFulfillSingleCondition( ) ) ||
233  ( !propagationIsForwards && hyrbidTerminationCondition->getFulfillSingleCondition( ) ) )
234  {
235  endState = endStates[ maximumTimeIndex ];
236  endTime = endTimes[ maximumTimeIndex ];
237  }
238  else if( ( propagationIsForwards && hyrbidTerminationCondition->getFulfillSingleCondition( ) ) ||
239  ( !propagationIsForwards && !hyrbidTerminationCondition->getFulfillSingleCondition( ) ) )
240  {
241  endState = endStates[ minimumTimeIndex ];
242  endTime = endTimes[ minimumTimeIndex ];
243  }
244  else
245  {
246  throw std::runtime_error( "Error when propagating to exact final hybrid condition, case not recognized" );
247  }
248 
249 }

264 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
265 void getFinalStateForExactTerminationCondition(
266  const std::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > > integrator,
267  const std::shared_ptr< PropagationTerminationCondition > terminationCondition,
268  const TimeType secondToLastTime,
269  const TimeType lastTime,
270  const StateType& secondToLastState,
271  const StateType& lastState,
272  TimeType& endTime,
273  StateType& endState )
274 {
275  // Check type of termination condition
276  switch( terminationCondition->getTerminationType( ) )
277  {
278  case time_stopping_condition:
279  {
280  // Check input consistency
281  std::shared_ptr< FixedTimePropagationTerminationCondition > timeTerminationCondition =
282  std::dynamic_pointer_cast< FixedTimePropagationTerminationCondition >( terminationCondition );
283 
284  // Determine final time step and propagate
285  TimeStepType finalTimeStep = timeTerminationCondition->getStopTime( ) - secondToLastTime;
286 
287  integrator->rollbackToPreviousState( );
288  endState = integrator->performIntegrationStep( finalTimeStep );
289  endTime = integrator->getCurrentIndependentVariable( );
290 
291  break;
292  }
293  case cpu_time_stopping_condition:
294  {
295  // No exact final condition on CPU time is possible
296  std::cerr << "Error, cannot propagate to exact CPU time, returning state after condition violation." << std::endl;
297 
298  endTime = lastTime;
299  endState = lastState;
300  break;
301  }
302  case dependent_variable_stopping_condition:
303  {
304  integrator->rollbackToPreviousState( );
305 
306  std::shared_ptr< SingleVariableLimitPropagationTerminationCondition > dependentVariableTerminationCondition =
307  std::dynamic_pointer_cast< SingleVariableLimitPropagationTerminationCondition >( terminationCondition );
308  getFinalStateForExactDependentVariableTerminationCondition(
309  integrator, dependentVariableTerminationCondition, secondToLastTime, lastTime,
310  secondToLastState, lastState, endTime, endState );
311 
312  break;
313  }
314  case custom_stopping_condition:
315  {
316  // No exact final condition on custom condition is possible
317  std::cerr << "Error, cannot propagate to exact custom condition, returning state after condition violation." << std::endl;
318 
319  endTime = lastTime;
320  endState = lastState;
321  break;
322  }
323  case hybrid_stopping_condition:
324  {
325  std::shared_ptr< HybridPropagationTerminationCondition > hyrbidTerminationCondition =
326  std::dynamic_pointer_cast< HybridPropagationTerminationCondition >( terminationCondition );
327 
328  getFinalStateForExactHybridVariableTerminationCondition(
329  integrator, hyrbidTerminationCondition, secondToLastTime, lastTime,
330  secondToLastState, lastState, endTime, endState );
331  break;
332  }
333  default:
334  throw std::runtime_error( "Error when propagating to exact final condition, did not recognize termination time" );
335  }
336 }

353 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
354 void propagateToExactTerminationCondition(
355  const std::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > > integrator,
356  const std::shared_ptr< PropagationTerminationCondition > propagationTerminationCondition,
357  const TimeStepType timeStep,
358  const std::function< Eigen::VectorXd( ) > dependentVariableFunction,
359  std::map< TimeType, StateType >& solutionHistory,
360  std::map< TimeType, Eigen::VectorXd >& dependentVariableHistory,
361  const double currentCpuTime )
362 {
363  // Turn off step size control
364  integrator->setStepSizeControl( false );
365 
366  // Determine exact final time/state
367  TimeType endTime;
368  StateType endState;
369  getFinalStateForExactTerminationCondition(
370  integrator, propagationTerminationCondition,
371  integrator->getPreviousIndependentVariable( ),
372  integrator->getCurrentIndependentVariable( ),
373  integrator->getPreviousState( ),
374  integrator->getCurrentState( ),
375  endTime, endState );
376 
377  // Check if any dependent variables are saved. If so, remove last entry
378  bool recomputeDependentVariables = false;
379  if( dependentVariableHistory.size( ) > 0 )
380  {
381  if( dependentVariableHistory.rbegin( )->first == solutionHistory.rbegin( )->first )
382  {
383  if( timeStep > 0 )
384  {
385  dependentVariableHistory.erase( std::prev( dependentVariableHistory.end() ) );
386  }
387  else
388  {
389  dependentVariableHistory.erase( dependentVariableHistory.begin( ) );
390  }
391  recomputeDependentVariables = true;
392  }
393  }
394 
395  // Remove state entry last added, and enter converged final state
396  if( timeStep > 0 )
397  {
398  solutionHistory.erase( std::prev( solutionHistory.end() ) );
399  solutionHistory[ endTime ] = endState;
400  }
401  else
402  {
403  solutionHistory.erase( solutionHistory.begin( ) );
404  solutionHistory[ endTime ] = endState;
405  }
406 
407  // Recompute final dependent variables, if required
408  if( recomputeDependentVariables )
409  {
410  integrator->getStateDerivativeFunction( )( endTime, endState );
411  dependentVariableHistory[ endTime ] = dependentVariableFunction( );
412 
413  // Check stopping conditions to be able to save details
414  propagationTerminationCondition->checkStopCondition( endTime, currentCpuTime );
415  }
416 
417  // Turn step size control back on
418  integrator->setStepSizeControl( true );
419 }

444 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
445 std::shared_ptr< PropagationTerminationDetails > integrateEquationsFromIntegrator(
446  const std::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > > integrator,
447  const TimeStepType initialTimeStep,
448  const std::shared_ptr< PropagationTerminationCondition > propagationTerminationCondition,
449  std::map< TimeType, StateType >& solutionHistory,
450  std::map< TimeType, Eigen::VectorXd >& dependentVariableHistory,
451  std::map< TimeType, double >& cumulativeComputationTimeHistory,
452  const std::function< Eigen::VectorXd( ) > dependentVariableFunction = std::function< Eigen::VectorXd( ) >( ),
453  const std::function< void( StateType& ) > statePostProcessingFunction = std::function< void( StateType& ) >( ),
454  const int saveFrequency = TUDAT_NAN,
455  const TimeType printInterval = TUDAT_NAN,
456  const std::chrono::steady_clock::time_point initialClockTime = std::chrono::steady_clock::now( ) )
457 {
458  std::shared_ptr< PropagationTerminationDetails > propagationTerminationReason;
459 
460  // Get Initial state and time.
461  TimeType currentTime = integrator->getCurrentIndependentVariable( );
462  TimeType initialTime = currentTime;
463  StateType newState = integrator->getCurrentState( );
464 
465  // Initialization of numerical solutions for variational equations
466  solutionHistory.clear( );
467  solutionHistory[ currentTime ] = newState;
468 
469  dependentVariableHistory.clear( );
470  if( !( dependentVariableFunction == nullptr ) )
471  {
472  integrator->getStateDerivativeFunction( )( currentTime, newState );
473  dependentVariableHistory[ currentTime ] = dependentVariableFunction( );
474  }
475 
476  // CPU time
477  cumulativeComputationTimeHistory.clear( );
478  double currentCPUTime = std::chrono::duration_cast< std::chrono::nanoseconds >(
479  std::chrono::steady_clock::now( ) - initialClockTime ).count( ) * 1.0e-9;
480  cumulativeComputationTimeHistory[ currentTime ] = currentCPUTime;
481 
482  // Set initial time step and total integration time.
483  TimeStepType timeStep = initialTimeStep;
484  TimeType previousTime = currentTime;
485  TimeType previousPrintTime = TUDAT_NAN;
486 
487  int saveIndex = 0;
488 
489  propagationTerminationReason = std::make_shared< PropagationTerminationDetails >(
490  unknown_propagation_termination_reason );
491  bool breakPropagation = 0;
492 
493  // Perform numerical integration steps until end time reached.
494  do
495  {
496  try
497  {
498  if( ( newState.allFinite( ) == true ) && ( !newState.hasNaN( ) ) )
499  {
500  previousTime = currentTime;
501 
502  // Perform integration step.
503  newState = integrator->performIntegrationStep( timeStep );
504  if( statePostProcessingFunction != nullptr )
505  {
506  statePostProcessingFunction( newState );
507  integrator->modifyCurrentState( newState, true );
508  }
509 
510  // Check if the termination condition was reached during evaluation of integration sub-steps.
511  // If evaluation of the termination condition during integration sub-steps is disabled,
512  // this function returns always `false`.
513  // If the termination condition was reached, the last step could not be computed correctly because some
514  // of the integrator sub-steps were not computed. Thus, return immediately without saving the `newState`.
515  if( integrator->getPropagationTerminationConditionReached( ) )
516  {
517  propagationTerminationReason = std::make_shared< PropagationTerminationDetails >(
518  termination_condition_reached );
519  break;
520  }
521 
522  // Update epoch and step-size
523  currentTime = integrator->getCurrentIndependentVariable( );
524  timeStep = integrator->getNextStepSize( );
525 
526  // Save integration result in map
527  saveIndex++;
528  saveIndex = saveIndex % saveFrequency;
529  if( saveIndex == 0 )
530  {
531  solutionHistory[ currentTime ] = newState;
532 
533  if( !( dependentVariableFunction == nullptr ) )
534  {
535  integrator->getStateDerivativeFunction( )( currentTime, newState );
536  dependentVariableHistory[ currentTime ] = dependentVariableFunction( );
537  }
538  }
539  }
540  else
541  {
542  std::cerr << "Error, propagation terminated at t=" + std::to_string( static_cast< double >( currentTime ) ) +
543  ", found NaN/Inf entry, returning propagation data up to current time" << std::endl;
544  breakPropagation = true;
545  propagationTerminationReason = std::make_shared< PropagationTerminationDetails >(
546  nan_or_inf_detected_in_state );
547  }
548 
549  currentCPUTime = std::chrono::duration_cast< std::chrono::nanoseconds >(
550  std::chrono::steady_clock::now( ) - initialClockTime ).count( ) * 1.0e-9;
551  cumulativeComputationTimeHistory[ currentTime ] = currentCPUTime;
552 
553  // Print solutions
554  if( printInterval == printInterval )
555  {
556  if( !( previousPrintTime == previousPrintTime ) ||
557  std::fabs( static_cast< double >( currentTime - previousPrintTime ) ) > printInterval )
558  {
559  previousPrintTime = currentTime;
560  std::cout << "Current time: "<<currentTime<<"; time since initial time: "<<currentTime - initialTime<<std::endl;
561  std::cout << "Current time step: "<<timeStep<<std::endl;
562  std::cout << "Current state (transposed): "<<newState.transpose( ) <<std::endl<<std::endl;
563  }
564  }
565 
566  if( propagationTerminationCondition->checkStopCondition( static_cast< double >( currentTime ), currentCPUTime ) )
567  {
568  if( propagationTerminationCondition->getTerminateExactlyOnFinalCondition( ) )
569  {
570  propagateToExactTerminationCondition(
571  integrator, propagationTerminationCondition,
572  timeStep, dependentVariableFunction,
573  solutionHistory, dependentVariableHistory, currentCPUTime );
574  }
575 
576  // Set termination details
577  if( propagationTerminationCondition->getTerminationType( ) != hybrid_stopping_condition )
578  {
579  propagationTerminationReason = std::make_shared< PropagationTerminationDetails >(
580  termination_condition_reached,
581  propagationTerminationCondition->getTerminateExactlyOnFinalCondition( ) );
582  }
583  else
584  {
585  if( std::dynamic_pointer_cast< HybridPropagationTerminationCondition >( propagationTerminationCondition )
586  == nullptr )
587  {
588  throw std::runtime_error( "Error when saving termination reason, type is hybrid, but class is not." );
589  }
590  propagationTerminationReason = std::make_shared< PropagationTerminationDetailsFromHybridCondition >(
591  propagationTerminationCondition->getTerminateExactlyOnFinalCondition( ),
592  std::dynamic_pointer_cast< HybridPropagationTerminationCondition >(
593  propagationTerminationCondition ) );
594  }
595  breakPropagation = true;
596  }
597  }
598  catch( const std::exception& caughtException )
599  {
600  std::cerr << caughtException.what( ) << std::endl;
601  std::cerr << "Error, propagation terminated at t=" + std::to_string( static_cast< double >( currentTime ) ) +
602  ", returning propagation data up to current time." << std::endl;
603  breakPropagation = true;
604  propagationTerminationReason = std::make_shared< PropagationTerminationDetails >(
605  runtime_error_caught_in_propagation );
606  }
607  }
608  while( !breakPropagation );
609 
610  return propagationTerminationReason;
611 }

718  static std::shared_ptr< PropagationTerminationDetails > integrateEquations(
