55 template< typename ParameterType >
56 std::pair< SingleLinkObservationThreePartialList, std::shared_ptr< PositionPartialScaling > >
57 createEulerAngleObservablePartials(
58  const observation_models::LinkEnds eulerAngleLinkEnds,
59  const simulation_setup::NamedBodyMap& bodyMap,
60  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate )
61 
62 {
63  // Create scaling object, to be used for each partial created here (i.e. same scaling for different parameters but same
64  // observable).
65  std::shared_ptr< PositionPartialScaling > eulerAngleScaling = nullptr;
66 
67  SingleLinkObservationThreePartialList eulerAnglePartials;
68 
69  // Define start index and size of current parameter under consideration
70  int currentIndex = 0;
71  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
72 
73  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
74  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
75  parametersToEstimate->getEstimatedInitialStateParameters( );
76 
77  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
78  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
79  {
80  if( std::dynamic_pointer_cast< estimatable_parameters::InitialRotationalStateParameter< ParameterType > >(
81  initialDynamicalParameters.at( i ) ) != nullptr )
82  {
83  // Create partial (if needed)
84  std::shared_ptr< ObservationPartial< 3 > > currentObservablePartial =
85  createEulerAngleObservablePartialWrtCurrentRotationalState(
86  initialDynamicalParameters.at( i )->getParameterName( ) );
87 
88  // If partial exists, then dependency exists and parameter must be added.
89  if( currentObservablePartial != nullptr )
90  {
91  currentPair = std::pair< int, int >( currentIndex, 7 );
92  eulerAnglePartials[ currentPair ] = currentObservablePartial;
93  }
94  }
95  currentIndex += initialDynamicalParameters.at( i )->getParameterSize( );
96  }
97 
98  // Iterate over all double parameters that are to be estimated.
99  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParametersToEstimate =
100  parametersToEstimate->getDoubleParameters( );
101  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
102  parameterIterator = doubleParametersToEstimate.begin( );
103  parameterIterator != doubleParametersToEstimate.end( ); parameterIterator++ )
104  {
105  if( estimatable_parameters::isParameterRotationMatrixProperty( parameterIterator->second->getParameterName( ).first ) )
106  {
107  throw std::runtime_error(
108  "Error when making Euler angle partial, found kinematic rotation parameter: " +
109  std::to_string( estimatable_parameters::isParameterRotationMatrixProperty(
110  parameterIterator->second->getParameterName( ).first ) ) +
111  ". Computation requires derivatives of Euler angles w.r.t. rotation matrix entries, which is not yet implemented." );
112  }
113  }
114 
115  // Iterate over all vector parameters that are to be estimated.
116  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
117  vectorParametersToEstimate = parametersToEstimate->getVectorParameters( );
118  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
119  parameterIterator = vectorParametersToEstimate.begin( );
120  parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
121  {
122  if( estimatable_parameters::isParameterRotationMatrixProperty( parameterIterator->second->getParameterName( ).first ) )
123  {
124  throw std::runtime_error(
125  "Error when making Euler angle partial, found kinematic rotation parameter: " +
126  std::to_string( estimatable_parameters::isParameterRotationMatrixProperty(
127  parameterIterator->second->getParameterName( ).first ) ) +
128  ". Computation requires derivatives of Euler angles w.r.t. rotation matrix entries, which is not yet implemented." );
129  }
130  }
131 
132  return std::make_pair( eulerAnglePartials, eulerAngleScaling );
133 }

145 template< typename ParameterType >
146 std::map< observation_models::LinkEnds,
147 std::pair< SingleLinkObservationThreePartialList, std::shared_ptr< PositionPartialScaling > > >
148 createEulerAngleObservablePartials(
149  const std::vector< observation_models::LinkEnds > linkEnds,
150  const simulation_setup::NamedBodyMap& bodyMap,
151  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate )
152 {
153  std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationThreePartialList,
154  std::shared_ptr< PositionPartialScaling > > > eulerAnlgeObservablePartials;
155 
156  for( unsigned int i = 0; i < linkEnds.size( ); i++ )
157  {
158  if( linkEnds[ i ].count( observation_models::observed_body ) == 0 || linkEnds[ i ].size( ) != 1 )
159  {
160  throw std::runtime_error( "Error when making position observable partial, link ends are wrong" );
161  }
162 
163  eulerAnlgeObservablePartials[ linkEnds[ i ] ] = createEulerAngleObservablePartials(
164  linkEnds[ i ], bodyMap, parametersToEstimate );
165  }
166  return eulerAnlgeObservablePartials;
