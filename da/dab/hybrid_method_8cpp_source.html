28 std::pair< std::vector< double >, std::vector< double > > HybridMethod::performOptimisation( )
29 {
30  //Set seed for reproducible results
31  pagmo::random_device::set_seed( 456 );
32 
33  // Create object to compute the problem fitness
34  problem prob{ HybridMethodProblem( stateAtDeparture_, stateAtArrival_, maximumThrust_, specificImpulse_, timeOfFlight_,
35  initialGuessThrustModel_, initialAndFinalMEEcostatesBounds_,
36  optimisationSettings_->relativeToleranceConstraints_ )};
37 
38  std::vector< double > constraintsTolerance;
39  for ( unsigned int i = 0 ; i < ( prob.get_nec() + prob.get_nic() ) ; i++ )
40  {
41  constraintsTolerance.push_back( 1.0e-3 );
42  }
43  prob.set_c_tol( constraintsTolerance );
44 
45  algorithm algo = optimisationSettings_->optimisationAlgorithm_;
46 
47  unsigned long long populationSize = optimisationSettings_->numberOfIndividualsPerPopulation_;
48 
49  island island{ algo, prob, populationSize };
50 
51  // Evolve for 10 generations
52  for( int i = 0 ; i < optimisationSettings_->numberOfGenerations_ ; i++ )
53  {
54  island.evolve( );
55  while( island.status( ) != pagmo::evolve_status::idle &&
56  island.status( ) != pagmo::evolve_status::idle_error )
57  {
58  island.wait( );
59  }
60  island.wait_check( ); // Raises errors
61 
62  }
63 
64  championFitness_ = island.get_population().champion_f();
65  championDesignVariables_ = island.get_population().champion_x();
66 
67  std::pair< std::vector< double >, std::vector< double > > output;
68  output.first = championFitness_;
69  output.second = championDesignVariables_;
70 
71  return output;
72 
73 }

123 void HybridMethod::getThrustForceProfile(
124  std::vector< double >& epochsVector,
125  std::map< double, Eigen::VectorXd >& thrustProfile,
126  std::function< double ( const double ) > specificImpulseFunction )
127 {
128  thrustProfile.clear( );
129 
130  std::shared_ptr< simulation_setup::ThrustMagnitudeSettings > thrustMagnitudeSettings = hybridMethodModel_->getMEEcostatesBasedThrustMagnitudeSettings( );
131 
132  std::function< Eigen::Vector3d( ) > bodyFixedThrustDirection = simulation_setup::getBodyFixedThrustDirection(
133  thrustMagnitudeSettings, bodyMap_, bodyToPropagate_ );
134 
135  std::map< double, Eigen::Vector6d > trajectory;
136  getTrajectory( epochsVector, trajectory );
137 
138  for ( unsigned int i = 0 ; i < epochsVector.size() ; i++ )
139  {
140  if ( ( i > 0 ) && ( epochsVector[ i ] < epochsVector[ i - 1 ] ) )
141  {
142  throw std::runtime_error( "Error when retrieving the thrust profile of a hybrid method trajectory, "
143  "epochs are not provided in increasing order." );
144  }
145 
146  Eigen::Vector6d currentStateVector = trajectory[ epochsVector[ i ] ];
147 
148  std::function< Eigen::Vector6d( ) > thrustingBodyStateFunction = [ = ] ( )
149  {
150  return currentStateVector;
151  };
152 
153  std::function< Eigen::Vector6d( ) > centralBodyStateFunction = [ = ] ( )
154  {
155  return Eigen::Vector6d::Zero( );
156  };
157 
158  std::function< double( ) > centralBodyGravitationalParameterFunction = [ = ]( )
159  {
160  return bodyMap_[ centralBody_ ]->getGravityFieldModel( )->getGravitationalParameter( );
161  };
162 
163  std::function< double( ) > thrustingBodyMassFunction = std::bind( &simulation_setup::Body::getBodyMass, bodyMap_.at( bodyToPropagate_ ) );
164 
165 
166  propulsion::MeeCostatesBangBangThrustMagnitudeWrapper thrustMagnitudeWrapper = propulsion::MeeCostatesBangBangThrustMagnitudeWrapper(
167  thrustingBodyStateFunction, centralBodyStateFunction, centralBodyGravitationalParameterFunction,
168  hybridMethodModel_->getCostatesFunction_( ), maximumThrust_, specificImpulseFunction, thrustingBodyMassFunction);
169 
170  propulsion::MeeCostateBasedThrustGuidance thrustGuidance = propulsion::MeeCostateBasedThrustGuidance(
171  thrustingBodyStateFunction, centralBodyStateFunction, centralBodyGravitationalParameterFunction,
172  hybridMethodModel_->getCostatesFunction_( ), bodyFixedThrustDirection );
173 
174  thrustGuidance.updateCalculator( epochsVector[ i ] );
175  thrustMagnitudeWrapper.update( epochsVector[ i ] );
176 
177  thrustProfile[ epochsVector[ i ] ] = thrustMagnitudeWrapper.getCurrentThrustMagnitude( ) * thrustGuidance.getCurrentForceDirectionInPropagationFrame( );
178 
179  }
180 }

212 void HybridMethod::getThrustAccelerationProfile(
213  std::vector< double >& epochsVector,
214  std::map< double, Eigen::VectorXd >& thrustAccelerationProfile,
215  std::function< double ( const double ) > specificImpulseFunction,
216  std::shared_ptr<numerical_integrators::IntegratorSettings< double > > integratorSettings )
217 {
218  thrustAccelerationProfile.clear();
219 
220  std::map< double, Eigen::VectorXd > thrustProfile;
221  getThrustForceProfile( epochsVector, thrustProfile, specificImpulseFunction, integratorSettings );
222 
223  std::map< double, Eigen::VectorXd > massProfile;
224  getMassProfile( epochsVector, massProfile, specificImpulseFunction, integratorSettings );
225 
226  for ( unsigned int i = 0 ; i < epochsVector.size() ; i++ )
227  {
228  if ( ( i > 0 ) && ( epochsVector[ i ] < epochsVector[ i - 1 ] ) )
229  {
230  throw std::runtime_error( "Error when retrieving the thrust profile of a hybrid method trajectory, "
231  "epochs are not provided in increasing order." );
232  }
233 
234  Eigen::Vector3d currentThrustVector = thrustProfile[ epochsVector[ i ] ];
235  double currentMass = massProfile[ epochsVector[ i ] ][ 0 ];
236 
237  Eigen::Vector3d currentThrustAccelerationVector = currentThrustVector / currentMass;
238 
239  thrustAccelerationProfile[ epochsVector[ i ] ] = currentThrustAccelerationVector;
240 
241  }

296 
297  // Define backward translational state propagation settings.
298  translationalStatePropagatorSettings.first = std::make_shared< propagators::TranslationalStatePropagatorSettings< double > >
299  ( std::vector< std::string >{ centralBody_ }, accelerationModelMap,
300  std::vector< std::string >{ bodyToPropagate_ }, stateAtHalfOfTimeOfFlight,
301  terminationConditions.first, propagators::gauss_modified_equinoctial, dependentVariablesToSave );
302 
303  // Define forward translational state propagation settings.
304  translationalStatePropagatorSettings.second = std::make_shared< propagators::TranslationalStatePropagatorSettings< double > >
305  ( std::vector< std::string >{ centralBody_ }, accelerationModelMap,
306  std::vector< std::string >{ bodyToPropagate_ }, stateAtHalfOfTimeOfFlight,
