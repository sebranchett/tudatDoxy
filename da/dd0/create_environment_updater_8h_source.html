145 template< typename StateScalarType >
146 std::map< propagators::EnvironmentModelsToUpdate,
147 std::vector< std::string > > createEnvironmentUpdaterSettings(
148  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
149  const simulation_setup::NamedBodyMap& bodyMap,
150  const bool isPartOfMultiTypePropagation = false )
151 {
152  std::map< propagators::EnvironmentModelsToUpdate,
153  std::vector< std::string > > environmentModelsToUpdate;
154 
155  // Check dynamics type
156  switch( propagatorSettings->getStateType( ) )
157  {
158  case hybrid:
159  {
160  // Cast to derived type
161  std::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
162  std::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
163 
164  // Iterate over all propagation settings in hybrid model
165  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleAccelerationUpdateNeeds;
166 
167  for( typename std::map< IntegratedStateType,
168  std::vector< std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::const_iterator
169  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
170  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
171  {
172  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
173  {
174  // Create current propagation model from settings list (must not be hybrid).
175  if( typeIterator->first != hybrid )
176  {
177  singleAccelerationUpdateNeeds = createEnvironmentUpdaterSettings< StateScalarType >(
178  typeIterator->second.at( i ), bodyMap, true );
179 
180  // Add single model environment model update settings to full list
181  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
182  }
183  else
184  {
185  throw std::runtime_error( "Error when making environment updater type list, cannot handle hybrid "
186  "propagator inside hybrid propagator" );
187  }
188  }
189  }
190  break;
191  }
192  // Retrieve environment model settings for translational dynamics
193  case translational_state:
194  {
195  environmentModelsToUpdate = createTranslationalEquationsOfMotionEnvironmentUpdaterSettings(
196  std::dynamic_pointer_cast<
197  TranslationalStatePropagatorSettings< StateScalarType > >(
198  propagatorSettings )->getAccelerationsMap( ),
199  bodyMap );
200  break;
201  }
202  case rotational_state:
203  {
204  environmentModelsToUpdate = createRotationalEquationsOfMotionEnvironmentUpdaterSettings(
205  std::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings )->getTorqueModelsMap( ),
206  bodyMap );
207  break;
208  }
209  // Retrieve environment model settings for mass rate model
210  case body_mass_state:
211  {
212  environmentModelsToUpdate = createMassPropagationEnvironmentUpdaterSettings(
213  std::dynamic_pointer_cast<
214  MassPropagatorSettings< StateScalarType > >(
215  propagatorSettings )->getMassRateModelsMap( ), bodyMap );
216  break;
217  }
218  case custom_state:
219  {
220  break;
221  }
222  default:
223  {
224  throw std::runtime_error( "Error, cannot create environment updates for type " +
225  std::to_string( propagatorSettings->getStateType( ) ) );
226  }
227  }
228 
229  std::map< propagators::EnvironmentModelsToUpdate,
230  std::vector< std::string > > environmentModelsToUpdateForDependentVariables =
231  createEnvironmentUpdaterSettings( propagatorSettings->getDependentVariablesToSave( ), bodyMap );
232  addEnvironmentUpdates( environmentModelsToUpdate, environmentModelsToUpdateForDependentVariables );
233 
234  std::map< propagators::EnvironmentModelsToUpdate,
235  std::vector< std::string > > environmentModelsToUpdateForTerminationConditions =
236  createEnvironmentUpdaterSettings( propagatorSettings->getTerminationSettings( ), bodyMap );
237  addEnvironmentUpdates( environmentModelsToUpdate, environmentModelsToUpdateForTerminationConditions );
238 
239  // Remove variables from environment updates that are numerically propagated.
240  if( !isPartOfMultiTypePropagation )
241  {
242  removePropagatedStatesFomEnvironmentUpdates(
243  environmentModelsToUpdate, getIntegratedTypeAndBodyList( propagatorSettings ) );
244  checkValidityOfRequiredEnvironmentUpdates( environmentModelsToUpdate, bodyMap );
245  }
246 
247 
248  return environmentModelsToUpdate;
249 
250 }

273 template< typename StateScalarType, typename TimeType >
274 std::shared_ptr< propagators::EnvironmentUpdater< StateScalarType, TimeType > >
275 createEnvironmentUpdaterForDynamicalEquations(
276  const std::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
277  const simulation_setup::NamedBodyMap& bodyMap )
278 {
279  // Create environment update settings.
280  std::map< IntegratedStateType,
281  std::vector< std::pair< std::string, std::string > > > integratedTypeAndBodyList =
282  getIntegratedTypeAndBodyList< StateScalarType >( propagatorSettings );
283 
284  std::map< propagators::EnvironmentModelsToUpdate,
285  std::vector< std::string > > environmentModelsToUpdate =
286  createEnvironmentUpdaterSettings< StateScalarType >( propagatorSettings, bodyMap );
287 
288  // Create and return environment updater object.
289  return std::make_shared< EnvironmentUpdater< StateScalarType, TimeType > >(
290  bodyMap, environmentModelsToUpdate, integratedTypeAndBodyList );
