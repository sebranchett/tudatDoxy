150 template< typename IndependentVariableType, typename DependentVariableType >
151 std::shared_ptr< numerical_quadrature::NumericalQuadrature< IndependentVariableType, DependentVariableType > > createQuadrature(
152  std::function< DependentVariableType( const IndependentVariableType ) > derivativeFunction,
153  std::shared_ptr< QuadratureSettings< IndependentVariableType > > quadratureSettings,
154  IndependentVariableType finalIndependentVariable = TUDAT_NAN )
155 {
156  // Declare eventual output.
157  std::shared_ptr< NumericalQuadrature < IndependentVariableType, DependentVariableType > > quadrature;
158 
159  // Retrieve requested type of quadrature.
160  switch( quadratureSettings->quadratureType_ )
161  {
162  case gaussian:
163  {
164  // Cast dynamic pointer on gaussian quadrature settings.
165  std::shared_ptr< GaussianQuadratureSettings< double > > gaussianQuadratureSettings =
166  std::dynamic_pointer_cast< GaussianQuadratureSettings < double > >( quadratureSettings );
167 
168  // Create gaussian quadrature.
169  quadrature = std::make_shared< GaussianQuadrature < IndependentVariableType, DependentVariableType > >
170  ( derivativeFunction, gaussianQuadratureSettings->initialIndependentVariable_, finalIndependentVariable,
171  gaussianQuadratureSettings->numberOfNodes_ ) ;
172  break;
173  }
174  case trapezoidal:
175  {
176  // Cats dynamic pointer on trapezoid quadrature settings.
177  std::shared_ptr< TrapezoidQuadratureSettings< double > > trapezoidQuadratureSettings =
178  std::dynamic_pointer_cast< TrapezoidQuadratureSettings< double > >( quadratureSettings );
179 
180  // Retrieve independent variables at which the values of the derivative function must be given.
181  std::vector< IndependentVariableType > independentVariables = trapezoidQuadratureSettings->independentVariables_;
182 
183  // Evaluate derivative function at each of the independent variable values.
184  std::vector< DependentVariableType > dependentVariables;
185  for ( unsigned int currentIndependentValue = 0 ; currentIndependentValue < independentVariables.size() ; currentIndependentValue++ )
186  {
187  dependentVariables.push_back( derivativeFunction( currentIndependentValue ) );
188  }
189 
190  // Create Trapezoid quadrature.
191  quadrature = std::make_shared< TrapezoidNumericalQuadrature < IndependentVariableType, DependentVariableType > >
192  ( independentVariables, dependentVariables ) ;
193  break;
194  }
195  default:
196  throw std::runtime_error( "Error, quadrature " + std::to_string( quadratureSettings->quadratureType_ ) + " not found." );
197  }
198 
199  // Check that assignment of quadrature went well.
200  if ( quadrature == nullptr )
201  {
202  throw std::runtime_error( "Error while creating quadrature. The resulting quadrature pointer is null." );
203  }
204 
205  // Return numerical quadrature object.
206  return quadrature;
207 }
