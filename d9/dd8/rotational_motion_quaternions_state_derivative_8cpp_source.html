22 Eigen::Matrix4d getQuaterionToQuaternionRateMatrix( const Eigen::Vector3d& angularVelocityVectorInBodyFixedFrame )
23 {
24  Eigen::Matrix4d conversionMatrix = Eigen::Matrix4d::Zero( );
25  conversionMatrix( 1, 0 ) = angularVelocityVectorInBodyFixedFrame( 0 );
26  conversionMatrix( 2, 0 ) = angularVelocityVectorInBodyFixedFrame( 1 );
27  conversionMatrix( 3, 0 ) = angularVelocityVectorInBodyFixedFrame( 2 );
28 
29  conversionMatrix( 2, 1 ) = -angularVelocityVectorInBodyFixedFrame( 2 );
30  conversionMatrix( 3, 1 ) = angularVelocityVectorInBodyFixedFrame( 1 );
31 
32  conversionMatrix( 3, 2 ) = -angularVelocityVectorInBodyFixedFrame( 0 );
33 
34  conversionMatrix( 0, 1 ) = -angularVelocityVectorInBodyFixedFrame( 0 );
35  conversionMatrix( 0, 2 ) = -angularVelocityVectorInBodyFixedFrame( 1 );
36  conversionMatrix( 0, 3 ) = -angularVelocityVectorInBodyFixedFrame( 2 );
37 
38  conversionMatrix( 1, 2 ) = angularVelocityVectorInBodyFixedFrame( 2 );
39  conversionMatrix( 1, 3 ) = -angularVelocityVectorInBodyFixedFrame( 1 );
40 
41  conversionMatrix( 2, 3 ) = angularVelocityVectorInBodyFixedFrame( 0 );
42 
43  return 0.5 * ( conversionMatrix );
