120 std::shared_ptr< RotationMatrixPartial > createRotationMatrixPartialsWrtStateParameter(
121  const simulation_setup::NamedBodyMap& bodyMap,
122  const std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::Matrix<
123  InitialStateParameterType, Eigen::Dynamic, 1 > > > parameterToEstimate )
124 {
125  using namespace simulation_setup;
126  using namespace ephemerides;
127 
128  // Declare return object.
129  std::shared_ptr< RotationMatrixPartial > rotationMatrixPartial;
130 
131  // Get body for rotation of which partial is to be created.
132  std::shared_ptr< Body > currentBody = bodyMap.at( parameterToEstimate->getParameterName( ).second.first );
133 
134  // Check for which rotation model parameter the partial object is to be created.
135  switch( parameterToEstimate->getParameterName( ).first )
136  {
137  case estimatable_parameters::initial_rotational_body_state:
138 
139  // Create rotation matrix partial object
140  rotationMatrixPartial = std::make_shared< RotationMatrixPartialWrtQuaternion >(
141  std::bind( &Body::getCurrentRotationToGlobalFrame, currentBody ) );
142  break;
143 
144  default:
145  std::string errorMessage = "Warning, rotation matrix partial not implemented for state parameter " +
146  std::to_string( parameterToEstimate->getParameterName( ).first );
147  throw std::runtime_error( errorMessage );
148  break;
149  }
150 
151  return rotationMatrixPartial;
152 
153 }

189 template< typename ParameterType >
190 RotationMatrixPartialNamedList createRotationMatrixPartials(
191  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
192  const std::string& bodyName, const simulation_setup::NamedBodyMap& bodyMap )
193 
194 {
195  //std::vector< std::shared_ptr< EstimatableParameter< Eigen:: > > >
196  //getEstimatedInitialStateParameters( )
197 
198  // Declare map to return.
199  std::map< std::pair< estimatable_parameters::EstimatebleParametersEnum, std::string >,
200  std::shared_ptr< RotationMatrixPartial > >
201  rotationMatrixPartials;
202 
203  // Retrieve double and vector parameters from total set of parameters.
204  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter<
205  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > stateParameters =
206  parametersToEstimate->getInitialStateParameters( );
207  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParameters =
208  parametersToEstimate->getDoubleParameters( );
209  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > > vectorParameters =
210  parametersToEstimate->getVectorParameters( );
211 
212  std::shared_ptr< RotationMatrixPartial > rotationMatrixPartialWrtTranslationalState =
213  createRotationMatrixPartialsWrtTranslationalState( bodyMap.at( bodyName ) );
214  if( rotationMatrixPartialWrtTranslationalState != nullptr )
215  {
216  rotationMatrixPartials[ std::make_pair(
217  estimatable_parameters::initial_body_state, "" ) ] = rotationMatrixPartialWrtTranslationalState;
218  }
219 
220  for( auto parameterIterator = stateParameters.begin( ); parameterIterator != stateParameters.end( );
221  parameterIterator++ )
222  {
223  // Check parameter is rotational property of requested body.
224  if( ( parameterIterator->second->getParameterName( ).second.first == bodyName ) &&
225  ( estimatable_parameters::isParameterRotationMatrixProperty(
226  parameterIterator->second->getParameterName( ).first ) ) )
227  {
228  // Create partial object.
229  rotationMatrixPartials[ std::make_pair(
230  parameterIterator->second->getParameterName( ).first,
231  parameterIterator->second->getSecondaryIdentifier( )) ] =
232  createRotationMatrixPartialsWrtStateParameter( bodyMap, parameterIterator->second );
233  }
234  }
235 
236  // Iterate over double parameters.
237  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
238  parameterIterator = doubleParameters.begin( ); parameterIterator != doubleParameters.end( ); parameterIterator++ )
239  {
240  // Check parameter is rotational property of requested body.
241  if( ( parameterIterator->second->getParameterName( ).second.first == bodyName ) &&
242  ( estimatable_parameters::isParameterRotationMatrixProperty(
243  parameterIterator->second->getParameterName( ).first ) ) )
244  {
245  // Create partial object.
246  rotationMatrixPartials[ std::make_pair(
247  parameterIterator->second->getParameterName( ).first,
248  parameterIterator->second->getSecondaryIdentifier( )) ] =
249  createRotationMatrixPartialsWrtParameter( bodyMap, parameterIterator->second );
250  }
251  }
252 
253  // Iterate over vector parameters.
254  for( std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
255  parameterIterator = vectorParameters.begin( ); parameterIterator != vectorParameters.end( ); parameterIterator++ )
256  {
257  // Check parameter is rotational property of requested body.
258  if( ( parameterIterator->second->getParameterName( ).second.first == bodyName ) &&
259  ( estimatable_parameters::isParameterRotationMatrixProperty(
260  parameterIterator->second->getParameterName( ).first ) ) )
261  {
262  // Create partial object.
263  rotationMatrixPartials[ std::make_pair(
264  parameterIterator->second->getParameterName( ).first,
265  parameterIterator->second->getSecondaryIdentifier( )) ] =
266  createRotationMatrixPartialsWrtParameter( bodyMap, parameterIterator->second );
267  }
268  }
269 
270  return rotationMatrixPartials;
271 }

305 template< typename ParameterType >
306 std::pair< SingleLinkObservationThreePartialList, std::shared_ptr< PositionPartialScaling > >
307 createPositionObservablePartials(
308  const observation_models::LinkEnds positionObservableLinkEnds,
309  const simulation_setup::NamedBodyMap& bodyMap,
310  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate )
311 
312 {
313  // Create scaling object, to be used for each partial created here (i.e. same scaling for different parameters but same
314  // observable).
315  std::shared_ptr< PositionObservationScaling > positionObservableScaling =
316  std::make_shared< PositionObservationScaling >( );
317 
318  SingleLinkObservationThreePartialList positionObservablePartials;
319 
320  // Define start index and size of current parameter under consideration
321  int currentIndex = 0;
322  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
323 
324  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
325  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
326  parametersToEstimate->getEstimatedInitialStateParameters( );
327 
328  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
329  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
330  {
331  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state ||
332  initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
333  {
334  // Retrieve name of body w.r.t. position of which partial is to be taken.
335  std::string acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
336 
337  // Create partial (if needed)
338  std::shared_ptr< PositionObervationPartial > currentObservablePartial =
339  createPositionObservablePartialWrtPosition(
340  positionObservableLinkEnds, bodyMap, acceleratedBody, positionObservableScaling );
341 
342  // If partial exists, then dependency exists and parameter must be added.
343  if( currentObservablePartial != nullptr )
344  {
345  currentPair = std::pair< int, int >( currentIndex, 6 );
346  positionObservablePartials[ currentPair ] = currentObservablePartial;
347  }
348 
349  currentIndex += 6;
350  }
351  }
352 
353  return std::make_pair( positionObservablePartials, positionObservableScaling );
354 }
