172  DependentVariableType interpolate( const std::vector< IndependentVariableType >& independentValuesToInterpolate )
173  {
174  // Check whether size of independent variable vector is correct
175  if ( independentValuesToInterpolate.size( ) != NumberOfDimensions )
176  {
177  throw std::runtime_error( "Error in multi-dimensional interpolator. The number of independent variables "
178  "provided is incompatible with the previous definition. Provided: " +
179  std::to_string( independentValuesToInterpolate.size( ) ) + ". Needed: " +
180  std::to_string( NumberOfDimensions ) );
181  }
182 
183  // Create local copy of current independent variables
184  std::vector< IndependentVariableType > localIndependentValuesToInterpolate = independentValuesToInterpolate;
185 
186  // Check that independent variables are in range
187  bool useValue = false;
188  DependentVariableType currentDependentVariable;
189  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
190  {
191  this->checkBoundaryCase( i, useValue, localIndependentValuesToInterpolate.at( i ), currentDependentVariable );
192  if ( useValue )
193  {
194  return currentDependentVariable;
195  }
196  }
197 
198  // Determine the nearest lower neighbours.
199  std::vector< int > nearestLowerIndices;
200  nearestLowerIndices.resize( NumberOfDimensions );
201  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
202  {
203  nearestLowerIndices[ i ] = lookUpSchemes_[ i ]->findNearestLowerNeighbour(
204  localIndependentValuesToInterpolate[ i ] );
205  }
206 
207  // Initialize function evaluation indices to -1 for debugging purposes.
208  boost::array< unsigned int, NumberOfDimensions > interpolationIndices;
209  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
210  {
211  interpolationIndices[ i ] = -1;
212  }
213 
214  // Call first step of interpolation, this function calls itself at subsequent independent
215  // variable dimensions to evaluate and properly scale dependent variable table values at
216  // all 2^n grid edges.
217  return performRecursiveInterpolationStep( 0, localIndependentValuesToInterpolate,
218  interpolationIndices, nearestLowerIndices );
219  }

231  void makeLookupSchemes( const AvailableLookupScheme selectedScheme )
232  {
233  lookUpSchemes_.resize( NumberOfDimensions );
234  // Find which type of scheme is used.
235  switch( selectedScheme )
236  {
237  case binarySearch:
238 
239  for( unsigned int i = 0; i < NumberOfDimensions; i++ )
240  {
241  // Create binary search look up scheme.
242  lookUpSchemes_[ i ] = std::shared_ptr< LookUpScheme< IndependentVariableType > >
243  ( new BinarySearchLookupScheme< IndependentVariableType >(
244  independentValues_[ i ] ) );
245  }
246 
247  break;
248 
249  case huntingAlgorithm:
250 
251  for( unsigned int i = 0; i < NumberOfDimensions; i++ )
252  {
253  // Create hunting scheme, which uses an intial guess from previous look-ups.
254  lookUpSchemes_[ i ] = std::shared_ptr< LookUpScheme< IndependentVariableType > >
255  ( new HuntingAlgorithmLookupScheme< IndependentVariableType >(
256  independentValues_[ i ] ) );
257  }
258 
259  break;
260 
261  default:
262 
263  throw std::runtime_error( "Warning: lookup scheme not found when making scheme for 1-D interpolator" );
264  }
265  }

286  DependentVariableType performRecursiveInterpolationStep(
287  const unsigned int currentDimension,
288  const std::vector< IndependentVariableType >& independentValuesToInterpolate,
289  boost::array< unsigned int, NumberOfDimensions > currentArrayIndices,
290  const std::vector< int >& nearestLowerIndices )
291  {
292  IndependentVariableType upperFraction, lowerFraction;
293  DependentVariableType upperContribution, lowerContribution;
294 
295  // Calculate fractions of data points above and below independent
296  // variable value to be added to interpolated value.
297  upperFraction = ( independentValuesToInterpolate[ currentDimension ] -
298  independentValues_[ currentDimension ]
299  [ nearestLowerIndices[ currentDimension ] ] ) /
300  ( independentValues_[ currentDimension ]
301  [ nearestLowerIndices[ currentDimension ] + 1 ] -
302  independentValues_[ currentDimension ]
303  [ nearestLowerIndices[ currentDimension ] ] );
304  lowerFraction = -( independentValuesToInterpolate[ currentDimension ] -
305  independentValues_[ currentDimension ]
306  [ nearestLowerIndices[ currentDimension ] + 1 ] ) /
307  ( independentValues_[ currentDimension ]
308  [ nearestLowerIndices[ currentDimension ] + 1 ] -
309  independentValues_[ currentDimension ]
310  [ nearestLowerIndices[ currentDimension ] ] );
311 
312  // If at top dimension, call dependent variable data.
313  if ( currentDimension == NumberOfDimensions - 1 )
314  {
315  currentArrayIndices[ NumberOfDimensions - 1 ] = nearestLowerIndices[ currentDimension ];
316  lowerContribution = dependentData_( currentArrayIndices );
317  currentArrayIndices[ NumberOfDimensions - 1 ] = nearestLowerIndices[ currentDimension ]
318  + 1;
319  upperContribution = dependentData_( currentArrayIndices );
320  }
321 
322  // If at lower dimension, update currentArrayIndices and call function with
323  // currentDimension++.
324  else
325  {
326  currentArrayIndices[ currentDimension ] = nearestLowerIndices[ currentDimension ];
327  lowerContribution = performRecursiveInterpolationStep(
328  currentDimension + 1, independentValuesToInterpolate,
329  currentArrayIndices, nearestLowerIndices );
330  currentArrayIndices[ currentDimension ] = nearestLowerIndices[ currentDimension ] + 1;
331  upperContribution = performRecursiveInterpolationStep(
332  currentDimension + 1, independentValuesToInterpolate,
333  currentArrayIndices, nearestLowerIndices );
334  }
335 
336  // Return interpolated value.
337  DependentVariableType returnValue = upperFraction * upperContribution +
338  lowerFraction * lowerContribution;
339  return returnValue;
340  }
