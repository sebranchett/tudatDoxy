47 std::map< double, Eigen::Vector6d > performCR3BPIntegration(
48  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
49  const double massParameter,
50  const Eigen::Vector6d& initialState,
51  const double finalTime,
52  const bool propagateToExactFinalTime )
53 {
54  // Create integrator object
55  std::shared_ptr< numerical_integrators::NumericalIntegrator< double, Eigen::Vector6d > > integrator =
56  createCR3BPIntegrator( integratorSettings, massParameter, initialState );
57 
58  // Initialize return data map
59  std::map< double, Eigen::Vector6d > stateHistory;
60 
61  // Store initial state and time
62  double currentTime = integratorSettings->initialTime_;
63  double secondToLastTime = integratorSettings->initialTime_;
64 
65  Eigen::Vector6d currentState = initialState;
66  stateHistory[ currentTime ] = currentState;
67 
68  // Integrate to final time
69  double timeStep = integratorSettings->initialTimeStep_;
70  while( currentTime <= finalTime )
71  {
72  secondToLastTime = currentTime;
73  currentState = integrator->performIntegrationStep( timeStep );
74  currentTime = integrator->getCurrentIndependentVariable( );
75  timeStep = integrator->getNextStepSize( );
76  stateHistory[ currentTime ] = currentState;
77  }
78 
79  if( propagateToExactFinalTime )
80  {
81  // Determine final time step and propagate
82  double finalTimeStep = finalTime - secondToLastTime;
83 
84  stateHistory.erase( currentTime );
85  integrator->rollbackToPreviousState( );
86 
87  currentState = integrator->performIntegrationStep( finalTimeStep );
88  currentTime = integrator->getCurrentIndependentVariable( );
89  stateHistory[ currentTime ] = currentState;
90  }
91  return stateHistory;
92 
93 }
