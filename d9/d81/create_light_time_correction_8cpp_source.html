36  switch( correctionSettings->getCorrectionType( ) )
37  {
38  case first_order_relativistic:
39  {
40  // Check input consistency
41  if( std::dynamic_pointer_cast< FirstOrderRelativisticLightTimeCorrectionSettings >( correctionSettings ) != nullptr )
42  {
43  // Retrieve list of bodies causing light time perturbation
44  std::vector< std::string > perturbingBodies =
45  std::dynamic_pointer_cast< FirstOrderRelativisticLightTimeCorrectionSettings >( correctionSettings )->
46  getPerturbingBodies( );
47 
48  std::vector< std::function< Eigen::Vector6d( const double ) > > perturbingBodyStateFunctions;
49  std::vector< std::function< double( ) > > perturbingBodyGravitationalParameterFunctions;
50 
51  // Retrieve mass and state functions for each perturbing body.
52  for( unsigned int i = 0; i < perturbingBodies.size( ); i++ )
53  {
54  if( bodyMap.count( perturbingBodies[ i ] ) == 0 )
55  {
56  throw std::runtime_error(
57  "Error when making 1st order relativistic light time correction, could not find body " +
58  perturbingBodies.at( i ) );
59  }
60  else
61  {
62  // Set state function.
63  perturbingBodyStateFunctions.push_back(
64  std::bind( &simulation_setup::Body::getStateInBaseFrameFromEphemeris< double, double >,
65  bodyMap.at( perturbingBodies[ i ] ), std::placeholders::_1 ) );
66 
67  // Set gravitational parameter function.
68  perturbingBodyGravitationalParameterFunctions.push_back(
69  std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
70  bodyMap.at( perturbingBodies[ i ] )->
71  getGravityFieldModel( ) ) );
72  }
73  }
74 
75  // Create light-time correction function
76  lightTimeCorrection = std::make_shared< FirstOrderLightTimeCorrectionCalculator >(
77  perturbingBodyStateFunctions, perturbingBodyGravitationalParameterFunctions, perturbingBodies,
78  transmitter.first, receiver.first,
79  std::bind( &relativity::PPNParameterSet::getParameterGamma, relativity::ppnParameterSet ) );
80 
81  }
82  else
83  {
84  throw std::runtime_error(
85  "Error, correction settings type (1st order relativistic) does not coincide with data type." );
86  }
87

95 
96  break;
97  }

99  }
