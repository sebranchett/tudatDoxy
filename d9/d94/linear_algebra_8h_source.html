143 static inline void flipMatrixRows( Eigen::MatrixXd& matrixToFlip )
144 {
145  // Loop through each row.
146  for ( int i = 0; i < static_cast< int >( matrixToFlip.rows( ) / 2 ); i++ )
147  {
148  // Save top row in a temporary matrix.
149  const Eigen::MatrixXd temporaryRow = matrixToFlip.row( i );
150 
151  // Set top row to bottom row.
152  matrixToFlip.row( i ) = matrixToFlip.row( matrixToFlip.rows( ) - 1 - i );
153 
154  // Set bottom row to temporarily stored matrix.
155  matrixToFlip.row( matrixToFlip.rows( ) - 1 - i ) = temporaryRow;
156  }
157 }

185 template< typename StateScalarType, int NumberOfRows, int NumberOfColumns >
186 bool doesMatrixHaveNanEntries( const Eigen::Matrix< StateScalarType, NumberOfRows, NumberOfColumns > matrixToCheck )
187 {
188  bool areNanEntriesPresent = false;
189  for( int i = 0; i < matrixToCheck.rows( ); i++ )
190  {
191  for( int j = 0; j < matrixToCheck.cols( ); j++ )
192  {
193  if( !( matrixToCheck( i, j ) == matrixToCheck( i, j ) ) )
194  {
195  areNanEntriesPresent = true;
196  }
197  }
198  }
199  return areNanEntriesPresent;
200 }
