50 Eigen::Vector4d quaternionProduct( const Eigen::Vector4d& firstQuaternion, const Eigen::Vector4d& secondQuaternion )
51 {
52  Eigen::Vector4d resultantQuaternion;
53  resultantQuaternion[ 0 ] = firstQuaternion[ 0 ] * secondQuaternion[ 0 ] -
54  firstQuaternion.segment( 1, 3 ).dot( secondQuaternion.segment( 1, 3 ) );
55  resultantQuaternion.segment( 1, 3 ) = firstQuaternion[ 0 ] * secondQuaternion.segment( 1, 3 ) +
56  secondQuaternion[ 0 ] * firstQuaternion.segment( 1, 3 ) +
57  getCrossProductMatrix( firstQuaternion.segment( 1, 3 ) ) * secondQuaternion.segment( 1, 3 );
58  return resultantQuaternion;

81 double computeCosineOfAngleBetweenVectors( const Eigen::VectorXd& vector0,
82  const Eigen::VectorXd& vector1 )
83 {
84  if( !( vector0.size( ) == vector1.size( ) ) )
85  {
86  throw std::runtime_error( "Error when computing angle between vectors; size is incompatible" );
87  }
88 
89  // Get the cosine of the angle by dotting the normalized vectors.
90  double dotProductOfNormalizedVectors = vector0.normalized( ).dot( vector1.normalized( ) );
91 
92  // Explicitly define the extreme cases, which can give problems with the acos function.
93  if ( dotProductOfNormalizedVectors >= 1.0 )
94  {
95  return 1.0;
96  }
97 
98  else if ( dotProductOfNormalizedVectors <= -1.0 )
99  {
100  return -1.0;
101  }
102  // Determine the actual angle.
103  else
104  {
105  return dotProductOfNormalizedVectors;
106  }
107 }

110 double computeAngleBetweenVectors( const Eigen::VectorXd& vector0, const Eigen::VectorXd& vector1 )
111 {
112  // Determine the cosine of the angle by using another routine.
113  double dotProductOfNormalizedVectors = computeCosineOfAngleBetweenVectors( vector0, vector1 );
114 
115  // Return arccosine of the above, which is effectively the angle.
116  return std::acos( dotProductOfNormalizedVectors );
117 }

147  const Eigen::Vector3d& unnormalizedVector )
148 {
149  double normOfVector = unnormalizedVector.norm( );
150 
151  return ( Eigen::Matrix3d::Identity( ) / normOfVector - unnormalizedVector * unnormalizedVector.transpose( ) /
152  ( normOfVector * normOfVector * normOfVector ) ) * partialOfUnnormalizedVector;
