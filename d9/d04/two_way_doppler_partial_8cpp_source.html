21 void TwoWayDopplerScaling::update( const std::vector< Eigen::Vector6d >& linkEndStates,
22  const std::vector< double >& times,
23  const observation_models::LinkEndType fixedLinkEnd,
24  const Eigen::VectorXd currentObservation )
25 {
26  Eigen::Vector3d currentRangeVector;
27 
28  // Define lists of link end states and tines to be used for eah one-way link.
29  std::vector< Eigen::Vector6d > singleLinkEndStates;
30  singleLinkEndStates.resize( 2 );
31  std::vector< double > singleLinkTimes;
32  singleLinkTimes.resize( 2 );
33 
34  double finiteDifferenceTimeStep = 60.0;
35  double uplinkDoppler = TUDAT_NAN, downlinkDoppler = TUDAT_NAN;
36  double upperturbedDoppler = TUDAT_NAN, downperturbedDoppler = TUDAT_NAN;
37  double observationTime = TUDAT_NAN;
38  // Find index in link ends for fixed (reference) link ends
39  observation_models::LinkEndType referenceLinkEnd;
40  int fixedLinkEndIndex = observation_models::getNWayLinkIndexFromLinkEndType(
41  fixedLinkEnd, 3 );
42  {
43  singleLinkEndStates[ 0 ] = linkEndStates.at( 0 );
44  singleLinkEndStates[ 1 ] = linkEndStates.at( 1 );
45  singleLinkTimes[ 0 ] = times.at( 0 );
46  singleLinkTimes[ 1 ] = times.at( 1 );
47 
48  if( fixedLinkEndIndex == 0 )
49  {
50  referenceLinkEnd = observation_models::transmitter;
51  observationTime = times.at( 0 );
52  }
53  else
54  {
55  referenceLinkEnd = observation_models::receiver;
56  observationTime = times.at( 1 );
57  }
58 
59  upperturbedDoppler = uplinkDopplerModel_( observationTime + finiteDifferenceTimeStep, referenceLinkEnd );
60  downperturbedDoppler = uplinkDopplerModel_( observationTime - finiteDifferenceTimeStep, referenceLinkEnd );
61  uplinkOneWayDopplerTimeDerivative_ = ( upperturbedDoppler - downperturbedDoppler ) / ( 2.0 * finiteDifferenceTimeStep );
62  uplinkDoppler = uplinkDopplerModel_( observationTime, referenceLinkEnd );
63 
64  // Update current one-way range scaling
65  uplinkDopplerScaling_->update( singleLinkEndStates, singleLinkTimes, referenceLinkEnd );
66  uplinkRangeScaling_->update( singleLinkEndStates, singleLinkTimes, referenceLinkEnd );
67  }
68 
69  {
70  singleLinkEndStates[ 0 ] = linkEndStates.at( 2 );
71  singleLinkEndStates[ 1 ] = linkEndStates.at( 3 );
72  singleLinkTimes[ 0 ] = times.at( 2 );
73  singleLinkTimes[ 1 ] = times.at( 3 );
74 
75  if( fixedLinkEndIndex == 2 )
76  {
77  referenceLinkEnd = observation_models::receiver;
78  observationTime = times.at( 3 );
79  }
80  else
81  {
82  referenceLinkEnd = observation_models::transmitter;
83  observationTime = times.at( 2 );
84  }
85 
86  upperturbedDoppler = downlinkDopplerModel_( observationTime + finiteDifferenceTimeStep, referenceLinkEnd );
87  downperturbedDoppler = downlinkDopplerModel_( observationTime - finiteDifferenceTimeStep, referenceLinkEnd );
88  downlinkOneWayDopplerTimeDerivative_ = ( upperturbedDoppler - downperturbedDoppler ) / ( 2.0 * finiteDifferenceTimeStep );
89  downlinkDoppler = downlinkDopplerModel_( observationTime, referenceLinkEnd );
90 
91  // Update current one-way range scaling
92  downlinkDopplerScaling_->update( singleLinkEndStates, singleLinkTimes, referenceLinkEnd );
93  downlinkRangeScaling_->update( singleLinkEndStates, singleLinkTimes, referenceLinkEnd );
94  }
95 
96  projectedRelativeVelocityRatios_[ 0 ] = downlinkDoppler + 1.0;
97  projectedRelativeVelocityRatios_[ 1 ] = uplinkDoppler + 1.0;

102 TwoWayDopplerPartial::TwoWayDopplerPartialReturnType TwoWayDopplerPartial::calculatePartial(
103  const std::vector< Eigen::Vector6d >& states,
104  const std::vector< double >& times,
105  const observation_models::LinkEndType linkEndOfFixedTime,
106  const Eigen::Vector1d& currentObservation )
107 {
108  TwoWayDopplerPartialReturnType completePartialSet;
109  int referenceStartLinkEndIndex = getNWayLinkIndexFromLinkEndType( linkEndOfFixedTime, numberOfLinkEnds_ );
110 
111  // Define states, times and reference link ends to be used for constutuent one-way range partials
112  std::vector< Eigen::Vector6d > subLinkStates;
113  subLinkStates.resize( 2 );
114  std::vector< double > subLinkTimes;
115  subLinkTimes.resize( 2 );
116  observation_models::LinkEndType subLinkReference;
117 
118  double currentPartialMultiplier = TUDAT_NAN;
119 
120  for( dopplerPartialIterator_ = dopplerPartialList_.begin( ); dopplerPartialIterator_ != dopplerPartialList_.end( );
121  dopplerPartialIterator_++ )
122  {
123  TwoWayDopplerPartialReturnType currentPartialSet;
124 
125  // Set link end times and states for current one-way range
126  subLinkStates[ 0 ] = states[ 2 * dopplerPartialIterator_->first ];
127  subLinkStates[ 1 ] = states[ 2 * dopplerPartialIterator_->first + 1 ];
128  subLinkTimes[ 0 ] = times[ 2 * dopplerPartialIterator_->first ];
129  subLinkTimes[ 1 ] = times[ 2 * dopplerPartialIterator_->first + 1 ];
130 
131  // Compute value by which one-way range should be scaled for inclusion into n-way range
132  currentPartialMultiplier = twoWayDopplerScaler_->getProjectedRelativeVelocityRatio( dopplerPartialIterator_->first );
133 
134  if( dopplerPartialIterator_->first >= referenceStartLinkEndIndex )
135  {
136  subLinkReference = observation_models::transmitter;
137  }
138  else
139  {
140  subLinkReference = observation_models::receiver;
141  }
142 
143 
144  // Compute one-way range partials
145  currentPartialSet = dopplerPartialIterator_->second->calculatePartial( subLinkStates, subLinkTimes, subLinkReference );
146 
147  // Scale partials by required amount and add to return map.
148  for( unsigned int i = 0; i < currentPartialSet.size( ); i++ )
149  {
150  currentPartialSet[ i ].first *= currentPartialMultiplier;
151  }
152  completePartialSet.insert( completePartialSet.end( ), currentPartialSet.begin( ), currentPartialSet.end( ) );
153 
154 
155  if( rangePartialList_.count( dopplerPartialIterator_->first ) > 0 )
156  {
157  if( ( linkEndOfFixedTime == observation_models::transmitter && dopplerPartialIterator_->first == 1 )||
158  ( linkEndOfFixedTime == observation_models::receiver && dopplerPartialIterator_->first == 0 ) )
159  {
160  currentPartialSet = rangePartialList_.at( dopplerPartialIterator_->first )->calculatePartial(
161  subLinkStates, subLinkTimes, subLinkReference );
162 
163  for( unsigned int i = 0; i < currentPartialSet.size( ); i++ )
164  {
165  currentPartialSet[ i ].first *= ( currentPartialMultiplier ) *
166  twoWayDopplerScaler_->getRelevantOneWayDopplerTimePartial( linkEndOfFixedTime ) /
167  physical_constants::SPEED_OF_LIGHT;
168  }
169  completePartialSet.insert( completePartialSet.end( ), currentPartialSet.begin( ), currentPartialSet.end( ) );
170  }
171  }
172  }
173 
174  return completePartialSet;
175 }
