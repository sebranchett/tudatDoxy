36 NRLMSISE00Input nrlmsiseInputFunction( const double altitude, const double longitude,
37  const double latitude, const double time,
38  const tudat::input_output::solar_activity::SolarActivityDataMap& solarActivityMap,
39  const bool adjustSolarTime,
40  const double localSolarTime ) {
41  using namespace tudat::input_output::solar_activity;
42 
43  // Declare input data class member
44  NRLMSISE00Input nrlmsiseInputData;
45 
46  // Julian dates
47  double julianDate = tudat::basic_astrodynamics::convertSecondsSinceEpochToJulianDay(
48  time, basic_astrodynamics::JULIAN_DAY_ON_J2000 );
49  double julianDay = std::floor( julianDate - 0.5 ) + 0.5;
50 
51  // Check if solar activity is found for current day.
52  if( solarActivityMap.count( julianDay ) == 0 )
53  {
54  std::string errorMessage = "Solar activity data could not be found for this julian date: "
55  + std::to_string( julianDay ) + " in nrlmsiseInputFunction";
56  throw std::runtime_error( errorMessage );
57  }
58  SolarActivityDataPtr solarActivity = solarActivityMap.at( julianDay );
59 
60 
61  // Compute julian date at the first of januari
62  double julianDate1Jan = tudat::basic_astrodynamics::convertCalendarDateToJulianDay(
63  solarActivity->year, 1, 1, 0, 0, 0.0 );
64 
65  nrlmsiseInputData.year = solarActivity->year; // int
66  nrlmsiseInputData.dayOfTheYear = julianDay - julianDate1Jan + 1;
67  nrlmsiseInputData.secondOfTheDay = time -
68  tudat::basic_astrodynamics::convertJulianDayToSecondsSinceEpoch( julianDay,
69  tudat::basic_astrodynamics::JULIAN_DAY_ON_J2000 );
70 
71  if( solarActivity->fluxQualifier == 1 )
72  { // requires adjustment
73  nrlmsiseInputData.f107 = solarActivity->solarRadioFlux107Adjusted;
74  nrlmsiseInputData.f107a = solarActivity->centered81DaySolarRadioFlux107Adjusted;
75  }
76  else
77  { // no adjustment required
78  nrlmsiseInputData.f107 = solarActivity->solarRadioFlux107Observed;
79  nrlmsiseInputData.f107a = solarActivity->centered81DaySolarRadioFlux107Observed;
80  }
81  nrlmsiseInputData.apDaily = solarActivity->planetaryEquivalentAmplitudeAverage;
82  nrlmsiseInputData.apVector = eigenToStlVector( solarActivity->planetaryEquivalentAmplitudeVector );
83 
84  // Compute local solar time
85  // Hrs since begin of the day at longitude 0 (GMT) + Hrs passed at current longitude
86  if( adjustSolarTime )
87  {
88  nrlmsiseInputData.localSolarTime = localSolarTime;
89  }
90  else
91  {
92  nrlmsiseInputData.localSolarTime = nrlmsiseInputData.secondOfTheDay / 3600.0
93  + longitude / ( tudat::mathematical_constants::PI / 12.0 );
94  }
95 
96  return nrlmsiseInputData;
97 }
