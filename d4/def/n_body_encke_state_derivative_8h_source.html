35 template< typename StateScalarType = double >
36 StateScalarType calculateEnckeQFunction( const StateScalarType qValue )
37 {
38  StateScalarType powerTerm = mathematical_constants::getFloatingInteger< StateScalarType >( 1 ) +
39  mathematical_constants::getFloatingInteger< StateScalarType >( 2 ) * qValue;
40  return mathematical_constants::getFloatingInteger< StateScalarType >( 1 ) - 1.0 / ( powerTerm * std::sqrt( powerTerm ) );
41 }

127  void calculateSystemStateDerivative(
128  const TimeType time,
129  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
130  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
131  {
132  stateDerivative.setZero( );
133 
134  // Retrieve Keplerian orbit state for each body.
135  calculateKeplerTrajectoryCartesianStates( time );
136 
137  // Get Cartesian state derivative for all bodies of Encke state (excluding central gravitational accelerations).
138  this->sumStateDerivativeContributions(
139  stateOfSystemToBeIntegrated, stateDerivative, true );
140 
141  // Initialize Encke algorithm variables.
142  StateScalarType qValue = 0.0;
143  StateScalarType qFunction = 0.0;
144  StateScalarType keplerianRadius = 0.0;
145  Eigen::Matrix< StateScalarType, 3, 1 > positionPerturbation = Eigen::Matrix< StateScalarType, 3, 1 >::Zero( );
146 
147  // Update state derivative for each body.
148  for( unsigned int i = 0; i < this->bodiesToBeIntegratedNumerically_.size( ); i++ )
149  {
150  // Get position perturbation.
151  positionPerturbation = stateOfSystemToBeIntegrated.segment( i * 6, 3 );
152 
153  // Get distance from central body, assuming purely Keplerian orbit.
154  keplerianRadius = currentKeplerianOrbitCartesianState_[ i ].segment( 0, 3 ).norm( );
155 
156  // Calculate Encke algorithm variables.
157  qValue = positionPerturbation.dot( currentKeplerianOrbitCartesianState_[ i ].segment( 0, 3 ) +
158  0.5 * positionPerturbation ) / ( keplerianRadius * keplerianRadius );
159  qFunction = calculateEnckeQFunction( qValue );
160 
161  // Update state derivative with Encke term.
162  stateDerivative.block( i * 6 + 3, 0, 3, 1 ) += static_cast< StateScalarType >(
163  centralBodyGravitationalParameters_[ i ]( ) ) /
164  ( keplerianRadius * keplerianRadius * keplerianRadius ) * (
165  ( positionPerturbation + currentKeplerianOrbitCartesianState_[ i ].segment( 0, 3 ) ) * qFunction -
166  positionPerturbation );
167  }

169  }
