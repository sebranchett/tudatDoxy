265  void computeAccelerationPartialWrtCurrentDragCoefficient( Eigen::MatrixXd& accelerationPartial )
266  {
267  Eigen::Quaterniond rotationToInertialFrame =
268  flightConditions_->getAerodynamicAngleCalculator( )->getRotationQuaternionBetweenFrames(
269  reference_frames::aerodynamic_frame, reference_frames::inertial_frame );
270 
271  double currentAirspeed = flightConditions_->getCurrentAirspeed( );
272  accelerationPartial =
273  rotationToInertialFrame * Eigen::Vector3d::UnitX( ) * (
274  -0.5 * flightConditions_->getCurrentDensity( ) * currentAirspeed * currentAirspeed *
275  flightConditions_->getAerodynamicCoefficientInterface( )->getReferenceArea( ) ) /
276  aerodynamicAcceleration_->getCurrentMass( );
277 
278  }

286  void computeAccelerationPartialWrtArcwiseDragCoefficient(
287  Eigen::MatrixXd& accelerationPartial,
288  const std::shared_ptr< estimatable_parameters::ArcWiseConstantDragCoefficient > parameter )
289  {
290  // Get partial w.r.t. rdrag coefficient
291  Eigen::MatrixXd partialWrtSingleParameter = Eigen::Vector3d::Zero( );
292  this->computeAccelerationPartialWrtCurrentDragCoefficient( partialWrtSingleParameter );
293 
294  // Retrieve current arc
295  std::shared_ptr< interpolators::LookUpScheme< double > > currentArcIndexLookUp =
296  parameter->getArcTimeLookupScheme( );
297  int currentArc = currentArcIndexLookUp->findNearestLowerNeighbour( currentTime_ );
298 
299  if( currentArc >= accelerationPartial.cols( ) )
300  {
301  throw std::runtime_error( "Error when getting arc-wise radiation pressure coefficient partials, data not consistent" );
302  }
303 
304  // Set partial
305  accelerationPartial.setZero( 3, parameter->getNumberOfArcs( ) );
306  accelerationPartial.block( 0, currentArc, 3, 1 ) = partialWrtSingleParameter;
307  }
