39 Capsule::Capsule( const double noseRadius,
40  const double middleRadius,
41  const double rearLength,
42  const double rearAngle,
43  const double sideRadius )
44 {
45  using std::sin;
46  using std::cos;
47  using mathematical_constants::PI;
48 
49  // Call set functions for number of single and composite surface geometries
50  // with predetermined values.
51  setNumberOfCompositeSurfaceGeometries( 0 );
52  setNumberOfSingleSurfaceGeometries( 4 );
53 
54  // Set member shape variables
55  noseRadius_ = noseRadius;
56  middleRadius_ = middleRadius;
57  rearLength_ = rearLength;
58  rearAngle_ = rearAngle;
59  sideRadius_ = sideRadius;
60 
61  // Determine and set extent of spherical nose part.
62  double noseSphereAngle_ = asin( ( middleRadius_ - sideRadius_ )
63  / ( noseRadius_ - sideRadius_ ) );
64 
65  // Create nose sphere.
66  std::shared_ptr< SphereSegment > noseSphere_ = std::make_shared< SphereSegment >(
67  noseRadius_, 0, 2 * PI, 0, noseSphereAngle_ );
68 
69  // Declare translation vector.
70  Eigen::VectorXd translationVector_ = Eigen::VectorXd( 3 );
71  translationVector_( 2 ) = 0.0;
72  translationVector_( 1 ) = 0.0;
73  translationVector_( 0 ) = - noseRadius_ * cos( noseSphereAngle_ );
74 
75  // Set nose translation vector.
76  noseSphere_->setOffset( translationVector_ );
77 
78  // Set noseSphere_ in singleSurfaceList_.
79  setSingleSurfaceGeometry( noseSphere_, 0 );
80 
81  //Calculate noseSphere volume
82  double noseSphereHeight_ = noseRadius_ * (1.0 - cos( noseSphereAngle_ ));
83  double noseSphereVolume_ = (PI / 3.0) * pow(noseSphereHeight_, 2.0) * (
84  3.0 * noseRadius - noseSphereHeight_ );
85 
86  // Create rear cone, fully revolved.
87  std::shared_ptr< ConicalFrustum > cone_ = std::make_shared< ConicalFrustum >(
88  rearAngle_, middleRadius_ - sideRadius_ * ( 1.0 - cos( rearAngle_ ) ),
89  rearLength_ );
90 
91  // Set translation vector of cone.
92  translationVector_( 0 ) = -sideRadius_ * ( sin( PI / 2.0 - noseSphereAngle_ )
93  + sin ( -rearAngle_ ) );
94  cone_->setOffset( translationVector_ );
95 
96  // Set cone in singleSurfaceList_.
97  setSingleSurfaceGeometry( cone_, 1 );
98 
99  // Obtain start radius of cone.
100  double startRadius_ = cone_->getStartRadius( );
101 
102  // Calculate end radius of cone.
103  double endRadius_ = startRadius_ + rearLength_ * tan( rearAngle_ );
104 
105  // Calculate volume of cone.
106  double fullConeLength_ = rearLength_ + endRadius_ / tan( -rearAngle_ );
107  double coneVolume_ = (PI / 3.0) *
108  ( pow(startRadius_, 2.0) * fullConeLength_ -
109  pow(endRadius_, 2.0) * ( fullConeLength_ - rearLength_ ));
110 
111  // Calculate rear sphere radius.
112  double rearNoseRadius_ = endRadius_ / cos( -rearAngle_ );
113 
114  // Create rear sphere ( "end cap" ), fully revolved.
115  std::shared_ptr< SphereSegment > rearSphere_ = std::make_shared< SphereSegment >(
116  rearNoseRadius_, 0.0, 2.0 * PI, PI / 2.0 - rearAngle_, PI );
117 
118  // Set translation vector of rear sphere.
119  translationVector_( 0 ) = ( rearNoseRadius_ * sin( -rearAngle_ ) ) - rearLength_
120  - ( sideRadius_ * ( sin( PI / 2.0 - noseSphereAngle_ ) + sin ( -rearAngle_ ) ) );
121  rearSphere_->setOffset( translationVector_ );
122  setSingleSurfaceGeometry( rearSphere_, 2 );
123 
124  // Calcualte volume of rear sphere
125  double rearSphereHeight_ = rearNoseRadius_ * ( 1.0 - cos( (PI / 2.0) + rearAngle));
126  double rearSphereVolume_ = (PI / 3.0 ) * pow( rearSphereHeight_, 2.0 ) * (
127  3.0 * rearNoseRadius_ - rearSphereHeight_);
128 
129 
130  // Create torus section of capsule.
131  double torusMajorRadius_ = ( noseRadius_ - sideRadius_ ) * sin( noseSphereAngle_ );
132  std::shared_ptr< Torus > torus_ = std::make_shared< Torus >(
133  torusMajorRadius_, sideRadius_, 0.0, 2.0 * PI, PI / 2.0 - noseSphereAngle_, rearAngle_ );
134 
135  //Calculate torus volume
136  double minimumDiskRadius_ = middleRadius_ - sideRadius_;
137  double integrationLowerLimit_ = -1.0 * sideRadius_ * cos( noseSphereAngle_ );
138  double integrationUpperLimit_ = sideRadius_ * sin( -rearAngle_ );
139  double torusVolume_ = PI * (
140  pow( minimumDiskRadius_, 2.0 ) * (integrationUpperLimit_ - integrationLowerLimit_ )
141  + minimumDiskRadius_ * pow( sideRadius_, 2.0 ) * (
142  (asin( integrationUpperLimit_ / sideRadius_) + 0.5* sin( 2* asin(integrationUpperLimit_ / sideRadius_)))
143  - (asin( integrationLowerLimit_ / sideRadius_) + 0.5* sin( 2* asin(integrationLowerLimit_ / sideRadius_)))
144  )
145  + pow( sideRadius_, 2.0 ) * ( integrationUpperLimit_ - integrationLowerLimit_ )
146  - (pow( integrationUpperLimit_, 3.0 ) - pow( integrationLowerLimit_, 3.0 )) / 3.0
147  );
148 
149 
150  //Calculate total capsule volume
151  capsuleVolume_ = noseSphereVolume_ + coneVolume_ + rearSphereVolume_ + torusVolume_;
152 
153  //Calculate total capsule length
154  totalLength_ = noseSphereHeight_ - integrationLowerLimit_ + integrationUpperLimit_ +
155  rearLength_ + rearSphereHeight_;
156 
157  //Calculate frontal area
158  frontalArea_ = PI * pow( middleRadius_, 2.0 );
159 
160  // Set translation vector of rear sphere.
161  translationVector_( 0 ) = -cos( noseSphereAngle_ ) * sideRadius_;
162  torus_->setOffset( translationVector_ );
163  setSingleSurfaceGeometry( torus_, 3 );
164 
165  // Set rotation matrix fo each part to be compatible with flow direction in
166  // aerodynamic analysis.
167  Eigen::MatrixXd rotationMatrix = Eigen::MatrixXd( 3, 3 );
168  double angle_ = PI / 2.0;
169  rotationMatrix( 0, 0 ) = cos( angle_ );
170  rotationMatrix( 0, 1 ) = 0.0;
171  rotationMatrix( 0, 2 ) = sin( angle_ );
172  rotationMatrix( 1, 0 ) = 0.0;
173  rotationMatrix( 1, 1 ) = 1.0;
174  rotationMatrix( 1, 2 ) = 0.0;
175  rotationMatrix( 2, 0 ) = -sin( angle_ );
176  rotationMatrix( 2, 1 ) = 0.0;
177  rotationMatrix( 2, 2 ) = cos( angle_ );
178 
179  // Set rotation matrix for single surface geometries.
180  for ( unsigned i = 0; i < numberOfSingleSurfaceGeometries_ ; i++ )
181  {
182  singleSurfaceGeometryList_[ i ]->setRotationMatrix( rotationMatrix );
183  }
184 }
