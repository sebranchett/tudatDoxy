494 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
495 StateType
496 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
497 ::performIntegrationStep( const TimeStepType stepSize )
498 {
499  // Define and allocated vector for the number of stages.
500  currentStateDerivatives_.clear( );
501  currentStateDerivatives_.reserve( this->coefficients_.cCoefficients.rows( ) );
502 
503  // Define lower and higher order estimates.
504  StateType lowerOrderEstimate( this->currentState_ ), higherOrderEstimate( this->currentState_ );
505 
506  // Compute the k_i state derivatives per stage.
507  for ( int stage = 0; stage < this->coefficients_.cCoefficients.rows( ); stage++ )
508  {
509  // Compute the intermediate state to pass to the state derivative for this stage.
510  StateType intermediateState( this->currentState_ );
511 
512  // Compute the intermediate state.
513  for ( int column = 0; column < stage; column++ )
514  {
515  intermediateState += stepSize * this->coefficients_.aCoefficients( stage, column ) *
516  currentStateDerivatives_[ column ];
517  }
518 
519  // Compute the state derivative.
520  const IndependentVariableType time = this->currentIndependentVariable_ +
521  this->coefficients_.cCoefficients( stage ) * stepSize;
522  currentStateDerivatives_.push_back( this->stateDerivativeFunction_( time, intermediateState ) );
523 
524  // Check if propagation should terminate because the propagation termination condition has been reached
525  // while computing the intermediate state.
526  // If so, return immediately the current state (not recomputed yet), which will be discarded.
527  if ( this->propagationTerminationFunction_( static_cast< double >( time ), TUDAT_NAN ) )
528  {
529  this->propagationTerminationConditionReachedDuringStep_ = true;
530  return this->currentState_;
531  }
532 
533  // Update the estimate.
534  lowerOrderEstimate += this->coefficients_.bCoefficients( 0, stage ) * stepSize *
535  currentStateDerivatives_[ stage ];
536  higherOrderEstimate += this->coefficients_.bCoefficients( 1, stage ) * stepSize *
537  currentStateDerivatives_[ stage ];
538  }
539 
540  // Determine if the error was within bounds and compute a new step size.
541  if ( computeNextStepSizeAndValidateResult( lowerOrderEstimate,
542  higherOrderEstimate, stepSize ) )
543  {
544  // Accept the current step.
545  this->lastIndependentVariable_ = this->currentIndependentVariable_;
546  this->lastState_ = this->currentState_;
547  this->currentIndependentVariable_ += stepSize;
548 
549  switch ( this->coefficients_.orderEstimateToIntegrate )
550  {
551  case RungeKuttaCoefficients::lower:
552  this->currentState_ = lowerOrderEstimate;
553  return this->currentState_;
554 
555  case RungeKuttaCoefficients::higher:
556  this->currentState_ = higherOrderEstimate;
557  return this->currentState_;
558 
559  default: // The default case will never occur because OrderEstimateToIntegrate is an enum.
560  throw std::runtime_error( "Order estimate to integrate is invalid." );
561  }
562  }
563  else
564  {
565  // Reject current step.
566  return performIntegrationStep( this->stepSize_ );
567  }
568 }

571 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
572 bool
573 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
574 ::computeNextStepSizeAndValidateResult(
575  const StateType& lowerOrderEstimate, const StateType& higherOrderEstimate,
576  const TimeStepType stepSize )
577 {
578  if( useStepSizeControl_ )
579  {
580  // Compute new step size using new step size function, which also returns whether the
581  // relative error is within bounds or not.
582  std::pair< TimeStepType, bool > newStepSizePair = this->newStepSizeFunction_(
583  stepSize, std::make_pair( this->coefficients_.lowerOrder, this->coefficients_.higherOrder ),
584  this->safetyFactorForNextStepSize_, std::make_pair( this->minimumFactorDecreaseForNextStepSize_,
585  this->maximumFactorIncreaseForNextStepSize_ ),
586  this->relativeErrorTolerance_, this->absoluteErrorTolerance_, lowerOrderEstimate, higherOrderEstimate );
587 
588  // Check whether change in stepsize does not exceed bounds.
589  // If the stepsize is reduced to less than the prescibed minimum factor, set to minimum factor.
590  // If the stepsize is increased to more than the prescribed maximum factor, set to maximum
591  // factor. These bounds are necessary to prevent the stepsize changes from aliasing
592  // with the dynamics of the system of ODEs.
593  // Also check if maximum step size is exceeded and step next step size to maximum if necessary.
594  // Typically used bounds can be found in (Burden and Faires, 2001).
595  if ( newStepSizePair.first / stepSize <= this->minimumFactorDecreaseForNextStepSize_ )
596  {
597  this->stepSize_ = stepSize * this->minimumFactorDecreaseForNextStepSize_;
598  }
599 
600  else if ( newStepSizePair.first / stepSize >= maximumFactorIncreaseForNextStepSize_ )
601  {
602  this->stepSize_ = stepSize * this->maximumFactorIncreaseForNextStepSize_;
603  }
604 
605  else
606  {
607  this->stepSize_ = newStepSizePair.first;
608  }
609 
610  // Check if minimum step size is violated and throw exception if necessary.
611  if ( std::fabs( this->stepSize_ ) < std::fabs( this->minimumStepSize_ ) )
612  {
613  throw MinimumStepSizeExceededError( std::fabs( this->minimumStepSize_ ),
614  std::fabs( this->stepSize_ ) );
615  }
616  else if( std::fabs( this->stepSize_ ) > std::fabs( this->maximumStepSize_ ) )
617  {
618  this->stepSize_ = stepSize / std::fabs( stepSize ) * std::fabs( this->maximumStepSize_ );
619  }
620 
621  if( stepSize * this->stepSize_ < 0 )
622  {
623  throw std::runtime_error( "Error during step size control, step size flipped sign" );
624  }
625 
626  // Check if computed error in state is too large and reject step if true.
627  return newStepSizePair.second;
628  }
629  else
630  {
631  this->stepSize_ = stepSize;
632  return true;
633  }
634 }

640 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
641 std::pair< TimeStepType, bool >
642 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
643 ::computeNewStepSize(
644  const TimeStepType stepSize,
645  const std::pair< TimeStepType, TimeStepType >& orders,
646  const TimeStepType safetyFactorForNextStepSize,
647  const std::pair< TimeStepType, TimeStepType >& minimumAndMaximumFactorsForNextStepSize,
648  const StateType& relativeErrorTolerance,
649  const StateType& absoluteErrorTolerance,
650  const StateType& lowerOrderEstimate,
651  const StateType& higherOrderEstimate )
652 {
653  TUDAT_UNUSED_PARAMETER( minimumAndMaximumFactorsForNextStepSize );
654 
655  // Compute the truncation error based on the higher and lower order estimates.
656  const StateType truncationError_ =
657  ( higherOrderEstimate - lowerOrderEstimate ).array( ).abs( );
658 
659  // Compute error tolerance based on relative and absolute error tolerances.
660  const StateType errorTolerance_ =
661  ( higherOrderEstimate.array( ).abs( ) *
662  relativeErrorTolerance.array( ) ).matrix( )
663  + absoluteErrorTolerance;
664 
665  // Compute relative truncation error. This will indicate if the current step satisfies the
666  // required tolerances.
667  const StateType relativeTruncationError_ = truncationError_.array( ) /
668  errorTolerance_.array( );
669 
670  // Compute the maximum error based on the largest coefficient in the relative truncation error
671  // matrix.
672  const typename StateType::Scalar maximumErrorInState_
673  = relativeTruncationError_.array( ).abs( ).maxCoeff( );
674 
675  // Compute the new step size. This is based off of the equation given in
676  // (Montenbruck and Gill, 2005).
677  const TimeStepType newStepSize = safetyFactorForNextStepSize * stepSize
678  * std::pow( 1.0 / maximumErrorInState_, 1.0 / orders.second );
679 
680  // Check if the current state can be accepted.
681  const bool isIntegrationStepAccepted = maximumErrorInState_ <= 1.0;
682 
683  // Return the computed new step size and whether the current step taken is acceptable. If it
684  // isn't, the step will be recomputed with the computed new step size.
685  return std::make_pair( newStepSize, isIntegrationStepAccepted );
686 }
