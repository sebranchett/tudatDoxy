440  {

442  Eigen::Matrix< ParameterScalar, Eigen::Dynamic, 1 >::Zero( totalParameterSetSize_ );
443 
444  int currentStartIndex = 0;
445 
446  // Retrieve initial state parameter values.
447  for( unsigned int i = 0; i < estimateInitialStateParameters_.size( ); i++ )
448  {
449  parameterValues.segment( currentStartIndex, estimateInitialStateParameters_[ i ]->getParameterSize( ) ) =
450  estimateInitialStateParameters_[ i ]->getParameterValue( ).template cast< ParameterScalar >( );
451  currentStartIndex += estimateInitialStateParameters_[ i ]->getParameterSize( );
452  }
453 
454  // Retrieve double parameter values.

480  {
481  // Check input consistency
482  if( newParameterValues.rows( ) != totalParameterSetSize_ )
483  {
484  throw std::runtime_error( "Error when resetting parameters of parameter set, given vector has size " +
485  std::to_string( newParameterValues.rows( ) ) +
486  ", while internal size is " + std::to_string( totalParameterSetSize_ ) );
487  }
488  else
489  {
490  int currentStartIndex = 0;
491 
492  for( unsigned int i = 0; i < estimateInitialStateParameters_.size( ); i++ )
493  {
494  estimateInitialStateParameters_[ i ]->setParameterValue(
495  newParameterValues.segment( currentStartIndex, estimateInitialStateParameters_[ i ]->getParameterSize( ) ).
496  template cast< InitialStateParameterType >( ) );
497  currentStartIndex += estimateInitialStateParameters_[ i ]->getParameterSize( );
498  }
499 
500  // Set double parameter values.
501  for( unsigned int i = 0; i < estimatedDoubleParameters_.size( ); i++ )
502  {
503  estimatedDoubleParameters_[ i ]->setParameterValue( static_cast< double >( newParameterValues( currentStartIndex ) ) );
504  currentStartIndex++;
505  }
506 
507  // Set vector parameter values.
508 
509  for( unsigned int i = 0; i < estimatedVectorParameters_.size( ); i++ )
510  {
511  estimatedVectorParameters_[ i ]->setParameterValue(
512  newParameterValues.segment( currentStartIndex, estimatedVectorParameters_[ i ]->getParameterSize( ) ).
513  template cast< double >( ) );

676  void getConstraints( Eigen::MatrixXd& constraintStateMultiplier, Eigen::VectorXd& constraintRightHandSide )
677  {
678  // Resize constraint elements
679  constraintStateMultiplier.setZero( totalConstraintSize_, estimatedParameterSetSize_ );
680  constraintRightHandSide.setZero( totalConstraintSize_, 1 );
681 
682  // Iterate over all state parameters
683  int currentConstraintRow = 0;
684  int currentConstraintSize = 0;
685  for( auto parameterIterator = initialStateParameters_.begin( ); parameterIterator != initialStateParameters_.end( );
686  parameterIterator++ )
687  {
688  // Add constraint if of non-zero size
689  currentConstraintSize = parameterIterator->second->getConstraintSize( );
690  if( currentConstraintSize > 0 )
691  {
692  constraintStateMultiplier.block(
693  currentConstraintRow, parameterIterator->first, currentConstraintSize,
694  parameterIterator->second->getParameterSize( ) ) =
695  parameterIterator->second->getConstraintStateMultipler( );
696  constraintRightHandSide.segment( currentConstraintRow, currentConstraintSize ) =
697  parameterIterator->second->getConstraintRightHandSide( );
698 
699  currentConstraintRow += currentConstraintSize;
