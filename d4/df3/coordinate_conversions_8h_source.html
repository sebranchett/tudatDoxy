49 template< typename StateScalarType = double >
50 Eigen::Matrix< StateScalarType, 3, 1 > convertSphericalToCartesian(
51  const Eigen::Matrix< StateScalarType, 3, 1 >& sphericalCoordinates )
52 {
53  // Create local variables.
54  double radius_ = sphericalCoordinates( 0 );
55  double zenithAngle_ = sphericalCoordinates( 1 );
56  double azimuthAngle_ = sphericalCoordinates( 2 );
57 
58  // Declaring sine which has multiple usages to save computation time.
59  double sineOfZenithAngle_ = std::sin( sphericalCoordinates( 1 ) );
60 
61  // Create output Vector
62  Eigen::Matrix< StateScalarType, 3, 1 > convertedCartesianCoordinates_ = Eigen::Matrix< StateScalarType, 3, 1 >::Zero( 3 );
63 
64  // Perform transformation.
65  convertedCartesianCoordinates_( 0 ) = radius_ * std::cos( azimuthAngle_ ) * sineOfZenithAngle_;
66  convertedCartesianCoordinates_( 1 ) = radius_ * std::sin( azimuthAngle_ ) * sineOfZenithAngle_;
67  convertedCartesianCoordinates_( 2 ) = radius_ * std::cos( zenithAngle_ );
68 
69  return convertedCartesianCoordinates_;
70 }

87 template< typename ScalarType = double >
88 Eigen::Matrix< ScalarType, 3, 1 > convertCartesianToSpherical( const Eigen::Matrix< ScalarType, 3, 1 > & cartesianCoordinates )
89 
90 {
91  // Create output Vector3d.
92  Eigen::Matrix< ScalarType, 3, 1 > convertedSphericalCoordinates_ = Eigen::Matrix< ScalarType, 3, 1 >::Zero( );
93 
94  // Compute transformation of Cartesian coordinates to spherical coordinates.
95  convertedSphericalCoordinates_( 0 ) = cartesianCoordinates.norm( );
96 
97  // Check if coordinates are at origin.
98  if ( convertedSphericalCoordinates_( 0 ) < std::numeric_limits< ScalarType >::epsilon( ) )
99  {
100  convertedSphericalCoordinates_( 1 ) = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
101  convertedSphericalCoordinates_( 2 ) = mathematical_constants::getFloatingInteger< ScalarType >( 0 );
102  }
103  // Else compute coordinates using trigonometric relationships.
104  else
105  {
106  convertedSphericalCoordinates_( 1 ) = std::acos( cartesianCoordinates( 2 )
107  / convertedSphericalCoordinates_( 0 ) );
108  convertedSphericalCoordinates_( 2 ) = std::atan2( cartesianCoordinates( 1 ),
109  cartesianCoordinates( 0 ) );
110  }
111 
112  return convertedSphericalCoordinates_;
113 }
