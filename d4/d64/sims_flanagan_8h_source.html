51  SimsFlanagan(
52  const Eigen::Vector6d& stateAtDeparture,
53  const Eigen::Vector6d& stateAtArrival,
54  const double centralBodyGravitationalParameter,
55  const double initialSpacecraftMass,
56  const double maximumThrust,
57  const std::function< double ( const double ) > specificImpulseFunction,
58  const int numberSegments,
59  const double timeOfFlight,
60  std::shared_ptr< simulation_setup::OptimisationSettings > optimisationSettings ) :
61  LowThrustLeg( stateAtDeparture, stateAtArrival, timeOfFlight, initialSpacecraftMass, true ),
62  centralBodyGravitationalParameter_( centralBodyGravitationalParameter ),
63  maximumThrust_( maximumThrust ),
64  specificImpulseFunction_( specificImpulseFunction ),
65  numberSegments_( numberSegments ),
66  optimisationSettings_( optimisationSettings )
67  {
68  // Calculate number of segments for both the forward propagation (from departure to match point)
69  // and the backward propagation (from arrival to match point).
70  numberSegmentsForwardPropagation_ = ( numberSegments_ + 1 ) / 2;
71  numberSegmentsBackwardPropagation_ = numberSegments_ / 2;
72 
73  // Convert the thrust model proposed as initial guess into simplified thrust model adapted to the Sims-Flanagan method.
74  if ( optimisationSettings_->initialGuessThrustModel_.first.size( ) != 0 )
75  {
76  if ( optimisationSettings_->initialGuessThrustModel_.first.size( ) !=
77  static_cast< unsigned int >( 3 * numberSegments ) )
78  {
79  throw std::runtime_error(
80  "Error when providing an initial guess for Sims-Flanagan, size of the thrust model initial guess unconsistent with number of segments" );
81  }
82  else
83  {
84  initialGuessThrustModel_.first = optimisationSettings_->initialGuessThrustModel_.first;
85  }
86 
87  }
88  else
89  {
90  initialGuessThrustModel_.first = std::vector< double >( );
91  }
92  initialGuessThrustModel_.second = optimisationSettings_->initialGuessThrustModel_.second;
93 
94  // Perform optimisation
95  std::pair< std::vector< double >, std::vector< double > > bestIndividual = performOptimisation( );
96  championFitness_ = bestIndividual.first;
97  championDesignVariables_ = bestIndividual.second;
98 
99  // Transform best design variables into vector of throttles.
100  std::vector< Eigen::Vector3d > throttles;
101  for ( int i = 0 ; i < numberSegments ; i++ )
102  {
103  throttles.push_back( ( Eigen::Vector3d( ) << championDesignVariables_[ i * 3 ], championDesignVariables_[ i * 3 + 1 ],
104  championDesignVariables_[ i * 3 + 2 ] ).finished( ) );
105  }
106 
107  // Create Sims-Flanagan leg from the best optimisation individual.
108  simsFlanaganModel_ = std::make_shared< SimsFlanaganModel >(
109  stateAtDeparture_, stateAtArrival_,
110  centralBodyGravitationalParameter_, initialMass_, maximumThrust_, specificImpulseFunction_,
111  timeOfFlight_, throttles );
112 
113  }
