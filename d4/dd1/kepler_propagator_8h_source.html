64 template< typename ScalarType = double >
65 Eigen::Matrix< ScalarType, 6, 1 > propagateKeplerOrbit(
66  const Eigen::Matrix< ScalarType, 6, 1 >& initialStateInKeplerianElements,
67  const ScalarType propagationTime,
68  const ScalarType centralBodyGravitationalParameter,
69  std::shared_ptr< root_finders::RootFinderCore< ScalarType > > aRootFinder =
70  std::shared_ptr< root_finders::RootFinderCore< ScalarType > >( ) )
71 {
72  // Create final state in Keplerian elements.
73  Eigen::Matrix< ScalarType, 6, 1 > finalStateInKeplerianElements =
74  initialStateInKeplerianElements;
75 
76  // Check if eccentricity is valid.
77  if ( initialStateInKeplerianElements( eccentricityIndex ) <
78  mathematical_constants::getFloatingInteger< ScalarType >( 0 ) )
79  {
80  throw std::runtime_error( "Eccentricity is invalid (smaller than 0)." );
81  }
82 
83  // Check if orbit is elliptical.
84  else if ( initialStateInKeplerianElements( eccentricityIndex ) <
85  mathematical_constants::getFloatingInteger< ScalarType >( 1 ) )
86  {
87  // Convert initial true anomaly to eccentric anomaly.
88  const ScalarType initialEccentricAnomaly =
89  convertTrueAnomalyToEccentricAnomaly< ScalarType >(
90  initialStateInKeplerianElements( trueAnomalyIndex ),
91  initialStateInKeplerianElements( eccentricityIndex ) );
92 
93  // Convert initial eccentric anomaly to mean anomaly.
94  const ScalarType initialMeanAnomaly =
95  convertEccentricAnomalyToMeanAnomaly< ScalarType >(
96  initialEccentricAnomaly,
97  initialStateInKeplerianElements( eccentricityIndex ) );
98 
99  // Compute change of mean anomaly between start and end of propagation.
100  const ScalarType meanAnomalyChange =
101  convertElapsedTimeToEllipticalMeanAnomalyChange< ScalarType >(
102  propagationTime, centralBodyGravitationalParameter,
103  initialStateInKeplerianElements( semiMajorAxisIndex ) );
104 
105  // Compute eccentric anomaly for mean anomaly.
106  const ScalarType finalEccentricAnomaly =
107  convertMeanAnomalyToEccentricAnomaly< ScalarType >(
108  initialStateInKeplerianElements( eccentricityIndex ),
109  initialMeanAnomaly + meanAnomalyChange, true,
110  TUDAT_NAN, aRootFinder );
111 
112  // Compute true anomaly for computed eccentric anomaly.
113  finalStateInKeplerianElements( trueAnomalyIndex ) =
114  convertEccentricAnomalyToTrueAnomaly< ScalarType >(
115  finalEccentricAnomaly, finalStateInKeplerianElements( eccentricityIndex ) );
116  }
117 
118  // Check if the orbit is hyperbolic.
119  else if ( initialStateInKeplerianElements( eccentricityIndex ) >
120  mathematical_constants::getFloatingInteger< ScalarType >( 1 ))
121  {
122  // Convert initial true anomaly to hyperbolic eccentric anomaly.
123  const ScalarType initialHyperbolicEccentricAnomaly =
124  convertTrueAnomalyToHyperbolicEccentricAnomaly(
125  initialStateInKeplerianElements( trueAnomalyIndex ),
126  initialStateInKeplerianElements( eccentricityIndex ) );
127 
128  // Convert initial hyperbolic eccentric anomaly to the hyperbolic mean anomaly.
129  const ScalarType initialHyperbolicMeanAnomaly =
130  convertHyperbolicEccentricAnomalyToMeanAnomaly(
131  initialHyperbolicEccentricAnomaly,
132  initialStateInKeplerianElements( eccentricityIndex ) );
133 
134  // Compute change of mean anomaly because of the propagation time.
135  const ScalarType hyperbolicMeanAnomalyChange =
136  convertElapsedTimeToHyperbolicMeanAnomalyChange(
137  propagationTime, centralBodyGravitationalParameter,
138  initialStateInKeplerianElements( semiMajorAxisIndex ) );
139 
140  // Compute hyperbolic eccentric anomaly for mean anomaly.
141  const ScalarType finalHyperbolicEccentricAnomaly =
142  convertMeanAnomalyToHyperbolicEccentricAnomaly(
143  initialStateInKeplerianElements( eccentricityIndex ),
144  initialHyperbolicMeanAnomaly + hyperbolicMeanAnomalyChange );
145 
146  // Compute true anomaly for computed hyperbolic eccentric anomaly.
147  finalStateInKeplerianElements( trueAnomalyIndex ) =
148  convertHyperbolicEccentricAnomalyToTrueAnomaly(
149  finalHyperbolicEccentricAnomaly,
150  finalStateInKeplerianElements( eccentricityIndex ) );
151  }
152 
153  // In this case the eccentricity has to be 1.0, hence the orbit is parabolic.
154  else
155  {
156  throw std::runtime_error( "Parabolic orbits are not (yet) supported." );
157  }
158 
159  return finalStateInKeplerianElements;
160 }
