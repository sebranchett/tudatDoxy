20 std::vector< std::function< double( ) > > removeCentralGravityAccelerations(
21  const std::vector< std::string >& centralBodies, const std::vector< std::string >& bodiesToIntegrate,
22  basic_astrodynamics::AccelerationMap& accelerationModelsPerBody )
23 {
24  using namespace basic_astrodynamics;
25  using namespace gravitation;
26 
27  std::vector< std::function< double( ) > > centralBodyGravitationalParameters;
28  centralBodyGravitationalParameters.resize( bodiesToIntegrate.size( ) );
29 
30  std::vector< std::shared_ptr< AccelerationModel< Eigen::Vector3d > > > listOfAccelerations;
31 
32  // Iterate over all central bodies.
33  for( unsigned int i = 0; i < centralBodies.size( ); i++ )
34  {
35  // Check if current central body is exerting any accelerations on current body.
36  if( accelerationModelsPerBody[ bodiesToIntegrate.at( i ) ].count( centralBodies.at( i ) ) == 0 )
37  {
38  std::string errorMessage =
39  "Error, cannot remove central point gravity of body " + bodiesToIntegrate.at( i ) +
40  " with central body " + centralBodies.at( i ) + " no accelerations due to requested central body found.";
41  throw std::runtime_error( errorMessage );
42  }
43  else
44  {
45  // Find central acceleration to central body, i.e. iterate over list of accelerations and find
46  // central acceleration candiates (should be 1)
47  int lastCandidate = -1;
48  int numberOfCandidates = 0;
49  bool isLastCandidateSphericalHarmonic = 0;
50  listOfAccelerations =
51  accelerationModelsPerBody[ bodiesToIntegrate.at( i ) ][ centralBodies.at( i ) ];
52 
53  for( unsigned int j = 0; j < listOfAccelerations.size( ); j++ )
54  {
55  // Get type of current acceleration.
56  AvailableAcceleration currentAccelerationType = getAccelerationModelType( listOfAccelerations[ j ] );
57 
58  // If central gravity, set as central acceleration candidate.
59  if( currentAccelerationType == central_gravity )
60  {
61  numberOfCandidates++;
62  lastCandidate = j;
63  }
64  else if( currentAccelerationType == spherical_harmonic_gravity )
65  {
66  isLastCandidateSphericalHarmonic = 1;
67  numberOfCandidates++;
68  lastCandidate = j;
69  }
70  else if( ( currentAccelerationType == third_body_central_gravity ) ||
71  ( currentAccelerationType == third_body_spherical_harmonic_gravity ) )
72  {
73  std::string errorMessage =
74  "Error when removing central body point gravity term, removal of 3rd body accelerations (of " +
75  centralBodies.at( i ) +
76  " on " + bodiesToIntegrate.at( i ) + ",) not yet supported";
77  throw std::runtime_error( errorMessage );
78  }
79  }
80 
81  // If no or multiple central acceleration candidates were found, give error.
82  if( numberOfCandidates == 0 )
83  {
84  std::string errorMessage =
85  "Error when removing central body point gravity term on body " +
86  bodiesToIntegrate.at( i ) +
87  " with central body " + centralBodies.at( i ) + ", no central gravity found.";
88  throw std::runtime_error( errorMessage );
89 
90  }
91  else if( numberOfCandidates != 1 )
92  {
93  std::string errorMessage =
94  "Error when removing central body point gravity term on body " +
95  bodiesToIntegrate.at( i ) +
96  " with central body " + centralBodies.at( i ) + ", multiple central gravities found.";
97  throw std::runtime_error( errorMessage );
98  }
99  else
100  {
101  if( !isLastCandidateSphericalHarmonic )
102  {
103  // Set central body gravitational parameter (used for Kepler orbit propagation)
104  centralBodyGravitationalParameters.at( i ) =
105  std::dynamic_pointer_cast< CentralGravitationalAccelerationModel3d >(
106  listOfAccelerations.at( lastCandidate ) )->getGravitationalParameterFunction( );
107 
108  // Remove central acceleration from list of accelerations that are evaluated at each time step.
109  listOfAccelerations.erase( listOfAccelerations.begin( ) + lastCandidate,
110  listOfAccelerations.begin( ) + lastCandidate + 1 );
111  accelerationModelsPerBody[ bodiesToIntegrate.at( i ) ][ centralBodies.at( i ) ] =
112  listOfAccelerations;
113  }
114  else
115  {
116  std::shared_ptr< SphericalHarmonicsGravitationalAccelerationModel > originalAcceleration =
117  std::dynamic_pointer_cast< SphericalHarmonicsGravitationalAccelerationModel >(
118  listOfAccelerations.at( lastCandidate ) );
119  centralBodyGravitationalParameters.at( i ) = originalAcceleration->getGravitationalParameterFunction( );
120 
121  // Create 'additional' acceleration model which subtracts central gravity term.
122  accelerationModelsPerBody[ bodiesToIntegrate.at( i ) ][ centralBodies.at( i ) ].push_back(
123  std::make_shared< CentralGravitationalAccelerationModel3d >
124  ( originalAcceleration->getStateFunctionOfBodyExertingAcceleration( ),
125  originalAcceleration->getGravitationalParameterFunction( ),
126  originalAcceleration->getStateFunctionOfBodyUndergoingAcceleration( ) ) );
127  }
128  }
129  }
130  }
131  return centralBodyGravitationalParameters;
132 }

135 std::vector< std::string > determineEphemerisUpdateorder( std::vector< std::string > integratedBodies,
136  std::vector< std::string > centralBodies,
137  std::vector< std::string > ephemerisOrigins )
138 {
139 
140  std::vector< std::string > updateOrder;
141 
142  // Declare variables
143  bool isFinished = 0;
144  int currentIndex = 0;
145  int counter = 0;
146  std::vector< std::string >::const_iterator centralBodyIterator;
147  std::vector< std::string >::const_iterator ephemerisOriginIterator;
148 
149  // Continue iterating until all integratedBodies have been handled.
150  while( !isFinished )
151  {
152  // Check if current central body or ephemeris origin is integratedBodies
153  centralBodyIterator = std::find(
154  integratedBodies.begin( ), integratedBodies.end( ),
155  centralBodies.at( currentIndex ) );
156  ephemerisOriginIterator = std::find(
157  integratedBodies.begin( ), integratedBodies.end( ),
158  ephemerisOrigins.at( currentIndex ) );
159 
160  // If neither is in list, there is no dependency, and current body can be added to update
161  // list.
162  if( centralBodyIterator == integratedBodies.end( )
163  && ephemerisOriginIterator == integratedBodies.end( ) )
164  {
165  // Add to list.
166  updateOrder.push_back( integratedBodies.at( currentIndex ) );
167 
168  // Remove from list of handled bodies.
169  integratedBodies.erase( integratedBodies.begin( ) + currentIndex );
170  centralBodies.erase( centralBodies.begin( ) + currentIndex );
171  ephemerisOrigins.erase( ephemerisOrigins.begin( ) + currentIndex );
172 
173  // Handle first entry at next iteration.
174  currentIndex = 0;
175 
176  // Check if any bodies left.
177  if( integratedBodies.size( ) == 0 )
178  {
179  isFinished = 1;
180  }
181  }
182  else
183  {
184  // If both are found in list, start next iteration at whichever is first in list.
185  if( centralBodyIterator != integratedBodies.end( )
186  && ephemerisOriginIterator != integratedBodies.end( ) )
187  {
188 
189  currentIndex = std::min(
190  std::distance(
191  integratedBodies.begin( ),
192  std::find( integratedBodies.begin( ), integratedBodies.end( ),
193  centralBodies.at( currentIndex ) ) ),
194  std::distance(
195  integratedBodies.begin( ),
196  std::find( integratedBodies.begin( ), integratedBodies.end( ),
197  ephemerisOrigins.at( currentIndex ) ) ));
198 
199  }
200  // If only central body is found, start with this central body in next iteration
201  else if( centralBodyIterator != integratedBodies.end( ) )
202  {
203  currentIndex = std::distance(
204  integratedBodies.begin( ),
205  std::find( integratedBodies.begin( ), integratedBodies.end( ),
206  centralBodies.at( currentIndex ) ) );
207  }
208  // If only ephemeris origin is found, start with this ephemeris origin in next iteration
209  else if( ephemerisOriginIterator != integratedBodies.end( ) )
210  {
211  currentIndex = std::distance(
212  integratedBodies.begin( ),
213  std::find( integratedBodies.begin( ), integratedBodies.end( ),
214  ephemerisOrigins.at( currentIndex ) ) );
215  }
216  }
217 
218  // Check to break circular dependency that occurs for inadmissible input data.
219  counter++;
220  if( counter > 10000 )
221  {
222  throw std::runtime_error( "Warning, ephemeris update order determination now at iteration " +
223  std::to_string( counter ) );
224  }
225  }
226 
227  return updateOrder;
228 }
