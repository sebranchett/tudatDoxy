22 Eigen::Matrix< double, 3, 4 > getPartialDerivativeOfSphericalHarmonicGravitationalTorqueWrtQuaternion(
23  const Eigen::Matrix3d& bodyFixedRelativePositionCrossProductMatrix,
24  const Eigen::Matrix3d& bodyFixedPotentialGradientPositionPartial,
25  const Eigen::Matrix3d& bodyFixedPotentialGradientCrossProductMatrix,
26  const Eigen::Vector3d& inertialRelativePosition,
27  const std::vector< Eigen::Matrix3d > derivativeOfRotationMatrixWrtQuaternions )
28 {
29  Eigen::Matrix< double, 3, 4 > partialDerivative = Eigen::Matrix< double, 3, 4 >::Zero( );
30  for( unsigned int i = 0; i < derivativeOfRotationMatrixWrtQuaternions.size( ); i++ )
31  {
32  partialDerivative.block( 0, i, 3, 1 ) =
33  derivativeOfRotationMatrixWrtQuaternions.at( i ).transpose( ) * inertialRelativePosition;
34  }
35  return ( bodyFixedRelativePositionCrossProductMatrix * bodyFixedPotentialGradientPositionPartial -
36  bodyFixedPotentialGradientCrossProductMatrix ) * partialDerivative;
37 }

102 void SphericalHarmonicGravitationalTorquePartial::update( const double currentTime )
103 {
104  if( !( currentTime_ == currentTime ) )
105  {
106  torqueModel_->updateMembers( currentTime );
107  accelerationPartial_->update( currentTime );
108 
109  currentRotationToBodyFixedFrame_ =
110  torqueModel_->getSphericalHarmonicAcceleration( )->getCurrentRotationToIntegrationFrameMatrix( ).transpose( );
111 
112  currentBodyFixedRelativePosition_ = torqueModel_->getSphericalHarmonicAcceleration( )->getCurrentRelativePosition( );
113  currentBodyFixedRelativePositionCrossProductMatrix_ = linear_algebra::getCrossProductMatrix(
114  currentBodyFixedRelativePosition_ );
115  currentBodyFixedPotentialGradient_ = torqueModel_->getSphericalHarmonicAcceleration( )->getAccelerationInBodyFixedFrame( );
116  currentBodyFixedPotentialGradientCrossProductMatrix_ = linear_algebra::getCrossProductMatrix(
117  currentBodyFixedPotentialGradient_ );
118 
119  currentQuaternionVector_ = linear_algebra::convertQuaternionToVectorFormat(
120  Eigen::Quaterniond( currentRotationToBodyFixedFrame_.transpose( ) ) );
121  linear_algebra::computePartialDerivativeOfRotationMatrixWrtQuaternion(
122  currentQuaternionVector_, currentRotationMatrixDerivativesWrtQuaternion_ );
123 
124  currentPartialDerivativeWrtQuaternion_ =
125  getPartialDerivativeOfSphericalHarmonicGravitationalTorqueWrtQuaternion(
126  currentBodyFixedRelativePositionCrossProductMatrix_,
127  accelerationPartial_->getCurrentBodyFixedPartialWrtPosition( ),
128  currentBodyFixedPotentialGradientCrossProductMatrix_,
129  torqueModel_->getSphericalHarmonicAcceleration( )->getCurrentInertialRelativePosition( ),
130  currentRotationMatrixDerivativesWrtQuaternion_ );
131 
132  currentParameterPartialPreMultiplier_ = currentBodyFixedRelativePositionCrossProductMatrix_ *
