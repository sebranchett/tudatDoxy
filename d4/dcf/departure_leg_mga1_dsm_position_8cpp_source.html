139 void DepartureLegMga1DsmPosition::calculateDsmLocation( )
140 {
141  // Calculate the required unit vectors
142  const Eigen::Vector3d unitVector1 = departureBodyPosition_ / departureBodyPosition_.norm( ) ;
143  const Eigen::Vector3d unitVector3 = unitVector1.cross( departureBodyVelocity_ ) /
144  ( unitVector1.cross( departureBodyVelocity_ ) ).norm( );
145  const Eigen::Vector3d unitVector2 = unitVector3.cross( unitVector1 );
146 
147  // Calculate the absolute DSM radius.
148  const double absoluteRadiusDsm = dimensionlessRadiusDsm_ * departureBodyPosition_.norm( );
149 
150  // Calculate the radius in the central body reference frame.
151  dsmLocation_ = std::cos( inPlaneAngle_ ) * std::cos( outOfPlaneAngle_ ) * absoluteRadiusDsm *
152  unitVector1 +
153  std::sin( inPlaneAngle_ ) * std::cos( outOfPlaneAngle_ ) * absoluteRadiusDsm *
154  unitVector2 +
155  std::sin( outOfPlaneAngle_ ) * absoluteRadiusDsm * unitVector3;
