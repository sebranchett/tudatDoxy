26 double computeLinearInterpolation( const Eigen::VectorXd& sortedIndependentVariables,
27  const Eigen::VectorXd& associatedDependentVariables,
28  const double targetIndependentVariableValue )
29 {
30  // Declare local variables.
31  // Declare nearest neighbor.
32  int nearestNeighbor;
33  double locationTargetIndependentVariableValueInInterval;
34 
35  // Compute nearest neighbor in sorted vector of independent variables.
36  // Result is always to the left of the target independent variable value.
37  nearestNeighbor = basic_mathematics::computeNearestLeftNeighborUsingBinarySearch(
38  sortedIndependentVariables, targetIndependentVariableValue );
39 
40  // Compute location of target independent variable value in interval
41  // between nearest neighbors.
42  locationTargetIndependentVariableValueInInterval
43  = ( targetIndependentVariableValue
44  - sortedIndependentVariables[ nearestNeighbor ] )
45  / ( sortedIndependentVariables[ nearestNeighbor + 1 ]
46  - sortedIndependentVariables[ nearestNeighbor ] );
47 
48  // Return the computed value of the dependent variable.
49  return ( associatedDependentVariables[ nearestNeighbor ]
50  * ( 1 - locationTargetIndependentVariableValueInInterval )
51  + associatedDependentVariables[ nearestNeighbor + 1 ]
52  * locationTargetIndependentVariableValueInInterval );
53 }
