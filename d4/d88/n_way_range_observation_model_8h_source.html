104  Eigen::Matrix< ObservationScalarType, 1, 1 > computeIdealObservationsWithLinkEndData(
105  const TimeType time,
106  const LinkEndType linkEndAssociatedWithTime,
107  std::vector< double >& linkEndTimes,
108  std::vector< Eigen::Matrix< double, 6, 1 > >& linkEndStates )
109  {
110  // Initialize total light-time/single-leg light-time
111  ObservationScalarType totalLightTime =
112  mathematical_constants::getFloatingInteger< ObservationScalarType >( 0 );
113  ObservationScalarType currentLightTime;
114  StateType currentReceiverStateOutput, currentTransmitterStateOutput;
115 
116  // Resize link-end states/times
117  linkEndTimes.clear( );
118  linkEndStates.clear( );
119  linkEndTimes.resize( 2 * ( numberOfLinkEnds_ - 1 ) );
120  linkEndStates.resize( 2 * ( numberOfLinkEnds_ - 1 ) );
121 
122  // Retrieve retransmission delays
123  if( !( retransmissionDelays_ == nullptr ) )
124  {
125  currentRetransmissionDelays_ = retransmissionDelays_( time );
126  if( currentRetransmissionDelays_.size( ) != static_cast< unsigned int >( numberOfLinkEnds_ - 2 ) )
127  {
128  throw std::runtime_error( "Error when calculating n-way range, retransmission delay vector size is inconsistent" );
129  }
130  }
131  else
132  {
133  for( int i = 0; i < numberOfLinkEnds_; i++ )
134  {
135  currentRetransmissionDelays_.push_back( 0.0 );
136  }
137  }
138 
139  // Retrieve index of link end where to start.
140  int startLinkEndIndex = getNWayLinkIndexFromLinkEndType( linkEndAssociatedWithTime, numberOfLinkEnds_ );
141  int currentDownIndex = startLinkEndIndex;
142 
143  // Define 'current time'
144  TimeType currentLinkEndStartTime = time;
145 
146  // Move 'backwards' from reference link end to transmitter.
147  while( currentDownIndex > 0 )
148  {
149  currentLightTime = lightTimeCalculators_.at( currentDownIndex - 1 )->calculateLightTimeWithLinkEndsStates(
150  currentReceiverStateOutput, currentTransmitterStateOutput,
151  currentLinkEndStartTime, 1 );
152 
153  // Add link-end times/states for current leg.
154  linkEndStates[ 2 * ( currentDownIndex - 1 ) + 1 ] = currentReceiverStateOutput.template cast< double >( );
155  linkEndStates[ 2 * ( currentDownIndex - 1 ) ] = currentTransmitterStateOutput.template cast< double >( );
156  linkEndTimes[ 2 * ( currentDownIndex - 1 ) + 1 ] = currentLinkEndStartTime;
157  linkEndTimes[ 2 * ( currentDownIndex - 1 )] = currentLinkEndStartTime - currentLightTime;
158 
159 
160  // If an additional leg is required, retrieve retransmission delay and update current time
161  currentLinkEndStartTime -= currentLightTime;
162  if( currentDownIndex > 1 )
163  {
164  currentLightTime += currentRetransmissionDelays_.at( currentDownIndex - 2 );
165  }
166 
167  // Add computed light-time to total time and move to next leg
168  totalLightTime += currentLightTime;
169  currentDownIndex--;
170  }
171 
172  int currentUpIndex = startLinkEndIndex;
173 
174  // If start is not at transmitter, compute and add retransmission delay.
175  if( ( startLinkEndIndex != 0 ) && ( startLinkEndIndex != numberOfLinkEnds_ - 1 ) )
176  {
177  currentLinkEndStartTime = time + currentRetransmissionDelays_.at( startLinkEndIndex - 1 );
178  totalLightTime += currentRetransmissionDelays_.at( startLinkEndIndex - 1 );
179  }
180 
181  // Move 'forwards' from reference link end to receiver.
182  while( currentUpIndex < static_cast< int >( lightTimeCalculators_.size( ) ) )
183  {
184  currentLightTime = lightTimeCalculators_.at( currentUpIndex )->calculateLightTimeWithLinkEndsStates(
185  currentReceiverStateOutput, currentTransmitterStateOutput,
186  currentLinkEndStartTime, 0 );
187 
188  // Add link-end times/states for current leg.
189  linkEndStates[ 2 * currentUpIndex + 1 ] = currentReceiverStateOutput.template cast< double >( );
190  linkEndStates[ 2 * currentUpIndex ] = currentTransmitterStateOutput.template cast< double >( );
191  linkEndTimes[ 2 * currentUpIndex + 1 ] = currentLinkEndStartTime + currentLightTime;
192  linkEndTimes[ 2 * currentUpIndex ] = currentLinkEndStartTime;
193 
194  // If an additional leg is required, retrieve retransmission delay and update current time
195  currentLinkEndStartTime += currentLightTime;
196  if( currentUpIndex < static_cast< int >( lightTimeCalculators_.size( ) ) - 1 )
197  {
198  currentLightTime += currentRetransmissionDelays_.at( currentUpIndex );
199  }
200 
201  // Add computed light-time to total time and move to next leg
202  totalLightTime += currentLightTime;
203  currentUpIndex++;
204  }
205 
206  // Return total range observation.
207  return ( Eigen::Matrix< ObservationScalarType, 1, 1 >(
208  ) << totalLightTime * physical_constants::getSpeedOfLight< ObservationScalarType >( ) ).finished( );
209  }
