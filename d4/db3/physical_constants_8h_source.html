39 template<typename T, typename U>
40 T constexpr compile_time_pow(T base, U exponent) {
41  static_assert(std::is_integral<U>(), "exponent must be integral");
42  return exponent == 0 ? 1 : base * compile_time_pow(base, exponent - 1);
43 }
