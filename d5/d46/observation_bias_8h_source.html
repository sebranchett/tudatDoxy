656 template< int ObservationSize >
657 ObservationBiasTypes getObservationBiasType(
658  const std::shared_ptr< ObservationBias< ObservationSize > > biasObject )
659 {
660  ObservationBiasTypes biasType;
661 
662  // Check available bias types
663  if( std::dynamic_pointer_cast< ConstantObservationBias< ObservationSize > >( biasObject ) != nullptr )
664  {
665  biasType = constant_absolute_bias;
666  }
667  else if( std::dynamic_pointer_cast< ConstantArcWiseObservationBias< ObservationSize > >( biasObject ) != nullptr )
668  {
669  biasType = arc_wise_constant_absolute_bias;
670  }
671  else if( std::dynamic_pointer_cast< ConstantRelativeObservationBias< ObservationSize > >( biasObject ) != nullptr )
672  {
673  biasType = constant_relative_bias;
674  }
675  else if( std::dynamic_pointer_cast< ConstantRelativeArcWiseObservationBias< ObservationSize > >( biasObject ) != nullptr )
676  {
677  biasType = arc_wise_constant_relative_bias;
678  }
679  else if( std::dynamic_pointer_cast< MultiTypeObservationBias< ObservationSize > >( biasObject ) != nullptr )
680  {
681  biasType = multiple_observation_biases;
682  }
683  else
684  {
685  std::string errorMessage = "Error, did not recognize observation bias when retrieveing bias type";
686  throw std::runtime_error( errorMessage );
687  }
688  return biasType;
689 }
