21 Eigen::Matrix3d calculatePartialOfPointMassGravityWrtPositionOfAcceleratedBody(
22  const Eigen::Vector3d& acceleratedBodyPosition,
23  const Eigen::Vector3d& acceleratingBodyPositions,
24  double gravitationalParameter )
25 {
26  // Calculate relative position
27  Eigen::Vector3d relativePosition = acceleratedBodyPosition - acceleratingBodyPositions;
28 
29  // Calculate partial (Montenbruck & Gill, Eq. 7.56)
30  double relativePositionNorm = relativePosition.norm( );
31  double invSquareOfPositionNorm = 1.0 / ( relativePositionNorm * relativePositionNorm );
32  double invCubeOfPositionNorm = invSquareOfPositionNorm / relativePositionNorm;
33  Eigen::Matrix3d partialMatrix = -gravitationalParameter *
34  ( Eigen::Matrix3d::Identity( ) * invCubeOfPositionNorm -
35  ( 3.0 * invSquareOfPositionNorm * invCubeOfPositionNorm ) * relativePosition * relativePosition.transpose( ) );
36 
37  return partialMatrix;
38 }

41 Eigen::Vector3d computePartialOfCentralGravityWrtGravitationalParameter( const Eigen::Vector3d& acceleratedBodyPosition,
42  const Eigen::Vector3d& acceleratingBodyPosition)
43 {
44  // Calculate relative position
45  Eigen::Vector3d relativePosition = acceleratingBodyPosition - acceleratedBodyPosition;
46 
47  // Calculate partial (Montenbruck & Gill, Eq. 7.76)
48  double positionNorm = relativePosition.norm( );
49  Eigen::Vector3d partialMatrix = relativePosition / ( positionNorm * positionNorm * positionNorm );
50  return partialMatrix;
51 }

100 CentralGravitationPartial::getGravitationalParameterPartialFunction(

102 {
103  std::function< void( Eigen::MatrixXd& ) > partialFunction;
104  int numberOfColumns = 0;
105 
106  // Check if parameter is gravitational parameter.
107  if( parameterId.first == estimatable_parameters::gravitational_parameter )
108  {
109  // Check if parameter body is central body.
110  if( parameterId.second.first == acceleratingBody_ )
111  {
112  partialFunction = std::bind( &CentralGravitationPartial::wrtGravitationalParameterOfCentralBody,
113  this, std::placeholders::_1 );
114  numberOfColumns = 1;
115 
116  }
117 
118  // Check if parameter body is accelerated body, and if the mutual acceleration is used.
119  if( parameterId.second.first == acceleratedBody_ )
120  {
121  if( accelerationUsesMutualAttraction_ )
122  {
123  partialFunction = std::bind( &CentralGravitationPartial::wrtGravitationalParameterOfCentralBody,
124  this, std::placeholders::_1 );
125  numberOfColumns = 1;
126  }
127  }
128  }
