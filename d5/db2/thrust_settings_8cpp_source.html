22 std::vector< std::function< double( ) > > getPropulsionInputVariables(
23  const std::shared_ptr< Body > bodyWithGuidance,
24  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
25  const std::vector< std::function< double( ) > > guidanceInputFunctions )
26 {
27  std::vector< std::function< double( ) > > inputFunctions;
28  std::shared_ptr< aerodynamics::FlightConditions > vehicleFlightConditions =
29  bodyWithGuidance->getFlightConditions( );
30 
31  std::shared_ptr< aerodynamics::AtmosphericFlightConditions > vehicleAtmosphericFlightConditions =
32  std::dynamic_pointer_cast< aerodynamics::AtmosphericFlightConditions >(
33  bodyWithGuidance->getFlightConditions( ) );
34 
35  // Iterate over all dependent variables and create requested function.
36  unsigned int numberOfCustomInputs = 0;
37  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
38  {
39  switch( independentVariables.at( i ) )
40  {
41  case propulsion::altitude_dependent_thrust:
42  inputFunctions.push_back(
43  std::bind( &aerodynamics::FlightConditions::getCurrentAltitude, vehicleFlightConditions ) );
44  break;
45  case propulsion::density_dependent_thrust:
46  if( vehicleAtmosphericFlightConditions == nullptr )
47  {
48  throw std::runtime_error( "Error when making propulsion input variables, atmospheric flight conditions required, but not found" );
49  }
50  inputFunctions.push_back(
51  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentDensity, vehicleAtmosphericFlightConditions ) );
52  break;
53  case propulsion::dynamic_pressure_dependent_thrust:
54  if( vehicleAtmosphericFlightConditions == nullptr )
55  {
56  throw std::runtime_error( "Error when making propulsion input variables, atmospheric flight conditions required, but not found" );
57  }
58  inputFunctions.push_back(
59  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentDynamicPressure, vehicleAtmosphericFlightConditions ) );
60  break;
61  case propulsion::mach_number_dependent_thrust:
62  if( vehicleAtmosphericFlightConditions == nullptr )
63  {
64  throw std::runtime_error( "Error when making propulsion input variables, atmospheric flight conditions required, but not found" );
65  }
66  inputFunctions.push_back(
67  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentMachNumber, vehicleAtmosphericFlightConditions ) );
68  break;
69  case propulsion::pressure_dependent_thrust:
70  if( vehicleAtmosphericFlightConditions == nullptr )
71  {
72  throw std::runtime_error( "Error when making propulsion input variables, atmospheric flight conditions required, but not found" );
73  }
74  inputFunctions.push_back(
75  std::bind( &aerodynamics::AtmosphericFlightConditions::getCurrentPressure, vehicleAtmosphericFlightConditions ) );
76  break;
77  case propulsion::guidance_input_dependent_thrust:
78  inputFunctions.push_back( guidanceInputFunctions.at( numberOfCustomInputs ) );
79  numberOfCustomInputs++;
80  break;
81  case propulsion::throttle_dependent_thrust:
82  if( guidanceInputFunctions.size( ) >= numberOfCustomInputs )
83  {
84  throw std::runtime_error( "Error when creating propulsion indput dependent variables, insufficient user-defined inputs found" );
85  }
86  inputFunctions.push_back( guidanceInputFunctions.at( numberOfCustomInputs ) );
87  numberOfCustomInputs++;
88  break;
89  default:
90  throw std::runtime_error( "Error when getting parameterized thrust input variables, variable " +
91  std::to_string( independentVariables.at( i ) ) + "not found" );
92  }
93  }
94 
95  // Check input consistency
96  if( numberOfCustomInputs != guidanceInputFunctions.size( ) )
97  {
98  std::cerr << "Warning when creating propulsion indput dependent variables, not all user-defined inputs have been parsed" << std::endl;
99  }
100 
101  return inputFunctions;
102 }

115 void ParameterizedThrustMagnitudeSettings::parseInputDataAndCheckConsistency(
116  const std::shared_ptr< interpolators::Interpolator< double, double > > thrustMagnitudeInterpolator,
117  const std::shared_ptr< interpolators::Interpolator< double, double > > specificImpulseInterpolator )
118 {
119  int numberOfUserSpecifiedThrustInputs =
120  std::count( thrustIndependentVariables_.begin( ), thrustIndependentVariables_.end( ),
121  propulsion::guidance_input_dependent_thrust );
122  int numberOfMaximumThrustMultipliers =
123  std::count( thrustIndependentVariables_.begin( ), thrustIndependentVariables_.end( ),
124  propulsion::throttle_dependent_thrust );
125 
126  // Check consistency of user-defined thrust input.
127  if( ( numberOfUserSpecifiedThrustInputs + numberOfMaximumThrustMultipliers ) !=
128  static_cast< int >( thrustGuidanceInputVariables_.size( ) ) )
129  {
130  throw std::runtime_error( "Error in parameterized thrust settings, inconsistent number of user-defined input variables for thrust" );
131  }
132 
133  if( thrustMagnitudeInterpolator->getNumberOfDimensions( ) !=
134  ( static_cast< int >( thrustIndependentVariables_.size( ) ) - numberOfMaximumThrustMultipliers ) )
135  {
136  throw std::runtime_error( "Error in parameterized thrust settings, thrust interpolator size has inconsistent size" );
137  }
138 
139  if( numberOfMaximumThrustMultipliers > 1 )
140  {
141  throw std::runtime_error( "Error in parameterized thrust settings, only 1 maximum thrust multiplier may be defined." );
142  }
143 
144  // Parse maximum thrust multiplier function
145  if( numberOfMaximumThrustMultipliers == 1 )
146  {
147  std::vector< propulsion::ThrustIndependentVariables >::iterator findIterator =
148  std::find( thrustIndependentVariables_.begin( ), thrustIndependentVariables_.end( ),
149  propulsion::throttle_dependent_thrust );
150  int entryForMaximumThrustMultiplierInIndependentVariables =
151  std::distance( thrustIndependentVariables_.begin( ), findIterator );
152  int entryForMaximumThrustMultiplierInGuidanceFunctions = 0;
153 
154  for( unsigned int i = 0; i < thrustIndependentVariables_.size( ); i++ )
155  {
156  if( thrustIndependentVariables_.at( i ) == propulsion::guidance_input_dependent_thrust )
157  {
158  entryForMaximumThrustMultiplierInGuidanceFunctions++;
159  }
160  else if( thrustIndependentVariables_.at( i ) == propulsion::throttle_dependent_thrust)
161  {
162  break;
163  }
164  }
165 
166  std::function< double( const std::vector< double >& ) > maximumThrustMagnitudeFunction =
167  thrustMagnitudeFunction_;
168  thrustMagnitudeFunction_ = std::bind(
169  &multiplyMaximumThrustByScalingFactor, maximumThrustMagnitudeFunction,
170  thrustGuidanceInputVariables_.at( entryForMaximumThrustMultiplierInGuidanceFunctions ), std::placeholders::_1 );
171  thrustIndependentVariables_.erase( thrustIndependentVariables_.begin( ) +
172  entryForMaximumThrustMultiplierInIndependentVariables);
173  thrustGuidanceInputVariables_.erase( thrustGuidanceInputVariables_.begin( ) +
174  entryForMaximumThrustMultiplierInGuidanceFunctions );
175 
176  }
177 
178  if( specificImpulseInterpolator != nullptr )
179  {
180  // Check consistency of user-defined specific impulse input.
181  int numberOfUserSpecifiedSpecificImpulseInputs =
182  std::count( specificImpulseDependentVariables_.begin( ), specificImpulseDependentVariables_.end( ),
183  propulsion::guidance_input_dependent_thrust );
184  int numberOfMaximumThrustMultipliersForSpecificImpulse =
185  std::count( specificImpulseDependentVariables_.begin( ), specificImpulseDependentVariables_.end( ),
186  propulsion::throttle_dependent_thrust );
187 
188  if( numberOfUserSpecifiedSpecificImpulseInputs !=
189  static_cast< int >( specificImpulseGuidanceInputVariables_.size( ) ) )
190  {
191  throw std::runtime_error( "Error in parameterized thrust settings, inconsistent number of user-defined input variables for specific impulse " );
192  }
193  if( specificImpulseInterpolator->getNumberOfDimensions( ) != static_cast< int >(
194  specificImpulseDependentVariables_.size( ) ) )
195  {
196  throw std::runtime_error( "Error in parameterized thrust settings, specific impulse interpolator size has inconsistent size" );
197  }
198  if( numberOfMaximumThrustMultipliersForSpecificImpulse != 0 )
199  {
200  throw std::runtime_error( "Error in parameterized thrust settings, no maximum thrust multiplier may be defined for specific impulse." );
201  }
202  }
203  else
204  {
205  if( specificImpulseDependentVariables_.size( ) > 0 || specificImpulseGuidanceInputVariables_.size( ) > 0 )
206  {
207  throw std::runtime_error( "Error in parameterized thrust settings, no specific impulse interpolator define, but inptu data has been provided." );
208  }
209  }
210 }

213 std::shared_ptr< interpolators::Interpolator< double, double > > readCoefficientInterpolatorFromFile(
214  const std::string coefficientFile )
215 {
216  std::shared_ptr< interpolators::Interpolator< double, double > > coefficientInterpolator;
217  int numberOfThrustIndependentVariables = input_output::getNumberOfIndependentVariablesInCoefficientFile(
218  coefficientFile );
219  if( numberOfThrustIndependentVariables == 1 )
220  {
221  std::pair< boost::multi_array< double, 1 >, std::vector< std::vector< double > > > coefficientData =
222  input_output::MultiArrayFileReader< 1 >::readMultiArrayAndIndependentVariables(
223  coefficientFile );
224  coefficientInterpolator = std::make_shared< interpolators::MultiLinearInterpolator< double, double, 1 > >(
225  coefficientData.second, coefficientData.first );
226  }
227  else if( numberOfThrustIndependentVariables == 2 )
228  {
229  std::pair< boost::multi_array< double, 2 >, std::vector< std::vector< double > > > coefficientData =
230  input_output::MultiArrayFileReader< 2 >::readMultiArrayAndIndependentVariables(
231  coefficientFile );
232  coefficientInterpolator = std::make_shared< interpolators::MultiLinearInterpolator< double, double, 2 > >(
233  coefficientData.second, coefficientData.first );
234  }
235  else if( numberOfThrustIndependentVariables == 3 )
236  {
237  std::pair< boost::multi_array< double, 3 >, std::vector< std::vector< double > > > coefficientData =
238  input_output::MultiArrayFileReader< 3 >::readMultiArrayAndIndependentVariables(
239  coefficientFile );
240  coefficientInterpolator = std::make_shared< interpolators::MultiLinearInterpolator< double, double, 3 > >(
241  coefficientData.second, coefficientData.first );
242  }
243  else
244  {
245  throw std::runtime_error( "Error, reading of thrust magnitude files of mroe than 3 independent variables not yet implemented" );
246  }
247  return coefficientInterpolator;
248 }

252  const std::shared_ptr< ThrustInputParameterGuidance > thrustInputParameterGuidance,
253  const std::shared_ptr< interpolators::Interpolator< double, double > > thrustMagnitudeInterpolator,
254  const std::vector< propulsion::ThrustIndependentVariables > thrustIndependentVariables,
255  const std::shared_ptr< interpolators::Interpolator< double, double > > specificImpulseInterpolator,
256  const std::vector< propulsion::ThrustIndependentVariables > specificImpulseDependentVariables )
257 {
258  // Create guidance input functions.
259  std::vector< std::function< double( ) > > thrustGuidanceInputVariables;
260  for( int i = 0; i < thrustInputParameterGuidance->getNumberOfThrustInputParameters( ); i++ )
261  {
262  thrustGuidanceInputVariables.push_back(
263  std::bind( &ThrustInputParameterGuidance::getThrustInputGuidanceParameter,
264  thrustInputParameterGuidance, i ) );
265  }
266 
267  // Create specific impulse input functions.
268  std::vector< std::function< double( ) > > specificImpulseGuidanceInputVariables;
269  for( int i = 0; i < thrustInputParameterGuidance->getNumberOfSpecificImpulseInputParameters( ); i++ )
270  {
271  specificImpulseGuidanceInputVariables.push_back(
272  std::bind( &ThrustInputParameterGuidance::getSpecificImpulseInputGuidanceParameter,
273  thrustInputParameterGuidance, i ) );
274  }
275 
276  // Create thrust magnitude settings object.
277  return std::make_shared< ParameterizedThrustMagnitudeSettings >(
278  thrustMagnitudeInterpolator,
279  thrustIndependentVariables,
280  specificImpulseInterpolator,
281  specificImpulseDependentVariables,
282  thrustGuidanceInputVariables,
283  specificImpulseGuidanceInputVariables,
284  std::bind( &ThrustInputParameterGuidance::update, thrustInputParameterGuidance, std::placeholders::_1 ) );
