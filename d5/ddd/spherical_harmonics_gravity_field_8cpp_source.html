32 double calculateSphericalHarmonicGravitationalPotential(
33  const Eigen::Vector3d& bodyFixedPosition, const double gravitationalParameter,
34  const double referenceRadius,
35  const Eigen::MatrixXd& cosineCoefficients,
36  const Eigen::MatrixXd& sineCoefficients,
37  std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache,
38  const int minimumumDegree,
39  const int minimumumOrder )
40 {
41  // Initialize (distance/reference radius)^n (n=ratioToPowerDegree)
42  double ratioToPowerDegree = 1.0;
43  double radiusRatio = referenceRadius / bodyFixedPosition.norm( );
44 
45  // Declare local variables used in calculation
46  double legendrePolynomial = 0.0;
47  double singleDegreeTerm = 0.0;
48 
49  // Determine body fixed spherical position of body udnergoing acceleration.
50  Eigen::Vector3d sphericalPositon =
51  coordinate_conversions::convertCartesianToSpherical( bodyFixedPosition );
52  double latitude = mathematical_constants::PI / 2.0 - sphericalPositon.y( );
53  double longitude = sphericalPositon.z( );
54 
55  double potential = 0.0;
56  int startDegree = 0;
57 
58  // Initialize value of potential to 1 (C_{0,0})
59  if( minimumumDegree == 0 )
60  {
61  potential = 1.0;
62  startDegree = 1;
63  }
64  else
65  {
66  startDegree = minimumumDegree;
67  ratioToPowerDegree *= basic_mathematics::raiseToIntegerPower< double >( radiusRatio, startDegree - 1 );
68  }
69 
70  basic_mathematics::LegendreCache& legendreCacheReference = *sphericalHarmonicsCache->getLegendreCache( );
71  legendreCacheReference.update( std::sin( latitude ) );
72 
73  // Iterate over all degrees
74  for( int degree = startDegree; degree < cosineCoefficients.rows( ); degree++ )
75  {
76  singleDegreeTerm = 0.0;
77 
78  // Iterate over all orders in current degree for which coefficients are provided.
79  for( int order = minimumumOrder; ( order < cosineCoefficients.cols( ) &&
80  order <= degree ); order++ )
81  {
82  // Calculate legendre polynomial (geodesy-normalized) at current degree and order
83  legendrePolynomial = basic_mathematics::computeGeodesyLegendrePolynomialFromCache(
84  degree, order, legendreCacheReference );
85 
86  // Calculate contribution to potential from current degree and order
87  singleDegreeTerm += legendrePolynomial * ( cosineCoefficients( degree, order ) *
88  std::cos( order * longitude ) +
89  sineCoefficients( degree, order ) *
90  std::sin( order * longitude ) );
91  }
92 
93  // Add potential contributions from current degree to toal value.
94  ratioToPowerDegree *= radiusRatio;
95  potential += singleDegreeTerm * ratioToPowerDegree;
96  }
97 
98  // Multiply by central term and return
99  return potential * gravitationalParameter / bodyFixedPosition.norm( );
100 }

103 Eigen::Matrix3d getInertiaTensor(
104  const double c20Coefficient,
105  const double c21Coefficient,
106  const double c22Coefficient,
107  const double s21Coefficient,
108  const double s22Coefficient,
109  const double scaledMeanMomentOfInertia,
110  const double bodyMass,
111  const double referenceRadius )
112 {
113  // Compute inertia tensor
114  double scalingConstant = bodyMass * referenceRadius * referenceRadius;
115  Eigen::Matrix3d inertiaTensor =
116  ( Eigen::Matrix3d( )<< c20Coefficient / 3.0 - 2.0 * c22Coefficient, -2.0 * s22Coefficient, -c21Coefficient,
117  -2.0 * s22Coefficient, c20Coefficient / 3.0 + 2.0 * c22Coefficient, -s21Coefficient,
118  -c21Coefficient, -s21Coefficient, -2.0 * c20Coefficient / 3.0 ).finished( );
119 
120  return scalingConstant * ( inertiaTensor + Eigen::Matrix3d::Identity( ) * scaledMeanMomentOfInertia );
121 }

139 Eigen::Matrix3d getInertiaTensor(
140  const std::shared_ptr< SphericalHarmonicsGravityField > sphericalHarmonicGravityField,
141  const double scaledMeanMomentOfInertia )
142 {
143  // Denormalize coefficients if needed, and compute inertia tensor
144  if( sphericalHarmonicGravityField->areCoefficientsGeodesyNormalized( ) )
145  {
146  Eigen::MatrixXd normalizedCosineCoefficients = Eigen::Matrix3d::Zero( );
147  Eigen::MatrixXd normalizedSineCoefficient = Eigen::Matrix3d::Zero( );
148  basic_mathematics::convertGeodesyNormalizedToUnnormalizedCoefficients(
149  sphericalHarmonicGravityField->getCosineCoefficients( ).block( 0, 0, 3, 3 ),
150  sphericalHarmonicGravityField->getSineCoefficients( ).block( 0, 0, 3, 3 ),
151  normalizedCosineCoefficients,
152  normalizedSineCoefficient );
153 
154  return getInertiaTensor(
155  normalizedCosineCoefficients, normalizedSineCoefficient,
156  scaledMeanMomentOfInertia,
157  sphericalHarmonicGravityField->getGravitationalParameter( ) / physical_constants::GRAVITATIONAL_CONSTANT,
158  sphericalHarmonicGravityField->getReferenceRadius( ) );
159  }
160  else
161  {
162  return getInertiaTensor(
163  sphericalHarmonicGravityField->getCosineCoefficients( ),
164  sphericalHarmonicGravityField->getSineCoefficients( ),
165  scaledMeanMomentOfInertia,

172 void getDegreeTwoSphericalHarmonicCoefficients(
173  const Eigen::Matrix3d inertiaTensor, const double bodyGravitationalParameter, const double referenceRadius,
174  const bool useNormalizedCoefficients,
175  Eigen::MatrixXd& cosineCoefficients, Eigen::MatrixXd& sineCoefficients, double& scaledMeanMomentOfInertia )
176 {
177  double scalingTerm = bodyGravitationalParameter * referenceRadius * referenceRadius /
178  physical_constants::GRAVITATIONAL_CONSTANT;
179 
180  cosineCoefficients.setZero( );
181  cosineCoefficients( 0, 0 ) = 1.0;
182  sineCoefficients.setZero( );
183 
184  cosineCoefficients( 2, 0 ) = ( 0.5 * inertiaTensor( 0, 0 ) + 0.5 * inertiaTensor( 1, 1 ) - inertiaTensor( 2, 2 ) ) / scalingTerm;
185  cosineCoefficients( 2, 2 ) = ( -0.25 * inertiaTensor( 0, 0 ) + 0.25 * inertiaTensor( 1, 1 ) ) / scalingTerm;
186  cosineCoefficients( 2, 1 ) = -inertiaTensor( 2, 0 ) / scalingTerm;
187  sineCoefficients( 2, 1 ) = -inertiaTensor( 2, 1 ) / scalingTerm;
188  sineCoefficients( 2, 2 ) = -0.5 * inertiaTensor( 1, 0 ) / scalingTerm;
189 
190  if( useNormalizedCoefficients )
191  {
192  basic_mathematics::geodesyNormalizeUnnormalizedCoefficients(
193  cosineCoefficients, sineCoefficients );
194  }
195 
196  scaledMeanMomentOfInertia = ( inertiaTensor( 0, 0 ) + inertiaTensor( 1, 1 ) + inertiaTensor( 2, 2 ) ) / 3.0;
197 }
