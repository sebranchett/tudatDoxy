160  void postProcessState( Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > unprocessedState )
161  {
162  // Loop over each body
163  Eigen::Matrix< StateScalarType, 3, 1 > exponentialMapVector;
164  StateScalarType exponentialMapMagnitude;
165  for( unsigned int i = 0; i < this->bodiesToPropagate_.size( ); i++ )
166  {
167  // Convert to/from shadow exponential map (SEM) (transformation is the same either way)
168  exponentialMapVector = unprocessedState.block( i * 7, 0, 3, 1 );
169  exponentialMapMagnitude = exponentialMapVector.norm( );
170  if ( exponentialMapMagnitude >= mathematical_constants::PI )
171  {
172  // Invert flag
173  unprocessedState.segment( i * 7 + 3, 1 ) = ( unprocessedState.block( i * 7 + 3, 0, 1, 1 ) -
174  Eigen::Matrix< StateScalarType, 1, 1 >::Ones( ) ).cwiseAbs( );
175 
176  // Convert to EM/SEM
177  exponentialMapVector *= ( 1.0 - ( 2.0 * mathematical_constants::PI / exponentialMapMagnitude ) );
178 
179  // Replace EM with SEM, or vice-versa
180  unprocessedState.segment( i * 7, 3 ) = exponentialMapVector;
181  }
182  }
183  }
