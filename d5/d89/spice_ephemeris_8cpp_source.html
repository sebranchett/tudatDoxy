25 SpiceEphemeris::SpiceEphemeris( const std::string& targetBodyName,
26  const std::string& observerBodyName,
27  const bool correctForStellarAberration,
28  const bool correctForLightTimeAberration,
29  const bool convergeLighTimeAberration,
30  const std::string& referenceFrameName,
31  const double referenceJulianDay )
32  : Ephemeris( observerBodyName, referenceFrameName ),
33  targetBodyName_( targetBodyName )
34 {
35  referenceDayOffSet_ = ( referenceJulianDay - basic_astrodynamics::JULIAN_DAY_ON_J2000 ) * physical_constants::JULIAN_DAY;
36 
37  // Check consistency of input.
38  if ( correctForLightTimeAberration == 0 && convergeLighTimeAberration == 1 )
39  {
40  throw std::runtime_error(
41  "Error, requested multiple iterations for light time correction, but not light time correction itself." );
42  }
43 
44  if ( correctForLightTimeAberration == 0 && correctForStellarAberration == 1 )
45  {
46 
47  throw std::runtime_error(
48  "Error, requested stellar aberration, but not light-time correction, this is not supprted by spice." );
49  }
50 
51  // Set aberration corrections variable.
52  aberrationCorrections_ = "";
53  if ( correctForLightTimeAberration && !convergeLighTimeAberration )
54  {
55  aberrationCorrections_.append( "LT" );
56  }
57 
58  else if ( correctForLightTimeAberration && convergeLighTimeAberration )
59  {
60  aberrationCorrections_.append( "CN" );
61  }
62 
63  else if ( !correctForLightTimeAberration && !correctForStellarAberration )
64  {
65  aberrationCorrections_.append( "NONE" );
66  }
67 
68  if ( correctForLightTimeAberration && correctForStellarAberration )
69  {
70  aberrationCorrections_.append( " +S" );
71  }
72 }
