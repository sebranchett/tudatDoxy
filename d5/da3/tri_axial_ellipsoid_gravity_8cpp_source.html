25 double calculateCosineTermForTriaxialEllipsoidSphericalHarmonicGravity(
26  const double aSquaredMinusCSquared, const double bSquaredMinusCSquared,
27  const double referenceRadius, const int degree, const int order )
28 {
29  // Initialize coefficient to zero
30  double powerSeries = 0.0;
31 
32  // Only non-zero terms are for even degree and order
33  if( !( ( degree % 2 != 0 || order % 2 != 0 ) ) )
34  {
35  using boost::math::factorial;
36 
37  // Calculate indices for algorithm
38  int l = degree / 2;
39  int m = order / 2;
40  int maximumIndex = std::floor( ( l - m ) / 2 );
41 
42  // Evaluate single term in power series of final equation of Boyce (1997)
43  for( int i = 0; i <= maximumIndex; i++ )
44  {
45  powerSeries +=
46  ( std::pow( ( aSquaredMinusCSquared - bSquaredMinusCSquared ) /
47  2.0, m + 2 * i ) *
48  std::pow( -( aSquaredMinusCSquared + bSquaredMinusCSquared ) /
49  2.0, l - m - 2 * i ) ) /
50  ( std::pow( 2.0, m + 2 * i ) * factorial< double >( l - m - 2 * i ) *
51  factorial< double >( m + i ) * factorial< double >( i ) );
52  }
53 
54  // Calculate multiplier of power series in final equation of Boyce (1997)
55  double multiplier = 3.0 / std::pow( referenceRadius, 2 * l ) *
56  ( factorial< double >( l ) * factorial< double >( 2 * l - 2 * m ) ) /
57  ( ( 2.0 * static_cast< double >( l ) + 3.0 ) * factorial< double >( 2 * l + 1 ) );
58  if( order != 0 )
59  {
60  multiplier *= 2.0;
61  }
62 
63  // Complete calculation of coefficient
64  powerSeries *= multiplier;
65  }
66  return powerSeries;
67 }

86 Eigen::MatrixXd createTriAxialEllipsoidSphericalHarmonicCosineCoefficients(
87  const double axisA, const double axisB, const double axisC,
88  const int maximumDegree, const int maximumOrder )
89 {
90  // Initialize vector to zeros
91  Eigen::MatrixXd cosineCoefficients = Eigen::MatrixXd::Zero(
92  maximumDegree + 1, maximumOrder + 1 );
93 
94  // Pre-calculate for single coefficient calculations.
95  double aSquaredMinusCSquared = axisA * axisA - axisC * axisC;
96  double bSquaredMinusCSquared = axisB * axisB - axisC * axisC;
97  double referenceRadius = calculateTriAxialEllipsoidReferenceRadius( axisA, axisB, axisC );
98 
99  // Iterate over all requested degrees.
100  for( int i = 0; i <= maximumDegree; i++ )
101  {
102  // Iterate over all requested orders.
103  for( int j = 0; ( ( j <= maximumOrder ) && ( j <= i ) ); j++ )
104  {
105  // Only even degree and order terms are non-zero
106  if( ( i % 2 == 0 ) && ( j % 2 == 0 ) )
107  {
108  // Calculate coefficient at single degree and order.
109  cosineCoefficients( i, j ) =
110  calculateCosineTermForTriaxialEllipsoidSphericalHarmonicGravity(
111  aSquaredMinusCSquared, bSquaredMinusCSquared, referenceRadius, i, j );
112  }
113  }
114  }
115 
116  return cosineCoefficients;
117 }
