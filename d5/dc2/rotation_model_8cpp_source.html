21 void to_json( nlohmann::json& jsonObject, const std::shared_ptr< RotationModelSettings >& rotationModelSettings )
22 {
23  if ( ! rotationModelSettings )
24  {
25  return;
26  }
27  using namespace json_interface;
28  using K = Keys::Body::RotationModel;
29 
30  const RotationModelType rotationModelType = rotationModelSettings->getRotationType( );
31  jsonObject[ K::type ] = rotationModelType;
32  jsonObject[ K::originalFrame ] = rotationModelSettings->getOriginalFrame( );
33  jsonObject[ K::targetFrame ] = rotationModelSettings->getTargetFrame( );
34 
35  switch ( rotationModelType )
36  {
37  case simple_rotation_model:
38  {
39  std::shared_ptr< SimpleRotationModelSettings > simpleRotationModelSettings =
40  std::dynamic_pointer_cast< SimpleRotationModelSettings >( rotationModelSettings );
41  assertNonnullptrPointer( simpleRotationModelSettings );
42  jsonObject[ K::initialOrientation ] = simpleRotationModelSettings->getInitialOrientation( );
43  jsonObject[ K::initialTime ] = simpleRotationModelSettings->getInitialTime( );
44  jsonObject[ K::rotationRate ] = simpleRotationModelSettings->getRotationRate( );
45  return;
46  }
47  case synchronous_rotation_model:
48  {
49  std::shared_ptr< SynchronousRotationModelSettings > synchronousRotationModelSettings =
50  std::dynamic_pointer_cast< SynchronousRotationModelSettings >( rotationModelSettings );
51  assertNonnullptrPointer( synchronousRotationModelSettings );
52 
53  jsonObject[ K::centralBodyName ] = synchronousRotationModelSettings->getCentralBodyName( );
54  return;
55  }
56  case spice_rotation_model:
57  return;
58  case gcrs_to_itrs_rotation_model:
59  {
60  std::cerr<<"Warning when writing GCRS to ITRS rotation model to JSON, only precession-nutation theory and base frame are saved, rest is kept at default values"<<std::endl;
61 
62  std::shared_ptr< GcrsToItrsRotationModelSettings > simpleRotationModelSettings =
63  std::dynamic_pointer_cast< GcrsToItrsRotationModelSettings >( rotationModelSettings );
64  assertNonnullptrPointer( simpleRotationModelSettings );
65  jsonObject[ K::precessionNutationTheory ] = simpleRotationModelSettings->getNutationTheory( );
66  return;
67  }
68  default:
69  handleUnimplementedEnumValue( rotationModelType, rotationModelTypes, unsupportedRotationModelTypes );
70  }
71 }

74 void from_json( const nlohmann::json& jsonObject, std::shared_ptr< RotationModelSettings >& rotationModelSettings )
75 {
76  using namespace json_interface;
77  using K = Keys::Body::RotationModel;
78 
79  // Base class settings
80  const RotationModelType rotationModelType = getValue< RotationModelType >( jsonObject, K::type );
81  const std::string originalFrame = getValue< std::string >( jsonObject, K::originalFrame );
82 
83  std::string targetFrame = "";
84  if( rotationModelType != gcrs_to_itrs_rotation_model )
85  {
86  targetFrame = getValue< std::string >( jsonObject, K::targetFrame );
87  }
88  else
89  {
90  targetFrame = "GCRS";
91  }
92 
93  switch ( rotationModelType ) {
94  case simple_rotation_model:
95  {
96  const double initialTime = getValue< double >( jsonObject, K::initialTime );
97 
98  // Get JSON object for initialOrientation (or create it if not defined)
99  nlohmann::json jsonInitialOrientation;
100  if ( isDefined( jsonObject, K::initialOrientation ) )
101  {
102  jsonInitialOrientation = getValue< nlohmann::json >( jsonObject, K::initialOrientation );
103  }
104  else
105  {
106  jsonInitialOrientation[ K::originalFrame ] = originalFrame;
107  jsonInitialOrientation[ K::targetFrame ] = targetFrame;
108  jsonInitialOrientation[ K::initialTime ] = initialTime;
109  }
110 
111  rotationModelSettings = std::make_shared< SimpleRotationModelSettings >(
112  originalFrame,
113  targetFrame,
114  getAs< Eigen::Quaterniond >( jsonInitialOrientation ),
115  initialTime,
116  getValue< double >( jsonObject, K::rotationRate ) );
117  return;
118  }
119  case synchronous_rotation_model:
120  {
121  const std::string centralBody = getValue< std::string >( jsonObject, K::centralBodyName );
122  rotationModelSettings = std::make_shared< SynchronousRotationModelSettings >(
123  centralBody, originalFrame, targetFrame );
124  return;
125  }
126 
127  case spice_rotation_model:
128  {
129  rotationModelSettings = std::make_shared< RotationModelSettings >(
130  rotationModelType, originalFrame, targetFrame );
131  return;
132  }
133  case gcrs_to_itrs_rotation_model:
134  {
135  rotationModelSettings = std::make_shared< GcrsToItrsRotationModelSettings >(
136  getValue< basic_astrodynamics::IAUConventions >(
137  jsonObject, K::precessionNutationTheory, basic_astrodynamics::iau_2006 ),
138  originalFrame );
139  return;
140  }
141  default:
142  handleUnimplementedEnumValue( rotationModelType, rotationModelTypes, unsupportedRotationModelTypes );
143  }
144 }
