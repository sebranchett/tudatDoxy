98 std::string getGlobalFrameOrigin( const NamedBodyMap& bodyMap )
99 {
100  std::string globalFrameOrigin = "SSB";
101 
102  for( NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( ); bodyIterator != bodyMap.end( ); bodyIterator++ )
103  {
104  if( bodyIterator->second->getIsBodyGlobalFrameOrigin( ) == -1 )
105  {
106  throw std::runtime_error( "Error, body " + bodyIterator->first + " does not have global frame origin set" );
107  }
108  else if( bodyIterator->second->getIsBodyGlobalFrameOrigin( ) == 1 )
109  {
110  if( globalFrameOrigin != "SSB" )
111  {
112  throw std::runtime_error( "Error, body " + bodyIterator->first + " found as global frame origin, but body " +
113  globalFrameOrigin + " has already been detected as global frame origin." );
114  }
115  else
116  {
117  globalFrameOrigin = bodyIterator->first;
118  }
119  }
120  }
121  return globalFrameOrigin;
122 }
