102 Eigen::Vector6d HybridMethodModel::propagateTrajectory( double initialTime, double finalTime, Eigen::Vector6d initialState, double initialMass )
103 {
104  // Re-initialise integrator settings.
105  integratorSettings_->initialTime_ = initialTime;
106 
107  bodyMap_[ bodyToPropagate_ ]->setConstantBodyMass( initialMass );
108 
109  // Acceleration from the central body.
110  std::map< std::string, std::vector< std::shared_ptr< simulation_setup::AccelerationSettings > > > bodyToPropagateAccelerations;
111  bodyToPropagateAccelerations[ centralBody_ ].push_back( std::make_shared< simulation_setup::AccelerationSettings >(
112  basic_astrodynamics::central_gravity ) );
113  bodyToPropagateAccelerations[ bodyToPropagate_ ].push_back( getMEEcostatesBasedThrustAccelerationSettings( ) );
114 
115  simulation_setup::SelectedAccelerationMap accelerationMap;
116  accelerationMap[ bodyToPropagate_ ] = bodyToPropagateAccelerations;
117 
118  // Create the acceleration map.
119  basic_astrodynamics::AccelerationMap accelerationModelMap = createAccelerationModelsMap(
120  bodyMap_, accelerationMap, std::vector< std::string >{ bodyToPropagate_ }, std::vector< std::string >{ centralBody_ } );
121 
122 
123  // Create mass rate models
124  std::map< std::string, std::shared_ptr< basic_astrodynamics::MassRateModel > > massRateModel;
125  massRateModel[ bodyToPropagate_ ] = createMassRateModel( bodyToPropagate_, std::make_shared< simulation_setup::FromThrustMassModelSettings >( 1 ),
126  bodyMap_, accelerationModelMap );
127 
128  // Ensure that the propagation stops when the required time of flight is required.
129  std::shared_ptr< propagators::PropagationTimeTerminationSettings > terminationSettings
130  = std::make_shared< propagators::PropagationTimeTerminationSettings >( finalTime, true );
131 
132 
133  // Define propagator settings.
134  std::shared_ptr< propagators::TranslationalStatePropagatorSettings< double > > translationalStatePropagatorSettings =
135  std::make_shared< propagators::TranslationalStatePropagatorSettings< double > >(
136  std::vector< std::string >{ centralBody_ }, accelerationModelMap, std::vector< std::string >{ bodyToPropagate_ },
137  initialState, terminationSettings, propagators::gauss_modified_equinoctial );
138 
139  // Create settings for propagating the mass of the vehicle.
140  std::shared_ptr< propagators::MassPropagatorSettings< double > > massPropagatorSettings
141  = std::make_shared< propagators::MassPropagatorSettings< double > >(
142  std::vector< std::string >{ bodyToPropagate_ }, massRateModel,
143  ( Eigen::Matrix< double, 1, 1 >( ) << bodyMap_[ bodyToPropagate_ ]->getBodyMass( ) ).finished( ),
144  terminationSettings );
145 
146  // Create list of propagation settings.
147  std::vector< std::shared_ptr< propagators::SingleArcPropagatorSettings< double > > > propagatorSettingsVector;
148  propagatorSettingsVector.push_back( translationalStatePropagatorSettings );
149  propagatorSettingsVector.push_back( massPropagatorSettings );
150 
151  // Hybrid propagation settings.
152  std::shared_ptr< propagators::PropagatorSettings< double > > propagatorSettings =
153  std::make_shared< propagators::MultiTypePropagatorSettings< double > >( propagatorSettingsVector, terminationSettings );
154 
155  integratorSettings_->initialTime_ = initialTime;
156 
157  // Propagate the trajectory.
158  propagators::SingleArcDynamicsSimulator< > dynamicsSimulator( bodyMap_, integratorSettings_, propagatorSettings );
159  Eigen::VectorXd propagationResult = dynamicsSimulator.getEquationsOfMotionNumericalSolution( ).rbegin( )->second;
160 
161 
162  // Retrieve state and mass of the spacecraft at the end of the propagation.
163  Eigen::Vector6d propagatedState = propagationResult.segment( 0, 6 );
164 
165  if ( finalTime == timeOfFlight_ )
166  {
167  massAtTimeOfFlight_ = propagationResult[ 6 ];
168  }
169 
170  return propagatedState;
171 }

175 std::map< double, Eigen::Vector6d > HybridMethodModel::propagateTrajectory(
176  std::vector< double > epochs, std::map< double, Eigen::Vector6d >& propagatedTrajectory )
177 {
178  // Initialise propagated state.
179  Eigen::Vector6d propagatedState = stateAtDeparture_;
180 
181  // Initialise mass of the spacecraft at departure.
182  bodyMap_[ bodyToPropagate_ ]->setConstantBodyMass( initialSpacecraftMass_ );
183  double currentMass = initialSpacecraftMass_;
184 
185 
186  for ( int epochIndex = 0 ; epochIndex < epochs.size( ) ; epochIndex++ )
187  {
188  double currentTime = epochs[ epochIndex ];
189  if ( epochIndex > 0 )
190  {
191  if ( currentTime < epochs[ epochIndex - 1 ] )
192  {
193  throw std::runtime_error( "Error when propagating trajectory with hybrid method, epochs at which the trajectory should be "
194  "computed are not in increasing order." );
195  }
196  }
197  if ( ( currentTime < 0.0 ) || ( currentTime > timeOfFlight_ ) )
198  {
199  throw std::runtime_error( "Error when propagating trajectory with hybrid method, epochs at which the trajectory should be "
200  "computed are not constrained between 0.0 and timeOfFlight." );
201  }
202 
203 
204  if ( epochIndex == 0 )
205  {
206  if ( currentTime > 0.0 )
207  {
208  propagatedState = propagateTrajectory( 0.0, currentTime, propagatedState, currentMass );
209  currentMass = bodyMap_[ bodyToPropagate_ ]->getBodyMass( );
210  }
211  propagatedTrajectory[ currentTime ] = propagatedState;
212  }
213  else
214  {
215  propagatedState = propagateTrajectory( epochs[ epochIndex - 1 ], currentTime, propagatedState, currentMass );
216  currentMass = bodyMap_[ bodyToPropagate_ ]->getBodyMass( );
217  propagatedTrajectory[ currentTime ] = propagatedState;
218  }
219 
220  }
221 
222  bodyMap_[ centralBody_ ]->setConstantBodyMass( initialSpacecraftMass_ );
223 
224  return propagatedTrajectory;
225 }

229 double HybridMethodModel::computeDeltaV( )
230 {
231 
232  // Compute (constant) mass rate.
233  double massRate = - maximumThrust_ /
234  ( specificImpulse_ * physical_constants::SEA_LEVEL_GRAVITATIONAL_ACCELERATION );
235 
236  // Compute time during which the engine was switched on.
237  double engineSwitchedOnDuration = ( massAtTimeOfFlight_ - initialSpacecraftMass_ ) / massRate;
238 
239  std::shared_ptr< numerical_quadrature::QuadratureSettings< double > > quadratureSettings_ =
240  std::make_shared< numerical_quadrature::GaussianQuadratureSettings < double > >( 0.0, 16 );
241 
242  // Thrust acceleration function to use quadrature.
243  // Define thrust acceleration as a function of time (to be integrated to compute the associated deltaV).
244  std::function< double( const double ) > thrustAcceleration = [ = ] ( const double currentTime ){
245 
246  // Compute current mass.
247  double currentMass = initialSpacecraftMass_ + massRate * currentTime;
248 
249  // Compute and return current thrust acceleration.
250  double currentThrustAcceleration = maximumThrust_ / currentMass;
251 
252  return currentThrustAcceleration;
253  };
254 
255  // Create numerical quadrature from quadrature settings.
256  std::shared_ptr< numerical_quadrature::NumericalQuadrature< double, double > > quadrature =
257  numerical_quadrature::createQuadrature( thrustAcceleration, quadratureSettings_, engineSwitchedOnDuration );
258 
259  // Compute deltaV analytically.
260  double deltaV = - specificImpulse_ * physical_constants::SEA_LEVEL_GRAVITATIONAL_ACCELERATION
261  * std::log( 1.0 + massRate / initialSpacecraftMass_ * engineSwitchedOnDuration );
262 
263  return deltaV;
264 }
