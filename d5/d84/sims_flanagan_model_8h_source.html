30  SimsFlanaganModel( const Eigen::Vector6d& stateAtDeparture,
31  const Eigen::Vector6d& stateAtArrival,
32  const double centralBodyGravitationalParameter,
33  const double initialSpacecraftMass,
34  const double maximumThrust,
35  const std::function< double ( const double ) > specificImpulseFunction,
36  const double timeOfFlight,
37  std::vector< Eigen::Vector3d >& throttles ):
38  stateAtDeparture_( stateAtDeparture ), stateAtArrival_( stateAtArrival ),
39  centralBodyGravitationalParameter_( centralBodyGravitationalParameter ),
40  initialSpacecraftMass_( initialSpacecraftMass ),
41  maximumThrust_( maximumThrust ),
42  specificImpulseFunction_( specificImpulseFunction ), timeOfFlight_( timeOfFlight ),
43  throttles_( throttles )
44  {
45 
46  // Retrieve number of segments from the size of the throttles vector.
47  numberSegments_ = throttles_.size();
48 
49  // Calculate number of segments for both the forward propagation (from departure to match point)
50  // and the backward propagation (from arrival to match point).
51  numberSegmentsForwardPropagation_ = ( numberSegments_ + 1 ) / 2;
52  numberSegmentsBackwardPropagation_ = numberSegments_ / 2;
53 
54  // Compute time at match point (half of the time of flight).
55  timeAtMatchPoint_ = timeOfFlight_ / 2.0;
56 
57  // Compute the times at the different nodes of the leg.
58  segmentDurationForwardPropagation_ = timeAtMatchPoint_ / numberSegmentsForwardPropagation_;
59  segmentDurationBackwardPropagation_ = timeAtMatchPoint_ / numberSegmentsBackwardPropagation_;
60  for ( int i = 0 ; i <= numberSegmentsForwardPropagation_ ; i++ )
61  {
62  timesAtNodes_.push_back( i * segmentDurationForwardPropagation_ );
63  }
64  for ( int i = 1 ; i <= numberSegmentsBackwardPropagation_ ; i++ )
65  {
66  timesAtNodes_.push_back( timeAtMatchPoint_ + i * segmentDurationBackwardPropagation_ );
67  }
68 
69  propagateMassToSegments( );
70 
71  // Initialise value of the total deltaV.
72  totalDeltaV_ = 0.0;
73  }
