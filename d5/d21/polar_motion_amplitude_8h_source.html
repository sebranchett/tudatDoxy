65  Eigen::VectorXd getParameterValue( )

70  ->getXpolarMotionCoefficients();
71  std::map< double, std::pair< double, double > > yPolarMotionCoefficients = rotationModel_->getPlanetaryOrientationAngleCalculator()
72  ->getYpolarMotionCoefficients();
73 
74  if ( xPolarMotionCoefficients.size() != yPolarMotionCoefficients.size() ){
75  throw std::runtime_error( "Error, unconsistent sizes when comparing x and y polar motion coefficients" );
76  }
77  else {
78 
79  for( std::map< double, std::pair< double, double > >::iterator coefficientsIterator = xPolarMotionCoefficients.begin( );
80  coefficientsIterator != xPolarMotionCoefficients.end( ); coefficientsIterator++ )
81  {
82  // Retrieve x polar motion coefficients
83  parameterValues.push_back( coefficientsIterator->second.first );
84  parameterValues.push_back( coefficientsIterator->second.second );
85 
86  // Retrieve y polar motion coefficients
87  parameterValues.push_back( yPolarMotionCoefficients[ coefficientsIterator->first ].first );
88  parameterValues.push_back( yPolarMotionCoefficients[ coefficientsIterator->first ].second );

109  // Retrieve current polar motion coefficients

117 
118  int currentCoefficientIndex = 0;
119 
120  for ( std::map< double, std::pair< double, double > >::iterator oldCoefficientIterator = oldXpolarMotionAmplitudeCoefficients.begin() ;
121  oldCoefficientIterator != oldXpolarMotionAmplitudeCoefficients.end() ; oldCoefficientIterator++ ){
122 
123  xPolarMotionAmplitudeCoefficients[ oldCoefficientIterator->first ]
124  = std::make_pair( parameterValue[ currentCoefficientIndex ], parameterValue[ currentCoefficientIndex + 1 ] );
125 
126  currentCoefficientIndex += 2;
127 
128  yPolarMotionAmplitudeCoefficients[ oldCoefficientIterator->first ]
129  = std::make_pair( parameterValue[ currentCoefficientIndex ], parameterValue[ currentCoefficientIndex + 1 ] );
130 
131  currentCoefficientIndex += 2;
132 
133  }
134 
135  // Reset x polar motion amplitude coefficients.
136  rotationModel_->getPlanetaryOrientationAngleCalculator()->resetXpolarMotionCoefficients( xPolarMotionAmplitudeCoefficients);
137 
138  // Reset y polar motion amplitude coefficients.
139  rotationModel_->getPlanetaryOrientationAngleCalculator()->resetYpolarMotionCoefficients( yPolarMotionAmplitudeCoefficients);
140 
141  }
