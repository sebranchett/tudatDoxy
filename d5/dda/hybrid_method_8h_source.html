33  HybridMethod(
34  const Eigen::Vector6d& stateAtDeparture,
35  const Eigen::Vector6d& stateAtArrival,
36  const double maximumThrust,
37  const double specificImpulse,
38  const double timeOfFlight,
39  std::shared_ptr< simulation_setup::OptimisationSettings > optimisationSettings,
40  const std::pair< double, double > initialAndFinalMEEcostatesBounds = std::make_pair( - 10.0, 10.0 ) ):
41  LowThrustLeg( stateAtDeparture, stateAtArrival, timeOfFlight ),
42  maximumThrust_( maximumThrust ),
43  specificImpulse_( specificImpulse ),
44  optimisationSettings_( optimisationSettings ),
45  initialAndFinalMEEcostatesBounds_( initialAndFinalMEEcostatesBounds )
46  {
47 
48  // Store initial spacecraft mass.
49  initialSpacecraftMass_ = bodyMap_[ bodyToPropagate_ ]->getBodyMass();
50 
51  // Convert the thrust model proposed as initial guess into simplified thrust model adapted to the hybrid method.
52  if ( optimisationSettings_->initialGuessThrustModel_.first.size( ) != 0 )
53  {
54  if ( optimisationSettings_->initialGuessThrustModel_.first.size( ) != 10 )
55  {
56  throw std::runtime_error( "Error when providing an initial guess for hybrid method, vector size"
57  "unconsistent with the 5 initial and 5 final MEE costate values which are required." );
58  }
59  else
60  {
61  initialGuessThrustModel_.first = optimisationSettings_->initialGuessThrustModel_.first;
62  }
63  }
64  else
65  {
66  initialGuessThrustModel_.first = std::vector< double>( );
67  }
68  initialGuessThrustModel_.second = optimisationSettings_->initialGuessThrustModel_.second;
69 
70  // Perform optimisation
71  std::pair< std::vector< double >, std::vector< double > > bestIndividual = performOptimisation( );
72  championFitness_ = bestIndividual.first;
73  championDesignVariables_ = bestIndividual.second;
74 
75 
76  // Transform vector of design variables into 3D vector of throttles.
77  Eigen::VectorXd initialCostates; initialCostates.resize( 5 );
78  Eigen::VectorXd finalCostates; finalCostates.resize( 5 );
79  for ( unsigned int i = 0 ; i < 5 ; i++ )
80  {
81  initialCostates[ i ] = championDesignVariables_[ i ];
82  finalCostates[ i ] = championDesignVariables_[ i + 5 ];
83  }
84 
85  bodyMap_[ bodyToPropagate_ ]->setConstantBodyMass( initialSpacecraftMass_ );
86 
87  // Create Sims-Flanagan leg from the best optimisation individual.
88  hybridMethodModel_ = std::make_shared< HybridMethodModel >(
89  stateAtDeparture_, stateAtArrival_, initialCostates, finalCostates, maximumThrust_, specificImpulse_, timeOfFlight_ );
90 
91  }
