29 template< typename StateScalarType >
30 void VariationalEquations::getBodyInitialStatePartialMatrix(
31  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic >& stateTransitionAndSensitivityMatrices,
32  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > currentMatrixDerivative )
33 {
34  setBodyStatePartialMatrix( );
35 
36  // Add partials of body positions and velocities.
37  currentMatrixDerivative.block( 0, 0, totalDynamicalStateSize_, numberOfParameterValues_ ) =
38  ( variationalMatrix_.template cast< StateScalarType >( ) * stateTransitionAndSensitivityMatrices );
39 
40  if( couplingEntriesToSuppress_ > 0 )
41  {
42  int numberOfStaticParameters = numberOfParameterValues_ - totalDynamicalStateSize_;
43  int numberOfUncoupledEntries = totalDynamicalStateSize_ - couplingEntriesToSuppress_;
44 
45  currentMatrixDerivative.block( couplingEntriesToSuppress_, totalDynamicalStateSize_, numberOfUncoupledEntries, numberOfStaticParameters ) =
46  variationalMatrix_.template cast< StateScalarType >( ).block(
47  couplingEntriesToSuppress_, couplingEntriesToSuppress_,
48  numberOfUncoupledEntries, numberOfUncoupledEntries ) *
49  stateTransitionAndSensitivityMatrices.block(
50  couplingEntriesToSuppress_, totalDynamicalStateSize_, numberOfUncoupledEntries, numberOfStaticParameters );
51  }

55 void VariationalEquations::setBodyStatePartialMatrix( )
56 {
57  // Initialize partial matrix
58  variationalMatrix_.setZero( );
59 
60  if( dynamicalStatesToEstimate_.count( propagators::translational_state ) > 0 )
61  {
62  int startIndex = stateTypeStartIndices_.at( propagators::translational_state );
63  for( unsigned int i = 0; i < dynamicalStatesToEstimate_.at( propagators::translational_state ).size( ); i++ )
64  {
65  variationalMatrix_.block( startIndex + i * 6, startIndex + i * 6 + 3, 3, 3 ).setIdentity( );
66  }
67  }
68 
69  if( dynamicalStatesToEstimate_.count( propagators::rotational_state ) > 0 )
70  {
71  Eigen::VectorXd rotationalStates = currentStatesPerTypeInConventionalRepresentation_.at(
72  propagators::rotational_state );
73 
74  int startIndex = stateTypeStartIndices_.at( propagators::rotational_state );
75  for( unsigned int i = 0; i < dynamicalStatesToEstimate_.at( propagators::rotational_state ).size( ); i++ )
76  {
77  variationalMatrix_.block( startIndex + i * 7, startIndex + i * 7 , 4, 4 ) =
78  getQuaterionToQuaternionRateMatrix( rotationalStates.segment( 7 * i + 4, 3 ) );
79  variationalMatrix_.block( startIndex + i * 7, startIndex + i * 7 + 4, 4, 3 ) =
80  getAngularVelocityToQuaternionRateMatrix( rotationalStates.segment( 7 * i, 4 ) );
81  }
82  }
83 
84  // Iterate over all bodies undergoing accelerations for which initial condition is to be estimated.
85  for( std::map< IntegratedStateType, std::vector< std::multimap< std::pair< int, int >,
86  std::function< void( Eigen::Block< Eigen::MatrixXd > ) > > > >::iterator
87  typeIterator = statePartialList_.begin( ); typeIterator != statePartialList_.end( ); typeIterator++ )
88  {
89  int startIndex = stateTypeStartIndices_.at( typeIterator->first );
90  int currentStateSize = getSingleIntegrationSize( typeIterator->first );
91  int entriesToSkipPerEntry = currentStateSize - getGeneralizedAccelerationSize( typeIterator->first );
92 
93  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
94  {
95  // Iterate over all bodies exerting an acceleration on this body.
96  for( statePartialIterator_ = typeIterator->second.at( i ).begin( );
97  statePartialIterator_ != typeIterator->second.at( i ).end( );
98  statePartialIterator_++ )
99  {
100  statePartialIterator_->second(
101  variationalMatrix_.block(
102  startIndex + entriesToSkipPerEntry + i* currentStateSize, statePartialIterator_->first.first,
103  currentStateSize - entriesToSkipPerEntry, statePartialIterator_->first.second ) );
104 
105  }
106  }
107  }
108 
109  for( unsigned int i = 0; i < statePartialAdditionIndices_.size( ); i++ )
110  {
111  variationalMatrix_.block( 0, statePartialAdditionIndices_.at( i ).second, totalDynamicalStateSize_, 3 ) +=
112  variationalMatrix_.block( 0, statePartialAdditionIndices_.at( i ).first, totalDynamicalStateSize_, 3 );
113  }
114 
115  for( unsigned int i = 0; i < inertiaTensorsForMultiplication_.size( ); i++ )
116  {
117  variationalMatrix_.block( inertiaTensorsForMultiplication_.at( i ).first, 0, 3, totalDynamicalStateSize_ ) =
118  ( inertiaTensorsForMultiplication_.at( i ).second( ).inverse( ) ) *
119  variationalMatrix_.block( inertiaTensorsForMultiplication_.at( i ).first, 0, 3, totalDynamicalStateSize_ ).eval( );
120  }

143 void VariationalEquations::setStatePartialFunctionList( )
144 {
145  std::pair< std::function< void( Eigen::Block< Eigen::MatrixXd > ) >, int > currentDerivativeFunction;
146 
147 
148  // Iterate over all state types
149  for( std::map< propagators::IntegratedStateType,
150  orbit_determination::StateDerivativePartialsMap >::iterator
151  stateDerivativeTypeIterator_ = stateDerivativePartialList_.begin( );
152  stateDerivativeTypeIterator_ != stateDerivativePartialList_.end( );
153  stateDerivativeTypeIterator_++ )
154  {
155  // Iterate over all bodies undergoing 'accelerations' for which initial state is to be estimated.
156  for( unsigned int i = 0; i < stateDerivativeTypeIterator_->second.size( ); i++ )
157  {
158  std::multimap< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
159  currentBodyPartialList;
160 
161  // Iterate over all 'accelerations' from single body on other single body
162  for( unsigned int j = 0; j < stateDerivativeTypeIterator_->second.at( i ).size( ); j++ )
163  {
164  for( std::map< propagators::IntegratedStateType,
165  std::vector< std::pair< std::string, std::string > > >::iterator
166  estimatedStateIterator = dynamicalStatesToEstimate_.begin( );
167  estimatedStateIterator != dynamicalStatesToEstimate_.end( );
168  estimatedStateIterator++ )
169  {
170  // Iterate over all bodies to see if body exerting acceleration is also to be estimated (cross-terms)
171  for( unsigned int k = 0; k < estimatedStateIterator->second.size( ); k++ )
172  {
173  currentDerivativeFunction = stateDerivativeTypeIterator_->second.at( i ).at( j )->
174  getDerivativeFunctionWrtStateOfIntegratedBody(
175  estimatedStateIterator->second.at( k ), estimatedStateIterator->first );
176 
177  // If function is not-empty: add to list.
178  if( currentDerivativeFunction.second != 0 )
179  {
180  currentBodyPartialList.insert(
181  std::make_pair(
182  std::make_pair( k * getSingleIntegrationSize( estimatedStateIterator->first ) +
183  stateTypeStartIndices_.at( estimatedStateIterator->first ),
184  getSingleIntegrationSize( estimatedStateIterator->first ) ),
185  currentDerivativeFunction.first ) );
186  }
187  }
188  }
189  }
190  statePartialList_[ stateDerivativeTypeIterator_->first ].push_back( currentBodyPartialList );
191  }
192  }
193 }
