23 Eigen::Matrix< double, 1, 6 > calculateNumericalPartialOfTrueAnomalyWrtState(
24  const Eigen::Vector6d& cartesianElements, const double gravitationalParameter,
25  const Eigen::Vector6d& cartesianStateElementPerturbations )
26 {
27  using namespace tudat::orbital_element_conversions;
28 
29  // Initialize partial to zero
30  Eigen::Matrix< double, 1, 6 > partial = Eigen::Matrix< double, 1, 6 >::Zero( );
31 
32  // Iterate over all six elements and calculate partials.
33  Eigen::Vector6d perturbedCartesianElements;
34  double upPerturbedTrueAnomaly, downPerturbedTrueAnomaly;
35  for( int i = 0; i < 6; i++ )
36  {
37  // Calculate true anomaly at up-perturbed cartesian element i
38  perturbedCartesianElements = cartesianElements;
39  perturbedCartesianElements( i ) += cartesianStateElementPerturbations( i );
40  upPerturbedTrueAnomaly = convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter )( trueAnomalyIndex );
41 
42  // Check validity of result
43  if( upPerturbedTrueAnomaly != upPerturbedTrueAnomaly )
44  {
45  std::cerr << "Error 1 in partial of true anomaly wrt cartesian state, element" << i
46  << ", keplerian state: " << std::endl
47  << convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter ).transpose( ) << std::endl
48  << perturbedCartesianElements.transpose( ) << std::endl;
49  }
50 
51  // Calculate true anomaly at down-perturbed cartesian element i
52  perturbedCartesianElements = cartesianElements;
53  perturbedCartesianElements( i ) -= cartesianStateElementPerturbations( i );
54  downPerturbedTrueAnomaly = convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter )( trueAnomalyIndex );
55 
56  // Check validity of result
57  if( downPerturbedTrueAnomaly != downPerturbedTrueAnomaly )
58  {
59  std::cerr << "Error 2 in partial of true anomaly wrt cartesian state, element" << i
60  << ", keplerian state: " << std::endl
61  << convertCartesianToKeplerianElements( perturbedCartesianElements, gravitationalParameter ).transpose( ) << std::endl
62  << perturbedCartesianElements.transpose( ) << std::endl;
63  }
64 
65  // Calculate central difference of term i
66  partial( 0, i ) = ( upPerturbedTrueAnomaly - downPerturbedTrueAnomaly ) / ( 2.0 * cartesianStateElementPerturbations( i ) );
67 
68  // Check validity of result
69  if( partial( 0, i ) != partial( 0, i ) )
70  {
71  std::cerr << "Error 2 in partial of true anomaly wrt cartesian state, element" << i << std::endl;
72  }
73  }
74 
75  return partial;
76 }

122 void EmpiricalAccelerationPartial::update( const double currentTime )
123 {
124  if( !( currentTime_ == currentTime ) )
125  {
126 
127  using namespace tudat::basic_mathematics;
128  using namespace tudat::linear_algebra;
129 
130  // Get current state and associated data.
131  Eigen::Vector6d currentState = empiricalAcceleration_->getCurrentState( );
132  Eigen::Vector3d angularMomentumVector = Eigen::Vector3d( currentState.segment( 0, 3 ) ).cross(
133  Eigen::Vector3d( currentState.segment( 3, 3 ) ) );
134  Eigen::Vector3d crossVector = angularMomentumVector.cross( Eigen::Vector3d( currentState.segment( 0, 3 ) ) );
135 
136  // Compute constituent geometric partials.
137  Eigen::Matrix3d normPositionWrtPosition = calculatePartialOfNormalizedVector( Eigen::Matrix3d::Identity( ), currentState.segment( 0, 3 ) );
138  Eigen::Matrix3d angularMomentumWrtPosition = -getCrossProductMatrix( currentState.segment( 3, 3 ) );
139  Eigen::Matrix3d angularMomentumWrtVelocity = getCrossProductMatrix( currentState.segment( 0, 3 ) );
140  Eigen::Matrix3d normAngularMomentumWrtPosition = calculatePartialOfNormalizedVector( angularMomentumWrtPosition, angularMomentumVector );
141  Eigen::Matrix3d normAngularMomentumWrtVelocity = calculatePartialOfNormalizedVector( angularMomentumWrtVelocity, angularMomentumVector );
142  Eigen::Matrix3d crossVectorWrtPosition = getCrossProductMatrix( angularMomentumVector ) -
143  angularMomentumWrtVelocity * angularMomentumWrtPosition;
144  Eigen::Matrix3d crossVectorWrtVelocity = -getCrossProductMatrix( currentState.segment( 0, 3 ) ) *
145  getCrossProductMatrix( currentState.segment( 0, 3 ) );
146  Eigen::Matrix3d normCrossVectorWrtPosition = calculatePartialOfNormalizedVector( crossVectorWrtPosition, crossVector );
147  Eigen::Matrix3d normCrossVectorWrtVelocity = calculatePartialOfNormalizedVector( crossVectorWrtVelocity, crossVector );
148 
149  // Retrieve current empirical acceleration
150  Eigen::Vector3d localAcceleration = empiricalAcceleration_->getCurrentLocalAcceleration( );
151 
152  // Compute partial derivative contribution of derivative of rotation matrix from RSW to inertial frame
153  currentPositionPartial_ = localAcceleration.x( ) * normPositionWrtPosition + localAcceleration.y( ) * normCrossVectorWrtPosition +
154  localAcceleration.z( ) * normAngularMomentumWrtPosition;
155  currentVelocityPartial_ = localAcceleration.y( ) * normCrossVectorWrtVelocity + localAcceleration.z( ) * normAngularMomentumWrtVelocity;
156 
157  // Compute partial derivative contribution of derivative of true anomaly
158  Eigen::Matrix< double, 1, 6 > localTrueAnomalyPartial = calculateNumericalPartialOfTrueAnomalyWrtState(
159  empiricalAcceleration_->getCurrentState( ),
160  empiricalAcceleration_->getCurrentGravitationalParameter( ), cartesianStateElementPerturbations );
161  Eigen::Matrix< double, 1, 6 > trueAnomalyPartial = localTrueAnomalyPartial;
162  currentPositionPartial_ += empiricalAcceleration_->getCurrentToInertialFrame( ) * (
163  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::sine_empirical ) *
164  std::cos( empiricalAcceleration_->getCurrentTrueAnomaly( ) ) -
165  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::cosine_empirical ) *
166  std::sin( empiricalAcceleration_->getCurrentTrueAnomaly( ) )
167  ) * trueAnomalyPartial.block( 0, 0, 1, 3 );
168  currentVelocityPartial_ +=
169  ( empiricalAcceleration_->getCurrentToInertialFrame( ) * (
170  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::sine_empirical ) *
171  std::cos( empiricalAcceleration_->getCurrentTrueAnomaly( ) ) -
172  empiricalAcceleration_->getCurrentAccelerationComponent( basic_astrodynamics::cosine_empirical ) *
173  std::sin( empiricalAcceleration_->getCurrentTrueAnomaly( ) ) ) ) * trueAnomalyPartial.block( 0, 3, 1, 3 );
174  currentTime_ = currentTime;
175 
176  // Check output.
177  if( currentPositionPartial_ != currentPositionPartial_ )
178  {
179  throw std::runtime_error( "Error, empirical acceleration position partials are NaN" );
180  }
181  if( currentVelocityPartial_ != currentVelocityPartial_ )
182  {
183  throw std::runtime_error( "Error, empirical acceleration velocity partials are NaN" );
184  }
185  }
186 }

190  std::shared_ptr< estimatable_parameters::ArcWiseEmpiricalAccelerationCoefficientsParameter > parameter,
191  Eigen::MatrixXd& partialDerivativeMatrix )

196  wrtEmpiricalAccelerationCoefficientFromIndices(

198 
199  // Retrieve arc of current time.
200  std::shared_ptr< interpolators::LookUpScheme< double > > currentArcIndexLookUp =
201  parameter->getArcTimeLookupScheme( );

212 void EmpiricalAccelerationPartial::wrtEmpiricalAccelerationCoefficientFromIndices(
213  const int numberOfAccelerationComponents,
214  const std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >& accelerationIndices,
215  Eigen::MatrixXd& partial )
216 {
217  // Retrieve rotation matrix to inertial frame
218  Eigen::Matrix3d rotationMatrix = Eigen::Matrix3d( empiricalAcceleration_->getCurrentToInertialFrame( ) );
219 
220  // Initialize partial derivatives
221  partial = Eigen::Matrix< double, 3, Eigen::Dynamic >::Zero( 3, numberOfAccelerationComponents );
222 
223  // Iterate over all terms, and set partial
224  int currentIndex = 0;
225  double multiplier = 0.0;
226  for( std::map< basic_astrodynamics::EmpiricalAccelerationFunctionalShapes, std::vector< int > >::const_iterator
227  indexIterator = accelerationIndices.begin( );
228  indexIterator != accelerationIndices.end( ); indexIterator++ )
229  {
230  // Get multiplier associated with functional shape of empirical acceleration
231  switch( indexIterator->first )
232  {
233  case basic_astrodynamics::constant_empirical:
234  multiplier = 1.0;
235  break;
236  case basic_astrodynamics::sine_empirical:
237  multiplier = std::sin( empiricalAcceleration_->getCurrentTrueAnomaly( ) );
238  break;
239  case basic_astrodynamics::cosine_empirical:
240  multiplier = std::cos( empiricalAcceleration_->getCurrentTrueAnomaly( ) );
241  break;
242  default:
243  throw std::runtime_error(
244  "Error when calculating partial w.r.t. empirical accelerations, could not find functional shape " );
245  }
246 
247  // Set partial value for current component and shape
248  for( unsigned int i = 0; i < indexIterator->second.size( ); i++ )
249  {
250  partial.block( 0, currentIndex, 3, 1 ) = multiplier * rotationMatrix.block ( 0, indexIterator->second.at( i ), 3, 1 );
251  currentIndex++;
252  }
253 
254  }
255 }
256 
257 }
