177  bool isStateDerivativeDependentOnIntegratedAdditionalStateTypes(

266  void wrtGravitationalParameterOfCentralBody( Eigen::MatrixXd& partialMatrix, const int addPartial = 0 )
267  {
268  if( gravitationalParameterFunction_( ) != 0.0 )
269  {
270  if( addPartial == 0 )
271  {
272  partialMatrix = accelerationFunction_( ) / gravitationalParameterFunction_( );
273  }
274  else if( addPartial == 1 )
275  {
276  partialMatrix += accelerationFunction_( ) / gravitationalParameterFunction_( );
277  }
278  else if( addPartial == -1 )
279  {
280  partialMatrix -= accelerationFunction_( ) / gravitationalParameterFunction_( );
281  }
282  else
283  {
284  throw std::runtime_error( "Error when adding partial of spherical harmonic acceleration w.r.t mu, inout is inconsistent" );
285  }
286  }
287  else
288  {
289  throw std::runtime_error( "Error cannot compute partial of spherical harminic gravity w.r.t mu for zero value" );
290  }
291  }
