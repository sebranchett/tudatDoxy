289 std::shared_ptr< BodySettings > getDefaultSingleBodySettings(
290  const std::string& bodyName,
291  const double initialTime,
292  const double finalTime,
293  const double timeStep )
294 {
295  std::shared_ptr< BodySettings > singleBodySettings = std::make_shared< BodySettings >( );
296 
297  // Get default settings for each of the environment models in the body.
298  singleBodySettings->atmosphereSettings = getDefaultAtmosphereModelSettings(
299  bodyName, initialTime, finalTime );
300  singleBodySettings->rotationModelSettings = getDefaultRotationModelSettings(
301  bodyName, initialTime, finalTime );
302 
303  if( ( !( initialTime == initialTime ) && ( finalTime == finalTime ) ) ||
304  ( ( initialTime == initialTime ) && !( finalTime == finalTime ) ) )
305  {
306  throw std::runtime_error( "Error when getting default body settings, only one input time is NaN" );
307  }
308  else if( !( initialTime == initialTime ) )
309  {
310  singleBodySettings->ephemerisSettings = getDefaultEphemerisSettings(
311  bodyName );
312  }
313  else
314  {
315  singleBodySettings->ephemerisSettings = getDefaultEphemerisSettings(
316  bodyName, initialTime, finalTime, timeStep );
317  }
318  singleBodySettings->gravityFieldSettings = getDefaultGravityFieldSettings(
319  bodyName, initialTime, finalTime );
320  singleBodySettings->shapeModelSettings = getDefaultBodyShapeSettings(
321  bodyName, initialTime, finalTime );
322 
323  return singleBodySettings;
324 }
