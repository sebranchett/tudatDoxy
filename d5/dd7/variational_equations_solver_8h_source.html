429 template< typename StateScalarType = double, typename TimeType = double >
430 bool checkMultiArcPropagatorSettingsAndParameterEstimationConsistency(
431  const std::shared_ptr< MultiArcPropagatorSettings< StateScalarType > > propagatorSettings,
432  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< StateScalarType > > parametersToEstimate,
433  const std::vector< double > arcStartTimes )
434 {
435  bool isInputConsistent = 1;
436 
437  // Get list of objets and associated bodies to estimate initial arc-wise translational states
438  typedef std::map< std::string, std::shared_ptr< estimatable_parameters::EstimatableParameter<
439  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > > > ArcWiseParameterList;
440  ArcWiseParameterList estimatedBodies = estimatable_parameters::getListOfBodiesWithTranslationalMultiArcStateToEstimate(
441  parametersToEstimate );
442 
443  // Iterate over all parameters and check consistency
444  for( typename ArcWiseParameterList::const_iterator parameterIterator = estimatedBodies.begin( ); parameterIterator !=
445  estimatedBodies.end( ); parameterIterator++ )
446  {
447  // Get arc start times of current parameter
448  std::vector< double > parameterArcStartTimes =
449  std::dynamic_pointer_cast< estimatable_parameters::
450  ArcWiseInitialTranslationalStateParameter< StateScalarType > >(
451  parameterIterator->second )->getArcStartTimes( );
452 
453  // Check if arc times are (almost) exactly the same
454  if( arcStartTimes.size( ) != parameterArcStartTimes.size( ) )
455  {
456  isInputConsistent = false;
457  throw std::runtime_error( "Error, arc times for " + parameterIterator->first + " have incompatible size with estimation" );
458  }
459  else
460  {
461  for( unsigned int i = 0; i < arcStartTimes.size( ); i++ )
462  {
463  if( std::fabs( arcStartTimes.at( i ) - parameterArcStartTimes.at( i ) ) >
464  std::max( 4.0 * parameterArcStartTimes.at( i ) * std::numeric_limits< double >::epsilon( ), 1.0E-12 ) )
465  {
466  isInputConsistent = false;
467  throw std::runtime_error( "Error, arc time for " + parameterIterator->first + " is incompatible with estimation" );
468  }
469  }
470  }
471  }
472 
473  std::map< IntegratedStateType, std::vector< std::string > > propagatedStateTypes;
474 
475  // Iterate over each arc in propagator settings and check consistency
476  for( int arc = 0; arc < propagatorSettings->getNmberOfArcs( ); arc++ )
477  {
478  // Check type of dynamics
479  switch( propagatorSettings->getSingleArcSettings( ).at( arc )->getStateType( ) )
480  {
481  case translational_state:
482  {
483  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > translationalPropagatorSettings =
484  std::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >(
485  propagatorSettings->getSingleArcSettings( ).at( arc ) );
486 
487  // Retrieve estimated and propagated translational states, and check equality.
488  std::vector< std::string > propagatedBodies = translationalPropagatorSettings->bodiesToIntegrate_;
489  if( arc == 0 )
490  {
491  propagatedStateTypes[ translational_state ] = propagatedBodies;
492  }
493  else
494  {
495  if( propagatedBodies.size( ) != propagatedStateTypes.at( translational_state ).size( ) )
496  {
497  isInputConsistent = false;
498  std::string errorMessage = "Error, propagated body vector sizes are inconsistent between arcs " +
499  std::to_string( propagatedBodies.size( ) ) + " " +
500  std::to_string( propagatedStateTypes[ translational_state ].size( ) ) +
501  " when checking multi-arc estimation/propagation consistency";
502  throw std::runtime_error( errorMessage );
503  }
504  else
505  {
506  for( unsigned int i = 0; i < propagatedBodies.size( ); i++ )
507  {
508  if( propagatedBodies.at( i ) != propagatedStateTypes[ translational_state ].at( i ) )
509  {
510  isInputConsistent = false;
511  std::string errorMessage = "Error, propagated body vector sizes are inconsistent between arcs at index " +
512  std::to_string( i ) + " " +
513  std::string( propagatedBodies.at( i ) ) + " " +
514  std::string( propagatedStateTypes[ translational_state ].at( i ) ) +
515  " when checking multi-arc estimation/propagation consistency";
516  throw std::runtime_error( errorMessage );
517  }
518  }
519 
520  }
521  }
522  break;
523  }
524  default:
525  std::string errorMessage = "Error, cannot yet check consistency of multi-arc propagator settings for type " +
526  std::to_string( propagatorSettings->getSingleArcSettings( ).at( arc )->getStateType( ) );
527  throw std::runtime_error( errorMessage );
528  }
529  }
530 
531  if( estimatedBodies.size( ) != propagatedStateTypes[ translational_state ].size( ) )
532  {
533  isInputConsistent = false;
534  std::string errorMessage = "Error, propagated body vector sizes are inconsistent " +
535  std::to_string( propagatedStateTypes[ translational_state ].size( ) ) + " " +
536  std::to_string( estimatedBodies.size( ) ) +
537  " when checking multi-arc estimation/propagation consistency";
538  throw std::runtime_error( errorMessage );
539 
540  for( unsigned int i = 0; i < propagatedStateTypes[ translational_state ].size( ); i++ )
541  {
542  if( estimatedBodies.count( propagatedStateTypes[ translational_state ].at( i ) ) == 0 )
543  {
544  isInputConsistent = false;
545  std::string errorMessage = "Error, propagated body " +
546  std::string( propagatedStateTypes[ translational_state ].at( i ) ) + " " +
547  " not found in estimated body list when checking multi-arc estimation/propagation consistency";
548  throw std::runtime_error( errorMessage );
549  }
550  }
551  }
552 
553  return isInputConsistent;
554 }
